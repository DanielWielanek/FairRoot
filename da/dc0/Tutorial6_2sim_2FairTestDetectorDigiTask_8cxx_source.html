66 {
67 
68  fDigiArray->Delete();

70  // fill the map

72  for (int ipnt = 0; ipnt < fPointArray->GetEntries(); ipnt++)
73  {
74  FairTestDetectorPoint* point = (FairTestDetectorPoint*)fPointArray->At(ipnt);

77  continue;

79 
80  Int_t xPad = CalcPad(point->GetX(), point->GetXOut());
81  Int_t yPad = CalcPad(point->GetY(), point->GetYOut());
82  Int_t zPad = CalcPad(point->GetZ(), point->GetZOut());
83 
84  Double_t timestamp = CalcTimeStamp(point->GetTime());
85 
86  FairTestDetectorDigi* digi = new ((*fDigiArray)[ipnt]) FairTestDetectorDigi(xPad, yPad, zPad, timestamp);
87  if (fTimeResolution > 0)
88  {
89  digi->SetTimeStampError(fTimeResolution / TMath::Sqrt(fTimeResolution));
90  }
91  else
92  {
93  digi->SetTimeStampError(0);
94  }
95 
96  digi->SetLink(FairLink("FairTestDetectorPoint", ipnt));
97  }
