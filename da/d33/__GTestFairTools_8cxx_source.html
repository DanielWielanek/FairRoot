179  bool CheckVerboseOutput(std::string logLevel, std::string outputString,
180  std::string verboseLevel, std::string resultString
181  ) {
182  // Example Output for logging level INFO and different verbosity levels
183  // verbosity Level:Output
184  // LOW :[INFO ] I am here.
185  // MEDIUM :[INFO ] [_TestFairTools.cxx::LogNoArguments:78] I am here.
186  // HIGH :[INFO ] [05.12.2011 12:00:10] [_TestFairTools.cxx::LogNoArguments:78] I am here.
187  std::string regexString = "\\[" + logLevel + "[ ]*][ ]";
188  if (verboseLevel.compare("HIGH") == 0) {
189  regexString = regexString + "\\[\\d{2}\\.\\d{2}\\.\\d{4}[ ]\\d{2}:\\d{2}:\\d{2}][ ]";
190  regexString = regexString + "\\[.*::.*:\\d{2}][ ]";
191  }
192  if (verboseLevel.compare("MEDIUM") == 0) {
193  regexString = regexString + "\\[.*::.*:\\d{2}][ ]";
194  }
195  regexString = regexString + outputString + ".*";
196 
197  const boost::regex e(regexString);
198  return regex_match(resultString, e);
199  }
