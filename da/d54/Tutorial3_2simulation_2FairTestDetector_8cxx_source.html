79 Bool_t FairTestDetector::ProcessHits(FairVolume* vol)
80 {
83  // Set parameters at entrance of volume. Reset ELoss.
84  if (gMC->IsTrackEntering())
85  {
86  fELoss = 0.;
87  fTime = gMC->TrackTime() * 1.0e09;
88  fLength = gMC->TrackLength();
89  gMC->TrackPosition(fPos);
90  gMC->TrackMomentum(fMom);
91  }
92 
93  // Sum energy loss for all steps in the active volume
94  fELoss += gMC->Edep();
95 
96  // Create FairTestDetectorPoint at exit of active volume
97  if (gMC->IsTrackExiting() || gMC->IsTrackStop() || gMC->IsTrackDisappeared())
98  {
99  fTrackID = gMC->GetStack()->GetCurrentTrackNumber();
100  fVolumeID = vol->getMCid();
101  gMC->TrackPosition(fPosOut);
102  gMC->TrackMomentum(fMomOut);
103  if (fELoss == 0.)
104  {
105  return kFALSE;
106  }
107  AddHit(fTrackID,
108  fVolumeID,
109  TVector3(fPos.X(), fPos.Y(), fPos.Z()),
110  TVector3(fMom.Px(), fMom.Py(), fMom.Pz()),
111  TVector3(fPosOut.X(), fPosOut.Y(), fPosOut.Z()),
112  TVector3(fMomOut.Px(), fMomOut.Py(), fMomOut.Pz()),
113  fTime,
114  fLength,
115  fELoss);
116 
117  // Increment number of FairTestDetector det points in TParticle
118  FairStack* stack = (FairStack*)gMC->GetStack();
119  stack->AddPoint(kTutDet);
120  }
121 
122  return kTRUE;
123 }

160 void FairTestDetector::ConstructGeometry()
161 {
166  FairGeoLoader* geoLoad = FairGeoLoader::Instance();
167  FairGeoInterface* geoFace = geoLoad->getGeoInterface();
168  FairTestDetectorGeo* Geo = new FairTestDetectorGeo();
169  Geo->setGeomFile(GetGeometryFileName());
170  geoFace->addGeoModule(Geo);
171 
172  Bool_t rc = geoFace->readSet(Geo);
173  if (rc)
174  {
175  Geo->create(geoLoad->getGeoBuilder());
176  }
177  TList* volList = Geo->getListOfVolumes();
178 
179  // store geo parameter
180  FairRun* fRun = FairRun::Instance();
181  FairRuntimeDb* rtdb = FairRun::Instance()->GetRuntimeDb();
182  FairTestDetectorGeoPar* par = (FairTestDetectorGeoPar*)(rtdb->getContainer("FairTestDetectorGeoPar"));
183  TObjArray* fSensNodes = par->GetGeoSensitiveNodes();
184  TObjArray* fPassNodes = par->GetGeoPassiveNodes();
185 
186  TListIter iter(volList);
187  FairGeoNode* node = NULL;
188  FairGeoVolume* aVol = NULL;
189 
190  while ((node = (FairGeoNode*)iter.Next()))
191  {
192  aVol = dynamic_cast<FairGeoVolume*>(node);
193  if (node->isSensitive())
194  {
195  fSensNodes->AddLast(aVol);
196  }
197  else
198  {
199  fPassNodes->AddLast(aVol);
200  }
201  }
202  par->setChanged();
203  par->setInputVersion(fRun->GetRunId(), 1);
204 
205  ProcessNodes(volList);
206 }
