138 void FairDbTutPar2::fill(UInt_t rid)
139 {
140  // Get Reader Meta Class
141  fParam_Reader=GetParamReader();
142 
143  // Define a Context
144  ValTimeStamp ts(rid);
145  ValCondition context(FairDbDetector::kGfi,DataType::kData,ts);
146 
147  // Activate reading for this Context
148  fParam_Reader->Activate(context, GetVersion());
149 
150  // By default use the latest row entry
151  // (Other rows would correspond to outdated versions)
152  Int_t numRows = fParam_Reader->GetNumRows();
153  if ( numRows > 1 ) { numRows = 1; }
154 
155  for (int i = 0; i < numRows; ++i) {
156  FairDbTutPar2* cgd = (FairDbTutPar2*) fParam_Reader->GetRow(i);
157  if (!cgd) { continue; }
158  fTopPitch = cgd->GetTopPitch();
159  fTopAnchor = cgd->GetTopAnchor();
160  fTopNrFE = cgd->GetNrTopFE();
161  fFeType = cgd->GetFeType();
162  }

167 void FairDbTutPar2::store(UInt_t rid)
168 {
169 
170  // Boolean IO test variable
171  Bool_t fail= kFALSE;
172 
173  // Create a unique statement on choosen DB entry
174  auto_ptr<FairDbStatement> stmtDbn(fMultConn->CreateStatement(GetDbEntry()));
175  if ( ! stmtDbn.get() ) {
176  cout << "-E- FairDbTutPar2::Store() Cannot create statement for Database_id: " << GetDbEntry()
177  << "\n Please check the FAIRDB_TSQL_* environment. Quitting ... " << endl;
178  exit(1);
179  }
180 
181  // Check if for this DB entry the table already exists.
182  // If not call the corresponding Table Definition Function
183  std::vector<std::string> sql_cmds;
184  TString atr(GetName());
185  atr.ToUpper();
186 
187  if (! fMultConn->GetConnection(GetDbEntry())->TableExists("FAIRDBTUTPAR") ) {
188  sql_cmds.push_back(FairDb::GetValDefinition("FAIRDBTUTPAR").Data());
189  sql_cmds.push_back(FairDbTutPar2::GetTableDefinition());
190  }
191 
192  // Packed SQL commands executed internally via SQL processor
193  std::vector<std::string>::iterator itr(sql_cmds.begin()), itrEnd(sql_cmds.end());
194  while( itr != itrEnd ) {
195  std::string& sql_cmd(*itr++);
196  stmtDbn->Commit(sql_cmd.c_str());
197  if ( stmtDbn->PrintExceptions() ) {
198  fail = true;
199  cout << "-E- FairDbTutPar2::Store() ******* Error Executing SQL commands *********** " << endl;
200  }
201 
202  }
203 
204  // Refresh list of tables in connected database
205  // for the choosen DB entry
206  fMultConn->GetConnection(GetDbEntry())->SetTableExists();
207 
208  // Writer Meta-Class Instance
209  fParam_Writer = GetParamWriter();
210 
211 
212  // Activate Writer Meta-Class with the proper
213  // Validity Time Interval (run_id encapsulated)
214 
215  // Writer Activate() arguments list
216  // <Arguments> <Type> <Comments>
217  //
218  // Validity Interval, ValInterval
219  // Composition , Int_t set via cont. factory
220  // Version , Int_t set via cont. factory
221  // DbEntry , Int_t set via cont. factory
222  // LogTitle , std::string set via cont. factory
223 
224  fParam_Writer->Activate(GetValInterval(rid),GetComboNo(), GetVersion(),GetDbEntry(),GetLogTitle());
225 
226  // Object to Table mapping
227  *fParam_Writer<< (*this);
228 
229  // Check for eventual IO problems
230  if ( !fParam_Writer->Close() ) {
231  fail = true;
232  cout << "-E- FairDbTutPar2::Store() ******** Cannot do IO on class: " << GetName() << endl;
233  }
