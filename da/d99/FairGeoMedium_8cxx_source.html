86 void FairGeoMedium::setNComponents(Int_t n)
87 {
88  // Sets the number of components in the material
89  if (n==0) { return; }
90  Int_t k=abs(n);
91  if (nComponents!=0 && k!=nComponents) {
92  delete [] ca;
93  delete [] cz;
94  delete [] cw;
95  nComponents=0;
96  }
97  if (nComponents==0) {
98  nComponents=k;
99  ca=new Double_t[k];
100  cz=new Double_t[k];
101  cw=new Double_t[k];
102  }
103  weightFac=(Int_t)(n/nComponents);
104 }

203 void FairGeoMedium::read(std::fstream& fin, Int_t aflag )
204 {
205  // Reads the parameters from file
206  autoflag=aflag;
207  Int_t n;
208  fin>>n;
209  setNComponents(n);
210  for(Int_t ik=0; ik<nComponents; ik++) {
211  fin>>ca[ik];
212  }
213  for(Int_t i=0; i<nComponents; i++) {
214  fin>>cz[i];
215  }
216  fin>>density;
217  if (nComponents==1) {
218  cw[0]=1.;
219  calcRadiationLength();
220  } else {
221  for(Int_t i=0; i<nComponents; i++) {
222  fin>>cw[i];
223  }
224  }
225  fin>>sensFlag>>fldFlag>>fld>>epsil ;
226  if (autoflag<1) { fin>>madfld>>maxstep>>maxde>>minstep; }
227  else {
228  //to use this feature one has to set TGeant3::SetAUTO(0), thus if the media does not
229  // defined these values one can force Geant3 to calculate them by given them a value
230  // of -1
231  madfld=-1;
232  maxstep=-1;
233  maxde=-1;
234  minstep=-1;
235  }
236  fin>>n;
237  setNpckov(n);
238  if (n>0) {
239  for(Int_t i=0; i<n; i++) {
240  fin>>ppckov[i]>>absco[i]>>effic[i]>>rindex[i];
241  }
242  }

292 Bool_t FairGeoMedium::calcRadiationLength()
293 {
294  // calculates radiation length
295  // formula in GEANT manual CONS110
296  if (cz[0]<1.e-6) { // VAKUUM$
297  radLen=1.e+16;
298  return kTRUE;
299  }
300  Double_t alpha=1/137.; // fine structure constant
301  Double_t fac=.1912821; // 4*((electron radius)**2)*(avogadro's number)
302  Double_t z, a, w, az2, fc, y, xi, x0i, amol=0., x0itot=0.;
303  if (weightFac>0) { amol=1.; }
304  else {
305  for (int i=0; i<nComponents; i++) {
306  amol+=cw[i]*ca[i];
307  if (amol==0.) {
308  Error("calcRadiationLength()","amol==0 for medium %s",fName.Data());
309  return kFALSE;
310  }
311  }
312  }
313  for (int i=0; i<nComponents; i++) {
314  z=cz[i];
315  a=ca[i];
316  if(weightFac>0) {
317  w=cw[i]/amol;
318  } else {
319  w=a*cw[i]/amol;
320  }
321  az2=alpha*alpha*z*z;
322  fc=az2 * (1./(1.+az2) + 0.20206 - 0.0369*az2 + 0.0083*az2*az2
323  - .002F*az2*az2*az2);
324  y=log(183./pow(z,1./3.)) - fc;
325  xi=(float)(log(1440./pow(z,2./3.)) / y);
326  x0i=fac*alpha/a*z*(z+xi)*y;
327  x0itot+=(x0i*w);
328  }
329  if (x0itot==0. || density==0.) {
330  Error("calcRadiationLength()","x0itot=0 or density=0 for medium %s",fName.Data());
331  return kFALSE;
332  }
333  radLen=1/density/x0itot;
334  return kTRUE;
335 }
