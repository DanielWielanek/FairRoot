308 Bool_t FairGeoNode::calcModuleTransform(FairGeoTransform& modTransform)
309 {
310  // Calculates the transformation relative to the detectors coordinate system
311  FairGeoNode* node=this;
312  modTransform=transform;
313  while (node&&!node->isModule()) {
314  FairGeoNode* pm=node->getMotherNode();
315  if (!pm) {
316  Error("calcModuleTransform","Mother volume of %s not found!",node->GetName());
317  return kFALSE;
318  }
319  modTransform.transFrom(pm->getTransform());
320  node=pm;
321  }
322  if (isTopNode()) { return kFALSE; }
323  else { return kTRUE; }

357  diff[1]=1;

360  } else {
361  diff[1]=1;
362  n++;
363  }
364  if (shape.CompareTo(rn.getShape())!=0) {
365  diff[2]=1;
366  n++;
367  }
368  Int_t np=rn.getNumPoints();
369  if (points&&nPoints==np) {
370  for (Int_t i=0; i<np; i++) {
371  FairGeoVector v1=*(getPoint(i));
372  FairGeoVector v2=*(rn.getPoint(i));
373  if (fabs(v1(0)-v2(0))>=0.001||fabs(v1(1)-v2(1))>=0.001||fabs(v1(1)-v2(1))>=0.001) {
374  diff[3]=1;
375  n++;
376  break;
377  }
378  }
379  } else {
380  diff[3]=1;
381  n++;
382  }
383  FairGeoVector v1=transform.getTransVector();
384  FairGeoVector v2=rn.getTransform().getTransVector();
385  if (fabs(v1(0)-v2(0))>=0.001||fabs(v1(1)-v2(1))>=0.001||fabs(v1(1)-v2(1))>=0.001) {
386  diff[4]=1;
387  n++;
388  }
389  FairGeoRotation r1=transform.getRotMatrix();

393  diff[5]=1;

398  if (n>0) {
