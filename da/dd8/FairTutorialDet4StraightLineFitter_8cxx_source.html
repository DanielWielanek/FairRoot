187 Bool_t FairTutorialDet4StraightLineFitter::IsGoodEvent()
188 {
189  // Check if each for the event there is maximum 1 hit per detector
190  // station. In the moment we create tracks with all hits in the
191  // event, so we have to check for this.
192  // In the end the algorithm should be able to work also with
193  // missing hits in some stations
194  FairTutorialDet4Hit* hit;
195  std::set<Int_t> detIdSet;
196  std::set<Int_t>::iterator it;
197 
198  Int_t nHits = fHits->GetEntriesFast();
199  for (Int_t iHit=0; iHit<nHits; ++iHit) {
200  hit = (FairTutorialDet4Hit*) fHits->At(iHit);
201  Int_t detId = hit->GetDetectorID();
202  it = detIdSet.find(detId);
203  if ( it == detIdSet.end() ) {
204  detIdSet.insert(detId);
205  } else {
206  // find hit in already used detector station
207  // this is not a good event
208  return kFALSE;
209  }
210  }
211  return kTRUE;
212 }
