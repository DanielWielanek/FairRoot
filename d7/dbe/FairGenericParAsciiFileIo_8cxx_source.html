73 const UChar_t *FairGenericParAsciiFileIo::readData(type t, const Char_t *format,
74  TString &line,
75  Int_t &length) {
76  // reads c-type single data and arrays
77  const Int_t st = sizeof(t);
78  const Int_t maxbuf = 8000;
79  const Int_t bufSizeExt = 10000 * st;
80  const Char_t d[] = " ";
81  Text_t buf[maxbuf];
82  TString s;
83  Int_t l = 0, bufSize = bufSizeExt;
84  UChar_t *val = new UChar_t[bufSize];
85  Ssiz_t m = line.Last('\\');
86  if (m < 0) {
87  sscanf(line.Data(), format, &t);
88  memcpy(&val[l], &t, st);
89  length = st;
90  } else {
91  do {
92  pFile->getline(buf, maxbuf);
93  if (buf[0] != '/' && buf[0] != '#') {
94  s = buf;
95  m = s.Last('\\');
96  if (m > 0) {
97  s = s(0, s.Length() - 2);
98  }
99  if ((bufSize - 1000) < l) {
100  bufSize += bufSizeExt;
101  UChar_t *va = new UChar_t[bufSize];
102  memcpy(va, val, l);
103  delete[] val;
104  val = va;
105  }
106  Char_t *ss = strtok((Char_t *)s.Data(), d);
107  while (ss != 0) {
108  sscanf(ss, format, &t);
109  memcpy(&val[l], &t, st);
110  l += st;
111  ss = strtok(NULL, d);
112  }
113  }
114  } while (buf[0] != '#' && !pFile->eof() && m > 0);
115  length = l;
116  }
117  return val;
118 }

136 Bool_t FairGenericParAsciiFileIo::readGenericSet(FairParGenericSet *pPar) {
137  // reads condition-stype parameter containers from ASCII file
138  if (!pFile) {
139  return kFALSE;
140  }
141  pFile->clear();
142  pFile->seekg(0, ios::beg);
143  Text_t *name = (Char_t *)pPar->GetName();
144 
145  if (!findContainer(name)) {
146  return kFALSE;
147  }
148 
149 
150  FairParamList *paramList = new FairParamList;
151  const Int_t maxbuf = 8000;
152  Text_t buf[maxbuf];
153  buf[0] = '\0';
154  TString s, pName, pVal, pType;
155  Ssiz_t n, m;
156  while (buf[0] != '#' && !pFile->eof()) {
157  pFile->getline(buf, maxbuf);
158  if (buf[0] != '/' && buf[0] != '#') {
159  s = buf;
160  n = s.First(':');
161  // <DB> Check if empty buffer
162  if (s.IsNull()) continue;
163  if (n == -1) {
164  Error("readCond(FairParGenericSet*)",
165  "%s:\n Missing backslash for parameter %s", name, pName.Data());
166  delete paramList;
167  return kFALSE;
168  }
169  pName = s(0, n);
170  s = s(n + 1, s.Length() - n - 1);
171  s = s.Strip(s.kLeading);
172  if (pName.CompareTo("author") == 0 ||
173  pName.CompareTo("description") == 0) {
174  m = s.Last('\\');
175  if (m <= 0) {
176  pVal = s;
177  } else {
178  pVal = s(0, m);
179  pVal += "\n";
180  while (m > 0) {
181  pFile->getline(buf, maxbuf);
182  if (buf[0] != '/') {
183  s = buf;
184  m = s.Last('\\');
185  if (m > 0) {
186  pVal += s(0, m);
187  pVal += "\n";
188  } else {
189  pVal += s;
190  }
191  }
192  }
193  }
194  if (pName.CompareTo("author") == 0) {
195  pPar->setAuthor(pVal.Data());
196  } else if (pName.CompareTo("description") == 0) {
197  pPar->setDescription(pVal.Data());
198  }
199  } else {
200  n = s.First(' ');
201  pType = s(0, n);
202  s = s(n + 1, s.Length() - n - 1);
203  s = s.Strip(s.kLeading);
204  if (pType.CompareTo("Text_t") == 0) {
205  m = s.Last('\\');
206  if (m < 0) {
207  pVal = s;
208  } else {
209  pVal = "";
210  while (m >= 0) {
211  pFile->getline(buf, maxbuf);
212  if (buf[0] != '/') {
213  s = buf;
214  m = s.Last('\\');
215  if (m > 0) {
216  pVal += s(0, m);
217  pVal += "\n";
218  } else {
219  pVal += s;
220  }
221  }
222  }
223  }
224  pVal = pVal.Strip(pVal.kLeading);
225  if (pVal.Length() > 0) {
226  paramList->add(pName.Data(), pVal.Data());
227  }
228  } else {
229  const UChar_t *val = 0;
230  Int_t length = 0;
231  if (pType.CompareTo("Int_t") == 0) {
232  Int_t v = 0;
233  val = readData(v, "%i", s, length);
234  } else if (pType.CompareTo("Float_t") == 0) {
235  Float_t v = 0.F;
236  val = readData(v, "%f", s, length);
237  } else if (pType.CompareTo("Double_t") == 0) {
238  Double_t v = 0.;
239  val = readData(v, "%lf", s, length);
240  } else if (pType.CompareTo("Char_t") == 0) {
241  Char_t v = '0';
242  val = readData(v, "%c", s, length);
243  } else {
244  Error("readCond(FairParGenericSet*)",
245  "%s:\n Parameter %s with unsupported type %s", name,
246  pName.Data(), pType.Data());
247  delete paramList;
248  return kFALSE;
249  }
250  FairParamObj *obj = new FairParamObj(pName.Data());
251  obj->setParamType(pType.Data());
252  UChar_t *pValue = obj->setLength(length);
253  memcpy(pValue, val, length);
254  paramList->getList()->Add(obj);
255  if (val) {
256  delete[] val;
257  }
258  }
259  }
260  }
261  }
262  Bool_t allFound = pPar->getParams(paramList);
263  if (allFound) {
264  pPar->setInputVersion(1, inputNumber);
265  pPar->setChanged();
266  printf("%s initialized from Ascii file\n", name);
267  } else {
268  pPar->setInputVersion(-1, inputNumber);
269  }
270  delete paramList;
271  return allFound;
272 }

274 Int_t FairGenericParAsciiFileIo::writeGenericSet(FairParGenericSet *pPar) {
275  // writes condition-stype parameter containers to ASCII file
276  if (pFile) {
277  // Int_t version=1;
278  const Text_t *name = pPar->GetName();
279  const Text_t *context = pPar->getParamContext();
280  *pFile << sepLine;
281  *pFile << "# Class: " << name << "\n# Context: " << context << "\n";
282  *pFile << sepLine;
283  *pFile << "[" << name << "]\n";
284  writeComment(pPar);
285  *pFile << "//-----------------------------------------------------------"
286  "-----------------\n";
287  FairParamList *paramList = new FairParamList;
288  pPar->putParams(paramList);
289  TList *pList = paramList->getList();
290  TIter next(pList);
291  FairParamObj *po;
292  while ((po = (FairParamObj *)next())) {
293  const Char_t *pType = po->getParamType();
294  UChar_t *pValue = po->getParamValue();
295  if (po->isBasicType() && strcmp(pType, "UChar_t") != 0) {
296  if (strcmp(pType, "Text_t") == 0) {
297  TString val((Char_t *)pValue, po->getLength());
298  val.ReplaceAll("\n", " \\\n");
299  *pFile << po->GetName() << ": " << pType << " \\\n " << val.Data()
300  << std::endl;
301  } else {
302  Int_t nParams = po->getNumParams();
303  if (nParams == 1) {
304  *pFile << po->GetName() << ": " << pType << " ";
305  } else {
306  *pFile << po->GetName() << ": " << pType << " \\\n ";
307  }
308  if (strcmp(pType, "Char_t") == 0) {
309  writeData((Char_t *)pValue, nParams);
310  } else if (strcmp(pType, "Int_t") == 0) {
311  writeData((Int_t *)pValue, nParams);
312  } else if (strcmp(pType, "Float_t") == 0) {
313  writeData((Float_t *)pValue, nParams);
314  } else if (strcmp(pType, "Double_t") == 0) {
315  writeData((Double_t *)pValue, nParams);
316  }
317  }
318  } else {
319  Error("writeCond(FairParGenericSet*)",
320  "Type %s of parameter %s not supported by ASCII I/O", pType,
321  po->GetName());
322  // version=-1;
323  }
324  }
325  *pFile << sepLine;
326  delete paramList;
327  pPar->setChanged(kFALSE);
328  return 1;
329  }
330  Error("writeCond(FairParGenericSet*)", "Output is not writable");
331  return -1;
332 }
