54 Bool_t FairGeoAsciiIo::open(const char* fname,const Text_t* status)
55 {
56  // Opens the file fname
57  close();
58  if (!file) { file=new std::fstream(); }
59  else { (file->clear()); }
60  if (!filedir.IsNull()) { filename=filedir+"/"+fname; }
61  else { filename=fname; }
62  filename=filename.Strip();
63  if (strcmp(status,"in")==0) {
64  file->open(filename,ios::in);
65  writable=kFALSE;
66  } else {
67  if (strcmp(status,"out")==0) {
68  file->open(filename,ios::in);
69  if (!isOpen()) {
70  file->close();
71  file->clear();
72  file->open(filename,ios::out);
73  writable=kTRUE;
74  } else {
75  file->close();
76  Error("open","Output file %s exists already and will not be recreated.",
77  filename.Data());
78  return kFALSE;
79  }
80  } else { Error("open","Invalid file option!"); }
81  }
82  if (file->rdbuf()->is_open()==0) {
83  Fatal("open","Failed to open file %s",filename.Data());
84  return kFALSE;
85  }
86  return kTRUE;

153 Bool_t FairGeoAsciiIo::readGeomConfig(FairGeoInterface* interface)
154 {
155  // Reads the GEANT configuration file
156  if (!isOpen() || writable || interface==0) { return kFALSE; }
157  TString buf(256);
158  TString simRefRun,historyDate;
159  Int_t k=-1;
160  while(!(*file).eof()) {
161  buf.ReadLine(*file);
162  buf=buf.Strip(buf.kBoth);
163  if (!buf.IsNull() && buf(0,2)!="//" && buf(0,1)!="*") {
164  if (buf.Contains(".geo")||buf.Contains("_gdb")) {
165  interface->addInputFile(buf.Data());
166  } else {
167  if (buf.Contains(".setup")) {
168  interface->addSetupFile(buf.Data());
169  } else {
170  if (buf.Contains("SimulRefRunDb:")) {
171  k=buf.Last(' ')+1;
172  if (k) { simRefRun=buf(k,buf.Length()-k); }
173  } else {
174  if (buf.Contains("HistoryDateDb:")) {
175  k=buf.First(' ')+1;
176  if (buf.Length()>k) {
177  historyDate=buf(k,buf.Length()-k);
178  historyDate=historyDate.Strip(historyDate.kBoth);
179  }
180  }
181  }
182  }
183  }
184  }
185  }
186  Bool_t rc=kTRUE;
187  FairGeoIo* oraIo=interface->getOraInput();
188  if (oraIo) {
189  if (historyDate.Length()>0) { rc=oraIo->setHistoryDate(historyDate.Data()); }
190  if (rc&&simRefRun.Length()>0) { rc=oraIo->setSimulRefRun(simRefRun.Data()); }
191  }
192  return rc;
193 }

195 Bool_t FairGeoAsciiIo::readDetectorSetup(FairGeoInterface* interface)
196 {
197  // Reads the detector setups, needed for create only subsets
198  if (!isOpen() || writable || interface==0) { return kFALSE; }
199  const Int_t maxbuf=256;
200  char buf[maxbuf];
201  TString s, det;
202  FairGeoSet* set=0;
203  Int_t maxModules=0, secNo=-1;
204  Int_t* mod=0;
205  const char d[]=" ";
206  while(!(*file).eof()) {
207  (*file).getline(buf,maxbuf);
208  if (strlen(buf)>=3 && buf[1]!='/') {
209  if (buf[0]=='[') {
210  set=0;
211  if (mod) {
212  delete [] mod;
213  mod=0;
214  }
215  s=buf;
216  Ssiz_t n=s.First(']');
217  det=s(1,n-1);
218  det.ToLower();
219  set=interface->findSet(det);
220  if (!set) {
221  Error("readDetectorSetup","Detector %s not found",det.Data());
222  if (mod) { delete [] mod; }
223  return kFALSE;
224  }
225  maxModules=set->getMaxModules();
226  mod=new Int_t[maxModules];
227  } else {
228  if (set&&mod) {
229  char* ss=strtok(buf,d);
230  if (ss&&strlen(ss)>3) {
231  secNo=(Int_t)(ss[3]-'0')-1;
232  for(Int_t i=0; i<maxModules&&mod; i++) {
233  ss=strtok(NULL,d);
234  if (ss) { sscanf(ss,"%i",&mod[i]); }
235  }
236  set->setModules(secNo,mod);
237  }
238  } else {
239  if (mod) { delete [] mod; }
240  return kFALSE;
241  }
242  }
243  }
244  }
245  if (mod) { delete [] mod; }
246  return kTRUE;
247 }
