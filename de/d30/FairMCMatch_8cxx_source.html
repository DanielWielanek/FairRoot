147 FairMCResult FairMCMatch::GetMCInfoForward(Int_t start, Int_t stop)
148 {
149  FairMCResult result(start, stop);

205 void FairMCMatch::FindStagesPointingToLinks(FairMultiLinkedData links, Int_t stop)
206 {
207  FairMultiLinkedData tempLinks;
208  tempLinks.SetPersistanceCheck(kFALSE);
209  for (int i = 0; i < links.GetNLinks(); i++) {
210  FairLink myLink = links.GetLink(i);
211  FairMultiLinkedData myNewLinks = FindStagesPointingToLink(myLink);
212  myNewLinks.SetPersistanceCheck(kFALSE);
213  if (myNewLinks.GetNLinks() == 0) {
214  fFinalStageML.AddLink(myLink, true);
215  } else {
216  for (int j = 0; j < myNewLinks.GetNLinks(); j++) {
217  if (myNewLinks.GetLink(j).GetType() == (Int_t)stop) {
218  fFinalStageML.AddLink(myNewLinks.GetLink(j), true);
219  } else if (myNewLinks.GetLink(j).GetType() > (Int_t)stop) {
220  fFinalStageML.AddLink(myLink, true);
221  } else {
222  tempLinks.AddLink(myNewLinks.GetLink(j), true);
223  }
224  }
225  }
226  }
227  if (tempLinks.GetNLinks() != 0) {
228  FindStagesPointingToLinks(tempLinks, stop);
229  }
230 }
