92 void PixelFitTracks::Exec(Option_t* /*opt*/) {

99  LOG(DEBUG) << "PixelFitTracks::Exec() EVENT " << fTNofEvents << " with " << fNTracks << " TRACKS" << FairLogger::endl;
100 
101  for ( Int_t itrack = 0 ; itrack < fNTracks ; itrack++ ) {
102  PixelTrack* curTrack = static_cast<PixelTrack*>(fTracks->At(itrack));
103 
104  const Int_t nofHits = curTrack->GetNofHits();
105  // Default initialize the arrays
106  Double_t* hitXPos = new Double_t[nofHits]();
107  Double_t* hitYPos = new Double_t[nofHits]();
108  Double_t* hitZPos = new Double_t[nofHits]();
109 
110  for ( Int_t ihit = 0 ; ihit < nofHits ; ihit++ ) {
111  PixelHit* curHit = static_cast<PixelHit*>(fHits->At(curTrack->GetHitIndex(ihit)));
112 
113  // LOG(INFO) << " HIT[" << curTrack->GetHitIndex(ihit) << "] = ( " << curHit->GetX() << " , " << curHit->GetY() << " , " << curHit->GetZ() << " )" << FairLogger::endl;
114  hitXPos[ihit] = curHit->GetX();
115  hitYPos[ihit] = curHit->GetY();
116  hitZPos[ihit] = curHit->GetZ();
117  }
118 
119  Double_t valX0 = 0., errX0 = 0., valAX = 0., errAX = 0.;
120  Double_t valY0 = 0., errY0 = 0., valAY = 0., errAY = 0.;
121 
122  LinearRegression(nofHits,hitZPos,hitXPos,valX0,errX0,valAX,errAX);
123  LinearRegression(nofHits,hitZPos,hitYPos,valY0,errY0,valAY,errAY);
124 
125  PixelTrack* fitTrack = new ((*fFitTracks)[fNFitTracks]) PixelTrack(valX0,valAX,valY0,valAY,
126  errX0,errAX,errY0,errAY);
127  for ( Int_t ihit = 0 ; ihit < nofHits ; ihit++ ) {
128  fitTrack->AddHitIndex(curTrack->GetHitIndex(ihit));
129  }
130 
131  fNFitTracks += 1;
132 
133  LOG(DEBUG) << "Track params: "
134  << " AX = " << curTrack->GetAX() << " += " << curTrack->GetAXErr()
135  << " X0 = " << curTrack->GetX0() << " += " << curTrack->GetX0Err() << FairLogger::endl
136  << " "

138  << " Y0 = " << curTrack->GetY0() << " += " << curTrack->GetY0Err() << FairLogger::endl;
139  LOG(DEBUG) << "Fitted params: "
140  << " AX = " << valAX << " += " << errAX
141  << " X0 = " << valX0 << " += " << errX0 << FairLogger::endl
142  << " "
143  << " AY = " << valAY << " += " << errAY
144  << " Y0 = " << valY0 << " += " << errY0 << FairLogger::endl;
145 
146  delete [] hitXPos;
147  delete [] hitYPos;
148  delete [] hitZPos;
149  }
150 
151  fTNofEvents += 1;

153  fTNofFitTracks += fNFitTracks;
154 }

157 // ----- Private method LinearRegression -------------------------------
158 Double_t PixelFitTracks::LinearRegression(Int_t nval, Double_t xval[], Double_t yval[],
159  Double_t& valA0, Double_t& errA0,
160  Double_t& valA1, Double_t& errA1 ) {
161  Double_t valN = static_cast<Double_t>(nval);
162  Double_t sumXY = 0.;
163  Double_t sumX = 0.;
164  Double_t sumY = 0.;
165  Double_t sumXX = 0.;
166  for ( Int_t ival = 0 ; ival < nval ; ival++ ) {
167  sumXY += xval[ival]*yval[ival];
168  sumX += xval[ival];
169  sumY += yval[ival];
170  sumXX += xval[ival]*xval[ival];
171  }
172  valA1 = (valN*sumXY-sumX*sumY)/(valN*sumXX-sumX*sumX);
173  valA0 = (sumY-valA1*sumX)/valN;
174  Double_t sumEE = 0.;
175  for ( Int_t ival = 0 ; ival < nval ; ival++ ) {
176  sumEE += (yval[ival]-valA0-valA1*xval[ival])*(yval[ival]-valA0-valA1*xval[ival]);
177  }
178  Double_t valS = TMath::Sqrt(sumEE/(valN-2.));
179  errA1 = valS*TMath::Sqrt(valN/(valN*sumXX-sumX*sumX));
180  errA0 = valS*TMath::Sqrt(sumXX/(valN*sumXX-sumX*sumX));
181  // cout << "A0 = " << valA0 << " +- " << errA0 << " / A1 = " << valA1 << " +- " << errA1 << " / S = " << valS << endl;
182  return valS;
183 }
