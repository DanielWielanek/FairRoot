129 void FairTutorialDet4HitProducerIdealMisalign::Exec(Option_t* opt)
130 {
131 
132  fHitArray->Clear();
133 
134  // Declare some variables
135  FairTutorialDet4Point* point = NULL;
136  Int_t detID = 0; // Detector ID
137  Int_t trackID = 0; // Track index
138  Double_t x, y, z; // Position
139  Double_t dx = 0.1; // Position error
140  Double_t tof = 0.; // Time of flight
141  TVector3 pos, dpos; // Position and error vectors
142 
143  // Loop over TofPoints
144  Int_t nHits = 0;
145  Int_t nPoints = fPointArray->GetEntriesFast();
146  for (Int_t iPoint=0; iPoint<nPoints; iPoint++) {
147  point = (FairTutorialDet4Point*) fPointArray->At(iPoint);
148  if ( ! point) { continue; }
149 
150  // Detector ID
151  detID = point->GetDetectorID();
152 
153  // MCTrack ID
154  trackID = point->GetTrackID();
155 
156  if(fDoMisalignment) {
157 
158  Float_t cosAlpha = TMath::Cos(fRotZ.At(detID));
159  Float_t sinAlpha = TMath::Sin(fRotZ.At(detID));
160 
161  // Determine hit position
162  x = (point->GetX()*cosAlpha + point->GetY()*sinAlpha)-fShiftX.At(detID);
163  y = (-point->GetX()*sinAlpha + point->GetY()*cosAlpha)-fShiftY.At(detID);
164  z = point->GetZ();
165 
166  LOG(DEBUG)<<"Pos before misalignment: "<< point->GetX() <<", "
167  << point->GetY() <<", "<< point->GetZ() <<FairLogger::endl;
168  LOG(DEBUG)<<"Pos after misalignment: "<< x <<", "
169  << y <<", "<< z <<FairLogger::endl;
170 
171  x = x + GetHitErr(0.1);
172  y = y + GetHitErr(0.1);
173 
174  LOG(DEBUG2)<<"Missallign hit by "<<fShiftX.At(detID)<<" cm in x- and "
175  << fShiftY.At(detID)<<" cm in y-direction."<<FairLogger::endl;
176 
177  // Time of flight
178  tof = point->GetTime();
179 
180  // Create new hit
181  pos.SetXYZ(x,y,z);
182  dpos.SetXYZ(dx, dx, 0.);
183  new ((*fHitArray)[nHits]) FairTutorialDet4Hit(detID, iPoint, pos, dpos);
184  nHits++;
185  } else {
186 
187  // Determine hit position
188  x = point->GetX();
189  y = point->GetY();
190  z = point->GetZ();
191 
192  LOG(INFO)<<"Position: "<<x<<", "<<y<<", "<<z<<FairLogger::endl;
193 
194  Double_t local[3] = {x, y, z};
195  Double_t global[3];
196 
197  fGeoHandler->LocalToGlobal(local, global, detID);
198 
199  x = global[0] + GetHitErr(0.1);
200  y = global[1] + GetHitErr(0.1);
201  z = global[2];
202 
203  LOG(INFO)<<"Position: "<<x<<", "<<y<<", "<<z<<FairLogger::endl;
204  LOG(INFO)<<"****"<<FairLogger::endl;
205  // Time of flight
206  tof = point->GetTime();
207 
208  // Create new hit
209  pos.SetXYZ(x,y,z);
210  dpos.SetXYZ(dx, dx, 0.);
211  new ((*fHitArray)[nHits]) FairTutorialDet4Hit(detID, iPoint, pos, dpos);
212  nHits++;
213 
214  }
215  }
216  // Event summary
217  LOG(DEBUG)<< "Create " << nHits << " TutorialDetHits out of "
218  << nPoints << " TutorilaDetPoints created." << FairLogger::endl;
219 
220 }

223 Double_t FairTutorialDet4HitProducerIdealMisalign::GetHitErr(Double_t sigma)
224 {
225  Double_t err = gRandom->Gaus(0, sigma);
226  return (TMath::Abs(err) < 3 * sigma) ? err : (err > 0) ? 3 * sigma : -3 * sigma;
227 }
