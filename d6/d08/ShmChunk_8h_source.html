35  void InitializeSegment(const std::string& op, const std::string& name, const size_t size = 0)
36  {
37  if (!fSegment)
38  {
39  try
40  {
41  if (op == "open_or_create")
42  {
43  fSegment = new boost::interprocess::managed_shared_memory(boost::interprocess::open_or_create, name.c_str(), size);
44  }
45  else if (op == "create_only")
46  {
47  fSegment = new boost::interprocess::managed_shared_memory(boost::interprocess::create_only, name.c_str(), size);
48  }
49  else if (op == "open_only")
50  {
51  int numTries = 0;
52  bool success = false;
53 
54  do
55  {
56  try
57  {
58  fSegment = new boost::interprocess::managed_shared_memory(boost::interprocess::open_only, name.c_str());
59  success = true;
60  }
61  catch (boost::interprocess::interprocess_exception& ie)
62  {
63  if (++numTries == 5)
64  {
65  LOG(ERROR) << "Could not open shared memory after " << numTries << " attempts, exiting!";
66  exit(EXIT_FAILURE);
67  }
68  else
69  {
70  LOG(DEBUG) << "Could not open shared memory segment on try " << numTries << ". Retrying in 1 second...";
71  LOG(DEBUG) << ie.what();
72 
73  boost::this_thread::sleep(boost::posix_time::milliseconds(1000));
74  }
75  }
76  }
77  while (!success);
78  }
79  else
80  {
81  LOG(ERROR) << "Unknown operation when initializing shared memory segment: " << op;
82  }
83  }
84  catch (std::exception& e)
85  {
86  LOG(ERROR) << "Exception during shared memory segment initialization: " << e.what() << ", application will now exit";
87  exit(EXIT_FAILURE);
88  }
89  }
90  }
