138 void FairRunOnline::Init()
139 {
140  LOG(INFO)<<"FairRunOnline::Init"<<FairLogger::endl;
141  if (fIsInitialized) {
142  LOG(FATAL) << "Error Init is already called before!" << FairLogger::endl;
143  exit(-1);
144  } else {
145  fIsInitialized = kTRUE;
146  }
147 
148  // FairGeoLoader* loader = new FairGeoLoader("TGeo", "Geo Loader");
149  // FairGeoInterface* GeoInterFace = loader->getGeoInterface();
150  // GeoInterFace->SetNoOfSets(ListOfModules->GetEntries());
151  // GeoInterFace->setMediaFile(MatFname.Data());
152  // GeoInterFace->readMedia();
153 
154  // Add a Generated run ID to the FairRunTimeDb
155  FairRunIdGenerator genid;
156  fRunId = genid.generateId();
157  fRtdb->addRun(fRunId);
158  fFileHeader->SetRunId(fRunId);
159  FairBaseParSet* par = dynamic_cast<FairBaseParSet*>(fRtdb->getContainer("FairBaseParSet"));
160  FairGeoParSet* geopar = dynamic_cast<FairGeoParSet*>(fRtdb->getContainer("FairGeoParSet"));
161  if (geopar) {
162  geopar->SetGeometry(gGeoManager);
163  }
164  if(fField) {
165  fField->Init();
166  fField->FillParContainer();
167  }
168  TList* containerList = fRtdb->getListOfContainers();
169  TIter next(containerList);
170  FairParSet* cont;
171  TObjArray* ContList= new TObjArray();
172  while ((cont=dynamic_cast<FairParSet*>(next()))) {
173  ContList->Add(new TObjString(cont->GetName()));
174  }
175  if (par) {
176  par->SetContListStr(ContList);
177  par->setChanged();
178  par->setInputVersion(fRunId,1);
179  }
180  if (geopar) {
181  geopar->setChanged();
182  geopar->setInputVersion(fRunId,1);
183  }
184 
185 fRootManager->WriteFileHeader(fFileHeader);
186 
187  fTask->SetParTask();
188  fRtdb->initContainers(fRunId);
189 
190  // InitContainers();
191  // --- Get event header from Run
192  fEvtHeader = dynamic_cast<FairEventHeader*> (FairRunOnline::Instance()->GetEventHeader());
193  if ( ! fEvtHeader ) {
194  LOG(FATAL) << "FairRunOnline::InitContainers:No event header in run!" << FairLogger::endl;
195  return;
196  }
197  LOG(INFO) << "FairRunOnline::InitContainers: event header at " << fEvtHeader << FairLogger::endl;
198  fRootManager->Register("EventHeader.", "Event", fEvtHeader, kTRUE);
199  fEvtHeader->SetRunId(fRunId);
200 
201  // Initialize the source
202  if(! fSource->Init()) {
203  LOG(FATAL) << "Initialization of data source failed!"
204  << FairLogger::endl;
205  exit(-1);
206  }
207 
208  // Now call the User initialize for Tasks
209  fTask->InitTask();
210 
211  // create the output tree after tasks initialisation
212  fOutFile->cd();
213  TTree* outTree =new TTree("cbmsim", "/cbmout", 99);
214  fRootManager->TruncateBranchNames(outTree, "cbmout");
215  fRootManager->SetOutTree(outTree);
216  fRootManager->WriteFolder();
217 }

261 Int_t FairRunOnline::EventLoop()
262 {
263  gSystem->IgnoreInterrupt();
264  gIsInterrupted = kFALSE;
265  signal(SIGINT, handler_ctrlc);
266 
267  fSource->Reset();
268  Int_t status = fSource->ReadEvent();
269  if(1 == status || 2 == status) {
270  return status;
271  }
272 
273 
274  Int_t tmpId = GetEventHeader()->GetRunId();
275 
276  if ( tmpId != fRunId ) {
277 // LOG(INFO) << "Call Reinit due to changed RunID" << FairLogger::endl;
278 // fRunId = tmpId;
279 // Reinit( fRunId );
280 // fTask->ReInitTask();
281  }
282 
283 
284  fRootManager->StoreWriteoutBufferData(fRootManager->GetEventTime());
285  fTask->ExecuteTask("");
286  fRootManager->Fill();
287  fRootManager->DeleteOldWriteoutBufferData();
288  fTask->FinishEvent();
289  fNevents += 1;
290  if(fGenerateHtml && 0 == (fNevents%fRefreshRate))
291  {
292  WriteObjects();
293  GenerateHtml();
294  }
295 
296  if(gIsInterrupted)
297  {
298  return 1;
299  }
300 
301  return 0;
302 }

306 void FairRunOnline::Run(Int_t nev, Int_t dummy)
307 {
308  fOutFile->cd();
309 
310  fNevents = 0;
311 
312  Int_t status;
313  if(nev < 0) {
314  while(kTRUE) {
315  status = EventLoop();
316  if(1 == status) {
317  break;
318  } else if(2 == status) {
319  continue;
320  }
321  if(gIsInterrupted)
322  {
323  break;
324  }
325  }
326  } else {
327  for (Int_t i = 0; i < nev; i++) {
328  status = EventLoop();
329  if(1 == status) {
330  break;
331  } else if(2 == status) {
332  i -= 1;
333  continue;
334  }
335  if(gIsInterrupted)
336  {
337  break;
338  }
339  }
340  }
341 
342  fRootManager->StoreAllWriteoutBufferData();
343  if (fAutomaticFinish) {
344  Finish();
345  }
346 }

412 void FairRunOnline::WriteObjects()
413 {
414  TDirectory *oldDir = gDirectory;
415 
416  TFile *file = new TFile(fHistFileName, "RECREATE");
417 
418  // Create iterator with the folder content
419  TIter iter(fFolder->GetListOfFolders());
420 
421  // Pointer to an object
422  TObject* object;
423 
424  // Class name of the object
425  TString className;
426 
427  // Histogram pointers
428  TH1* h1;
429  TH2* h2;
430 
431  // Loop over objects in the folder
432  while((object = iter())) {
433 
434  // Get class name
435  className = object->ClassName();
436 
437  // Recognise objects
438  if(0 == className.CompareTo("TH1F")) {
439  // If a histogram - plot it and save canvas
440  h1 = (TH1F*) object;
441  h1->Write();
442  } else if(0 == className.CompareTo("TH2F")) {
443  // If a histogram - plot it and save canvas
444  h2 = (TH2F*) object;
445  h2->Write();
446  } else if(0 == className.CompareTo("TH1D")) {
447  // If a histogram - plot it and save canvas
448  h1 = (TH1D*) object;
449  h1->Write();
450  } else if(0 == className.CompareTo("TH2D")) {
451  // If a histogram - plot it and save canvas
452  h2 = (TH2D*) object;
453  h2->Write();
454  }
455  }
456 
457  file->Close();
458 
459  oldDir->cd();
