98 void FairRunInfo::CreateAndFillHistograms(TList* histoList)
99 {
100  Int_t entries = fTime.size();
101  Double_t timePeriod = fTime.back()-fTime.front();
102 
103  TH1F* ResidentMemoryVsEvent = new TH1F("ResidentMemoryVsEvent","Resident Memory as function of Eventnumber;Event;Memory [MB]",entries, 0, entries);
104  TH1F* VirtualMemoryVsEvent = new TH1F("VirtualMemoryVsEvent","Virtual Memory as function of Eventnumber;Event;Memory [MB]",entries, 0, entries);
105  TH1F* ResidentMemoryVsTime = new TH1F("ResidentMemoryVsTime","Resident memory as function of Runtime;Time [s];Memory [MB]",(Int_t)(timePeriod*10), 0, (Int_t)timePeriod);
106  TH1F* VirtualMemoryVsTime = new TH1F("VirtualMemoryVsTime","Virtual memory as function of Runtime;Time [s];Memory [MB]",(Int_t)(timePeriod*10), 0, (Int_t)timePeriod);
107  TH1F* EventtimeVsEvent = new TH1F("EventtimeVsEvent","Runtime per Event as function of Event number;Event;Time [s]",entries-1, 1, entries);
108 
109  std::vector<Double_t> timeDiffSorted(fTimeDiff);
110  std::vector<Double_t>::iterator it;
111  std::sort(timeDiffSorted.begin(), timeDiffSorted.end());
112 
113  Double_t minTime = timeDiffSorted.front();
114  Double_t maxTime = timeDiffSorted.back();
115  timePeriod = maxTime - minTime;
116 
117  TH1F* TimePerEvent = new TH1F("TimePerEvent","Runtime;Time [s];Events",(Int_t)((timePeriod+20)*10), (Int_t)minTime-10, (Int_t)maxTime+10);
118 
119  Int_t counter = 0;
120  std::vector<Long_t>::iterator lit;
121  Double_t timeOffset = fTime.front()+1.;
122  for(lit=fResidentMemory.begin(); lit<fResidentMemory.end(); lit++) {
123  ResidentMemoryVsEvent->Fill(counter, *lit);
124  ResidentMemoryVsTime->Fill(fTime.at(counter)-timeOffset, *lit);
125  counter++;
126  }
127  histoList->AddLast(ResidentMemoryVsEvent);
128  histoList->AddLast(ResidentMemoryVsTime);
129 
130  counter = 0;
131  for(lit=fVirtualMemory.begin(); lit<fVirtualMemory.end(); lit++) {
132  VirtualMemoryVsEvent->Fill(counter, *lit);
133  VirtualMemoryVsTime->Fill(fTime.at(counter)-timeOffset, *lit);
134  counter++;
135  }
136  histoList->AddLast(VirtualMemoryVsEvent);
137  histoList->AddLast(VirtualMemoryVsTime);
138 
139  counter = 1;
140  for(it=fTimeDiff.begin(); it<fTimeDiff.end(); it++) {
141  TimePerEvent->Fill(*it);
142  EventtimeVsEvent->Fill(counter, *it);
143  counter++;
144  }
145  histoList->AddLast(TimePerEvent);
146  histoList->AddLast(EventtimeVsEvent);
147 }
