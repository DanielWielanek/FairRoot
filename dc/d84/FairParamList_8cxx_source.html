31 //
32 // FairParamObj
33 //
34 // Class for parameters stored in binary format in Oracle and used as list
35 // elements in FairParamList::paramList.
36 //
37 // The overloaded constructors and fill-functions accept single values or arrays of type
38 // UChar_t, Int_t, Float_t, Double_t, char_t, Text_t.
39 // The arguments for arrays are
40 // the name of the parameter
41 // the pointer to the array
42 // the length of the array
43 // The data are automatically converted to an UChar_t array.
44 // For classes also the class version is stored, for ROOT classes also the TStreamerInfo.
45 //
46 // -------------------------------------------------------------------------------------
47 //
48 // FairParamList
49 //
50 // Class for the generic Oracle and ASCII interface for parameter containers
51 // derived from HParCond
52 //
53 // The class contains a list to stores objects of type FairParamObj
54 // The list objects store the name, the value, the parameter type and some
55 // additional information depending for the object type.
56 //
57 // All add/addObject functions add an initialized parameter to the list for
58 // writing. The functions create a new list element and copy the data into
59 // this object.
60 // Add functions:
61 // 1. accepted basic types: Int_t, Float_t, Double_t, UChar_t
62 // 2. accepted ROOT arrays: TArrayI, TArrayF, TArrayD, TArrayC
63 // 3. accepted string: Text_t*
64 // This can be any text, for example also numbers in hexadecimal or
65 // scientific format. The number of characters must be specified (default 1).
66 // AddObject function:
67 // Accepts classes derived from TObject.
68 // The persistent data elements are streamed into an UChar_t array using the
69 // class streamer. For ROOT classes, for example histograms, the ROOT streamer
70 // info is stored in an additional binary array.
71 //
72 // All fill/fillObject functions convert the data in the list element back
73 // to the type of the parameter and copies them into the data element in the
74 // initialization process.
75 // 1. Single parameters of basic type:
76 // The functions return kFALSE, if the parameter is not in the list.
77 // 2. Arrays:
78 // a) If the array size is specified (return code Bool_t), the functions return
79 // kFALSE, if the number of data elements in the list objects is not the same.
80 // b) If the array size is not specified (return code Int_t), the array is
81 // recreated with the size of the number of data elements in the list object.
82 // The functions return the number of data elements or 0, if the parameter
83 // was not found.
84 // 3. Classes:
85 // The class version is checked and a warning printed, if it is not identical
86 // with the current version (typically class version of list object higher than
87 // version in the actual parameter container). The class streamer takes care
88 // for backward compatibility. A warning is printed, if the ROOT version is
89 // different from the current version.
90 // The function returns the number of bytes in the list object or 0, if the
91 // parameter was not found in the list.
92 //

535 void FairParamList::addObject(const Text_t* name,TObject* obj)
536 {
537  // Adds a TObject to the list, sets the class version and the streamer info for
538  // ROOT classes
539  if (!obj) { return; }
540  FairParamObj* o=new FairParamObj(name);
541  o->setParamType(obj->IsA()->GetName());
542  o->setClassVersion(obj->IsA()->GetClassVersion());
543  TFile* filesave=gFile;
544  FairParamTFile* paramFile=new FairParamTFile();
545  gFile=paramFile;
546  const Int_t bufsize=10000;
547 
548  TBufferFile* buffer=new TBufferFile(TBuffer::kWrite,bufsize);
549 
550  buffer->SetParent(paramFile);
551  buffer->MapObject(obj);
552  obj->Streamer(*buffer);
553  Int_t len=buffer->Length();
554  Char_t* buf=new char[len];
555  memcpy(buf,buffer->Buffer(),len);
556  o->setParamValue((UChar_t*)buf,len);
557  TArrayC* fClassIndex=paramFile->GetClassIndex();
558  if (fClassIndex&&fClassIndex->fArray[0] != 0) {
559  TIter next(gROOT->GetListOfStreamerInfo());
560  TStreamerInfo* info;
561  TList list;
562  while ((info=(TStreamerInfo*)next())) {
563  Int_t uid=info->GetNumber();
564  if (fClassIndex->fArray[uid]) { list.Add(info); }
565  }
566  if (list.GetSize()>0) {
567  list.Sort();
568  fClassIndex->fArray[0]=2; //to prevent adding classes in TStreamerInfo::TagFile
569 
570  TBufferFile* infoBuffer=new TBufferFile(TBuffer::kWrite,bufsize);
571 
572  infoBuffer->MapObject(&list);
573  list.Streamer(*infoBuffer);
574  Int_t infolen=infoBuffer->Length();
575  Char_t* infobuf=new char[infolen];
576  memcpy(infobuf,infoBuffer->Buffer(),infolen);
577  o->setStreamerInfo((UChar_t*)infobuf,infolen);
578  delete infoBuffer;
579  } else {
580  o->setStreamerInfo(0,0);
581  }
582  }
583  fClassIndex->fArray[0]=0;
584  delete paramFile;
585  paramList->Add(o);
586  delete buffer;
587  gFile=filesave;
588 }

599 {

642 Bool_t FairParamList::fill(const Text_t* name,Int_t* values,const Int_t nValues)
643 {
644  // Copies the data from the list object into the parameter array of type Int_t.
645  // The function returns an error, if the array size of the list object is not equal
646  // to nValues.
647  if (values==0) { return kFALSE; }
648  FairParamObj* o=(FairParamObj*)paramList->FindObject(name);
649  if (o!=0 && strcmp(o->getParamType(),"Int_t")==0) {
650  Int_t l=o->getLength();
651  Int_t n=o->getNumParams();
652  if (n==nValues) {
653  memcpy(values,o->getParamValue(),l);

785 Bool_t FairParamList::fillObject(const Text_t* name,TObject* obj)
786 {
787  // Fills the object obj (must exist!) via the Streamer and returns the class version.
788  // Prints a warning if the class version in the list objects differs from the actual
789  // class version.
790  if (!obj) { return 0; }
791  FairParamObj* o=(FairParamObj*)paramList->FindObject(name);
792  if (o!=0 && strcmp(o->getParamType(),obj->IsA()->GetName())==0) {
793  if (o->getClassVersion()!=obj->IsA()->GetClassVersion()) {
794  fLogger->Warning(MESSAGE_ORIGIN,"Read Class Version = %i does not match actual version = %i",o->getClassVersion(),obj->IsA()->GetClassVersion());
795  }
796  // Warning("FairParamList::fill",
797  // "\n Read Class Version = %i does not match actual version = %i",
798  // o->getClassVersion(),obj->IsA()->GetClassVersion());
799  TFile* filesave=gFile;
800  gFile=0;
801  TBufferFile* buf=0;
802 
803  Int_t len=o->getStreamerInfoSize();
804  if (len>0&&o->getStreamerInfo()!=0) {
805  buf=new TBufferFile(TBuffer::kRead,len);
806  memcpy(buf->Buffer(),(Char_t*)o->getStreamerInfo(),len);
807  buf->SetBufferOffset(0);
808  TList list;
809  buf->MapObject(&list);
810  list.Streamer(*buf);
811  delete buf;
812  TStreamerInfo* info;
813  TIter next(&list);
814  while ((info = (TStreamerInfo*)next())) {
815  if (info->IsA() != TStreamerInfo::Class()) {
816  fLogger->Warning(MESSAGE_ORIGIN,"Not a TStreamerInfo object");
817  // Warning("FairParamList::fill","not a TStreamerInfo object");
818  continue;
819  }
820  info->BuildCheck();
821  }
822  list.Clear(); //this will delete all TStreamerInfo objects with kCanDelete
823  }
824  len=o->getLength();
825  buf=new TBufferFile(TBuffer::kRead,len);
826  memcpy(buf->Buffer(),(Char_t*)o->getParamValue(),len);
827  buf->SetBufferOffset(0);
828  buf->MapObject(obj);
829  obj->Streamer(*buf);
830  delete buf;
831  gFile=filesave;
832  return len;
833  }
834  fLogger->Error(MESSAGE_ORIGIN,"Could not find parameter %s", name);
835  // Error("FairParamList::fill \nNot found: %s",name);
836  return 0;
