46  fSocket = nn_socket(AF_SP_RAW, GetConstant(type));

48  {
49  LOG(ERROR) << "failed creating socket #" << fId << ", reason: " << nn_strerror(errno);
50  exit(EXIT_FAILURE);
51  }
52  }
53  else
54  {
55  fSocket = nn_socket(AF_SP, GetConstant(type));
56  if (fSocket == -1)

63  nn_setsockopt(fSocket, NN_SUB, NN_SUB_SUBSCRIBE, NULL, 0);
64  }
65  }
