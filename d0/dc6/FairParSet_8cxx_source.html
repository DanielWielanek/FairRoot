46 Bool_t FairParSet::init()
47 {
48  // intitializes the container from an input in run time
49  // database. If this is not successful it is initialized from
50  // the second input. If this failes too, it returns an error.
51  // (calls internally the init function in the derived class)
52  FairRuntimeDb* rtdb=FairRuntimeDb::instance();
53  //FairRunAna* fRun =FairRunAna::Instance();
54 // cout << "-I- FairParSet::init() " << GetName() << endl;
55 
56  Bool_t allFound=kFALSE;
57  FairParIo* io=0;
58  if (rtdb) {
59  io=rtdb->getFirstInput();
60  if (io) { allFound=init(io); }
61  if (!allFound) {
62  io=rtdb->getSecondInput();
63  //cout << "-I FairParSet::init() 2 " << io << std::endl;
64  if (io) { allFound=init(io); }
65  } else { setInputVersion(-1,2); }
66  }
67  if (allFound) { return kTRUE; }
68  if (fLogger ) { fLogger->Error(MESSAGE_ORIGIN, "init() %s not initialized",GetName()); }
69  return kFALSE;
70 }
