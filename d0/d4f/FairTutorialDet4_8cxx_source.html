133 Bool_t FairTutorialDet4::ProcessHits(FairVolume* vol)
134 {
137  //Set parameters at entrance of volume. Reset ELoss.
138  if ( gMC->IsTrackEntering() ) {
139  fELoss = 0.;
140  fTime = gMC->TrackTime() * 1.0e09;
141  fLength = gMC->TrackLength();
142  gMC->TrackPosition(fPos);
143  gMC->TrackMomentum(fMom);
144  }
145 
146  // Sum energy loss for all steps in the active volume
147  fELoss += gMC->Edep();
148 
149  // Create FairTutorialDet4Point at exit of active volume
150  if ( gMC->IsTrackExiting() ||
151  gMC->IsTrackStop() ||
152  gMC->IsTrackDisappeared() ) {
153  fTrackID = gMC->GetStack()->GetCurrentTrackNumber();
154 // fVolumeID = vol->getMCid();
155  fVolumeID = fGeoHandler->GetUniqueDetectorId();
156  if (fELoss == 0. ) { return kFALSE; }
157 
158  // Save positions in local coordinate system, so transform the
159  // global coordinates into local ones.
160  Double_t master[3] = {fPos.X(), fPos.Y(), fPos.Z()};
161  Double_t local[3];
162 
163  gMC->Gmtod(master, local, 1);
164 
165 // AddHit(fTrackID, fVolumeID, TVector3(fPos.X(), fPos.Y(), fPos.Z()),
166  AddHit(fTrackID, fVolumeID, TVector3(local[0], local[1], local[2]),
167  TVector3(fMom.Px(), fMom.Py(), fMom.Pz()), fTime, fLength,
168  fELoss);
169 
170  // Increment number of tutorial det points in TParticle
171  FairStack* stack = (FairStack*) gMC->GetStack();
172  stack->AddPoint(kTutDet);
173  }
174 
175  return kTRUE;
176 }

236 void FairTutorialDet4::ConstructASCIIGeometry()
237 {
242  FairGeoLoader* geoLoad = FairGeoLoader::Instance();
243  FairGeoInterface* geoFace = geoLoad->getGeoInterface();
244  FairTutorialDet4Geo* Geo = new FairTutorialDet4Geo();
245  LOG(DEBUG)<<"Read Geo file "<<GetGeometryFileName()<<FairLogger::endl;
246  Geo->setGeomFile(GetGeometryFileName());
247  geoFace->addGeoModule(Geo);
248 
249  Bool_t rc = geoFace->readSet(Geo);
250  if (rc) { Geo->create(geoLoad->getGeoBuilder()); }
251  TList* volList = Geo->getListOfVolumes();
252 
253  // store geo parameter
254  FairRun* fRun = FairRun::Instance();
255  FairRuntimeDb* rtdb= FairRun::Instance()->GetRuntimeDb();
256  FairTutorialDet4GeoPar* par=(FairTutorialDet4GeoPar*)(rtdb->getContainer("FairTutorialDet4GeoPar"));
257  TObjArray* fSensNodes = par->GetGeoSensitiveNodes();
258  TObjArray* fPassNodes = par->GetGeoPassiveNodes();
259 
260  TListIter iter(volList);
261  FairGeoNode* node = NULL;
262  FairGeoVolume* aVol=NULL;
263 
264  while( (node = (FairGeoNode*)iter.Next()) ) {
265  aVol = dynamic_cast<FairGeoVolume*> ( node );
266  if ( node->isSensitive() ) {
267  fSensNodes->AddLast( aVol );
268  } else {
269  fPassNodes->AddLast( aVol );
270  }
271  }
272  par->setChanged();
273  par->setInputVersion(fRun->GetRunId(),1);
274 
275  ProcessNodes ( volList );
276 }
