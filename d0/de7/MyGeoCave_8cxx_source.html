47 Bool_t MyGeoCave::read(fstream& fin,FairGeoMedia* media)
48 {
49  // Reads the geometry from file
50  if (!media) { return kFALSE; }
51  const Int_t maxbuf=256;
52  char buf[maxbuf];
53  FairGeoNode* volu=0;
54  FairGeoMedium* medium;
55  Bool_t rc=kTRUE;
56  do {
57  fin.getline(buf,maxbuf);
58  if (buf[0]!='\0' && buf[0]!='/' && !fin.eof()) {
59  if (strcmp(buf,name)==0) {
60  volu=new FairGeoNode;
61  volu->SetName(buf);
62  volu->setVolumeType(kFairGeoTopNode);
63  volu->setActive();
64  fin.getline(buf,maxbuf);
65  TString shape(buf);
66  FairGeoBasicShape* sh=pShapes->selectShape(shape);
67  if (sh) { volu->setShape(sh); }
68  else { rc=kFALSE; }
69  fin.getline(buf,maxbuf);
70  medium=media->getMedium(buf);
71  if (!medium) {
72  medium=new FairGeoMedium();
73  media->addMedium(medium);
74  }
75  volu->setMedium(medium);
76  Int_t n=0;
77  if (sh) { n=sh->readPoints(&fin,volu); }
78  if (n<=0) { rc=kFALSE; }
79  } else { rc=kFALSE; }
80  }
81  } while (rc && !volu && !fin.eof());
82  if (volu && rc) {
83  volumes->Add(volu);
84  masterNodes->Add(new FairGeoNode(*volu));
85  } else {
86  delete volu;
87  volu=0;
88  rc=kFALSE;
89  }
90  return rc;
91 }
