69 Bool_t FairTutorialDet2::ProcessHits(FairVolume* vol)
70 {
73  //Set parameters at entrance of volume. Reset ELoss.
74  if ( gMC->IsTrackEntering() ) {
75  fELoss = 0.;
76  fTime = gMC->TrackTime() * 1.0e09;
77  fLength = gMC->TrackLength();
78  gMC->TrackPosition(fPos);
79  gMC->TrackMomentum(fMom);
80  }
81 
82  // Sum energy loss for all steps in the active volume
83  fELoss += gMC->Edep();
84 
85  // Create FairTutorialDet2Point at exit of active volume
86  if ( gMC->IsTrackExiting() ||
87  gMC->IsTrackStop() ||
88  gMC->IsTrackDisappeared() ) {
89  fTrackID = gMC->GetStack()->GetCurrentTrackNumber();
90  fVolumeID = vol->getMCid();
91  if (fELoss == 0. ) { return kFALSE; }
92  AddHit(fTrackID, fVolumeID, TVector3(fPos.X(), fPos.Y(), fPos.Z()),
93  TVector3(fMom.Px(), fMom.Py(), fMom.Pz()), fTime, fLength,
94  fELoss);
95 
96  // Increment number of tutorial det points in TParticle
97  FairStack* stack = (FairStack*) gMC->GetStack();
98  stack->AddPoint(kTutDet);
99  }
100 
101  return kTRUE;
102 }

139 void FairTutorialDet2::ConstructGeometry()
140 {
145  FairGeoLoader* geoLoad = FairGeoLoader::Instance();
146  FairGeoInterface* geoFace = geoLoad->getGeoInterface();
147  FairTutorialDet2Geo* Geo = new FairTutorialDet2Geo();
148  Geo->setGeomFile(GetGeometryFileName());
149  geoFace->addGeoModule(Geo);
150 
151  Bool_t rc = geoFace->readSet(Geo);
152  if (rc) { Geo->create(geoLoad->getGeoBuilder()); }
153  TList* volList = Geo->getListOfVolumes();
154 
155  // store geo parameter
156  FairRun* fRun = FairRun::Instance();
157  FairRuntimeDb* rtdb= FairRun::Instance()->GetRuntimeDb();
158  FairTutorialDet2GeoPar* par=(FairTutorialDet2GeoPar*)(rtdb->getContainer("FairTutorialDet2GeoPar"));
159  TObjArray* fSensNodes = par->GetGeoSensitiveNodes();
160  TObjArray* fPassNodes = par->GetGeoPassiveNodes();
161 
162  TListIter iter(volList);
163  FairGeoNode* node = NULL;
164  FairGeoVolume* aVol=NULL;
165 
166  while( (node = (FairGeoNode*)iter.Next()) ) {
167  aVol = dynamic_cast<FairGeoVolume*> ( node );
168  if ( node->isSensitive() ) {
169  fSensNodes->AddLast( aVol );
170  } else {
171  fPassNodes->AddLast( aVol );
172  }
173  }
174  par->setChanged();
175  par->setInputVersion(fRun->GetRunId(),1);
176 
177  ProcessNodes ( volList );
178 }
