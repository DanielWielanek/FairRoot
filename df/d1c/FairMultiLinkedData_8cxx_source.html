177 void FairMultiLinkedData::InsertLink(FairLink link)
178 {
179  if (FairLinkManager::Instance()->IsIgnoreType(link.GetType())){
180  return;
181  }
182 
183  std::set<FairLink>::iterator it = fLinks.find(link);
184  if (it != fLinks.end()) {
185  FairLink myTempLink = *it;
186  myTempLink.AddWeight(link.GetWeight());
187  fLinks.erase(it);
188  fLinks.insert(myTempLink);
189  } else {
190  fLinks.insert(link);
191  }
192  return;

195 void FairMultiLinkedData::InsertHistory(FairLink link)

197  FairRootManager* ioman = FairRootManager::Instance();

199 
200  if (fVerbose > 1)

202 
203  if (link.GetType() < 0)
204  return;
205  if (link.GetType() == ioman->GetBranchId("MCTrack"))
206  return;

209  if(link.GetEntry() != -1 && link.GetEntry() != ioman->GetEntryNr())
210  return;
211 
212  if (link.GetIndex() < 0) { //if index is -1 then this is not a TClonesArray so only the Object is returned
213  FairMultiLinkedData_Interface* interface = (FairMultiLinkedData_Interface*) ioman->GetObject(ioman->GetBranchName(link.GetType()));
214  pointerToLinks = interface->GetPointerToLinks();

217  if (link.GetIndex() < dataArray->GetEntriesFast()) {
218  FairMultiLinkedData_Interface* interface = (FairMultiLinkedData_Interface*) dataArray->At(link.GetIndex());
219  pointerToLinks = interface->GetPointerToLinks();
220 
221  }
222  }
223  if (pointerToLinks != 0){
224  std::set<FairLink> linkSet = pointerToLinks->GetLinks();

226  if (fVerbose > 1)
