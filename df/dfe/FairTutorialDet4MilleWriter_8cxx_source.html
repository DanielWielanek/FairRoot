107 Bool_t FairTutorialDet4MilleWriter::IsGoodEvent()
108 {
109  // Check if each for the event there is maximum 1 hit per detector
110  // station. In the moment we create tracks with all hits in the
111  // event, so we have to check for this.
112  // In the end the algorithm should be able to work also with
113  // missing hits in some stations
114  FairTutorialDet4Hit* hit;
115  std::set<Int_t> detIdSet;
116  std::set<Int_t>::iterator it;
117 
118  Int_t nHits = fHits->GetEntriesFast();
119  for (Int_t iHit=0; iHit<nHits; ++iHit) {
120  hit = (FairTutorialDet4Hit*) fHits->At(iHit);
121  Int_t detId = hit->GetDetectorID();
122  it = detIdSet.find(detId);
123  if ( it == detIdSet.end() ) {
124  detIdSet.insert(detId);
125  } else {
126  // find hit in already used detector station
127  // this is not a good event
128  return kFALSE;
129  }
130  }
131  return kTRUE;
132 }

134 void FairTutorialDet4MilleWriter::StraightLineShiftX(Option_t* option)
135 {
136 
137  const Int_t nLC = 2; // number of local parameters
138  // two for track x-coordinate
139  // x(z) = a1*z + a2
140  // dx(z)/da1 = z
141  // dx(z)/da2 = 1
142 
143 
144  const Int_t nGL = 1; // number of global parameters per point
145  // taken from millepede 1 dim example
146 
147  Float_t sigma = 0.1;
148 
149  Float_t* derLC = new Float_t[nLC]; // array of derivatives for local parameters
150  Float_t* derGL = new Float_t[nGL]; // array of derivatives for global parameters
151 
152  Int_t* label = new Int_t[nGL]; // array of labels
153 
154 
155  for (Int_t help = 0; help < nGL; help++) {
156  derGL[help] = 0;
157  }
158 
159  for (Int_t help = 0; help < nLC; help++) {
160  derLC[help] = 0;
161  }
162 
163  FairTrackParam* track = (FairTrackParam*) fTracks->At(0);
164 
165  //Extract Track parameters
166  Double_t OffX = track->GetX();
167  Double_t SlopeX = track->GetTx();
168 
169  Double_t residual;
170 
171  FairTutorialDet4Hit* hit;
172 
173  Int_t nHits = fHits->GetEntriesFast();
174  for (Int_t iHit=0; iHit<nHits; ++iHit) {
175  hit = (FairTutorialDet4Hit*) fHits->At(iHit);
176 
177  Float_t Z = hit->GetZ();
178  Float_t hitX = hit->GetX();
179  Float_t fitX = OffX + SlopeX * Z;
180  LOG(DEBUG)<<"hitX, fitX: "<< hitX<<" ,"<<fitX<<FairLogger::endl;
181 
182  label[0] = iHit+1;
183 
184  derGL[0] = -1;
185 
186  derLC[0] = 1;
187  derLC[1] = Z;
188 
189  residual = fitX - hitX;
190  LOG(DEBUG)<<"ResidualX: "<< residual<<FairLogger::endl;
191  //call to Mille Writer
192  fMille->mille(nLC,derLC,nGL,derGL,label,residual,sigma);
193 
194  }
195  fMille->end();
196 
197  delete[] derLC;
198  delete[] derGL;
199  delete[] label;
200 }
