161 void FairMQDevice::InitWrapper()
162 {
163  if (!fTransportFactory)
164  {
165  LOG(ERROR) << "Transport not initialized. Did you call SetTransport()?";
166  exit(EXIT_FAILURE);
167  }
168 
169  if (!fCmdSocket)
170  {
171  fCmdSocket = fTransportFactory->CreateSocket("pub", "device-commands", fNumIoThreads, fId);
172  fCmdSocket->Bind("inproc://commands");
173  }
174 
175  // Containers to store the uninitialized channels.
176  list<FairMQChannel*> uninitializedBindingChannels;
177  list<FairMQChannel*> uninitializedConnectingChannels;
178 
179  // Fill the uninitialized channel containers
180  for (auto mi = fChannels.begin(); mi != fChannels.end(); ++mi)
181  {
182  for (auto vi = (mi->second).begin(); vi != (mi->second).end(); ++vi)
183  {
184  if (vi->fMethod == "bind")
185  {
186  // set channel name: name + vector index
187  stringstream ss;
188  ss << mi->first << "[" << vi - (mi->second).begin() << "]";
189  vi->fChannelName = ss.str();
190  // if binding address is not specified, set it up to try getting it from the configured network interface
191  if (vi->fAddress == "unspecified" || vi->fAddress == "")
192  {
193  vi->fAddress = "tcp://" + FairMQ::tools::getInterfaceIP(fNetworkInterface) + ":1";
194  }
195  // fill the uninitialized list
196  uninitializedBindingChannels.push_back(&(*vi));
197  }
198  else if (vi->fMethod == "connect")
199  {
200  // set channel name: name + vector index
201  stringstream ss;
202  ss << mi->first << "[" << vi - (mi->second).begin() << "]";
203  vi->fChannelName = ss.str();
204  // fill the uninitialized list
205  uninitializedConnectingChannels.push_back(&(*vi));
206  }
207  else
208  {
209  LOG(ERROR) << "Cannot update configuration. Socket method (bind/connect) not specified.";
210  exit(EXIT_FAILURE);
211  }
212  }
213  }
214 
215  // Bind channels. Here one run is enough, because bind settings should be available locally
216  // If necessary this could be handled in the same way as the connecting channels
217  BindChannels(uninitializedBindingChannels);
218  // notify parent thread about completion of first validation.
219  {
220  lock_guard<mutex> lock(fInitialValidationMutex);
221  fInitialValidationFinished = true;
222  fInitialValidationCondition.notify_one();
223  }
224 
225  // go over the list of channels until all are initialized (and removed from the uninitialized list)
226  int numAttempts = 0;
227  int maxAttempts = fMaxInitializationAttempts;
228  while (!uninitializedConnectingChannels.empty())
229  {
230  ConnectChannels(uninitializedConnectingChannels);
231  if (++numAttempts > maxAttempts)
232  {
233  LOG(ERROR) << "could not connect all channels after " << maxAttempts << " attempts";
234  // TODO: goto ERROR state;
235  exit(EXIT_FAILURE);
236  }
237 
238  if (numAttempts != 0)
239  {
240  boost::this_thread::sleep(boost::posix_time::milliseconds(1000));
241  }
242  }
243 
244  Init();
245 
246  ChangeState(internal_DEVICE_READY);
247 
248  // notify parent thread about end of processing.
249  boost::lock_guard<boost::mutex> lock(fStateMutex);
250  fStateFinished = true;
251  fStateCondition.notify_one();
252 }

264 bool FairMQDevice::BindChannel(FairMQChannel& ch)
265 {
266  LOG(DEBUG) << "Initializing channel " << ch.fChannelName << " (" << ch.fType << ")";
267  // initialize the socket
268  ch.fSocket = fTransportFactory->CreateSocket(ch.fType, ch.fChannelName, fNumIoThreads, fId);
269  // set high water marks
270  ch.fSocket->SetOption("snd-hwm", &(ch.fSndBufSize), sizeof(ch.fSndBufSize));
271  ch.fSocket->SetOption("rcv-hwm", &(ch.fRcvBufSize), sizeof(ch.fRcvBufSize));
272 
273  // number of attempts when choosing a random port
274  int maxAttempts = 1000;
275  int numAttempts = 0;
276 
277  // initialize random generator
278  boost::random::mt19937 gen(getpid());
279  boost::random::uniform_int_distribution<> randomPort(fPortRangeMin, fPortRangeMax);
280 
281  LOG(DEBUG) << "Binding channel " << ch.fChannelName << " on " << ch.fAddress;
282 
283  // try to bind to the saved port. In case of failure, try random one.
284  while (!ch.fSocket->Bind(ch.fAddress))
285  {
286  LOG(DEBUG) << "Could not bind to configured (TCP) port, trying random port in range " << fPortRangeMin << "-" << fPortRangeMax;
287  ++numAttempts;
288 
289  if (numAttempts > maxAttempts)
290  {
291  LOG(ERROR) << "could not bind to any (TCP) port in the given range after " << maxAttempts << " attempts";
292  return false;
293  }
294 
295  size_t pos = ch.fAddress.rfind(":");
296  stringstream newPort;
297  newPort << static_cast<int>(randomPort(gen));
298  ch.fAddress = ch.fAddress.substr(0, pos + 1) + newPort.str();
299 
300  LOG(DEBUG) << "Binding channel " << ch.fChannelName << " on " << ch.fAddress;
301  }
302 
303  return true;
304 }

649 void FairMQDevice::LogSocketRates()
650 {
651  timestamp_t t0;
652  timestamp_t t1;
653 
654  timestamp_t msSinceLastLog;
655 
656  int numFilteredSockets = 0;
657  vector<FairMQSocket*> filteredSockets;
658  vector<string> filteredChannelNames;
659 
660  // iterate over the channels map
661  for (auto mi = fChannels.begin(); mi != fChannels.end(); ++mi)
662  {
663  // iterate over the channels vector
664  for (auto vi = (mi->second).begin(); vi != (mi->second).end(); ++vi)
665  {
666  if (vi->fRateLogging == 1)
667  {
668  filteredSockets.push_back(vi->fSocket);
669  stringstream ss;
670  ss << mi->first << "[" << vi - (mi->second).begin() << "]";
671  filteredChannelNames.push_back(ss.str());
672  ++numFilteredSockets;
673  }
674  }
675  }
676 
677  vector<unsigned long> bytesIn(numFilteredSockets);
678  vector<unsigned long> msgIn(numFilteredSockets);
679  vector<unsigned long> bytesOut(numFilteredSockets);
680  vector<unsigned long> msgOut(numFilteredSockets);
681 
682  vector<unsigned long> bytesInNew(numFilteredSockets);
683  vector<unsigned long> msgInNew(numFilteredSockets);
684  vector<unsigned long> bytesOutNew(numFilteredSockets);
685  vector<unsigned long> msgOutNew(numFilteredSockets);
686 
687  vector<double> mbPerSecIn(numFilteredSockets);
688  vector<double> msgPerSecIn(numFilteredSockets);
689  vector<double> mbPerSecOut(numFilteredSockets);
690  vector<double> msgPerSecOut(numFilteredSockets);
691 
692  int i = 0;
693  for (auto itr = filteredSockets.begin(); itr != filteredSockets.end(); ++itr)
694  {
695  bytesIn.at(i) = (*itr)->GetBytesRx();
696  bytesOut.at(i) = (*itr)->GetBytesTx();
697  msgIn.at(i) = (*itr)->GetMessagesRx();
698  msgOut.at(i) = (*itr)->GetMessagesTx();
699  ++i;
700  }
701 
702  t0 = get_timestamp();
703 
704  while (true)
705  {
706  try
707  {
708  t1 = get_timestamp();
709 
710  msSinceLastLog = (t1 - t0) / 1000.0L;
711 
712  i = 0;
713 
714  for (auto itr = filteredSockets.begin(); itr != filteredSockets.end(); itr++)
715  {
716  bytesInNew.at(i) = (*itr)->GetBytesRx();
717  mbPerSecIn.at(i) = (static_cast<double>(bytesInNew.at(i) - bytesIn.at(i)) / (1024. * 1024.)) / static_cast<double>(msSinceLastLog) * 1000.;
718  bytesIn.at(i) = bytesInNew.at(i);
719 
720  msgInNew.at(i) = (*itr)->GetMessagesRx();
721  msgPerSecIn.at(i) = static_cast<double>(msgInNew.at(i) - msgIn.at(i)) / static_cast<double>(msSinceLastLog) * 1000.;
722  msgIn.at(i) = msgInNew.at(i);
723 
724  bytesOutNew.at(i) = (*itr)->GetBytesTx();
