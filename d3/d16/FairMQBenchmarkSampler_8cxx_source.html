41 
42  int numSentMsgs = 0;
43 
44  unique_ptr<FairMQMessage> baseMsg(fTransportFactory->CreateMessage(fMsgSize));
45 
46  // store the channel reference to avoid traversing the map on every loop iteration
47  const FairMQChannel& dataOutChannel = fChannels.at("data-out").at(0);
48 
49  LOG(INFO) << "Starting the benchmark with message size of " << fMsgSize << " and number of messages " << fNumMsgs << ".";
50  boost::timer::auto_cpu_timer timer;
51 
52  while (CheckCurrentState(RUNNING))
53  {
54  unique_ptr<FairMQMessage> msg(fTransportFactory->CreateMessage());
55  msg->Copy(baseMsg);
56 
57  if (dataOutChannel.Send(msg) >= 0)
58  {
59  if (fNumMsgs > 0)
60  {
61  numSentMsgs++;
62  if (numSentMsgs >= fNumMsgs)
63  {
64  break;
65  }
66  }
67  }
68 
69  --fMsgCounter;
70 
71  while (fMsgCounter == 0) {
72  boost::this_thread::sleep(boost::posix_time::milliseconds(1));
73  }
74  }
75 
76  LOG(INFO) << "Sent " << numSentMsgs << " messages, leaving RUNNING state.";
77  LOG(INFO) << "Sending time: ";
78 
79  try
80  {
81  resetMsgCounter.interrupt();
82  resetMsgCounter.join();
83  }
