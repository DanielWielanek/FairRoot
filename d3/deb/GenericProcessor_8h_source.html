138  while ( fState == RUNNING )

141 
142  received = fPayloadInputs->at(0)->Receive(msg);
143  receivedMsgs++;
144 
145  // InputPolicy::DeSerializeMsg(msg) --> deserialize data of msg and fill output container
146  // TaskPolicy::ExecuteTask( ... ) --> process output container
147  TaskPolicy::ExecuteTask(InputPolicy::DeSerializeMsg(msg));
148 
149  // OutputPolicy::fMessage point to msg
150  OutputPolicy::SetMessage(msg);
151 
152  if (received > 0)
153  {
154  // TaskPolicy::GetOutputData() --> Get processed output container
155  // OutputPolicy::message(...) --> Serialize output container and fill fMessage
156  fPayloadOutputs->at(0)->Send(OutputPolicy::SerializeMsg(TaskPolicy::GetOutputData()));
157  sentMsgs++;
158  received = 0;
159  }
160 
161  if(msg)
162  msg->CloseMessage();
163  }
164 
165  MQLOG(INFO) << "Received " << receivedMsgs << " and sent " << sentMsgs << " messages!";
166 
167  try
168  {
169  rateLogger.interrupt();
170  rateLogger.join();
171  }
172  catch(boost::thread_resource_error& e)
173  {
174  MQLOG(ERROR) << e.what();
175  }
176 
177  FairMQDevice::Shutdown();
