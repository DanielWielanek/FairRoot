78 void FairLmdSource::AddPath(TString dir, TString wildCard)
79 {
80  FileStat_t buf;
81  if(1 == gSystem->GetPathInfo(dir.Data(), buf))
82  {
83  LOG(WARNING) << "FairLmdSource: not found: " << dir << FairLogger::endl;

86 
87  TRegexp *re = new TRegexp(wildCard.Data(), kTRUE);
88  TSystemDirectory *sdir = new TSystemDirectory("dir", dir.Data());
89  TString dname(dir);
90  if(! dname.EndsWith("/"))
91  {
92  dname += "/";
93  }
94  TList *list = sdir->GetListOfFiles();
95 
96  TIterator *iter = list->MakeIterator();
97  TSystemFile *file;
98  TString name;
99  while(NULL != (file = (TSystemFile*)iter->Next()))
100  {
101  name = file->GetName();
102  if(name.Contains(*re))
103  {
104  name = dname + name;
105  AddFile(name);
106  }
107  }
108 
109  list->Delete();
110 }
