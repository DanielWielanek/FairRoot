208  void ChannelParser(const boost::property_tree::ptree& tree, FairMQMap& channelMap, const string& formatFlag)
209  {
210  string channelKey;
211 
212  for (const auto& p : tree)
213  {
214  if (p.first == "channels")
215  {
216  for (const auto& q : p.second)
217  {
218  channelKey = q.second.get<string>("name");
219 
220  // try to get common properties to use for all subChannels
221  FairMQChannel commonChannel;
222  commonChannel.UpdateType(q.second.get<string>("type", commonChannel.GetType()));
223  commonChannel.UpdateMethod(q.second.get<string>("method", commonChannel.GetMethod()));
224  commonChannel.UpdateAddress(q.second.get<string>("address", commonChannel.GetAddress()));
225  commonChannel.UpdateProperty(q.second.get<string>("property", commonChannel.GetProperty()));
226  commonChannel.UpdateSndBufSize(q.second.get<int>("sndBufSize", commonChannel.GetSndBufSize()));
227  commonChannel.UpdateRcvBufSize(q.second.get<int>("rcvBufSize", commonChannel.GetRcvBufSize()));
228  commonChannel.UpdateRateLogging(q.second.get<int>("rateLogging", commonChannel.GetRateLogging()));
229 
230  LOG(DEBUG) << "\t[" << p.first << "] " << channelKey;
231 
232  // temporary FairMQChannel container
233  vector<FairMQChannel> channelList;
234  SocketParser(q.second.get_child(""), channelList, commonChannel);
235 
236  channelMap.insert(make_pair(channelKey, move(channelList)));
