426 // ----- Private method SelectTracks -----------------------------------
427 void MyProjStack::SelectTracks()
428 {
429 
430  // --> Clear storage map
431  fStoreMap.clear();
432 
433  // --> Check particles in the fParticle array
434  for (Int_t i=0; i<fNParticles; i++) {
435 
436  TParticle* thisPart = GetParticle(i);
437  Bool_t store = kTRUE;
438 
439  // --> Get track parameters
440  Int_t iMother = thisPart->GetMother(0);
441  TLorentzVector p;
442  thisPart->Momentum(p);
443  Double_t energy = p.E();
444  Double_t mass = p.M();
445 // Double_t mass = thisPart->GetMass();
446  Double_t eKin = energy - mass;
447 
448  // --> Calculate number of points
449  Int_t nPoints = 0;
450  for (Int_t iDet=kNewDetector; iDet<kSTOPHERE; iDet++) {
451  pair<Int_t, Int_t> a(i, iDet);
452  if ( fPointsMap.find(a) != fPointsMap.end() ) {
453  nPoints += fPointsMap[a];
454  }
455  }
456 
457  // --> Check for cuts (store primaries in any case)
458  if (iMother < 0) { store = kTRUE; }
459  else {
460  if (!fStoreSecondaries) { store = kFALSE; }
461  if (nPoints < fMinPoints) { store = kFALSE; }
462  if (eKin < fEnergyCut) { store = kFALSE; }
463  }
464 
465  // --> Set storage flag
466  fStoreMap[i] = store;
467 
468 
469  }
470 
471  // --> If flag is set, flag recursively mothers of selected tracks
472  if (fStoreMothers) {
473  for (Int_t i=0; i<fNParticles; i++) {
474  if (fStoreMap[i]) {
475  Int_t iMother = GetParticle(i)->GetMother(0);
476  while(iMother >= 0) {
477  fStoreMap[iMother] = kTRUE;
478  iMother = GetParticle(iMother)->GetMother(0);
479  }
480  }
481  }
482  }
483 
484 }
