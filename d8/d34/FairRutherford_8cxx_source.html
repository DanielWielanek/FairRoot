78 Bool_t FairRutherford::ProcessHits(FairVolume* vol)
79 {
82  //Set parameters at entrance of volume. Reset ELoss.
83  if ( gMC->IsTrackEntering() ) {
84  fELoss = 0.;
85  fTime = gMC->TrackTime() * 1.0e09;
86  fLength = gMC->TrackLength();
87  gMC->TrackPosition(fPos);
88  gMC->TrackMomentum(fMom);
89  }
90 
91  // Sum energy loss for all steps in the active volume
92  fELoss += gMC->Edep();
93 
94  // Create FairRutherfordPoint at exit of active volume
95  if ( gMC->IsTrackExiting() ||
96  gMC->IsTrackStop() ||
97  gMC->IsTrackDisappeared() ) {
98  fTrackID = gMC->GetStack()->GetCurrentTrackNumber();
99  fVolumeID = vol->getMCid();
100  if (fELoss == 0. ) { return kFALSE; }
101  AddHit(fTrackID, fVolumeID, TVector3(fPos.X(), fPos.Y(), fPos.Z()),
102  TVector3(fMom.Px(), fMom.Py(), fMom.Pz()), fTime, fLength,
103  fELoss);
104 
105  // Increment number of FairRutherford points in TParticle
106  FairStack* stack = (FairStack*) gMC->GetStack();
107  stack->AddPoint(kFairRutherford);
108  }
109 
110  return kTRUE;
111 }

148 void FairRutherford::ConstructGeometry()
149 {
154  FairGeoLoader* geoLoad = FairGeoLoader::Instance();
155  FairGeoInterface* geoFace = geoLoad->getGeoInterface();
156  FairRutherfordGeo* Geo = new FairRutherfordGeo();
157  Geo->setGeomFile(GetGeometryFileName());
158  geoFace->addGeoModule(Geo);
159 
160  Bool_t rc = geoFace->readSet(Geo);
161  if (rc) { Geo->create(geoLoad->getGeoBuilder()); }
162  TList* volList = Geo->getListOfVolumes();
163 
164  // store geo parameter
165  FairRun* fRun = FairRun::Instance();
166  FairRuntimeDb* rtdb= FairRun::Instance()->GetRuntimeDb();
167  FairRutherfordGeoPar* par=(FairRutherfordGeoPar*)(rtdb->getContainer("FairRutherfordGeoPar"));
168  TObjArray* fSensNodes = par->GetGeoSensitiveNodes();
169  TObjArray* fPassNodes = par->GetGeoPassiveNodes();
170 
171  TListIter iter(volList);
172  FairGeoNode* node = NULL;
173  FairGeoVolume* aVol=NULL;
174 
175  while( (node = (FairGeoNode*)iter.Next()) ) {
176  aVol = dynamic_cast<FairGeoVolume*> ( node );
177  if ( node->isSensitive() ) {
178  fSensNodes->AddLast( aVol );
179  } else {
180  fPassNodes->AddLast( aVol );
181  }
182  }
183  par->setChanged();
184  par->setInputVersion(fRun->GetRunId(),1);
185 
186  ProcessNodes ( volList );
187 }
