14 // FairGeoTube
15 //
16 // class for the GEANT shape TUBE
17 //
18 // The size of a TUBE is defined by 3 'points'.
19 // point 0: origin of starting circle of the tube;
20 // point 1: inner radius of starting circle,
21 // outer radius of starting circle;
22 // (z-component not used)
23 // point 2: origin of ending circle of the tube;
24 // Warning: The x- and y-values of point 0 and 2 have to be the same!!!!
25 // A rotation has to be desribed via the rotation matrix.
26 //
27 // The intrinsic coordinate system of a TUBE, which sits in the CAVE and is
28 // not rotated, is identical with the laboratory system.
29 //

76 Int_t FairGeoTube::readPoints(std::fstream* pFile,FairGeoVolume* volu)
77 {
78  // reads the 3 'points' decribed above from ascii file
79  // if the array of points is not existing in the volume it is created and
80  // the values are stored inside
81  // returns the number of points
82  if (!pFile) { return 0; }
83  if (volu->getNumPoints()!=nPoints) { volu->createPoints(nPoints); }
84  Double_t x,y,z;
85  const Int_t maxbuf=155;
86  Text_t buf[maxbuf];
87  for(Int_t i=0; i<nPoints; i++) {
88  pFile->getline(buf,maxbuf);
89  if (i!=1) {
90  sscanf(buf,"%lf%lf%lf",&x,&y,&z);
91  volu->setPoint(i,x,y,z);
92  } else {
93  sscanf(buf,"%lf%lf",&x,&y);
94  volu->setPoint(i,x,y,0.0);
95  }
96  }
97  return nPoints;
98 }

101 Bool_t FairGeoTube::writePoints(std::fstream* pFile,FairGeoVolume* volu)
102 {
103  // writes the 3 'points' decribed above to ascii file
104  if (!pFile) { return kFALSE; }
105  Text_t buf[155];
106  for(Int_t i=0; i<nPoints; i++) {
107  FairGeoVector& v=*(volu->getPoint(i));
108  if (i!=1) { sprintf(buf,"%9.3f%10.3f%10.3f\n",v(0),v(1),v(2)); }
109  else { sprintf(buf,"%9.3f%10.3f\n",v(0),v(1)); }
110  pFile->write(buf,strlen(buf));
111  }
112  return kTRUE;
113 }

127 TArrayD* FairGeoTube::calcVoluParam(FairGeoVolume* volu)
128 {
129  // calculates the parameters needed to create the shape TUBE
130  Double_t fac=10.;
131  FairGeoVector& v1=*(volu->getPoint(1));
132  param->AddAt(v1(0)/fac,0);
133  param->AddAt(v1(1)/fac,1);
134  FairGeoVector v=*(volu->getPoint(2)) - *(volu->getPoint(0));
135  param->AddAt(TMath::Abs(v(2))/fac/2.,2);
136  return param;
137 }
