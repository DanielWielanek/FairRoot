40 void FairMQProgOptions::ParseAll(const int argc, char** argv, bool allowUnregistered)
41 {
42 
43  // init description
44  InitOptionDescription();
45  // parse command line options
46  if (ParseCmdLine(argc, argv, fCmdLineOptions, fVarMap, allowUnregistered))
47  {
48  LOG(ERROR) << "Could not parse cmd options";
49  exit(EXIT_FAILURE);
50  }
51 
52  // if txt/INI configuration file enabled then parse it as well
53  if (fUseConfigFile)
54  {
55  // check if file exist
56  if (fs::exists(fConfigFile))
57  {
58  if (ParseCfgFile(fConfigFile.string(), fConfigFileOptions, fVarMap, allowUnregistered))
59  {
60  LOG(ERROR) << "Could not parse config";
61  exit(EXIT_FAILURE);
62  }
63  }
64  else
65  {
66  LOG(ERROR) << "config file '" << fConfigFile << "' not found";
67  exit(EXIT_FAILURE);
68  }
69  }
70 
71  // set log level before printing (default is 0 = DEBUG level)
72  std::string verbosity = GetValue<std::string>("verbose");
73  bool color = GetValue<bool>("log-color");
74  if (!color)
75  {
76  reinit_logger(false);
77  }
78 
79  if (fSeverityMap.count(verbosity))
80  {
81  set_global_log_level(log_op::operation::GREATER_EQ_THAN, fSeverityMap.at(verbosity));
82  }
83  else
84  {
85  LOG(ERROR) << " verbosity level '" << verbosity << "' unknown, it will be set to DEBUG";
86  set_global_log_level(log_op::operation::GREATER_EQ_THAN, fSeverityMap.at("DEBUG"));
87  }
88 
89  PrintOptions();
90 
91  // check if one of required MQ config option is there
92  auto parserOption_shptr = fMQParserOptions.options();
93  bool optionExists = false;
94  vector<string> MQParserKeys;
95  for (const auto& p : parserOption_shptr)
96  {
97  MQParserKeys.push_back(p->canonical_display_name());
98  if (fVarMap.count(p->canonical_display_name()))
99  {
100  optionExists = true;
101  break;
102  }
103  }
104 
105  if (!optionExists)
106  {
107  LOG(WARN) << "Options to configure FairMQ channels are not provided.";
108  LOG(WARN) << "Please provide the value for one of the following keys:";
109  for (const auto& p : MQParserKeys)
110  {
111  LOG(WARN) << p;
112  }
113  LOG(WARN) << "No channels will be created (You can create them manually).";
114  // return 1;
115  }
116  else
117  {
118  if (fVarMap.count("mq-config"))
119  {
120  LOG(DEBUG) << "mq-config: Using default XML/JSON parser";
121 
122  std::string file = fVarMap["mq-config"].as<std::string>();
123  std::string id;
124 
125  if (fVarMap.count("config-key"))
126  {
127  id = fVarMap["config-key"].as<std::string>();
128  }
129  else
130  {
131  id = fVarMap["id"].as<std::string>();
132  }
133 
134  std::string fileExtension = boost::filesystem::extension(file);
135 
136  std::transform(fileExtension.begin(), fileExtension.end(), fileExtension.begin(), ::tolower);
137 
138  if (fileExtension == ".json")
139  {
140  UserParser<FairMQParser::JSON>(file, id);
141  }
142  else
143  {
144  if (fileExtension == ".xml")
145  {
146  UserParser<FairMQParser::XML>(file, id);
147  }
148  else
149  {
150  LOG(ERROR) << "mq-config command line called but file extension '"
151  << fileExtension
152  << "' not recognized. Program will now exit";
153  exit(EXIT_FAILURE);
154  }
155  }
156  }
157  else if (fVarMap.count("config-json-string"))
158  {
159  LOG(DEBUG) << "config-json-string: Parsing JSON string";
160 
161  std::string id = fVarMap["id"].as<std::string>();
162 
163  std::string value = fairmq::ConvertVariableValue<fairmq::ToString>().Run(fVarMap.at("config-json-string"));
164  std::stringstream ss;
165  ss << value;
166  UserParser<FairMQParser::JSON>(ss, id);
167  }
168  else if (fVarMap.count("config-xml-string"))
169  {
170  LOG(DEBUG) << "config-json-string: Parsing XML string";
171 
172  std::string id = fVarMap["id"].as<std::string>();
173 
174  std::string value = fairmq::ConvertVariableValue<fairmq::ToString>().Run(fVarMap.at("config-xml-string"));
175  std::stringstream ss;
176  ss << value;
177  UserParser<FairMQParser::XML>(ss, id);
178  }
179  }
180 }
