52 void init_log_console(bool color/* = true*/)
53 {
54  FairMQ::Logger::sinkList.clear();

56  typedef sinks::synchronous_sink<sinks::text_ostream_backend> text_sink;
57 
58  logging::core::get()->remove_all_sinks();
59 
60  // CONSOLE - all severity except error
61  boost::shared_ptr<text_sink> sink = boost::make_shared<text_sink>();
62  // add "console" output stream to our sink
63  sink->locked_backend()->add_stream(boost::shared_ptr<std::ostream>(&std::clog, empty_deleter_t()));
64 
65  // specify the format of the log message
66  if (color)
67  {
68  sink->set_formatter(&init_log_formatter<tag_console>);
69  }
70  else
71  {
72  sink->set_formatter(&init_log_formatter<tag_file>);
73  }
74 
75  sink->set_filter(severity != SEVERITY_ERROR && severity < SEVERITY_NOLOG);
76  // add sink to the core
77  FairMQ::Logger::sinkList.push_back(sink);
78  logging::core::get()->add_sink(sink);
79 
80  // CONSOLE - only severity error
81  boost::shared_ptr<text_sink> sink_error = boost::make_shared<text_sink>();
82  sink_error->locked_backend()->add_stream(boost::shared_ptr<std::ostream>(&std::cerr, empty_deleter_t()));
83 
84  if (color)
85  {
86  sink_error->set_formatter(&init_log_formatter<tag_console>);

105 void init_log_file(const std::string& filename, custom_severity_level threshold, log_op::operation op, const std::string& id)
106 {
107  // add a text sink
108  std::string formatted_filename(filename);
109  formatted_filename += id;
110  formatted_filename += "_%Y-%m-%d_%H-%M-%S.%N.log";
111  boost::shared_ptr<sinks::text_file_backend> backend = boost::make_shared<sinks::text_file_backend>(
112  boost::log::keywords::file_name = formatted_filename,
113  boost::log::keywords::rotation_size = 10 * 1024 * 1024,
114  // rotate at midnight every day
115  boost::log::keywords::time_based_rotation = sinks::file::rotation_at_time_point(0, 0, 0),
116  // log collector,
117  // -- maximum total size of the stored log files is 1GB.
118  // -- minimum free space on the drive is 2GB
119  boost::log::keywords::max_size = 1000 * 1024 * 1024,
120  boost::log::keywords::min_free_space = 2000 * 1024 * 1024,
121  boost::log::keywords::auto_flush = true
122  //keywords::time_based_rotation = &is_it_time_to_rotate
123  );
124  typedef sinks::synchronous_sink<sinks::text_file_backend> sink_t;
125  boost::shared_ptr<sink_t> sink(new sink_t(backend));
126 
127  // specify the format of the log message
128  sink->set_formatter(&init_log_formatter<tag_file>);
129 
130  switch (op)
131  {
132  case log_op::operation::EQUAL :
133  sink->set_filter(severity == threshold);
134  break;
135  case log_op::operation::GREATER_THAN :
136  sink->set_filter(severity > threshold);
137  break;
138  case log_op::operation::GREATER_EQ_THAN :
139  sink->set_filter(severity >= threshold);
140  break;
141  case log_op::operation::LESS_THAN :
142  sink->set_filter(severity < threshold);
143  break;
144  case log_op::operation::LESS_EQ_THAN :
145  sink->set_filter(severity <= threshold);
146  break;
147  default:
148  break;
149  }
150 
151  logging::core::get()->add_sink(sink);
152 }
