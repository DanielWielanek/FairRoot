220  {
221  vector<T> DataBunch;
222  for(int i(0);i<dataset->numEntries();i++)

229  data.SetY( static_cast<Int_t>(round(dataset->get(i)->getRealValue("y")) ));

232  if(printval)

234  <<" y=" << data.GetY()
235  <<" z=" << data.GetZ()
236  <<" t=" << data.GetTimeStamp()
237  <<" tErr=" << data.GetTimeStampError();
238  DataBunch.push_back(data);
