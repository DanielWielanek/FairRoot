73 Bool_t NewDetector::ProcessHits(FairVolume* vol)
74 {
77  //Set parameters at entrance of volume. Reset ELoss.
78  if ( gMC->IsTrackEntering() ) {
79  fELoss = 0.;
80  fTime = gMC->TrackTime() * 1.0e09;
81  fLength = gMC->TrackLength();
82  gMC->TrackPosition(fPos);
83  gMC->TrackMomentum(fMom);
84  }
85 
86  // Sum energy loss for all steps in the active volume
87  fELoss += gMC->Edep();
88 
89  // Create NewDetectorPoint at exit of active volume
90  if ( gMC->IsTrackExiting() ||
91  gMC->IsTrackStop() ||
92  gMC->IsTrackDisappeared() ) {
93  fTrackID = gMC->GetStack()->GetCurrentTrackNumber();
94  fVolumeID = vol->getMCid();
95  if (fELoss == 0. ) { return kFALSE; }
96  AddHit(fTrackID, fVolumeID, TVector3(fPos.X(), fPos.Y(), fPos.Z()),
97  TVector3(fMom.Px(), fMom.Py(), fMom.Pz()), fTime, fLength,
98  fELoss);
99 
100  // Increment number of NewDetector det points in TParticle
101  FairStack* stack = (FairStack*) gMC->GetStack();
102  stack->AddPoint(kNewDetector);
103  }
104 
105  return kTRUE;
106 }

143 void NewDetector::ConstructGeometry()
144 {
149  FairGeoLoader* geoLoad = FairGeoLoader::Instance();
150  FairGeoInterface* geoFace = geoLoad->getGeoInterface();
151  NewDetectorGeo* Geo = new NewDetectorGeo();
152  Geo->setGeomFile(GetGeometryFileName());
153  geoFace->addGeoModule(Geo);
154 
155  Bool_t rc = geoFace->readSet(Geo);
156  if (rc) { Geo->create(geoLoad->getGeoBuilder()); }
157  TList* volList = Geo->getListOfVolumes();
158 
159  // store geo parameter
160  FairRun* fRun = FairRun::Instance();
161  FairRuntimeDb* rtdb= FairRun::Instance()->GetRuntimeDb();
162  NewDetectorGeoPar* par=(NewDetectorGeoPar*)(rtdb->getContainer("NewDetectorGeoPar"));
163  TObjArray* fSensNodes = par->GetGeoSensitiveNodes();
164  TObjArray* fPassNodes = par->GetGeoPassiveNodes();
165 
166  TListIter iter(volList);
167  FairGeoNode* node = NULL;
168  FairGeoVolume* aVol=NULL;
169 
170  while( (node = (FairGeoNode*)iter.Next()) ) {
171  aVol = dynamic_cast<FairGeoVolume*> ( node );
172  if ( node->isSensitive() ) {
173  fSensNodes->AddLast( aVol );
174  } else {
175  fPassNodes->AddLast( aVol );
176  }
177  }
178  par->setChanged();
179  par->setInputVersion(fRun->GetRunId(),1);
180 
181  ProcessNodes ( volList );
182 }
