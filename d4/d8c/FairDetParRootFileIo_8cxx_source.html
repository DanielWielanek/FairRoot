51 Bool_t FairDetParRootFileIo::read(FairParSet *pPar) {
52  // generic read function for parameter containers
53  Text_t *name = (char *)pPar->GetName();
54  Int_t version = findInputVersion(name);
55 
56  // cout << "-I- FairDetParRootFileIo# " << name << " : " << version << endl;
57 
58  if (version <= 0) {
59  pPar->setInputVersion(-1, inputNumber);
60  return kFALSE;
61  }
62 
63  if (pPar->getInputVersion(inputNumber) == version &&
64  pPar->getInputVersion(inputNumber) != -1) {
65  return kTRUE;
66  }
67 
68  TKey *key = dynamic_cast<TKey *>(gDirectory->GetKey(name, version));
69  if (key) {
70  pPar->clear();
71  key->Read(pPar);
72  pPar->setInputVersion(version, inputNumber);
73  pPar->setChanged();
74  cout << "Container " << pPar->GetName() << " initialized from ROOT file."
75  << endl;
76  return kTRUE;
77  }
78  pPar->setInputVersion(-1, inputNumber);
79  return kFALSE;
80 }

82 Int_t FairDetParRootFileIo::write(FairParSet *pPar) {
83  // writes a parameter container to the ROOT file and returns the new version
84  // number (returns -1 if the file is not writable)
85  if (pFile) {
86  pFile->cd();
87  if (pFile->IsWritable()) {
88  Text_t *name = (char *)pPar->GetName();
89  pPar->Write(name);
90  pPar->setChanged(kFALSE);
91  FairRuntimeDb::instance()->setVersionsChanged(kTRUE);
92  return getMaxVersion(name);
93  } else {
94  Error(":write", "Output is not writable");
95  }
96  }
97  Error(":write", "No output open");
98  return -1;
99 }

112 Int_t FairDetParRootFileIo::findInputVersion(Text_t *name) {
113  // finds the input version to initialize the container given by name;
114  // returns -1 if the version cannot be determined
115  FairParVersion *currVers =
116  FairRuntimeDb::instance()->getCurrentRun()->getParVersion(name);
117  Int_t v = currVers->getInputVersion(inputNumber);
118  if (v > 0) {
119  return v;
120  } // predefined
121  FairRtdbRun *r = pFile->getRun();
122  // cout << "-I- FairDetParRootFileIo::findInputVersion " << r << endl;
123  if (!r) {
124  return -1;
125  } // run not in ROOT file
126  FairParVersion *vers = r->getParVersion(name);
127  if (!vers) {
128  return -1;
129  } // container not in ROOT file
130  return vers->getRootVersion();
131 }
