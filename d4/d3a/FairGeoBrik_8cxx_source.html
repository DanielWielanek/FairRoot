63 TArrayD* FairGeoBrik::calcVoluParam(FairGeoVolume* volu)
64 {
65  // calculates the parameters needed to create the shape
66  if (!volu) { return 0; }
67  FairGeoVector v=*(volu->getPoint(5)) - *(volu->getPoint(3));
68  v.abs();
69  v*=(1/20.);
70  for(Int_t i=0; i<nParam; i++) { param->AddAt(v(i),i); }
71  return param;
72 }
