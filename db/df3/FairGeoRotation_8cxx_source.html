92 void FairGeoRotation::setEulerAngles(const Double_t a,const Double_t b,
93  const Double_t c)
94 {
95  // sets the matrix calculating the values from the given Euler angles
96  const double deg2rad=0.01745329252;
97  Double_t s0=TMath::Sin(a*deg2rad);
98  Double_t c0=TMath::Cos(a*deg2rad);
99  Double_t s1=TMath::Sin(b*deg2rad);
100  Double_t c1=TMath::Cos(b*deg2rad);
101  Double_t s2=TMath::Sin(c*deg2rad);
102  Double_t c2=TMath::Cos(c*deg2rad);
103  rot[0]=c0*c1*c2 - s0*s2;
104  rot[1]=-c0*c1*s2 - s0*c2;
105  rot[2]=c0*s1;
106  rot[3]=s0*c1*c2 + c0*s2;
107  rot[4]=c0*c2 - s0*c1*s2;
108  rot[5]=s0*s1;
109  rot[6]=-s1*c2;
110  rot[7]=s1*s2;
111  rot[8]=c1;
112 }

115 Double_t FairGeoRotation::diff2(const FairGeoRotation& r) const
116 {
117  // calculates the square of the difference between 2 matrices
118  Double_t s=0;
119  for(Int_t i=0; i<9; i++) {
120  Double_t d=rot[i]-r(i);
121  s+=d*d;
122  }
