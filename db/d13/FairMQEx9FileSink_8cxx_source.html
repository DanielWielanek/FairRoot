106 void FairMQEx9FileSink::Run()
107 {
108  while (CheckCurrentState(RUNNING))
109  {
110  FairMQParts parts;
111 
112  if (Receive(parts, fInputChannelName) >= 0)
113  {
114 
115  TObject* tempObjects[10];
116  for ( int ipart = 0 ; ipart < parts.Size() ; ipart++ )
117  {
118  Ex9TMessage tm(parts.At(ipart)->GetData(), parts.At(ipart)->GetSize());
119  tempObjects[ipart] = (TObject*)tm.ReadObject(tm.GetClass());
120  for ( unsigned int ibr = 0 ; ibr < fBranchNames.size() ; ibr++ )
121  {
122  if ( strcmp(tempObjects[ipart]->GetName(),fBranchNames[ibr].c_str()) == 0 )
123  {
124  fOutputObjects[ibr] = tempObjects[ipart];
125  fTree->SetBranchAddress(fBranchNames[ibr].c_str(),&fOutputObjects[ibr]);
126  }
127  }
128  }
129  fTree->Fill();
130 
131  if ( strcmp(fAckChannelName.data(),"") != 0 ) {
132  unique_ptr<FairMQMessage> msg(NewMessage());
133  Send(msg, fAckChannelName);
134  }
135  }
136  }
