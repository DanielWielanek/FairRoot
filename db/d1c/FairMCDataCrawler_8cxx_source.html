135 void FairMCDataCrawler::GetNextStage(FairMultiLinkedData& startStage, Int_t stopStage)
136 {
137  FairMultiLinkedData* tempStage;
138  for (int i = 0; i < startStage.GetNLinks(); i++) {
139  FairLink actualLink = startStage.GetLink(i);
140 // if (fVerbose > 0) { std::cout << "Actual Link: " << actualLink << std::endl; }
141  if (actualLink.GetType() < 0) {
142  if (fStoreAllEndpoints) { AddToFinalStage(actualLink, 1); }
143  if (fVerbose > 2) {
144  std::cout << "ActualLink.Type < 0" << std::endl;
145  std::cout << "FinalStage: " << fFinalStage << std::endl;
146  std::cout << "---------------------" << std::endl;
147  }
148  if (fVerbose > 1) {
149  std::cout << std::endl;
150  }
151  } else if (actualLink.GetType() == stopStage) {
152  AddToFinalStage(actualLink, 1);
153  if (fVerbose > 2) {
154  std::cout << "ActualLink.Type == stopStage" << std::endl;
155  std::cout << "FinalStage: " << fFinalStage << std::endl;
156  std::cout << "---------------------" << std::endl;
157  }
158  if (fVerbose > 1) {
159  std::cout << std::endl;
160  }
161  } else if (actualLink.GetType() == fUltimateStage) {
162  if (fStoreAllEndpoints) { AddToFinalStage(actualLink, 1); }
163  if (fVerbose > 2) {
164  std::cout << "ActualLink.Type == UltimateStage" << std::endl;
165  std::cout << "FinalStage: " << fFinalStage << std::endl;
166  std::cout << "---------------------" << std::endl;
167  }
168  if (fVerbose > 1) {
169  std::cout << std::endl;
170  }
171  } else if (actualLink.GetType() == FairRootManager::Instance()->GetBranchId("EventHeader.")) {
172  if (fStoreAllEndpoints) { AddToFinalStage(actualLink, 1); }
173  if (fVerbose > 2) {
174  std::cout << "ActualLink.Type == EventHeader" << std::endl;
175  std::cout << "FinalStage: " << fFinalStage << std::endl;
176  std::cout << "---------------------" << std::endl;
177  }
178  if (fVerbose > 1) {
179  std::cout << std::endl;
180  }
181  } else {
182  tempStage = GetEntry(actualLink);
183  if (tempStage == 0) {
184  if (fStoreAllEndpoints) { AddToFinalStage(actualLink, 1); }
185  if (fVerbose > 2) {
186  std::cout << "TempStage == 0" << std::endl;
187  std::cout << "FinalStage: " << fFinalStage << std::endl;
188  std::cout << "---------------------" << std::endl;

190  if (fVerbose > 1) {
191  std::cout << std::endl;
192  }
193  continue;
194  }
195  if (fVerbose > 0) {
196 // std::cout << "TempStage Start";
197  std::cout << " // ";
198  actualLink.Print();
199  std::cout << " --> " << *tempStage;
200  }
201  if (tempStage->GetNLinks() == 0) {
202  if (fStoreAllEndpoints) { AddToFinalStage(actualLink,1); }
203  if (fVerbose > 0) {
204  std::cout << "tempStage->GetNLinks == 0" << std::endl;
205  std::cout << "FinalStage: " << fFinalStage << std::endl;
206  std::cout << "---------------------" << std::endl;
207  }
208  } else {
209  if(fStoreIntermediate) { AddToFinalStage(actualLink, 1); }
210  double tempStageWeight = 1;
211  double startLinkWeight = startStage.GetLink(i).GetWeight();
212 
213  if (fVerbose > 2) {
214  //std::cout << "Tempstage " << tempStage.GetSource() << ": weight " << tempStageWeight << std::endl;
215  std::cout << "StartLinkWeight " << startLinkWeight << std::endl;
216  }
217  //std::cout << " StageWeight: " << tempStageWeight << " startLinkWeight: " << startLinkWeight;
218  tempStage->MultiplyAllWeights(tempStageWeight);
219 
220  if ((tempStageWeight * startLinkWeight) == 0) {
221  tempStage->MultiplyAllWeights(tempStageWeight);

223  if (fVerbose > 2) {

226  }

231  }
232  }
233 
234  }

236  // std::cout<< "TempStage Stop: " << *tempStage << std::endl;
237 
238  GetNextStage(*tempStage, stopStage);
