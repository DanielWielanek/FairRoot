120 int FairProgOptions::ParseCmdLine(const int argc, char** argv, const po::options_description& desc, po::variables_map& varmap, bool allowUnregistered)
121 {
122  // get options from cmd line and store in variable map
123  // here we use command_line_parser instead of parse_command_line, to allow unregistered and positional options
124  if (allowUnregistered)
125  {
126  po::command_line_parser parser{argc, argv};
127  parser.options(desc).allow_unregistered();
128  po::parsed_options parsedOptions = parser.run();
129  po::store(parsedOptions, varmap);
130  }
131  else
132  {
133  po::store(po::parse_command_line(argc, argv, desc), varmap);
134  }
135 
136  // call the virtual NotifySwitchOption method to handle switch options like e.g. "--help" or "--version"
137  // return 1 if switch options found in varmap
138  if (NotifySwitchOption())
139  {
140  return 1;
141  }
142 
143  po::notify(varmap);
144  return 0;
145 }

232  // Method to overload.
233  // -> loop over variable map and print its content
234  // -> In this example the following types are supported:

237 
238  MapVarValInfo_t mapinfo;

240  // get string length for formatting and convert varmap values into string

244  int maxLengthDefault = 0;

249  Max(maxLength1st, m.first.length());

264

268  maxLength2nd = 100;
269  }

271 
272  // maxLength2nd = 200;
273 
274  // formatting and printing
275 
276  LOG(INFO) << setfill ('*') << setw (totalLength + 3) << "*";// +3 because of string " = "
277  string PrintOptionsTitle = " Program options found ";

279  int leftSpaceLength = 0;
