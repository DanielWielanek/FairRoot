617 void FairMCApplication::Stepping()
618 {
619 // User actions at each step
620 // ---
621  // Work around for Fluka VMC, which does not call
622  // MCApplication::PreTrack()
623  static Int_t TrackId = 0;
624  if ( fMcVersion ==2 && gMC->GetStack()->GetCurrentTrackNumber() != TrackId ) {
625  PreTrack();
626  TrackId = gMC->GetStack()->GetCurrentTrackNumber();
627  }
628 
629 
630  // Check if the volume with id is in the volume multimap.
631  // If it is not in the map the volume is not a sensitive volume
632  // and we do not call nay of our ProcessHits functions.
633 
634  // If the volume is in the multimap, check in second step if the current
635  // copy is alredy inside the multimap.
636  // If the volume is not in the multimap add the copy of the volume to the
637  // multimap.
638  // In any case call the ProcessHits function for this specific detector.
639  Int_t copyNo;
640  Int_t id = gMC->CurrentVolID(copyNo);
641  Bool_t InMap =kFALSE;
642  fDisVol=0;
643  fDisDet=0;
644  Int_t fCopyNo=0;
645  fVolIter =fVolMap.find(id);
646  if (fVolIter!=fVolMap.end()) {
647  do {
648  fDisVol=fVolIter->second;
649  fCopyNo=fDisVol->getCopyNo();
650  if(copyNo==fCopyNo) {
651  fDisDet=fDisVol->GetDetector();
652  if (fDisDet) {
653  fDisDet->ProcessHits(fDisVol);
654  }
655  InMap=kTRUE;
656  break;
657  }
658  fVolIter++;
659  } while(fVolIter!=fVolMap.upper_bound(id));
660  // if(fDisVol && !InMap) { // fDisVolume is set previously, no check needed
661  if(!InMap) {
662  FairVolume* fNewV=new FairVolume( gMC->CurrentVolName(), id);
663  fNewV->setMCid(id);
664  fNewV->setModId(fDisVol->getModId());
665  fNewV->SetModule(fDisVol->GetModule());
666  fNewV->setCopyNo(copyNo);
667  fVolMap.insert(pair<Int_t, FairVolume* >(id, fNewV));
668  fDisDet=fDisVol->GetDetector();
669  if ( fDisDet) {
670  fDisDet->ProcessHits(fNewV);
671  }
672  }
673  }
674 
675  // If information about the tracks should be stored the information as to be
676  // stored for any step.
677  // Information about each single step has also to be stored for the other
678  // special run modes of the simulation which are used to store information
679  // about
680  // 1.) Radiation length in each volume
681  // 2.) Energy deposition in each volume
682  // 3.) Fluence of particles through a defined plane which can be anywhere
683  // in the geometry. This plane has not to be correlated with any real
684  // volume
685  if(fTrajAccepted) {
686  if(gMC->TrackStep() > fTrajFilter->GetStepSizeCut()) {
687  gMC->TrackPosition(fTrkPos);
688  fTrajFilter->GetCurrentTrk()->AddPoint(fTrkPos.X(), fTrkPos.Y(), fTrkPos.Z(), fTrkPos.T());
689  }
690  }
691  if(fRadLenMan) {
