98 void ParameterMQServer::Run()
99 {
100  string parameterName = "";
101  FairParGenericSet* par = nullptr;
102 
103  while (CheckCurrentState(RUNNING))
104  {
105  unique_ptr<FairMQMessage> req(fTransportFactory->CreateMessage());
106 
107  if (fChannels.at(fChannelName).at(0).Receive(req) > 0)
108  {
109  string reqStr(static_cast<char*>(req->GetData()), req->GetSize());
110  LOG(INFO) << "Received parameter request from client: \"" << reqStr << "\"";
111 
112  size_t pos = reqStr.rfind(",");
113  string newParameterName = reqStr.substr(0, pos);
114  int runId = stoi(reqStr.substr(pos + 1));
115  LOG(INFO) << "Parameter name: " << newParameterName;
116  LOG(INFO) << "Run ID: " << runId;
117 
118  LOG(INFO) << "Retrieving parameter...";
119  // Check if the parameter name has changed to avoid getting same container repeatedly
120  if (newParameterName != parameterName)
121  {
122  parameterName = newParameterName;
123  par = static_cast<FairParGenericSet*>(fRtdb->getContainer(parameterName.c_str()));
124  }
125  fRtdb->initContainers(runId);
126 
127  LOG(INFO) << "Sending following parameter to the client:";
128  if (par)
129  {
130  par->print();
131 
132  TMessage* tmsg = new TMessage(kMESS_OBJECT);
133  tmsg->WriteObject(par);
134 
135  unique_ptr<FairMQMessage> reply(fTransportFactory->CreateMessage(tmsg->Buffer(), tmsg->BufferSize(), free_tmessage, tmsg));
136 
137  fChannels.at(fChannelName).at(0).Send(reply);
138  }
139  else
140  {
141  LOG(ERROR) << "Parameter uninitialized!";
142  }
143  }
144  }
145 }
