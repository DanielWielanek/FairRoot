211 void FairRootManager::Register(const char* name, const char* folderName , TNamed* obj, Bool_t toFile)
212 {
213  FairMonitor::GetMonitor()->RecordRegister(name,folderName,toFile);
214 
215  // Security check. If the the name is equal the folder name there are problems with reading
216  // back the data. Instead of the object inside the folder the RootManger will return a pointer
217  // to the folder. To avoid such problems we check here if both strings are equal and stop the
218  // execution with some error message if this is the case.
219  if (strcmp (name, folderName) == 0 ) {
220  LOG(FATAL) << "The names for the object name "
221  << name << " and the folder name "
222  << folderName <<" are equal. This isn't allowed. So we stop the execution at this point. Pleae change either the name or the folder name."
223  << FairLogger::endl;
224  }
225 
226  if(toFile) {
227  TFolder* folder=0;
228  TFolder* f=0;
229  if(fCbmout==0) {
230  f=static_cast<TFolder*>(fCbmroot->FindObjectAny(folderName));
231  if(f==0) {
232  folder= fCbmroot->AddFolder(folderName,folderName);
233  } else {
234  folder=f;
235  }
236  } else {
237  f=static_cast<TFolder*>(fCbmout->FindObjectAny(folderName));
238  if(f==0) {
239  folder= fCbmout->AddFolder(folderName,folderName);
240  } else {
241  folder=f;
242  }
243  }
244  obj->SetName(name);
245  folder->Add(obj);
246  }
247  AddMemoryBranch(name, obj );
248  //cout << " FairRootManager::Register Adding branch:(Obj) " << name << " In folder : " << folderName << endl;
249 
250  AddBranchToList(name);
251 
252  if (toFile == kFALSE) {
253  FairLinkManager::Instance()->AddIgnoreType(GetBranchId(name));
254  }
255 }
