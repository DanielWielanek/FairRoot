425 {

428  // back the data. Instead of the object inside the folder the RootManger will return a pointer

430  // execution with some error message if this is the case.

432  LOG(FATAL) << "The names for the object name "

435  << FairLogger::endl;
436  }
437 
438  if(toFile) {
439  TFolder* folder=0;

442  f=(TFolder*)fCbmroot->FindObjectAny(folderName);
443  if(f==0) {
444  folder= fCbmroot->AddFolder(folderName,folderName);
445  } else {
446  folder=f;
447  }
448  } else {
449  f=(TFolder*)fCbmout->FindObjectAny(folderName);
450  if(f==0) {
451  folder= fCbmout->AddFolder(folderName,folderName);
452  } else {
453  folder=f;
454  }
455  }
456  obj->SetName(name);
457  folder->Add(obj);
458  }
459  AddMemoryBranch(name, obj );

461 
462  AddBranchToList(name);
463
