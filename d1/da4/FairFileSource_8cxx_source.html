241 void FairFileSource::AddFriendsToChain()
242 {
243  // Loop over all Friend files and extract the type. The type is defined by
244  // the tree which is stored in the file. If there is already a chain of with
245  // this type of tree then the file will be added to this chain.
246  // If there is no such chain it will be created.
247  //
248  // Check if the order of runids and the event numbers per runid for all
249  // friend chains is the same as the one defined by the input chain.
250  // TODO: Should the order be corrected or should the execution be stopped.
251  // The order in the input tree defined by the order in which the files have
252  // been added. A file is defined by the runid.
253 
254  // In the old way it was needed sometimes to add a freind file more
255  // than once. This is not needed any longer, so we remove deuplicates
256  // from the list and display a warning.
257  std::list<TString> friendList;
258  std::list<TString>::iterator iter1;
259  for(iter1 = fFriendFileList.begin();
260  iter1 != fFriendFileList.end(); iter1++) {
261  if (find(friendList.begin(), friendList.end(), (*iter1))
262  == friendList.end()) {
263  friendList.push_back(*iter1);
264  }
265  }
266  // TODO: print a warning if it was neccessary to remove a filname from the
267  // list. This can be chacked by comparing the size of both list
268 
269  TFile* temp = gFile;
270 
271  Int_t friendType = 1;
272  // Loop over all files which have been added as friends
273  for(iter1 = friendList.begin();
274  iter1 != friendList.end(); iter1++) {
275  std::list<TString>::iterator iter;
276  TString inputLevel;
277  // Loop over all already defined input levels to check if this type
278  // of friend tree is already added.
279  // If this type of friend tree already exist add the file to the
280  // then already existing friend chain. If this type of friend tree
281  // does not exist already create a new friend chain and add the file.
282  Bool_t inputLevelFound = kFALSE;
283  TFile* inputFile;
284  for ( iter = fInputLevel.begin(); iter !=fInputLevel.end(); iter++ ) {
285  inputLevel = (*iter);
286 
287  inputFile = new TFile((*iter1));
288  if (inputFile->IsZombie()) {
289  LOG(FATAL) << "Error opening the file " << (*iter).Data()
290  << " which should be added to the input chain or as friend chain" << FairLogger::endl;
291  }
292 
293  // Check if the branchlist is already stored in the map. If it is
294  // already stored add the file to the chain.
295  Bool_t isOk = CompareBranchList(inputFile, inputLevel);
296  if ( isOk ) {
297  inputLevelFound = kTRUE;
298  inputFile->Close();
299  continue;
300  }
301  inputFile->Close();
302  }
303  if (!inputLevelFound) {
304  inputLevel= Form("FriendTree_%i",friendType);
305  CreateNewFriendChain((*iter1), inputLevel);
306  friendType++;
307  }
308 
309  TChain* chain = (TChain*) fFriendTypeList[inputLevel];
310  chain->AddFile((*iter1), 1234567890, "cbmsim");
311  }
312  gFile=temp;
313 
314  // Check if all friend chains have the same runids and the same
315  // number of event numbers as the corresponding input chain
316  // CheckFriendChains();
317 
318  // Add all the friend chains which have been created to the
319  // main input chain.
320  std::map< TString, TChain* >::iterator mapIterator;
321  for (mapIterator = fFriendTypeList.begin();
322  mapIterator != fFriendTypeList.end(); mapIterator++ ) {
323 
324  TChain* chain = (TChain*)mapIterator->second;
325  fInChain->AddFriend(chain);
326  }
327 
328  // Print some output about the input structure
329  PrintFriendList();
330 
331 }

366 void FairFileSource::CheckFriendChains()
367 {
368  std::multimap< TString, std::multimap<TString,TArrayI> >::iterator it1;
369  std::multimap<TString,TArrayI> map1;
370 
371  // Get the structure from the input chain
372  it1=fRunIdInfoAll.find("InputChain");
373  map1 = it1->second;
374  std::vector<Int_t> runid;
375  std::vector<Int_t> events;
376  std::multimap<TString,TArrayI>::iterator it;
377  for ( it=map1.begin() ; it != map1.end(); it++ ) {
378  TArrayI bla = (*it).second;
379  runid.push_back(bla[0]);
380  events.push_back(bla[1]);
381  }
382 
383  // Now loop over all chains except the input chain and comapare the
384  // runids and event numbers.
385  // If there is a mismatch stop the execution.
386  Int_t errorFlag = 0;
387  TString inputLevel;
388  std::list<TString>::iterator listit;
389  for ( listit=fInputLevel.begin() ; listit != fInputLevel.end(); listit++ ) {
390  inputLevel = (*listit);
391  if ( !inputLevel.Contains("InputChain") ) {
392  it1=fRunIdInfoAll.find(inputLevel);
393  map1 = it1->second;
394  if ( runid.size() != map1.size()) {
395  errorFlag = 1;
396  // goto error_label;
397  break;
398  }
399  Int_t counter = 0;
400  for ( it=map1.begin() ; it != map1.end(); it++ ) {
401  TArrayI bla = (*it).second;
402  if ( (bla[0] != runid[counter]) || (bla[1] != events[counter]) ) {
403  errorFlag = 2;
404  // goto error_label;
405  break;
406  }
407  counter++;
408  }
409  if (errorFlag>0) {
410  break;
411  }
412  }
413  }
414 
415  // Use goto to leave double loop at once in case of error
416  // error_label:
417  if (errorFlag>0) {
418  LOG(ERROR) << "The input chain and the friend chain " << inputLevel.Data()
419  << " have a different structure:" << FairLogger::endl;
420  if (errorFlag == 1) {
421  LOG(ERROR) << "The input chain has the following runids and event numbers:" << FairLogger::endl;
422  for ( Int_t i=0; i<runid.size(); i++) {
423  LOG(ERROR) << " - Runid " << runid[i] << " with " << events[i]
424  << " events" << FairLogger::endl;
425  }
426  LOG(ERROR) << "The " << inputLevel.Data()
427  << " chain has the following runids and event numbers:"
428  << FairLogger::endl;
429  for ( it=map1.begin() ; it != map1.end(); it++ ) {
430  TArrayI bla = (*it).second;
431  LOG(ERROR) << " - Runid " << bla[0] << " with " << bla[1]
432  << " events" << FairLogger::endl;
433  }
434  }
435  if (errorFlag == 2) {
436  Int_t counter = 0;
437  for ( it=map1.begin() ; it != map1.end(); it++ ) {
438  TArrayI bla = (*it).second;
439  LOG(ERROR) << "Runid Input Chain, " << inputLevel.Data()
440  << " chain: " << bla[0] << ", " << runid[counter]
441  << FairLogger::endl;
442  LOG(ERROR) << "Event number Input Chain, " << inputLevel.Data()
443  << " chain: " << bla[1]
444  << ", " << events[counter] << FairLogger::endl;
445  counter++;
446  }
447  }
448  LOG(FATAL) << "Event structure mismatch" << FairLogger::endl;
449  }
450 }

501 Bool_t FairFileSource::CompareBranchList(TFile* fileHandle, TString inputLevel)
502 {
503  // fill a set with the original branch structure
504  // This allows to use functions find and erase
505  std::set<TString> branches;
506  std::list<TString>::const_iterator iter;
507  for(iter = fCheckInputBranches[inputLevel]->begin();
508  iter != fCheckInputBranches[inputLevel]->end(); iter++) {
509  branches.insert(*iter);
510  }
511 
512  // To do so we have to loop over the branches in the file and to compare
513  // the branches in the file with the information stored in
514  // fCheckInputBranches["InputChain"]. If both lists are equal everything
515  // is okay
516 
517  // Get The list of branches from the input file one by one and compare
518  // it to the reference list of branches which is defined for this tree.
519  // If a branch with the same name is found, this branch is removed from
520  // the list. If in the end no branch is left in the list everything is
521  // fine.
522  set<TString>::iterator iter1;
523  TList* list= dynamic_cast <TList*> (fileHandle->Get("BranchList"));
524  if(list) {
525  TObjString* Obj=0;
526  for(Int_t i =0; i< list->GetEntries(); i++) {
527  Obj=dynamic_cast <TObjString*> (list->At(i));
528  iter1=branches.find(Obj->GetString().Data());
529  if (iter1 != branches.end() ) {
530  branches.erase (iter1);
531  } else {
532  // Not found is an error because branch structure is
533  // different. It is impossible to add to tree with a
534  // different branch structure
535  return kFALSE;
536  }
537  }
538  }
539  // If the size of branches is !=0 after removing all branches also in the
540  // reference list, this is also a sign that both branch list are not the
541  // same
542  if (branches.size() != 0 ) {
543  return kFALSE;
544  }
545 
546  return kTRUE;
547 }
