60 Bool_t FairGeoOldAsciiIo::open(const char* fname,const Text_t* status)
61 {

63  close();
64  if (!file) { file=new std::fstream(); }
65  else { (file->clear()); }
66  if (!filedir.IsNull()) { filename=filedir+"/"+fname; }
67  else { filename=fname; }
68  filename=filename.Strip();

70  file->open(filename,ios::in);
71  writable=kFALSE;
72  } else {
73  if (strcmp(status,"out")==0) {
74  file->open(filename,ios::in);
75  if (!isOpen()) {
76  file->close();
77  file->clear();
78  file->open(filename,ios::out);
79  writable=kTRUE;
80  } else {
81  file->close();
82  Error("open","Output file %s exists already and will not be recreated.",
83  filename.Data());
84  return kFALSE;
85  }
86  } else { Error("open","Invalid file option!"); }
87  }
88  if (file->rdbuf()->is_open()==0) {
89  Error("open","Failed to open file %s",filename.Data());
90  return kFALSE;
91  }
92  return kTRUE;

127 Bool_t FairGeoOldAsciiIo::read(FairGeoSet* set,FairGeoMedia* media)
128 {
129  // Reads the geometry from file and converts it to the new format
130  if (!isOpen() || writable || set==0) { return kFALSE; }
131  std::fstream& fin=*file;
132  fin.clear();
133  fin.seekg(0,ios::beg);
134  FairGeoNode* volu=0;
135  Int_t sensitivity=0, na=0;
136  TList* volumes=set->getListOfVolumes();
137  FairGeoShapes* pShapes=set->getShapes();
138  while(!fin.eof()) {
139  // Read volumeName
140  TString volumeName = "";
141  fin >> volumeName;
142  if (fin.eof()) { break; }
143  volu=new FairGeoNode;
144  volu->SetName(volumeName);
145  // Read sensitivity
146  fin >> sensitivity;
147 // Why this additional integer only in trd file ????????????
148  if (sensitivity>0) { fin >> na; }
149  // Read motherName
150  TString motherName = "";
151  fin >> motherName;
152  FairGeoNode* mother=0;
153  if (motherName=="world") { mother=set->getMasterNode("cave"); }
154  else { mother=set->getVolume(motherName.Data()); }
155  volu->setMother(mother);
156  // Read position and rotation matrix
157  Double_t r[9], t[3];
158  fin>>t[0]>>t[1]>>t[2];
159  for(Int_t kk=0; kk<3; kk++) { t[kk]*=10.; }
160  for(Int_t i=0; i<9; i++) { fin>>r[i]; }
161  FairGeoTransform& tf=volu->getTransform();
162  tf.setRotMatrix(r);
163  tf.setTransVector(t);
164  // Read material
165  TString materialName = "";
166  fin >> materialName;
167  FairGeoMedium* medium=media->getMedium(materialName);
168  if (!medium) {
169  medium=new FairGeoMedium(materialName);
170  media->addMedium(medium);
171  }
172  volu->setMedium(medium);
173  // Read shape
174  TString type = "";
175  fin >> type;
176  FairGeoBasicShape* sh=pShapes->selectShape(type);
177  if (sh) { volu->setShape(sh); }
178  else {
179  cerr << "Shape "<<type<<" not supported."<<endl;
180  return kFALSE;
181  }
182  Int_t npar = sh->getNumParam();
183  Double_t* par = new Double_t[npar];
184  for (Int_t ik=0; ik<npar; ik++) {
185  fin >> par[ik];
186  }
187  Bool_t rc=calculateShapePoints(par,volu);
188  delete[] par;
189  if (!rc) {
190  cerr << "Conversion for shape "<<type<<" not implemented."<<endl;
191  return kFALSE;
192  }
193  // Check of volume end
194  TString control = "";
195  fin >> control;
196  if ( control != "#fi" && !fin.eof()) {
197  cerr << "End of File section is '" << control << "' instead of '#fi'." << endl;
198  return kFALSE;
199  }
200  volu->print();
201  volumes->Add(volu);
202  }
203  return kTRUE;
204 }

206 Bool_t FairGeoOldAsciiIo::calculateShapePoints(Double_t* par,FairGeoNode* volu)
207 {
208  FairGeoBasicShape* sh=volu->getShapePointer();
209  TString shName=sh->GetName();
210  Int_t n=sh->getNumPoints();
211  volu->createPoints(n);
212  Bool_t rc=kTRUE;
213  if (shName == "BOX ") {
214  Double_t x=par[0]*10.;
215  Double_t y=par[1]*10.;
216  Double_t z=par[2]*10.;
217  volu->setPoint(0, x,-y,-z);
218  volu->setPoint(1, x, y,-z);
219  volu->setPoint(2,-x, y,-z);
220  volu->setPoint(3,-x,-y,-z);
221  volu->setPoint(4, x,-y, z);
222  volu->setPoint(5, x, y, z);
223  volu->setPoint(6,-x, y, z);
224  volu->setPoint(7,-x,-y, z);
225  } else if (shName == "TUBE") {
226  Double_t z=par[2]*10.;
227  volu->setPoint(0,0.,0.,-z);
228  volu->setPoint(1,par[0]*10.,par[1]*10.,0.);
229  volu->setPoint(2,0.,0.,z);
230  } else if (shName == "TUBS") {
231  Double_t z=par[2]*10.;
232  Double_t a=par[3]/6.28318548*360;
233  volu->setPoint(0,0.,0.,-z);
234  volu->setPoint(1,par[0]*10.,par[1]*10.,0.);
235  volu->setPoint(2,0.,0.,z);
236  volu->setPoint(3,a,par[4]/6.28318548*360+a,0.);
237  } else if (shName == "CONE") {
238  Double_t z=par[4]*10.;
239  volu->setPoint(0,0.,0.,-z);
240  volu->setPoint(1,par[0]*10.,par[1]*10.,0.);
241  volu->setPoint(2,0.,0.,z);
242  volu->setPoint(3,par[2]*10.,par[3]*10.,0.);
243  } else if (shName == "CONS") {
244  Double_t z=par[4]*10.;
245  Double_t a=par[5]/6.28318548*360;
246  volu->setPoint(0,0.,0.,-z);
247  volu->setPoint(1,par[0]*10.,par[1]*10.,0.);
248  volu->setPoint(2,0.,0.,z);
249  volu->setPoint(3,par[2]*10.,par[3]*10.,0.);
250  volu->setPoint(4,a,par[6]/6.28318548*360+a,0.);
251  } else if (shName == "SPHE") {
252  Double_t a=par[4]/6.28318548*360;
253  volu->setPoint(0,par[0]*10.,par[1]*10.,0.);
254  volu->setPoint(1,par[1],par[2],0.);

259  volu->setPoint(1,par[0]*10.,par[1]*10.,0.);
260  volu->setPoint(2,0.,0.,z);
261  } else { rc=kFALSE; }
262  return rc;
263 }
