80 void FairPointSetDraw::Exec(Option_t* option)

86  q->SetOwnIds(kTRUE);

89  q->SetMarkerStyle(fStyle);
90  //std::cout << "fPointList: " << fPointList << " " << fPointList->GetEntries() << std::endl;
91  for (Int_t i=0; i<npoints; ++i) {
92  TObject* p=(TObject*)fPointList->At(i);
93  if(p!=0) {
94  TVector3 vec(GetVector(p));
95  q->SetNextPoint(vec.X(),vec.Y(), vec.Z());
96  q->SetPointId(GetValue(p, i));
97  }
98  }
99  gEve->AddElement(q);
100  gEve->Redraw3D(kFALSE);
101  fq=q;
