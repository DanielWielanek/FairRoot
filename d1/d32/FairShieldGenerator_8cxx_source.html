79 Bool_t FairShieldGenerator::ReadEvent(FairPrimaryGenerator* primGen)
80 {
81 
82  // Check for input file
83  if ( ! fInputFile->is_open() ) {
84  LOG(ERROR) << "FairShieldGenerator: Input file not open!"
85  << FairLogger::endl;
86  return kFALSE;
87  }
88 
89 
90  // Define event variable to be read from file
91  Int_t eventId = 0;
92  Int_t nTracks = 0;
93  Double_t pBeam = 0.;
94  Double_t b = 0.;
95 
96  // Define track variables to be read from file
97  Int_t iPid = 0;
98  Int_t iMass = 0;
99  Int_t iCharge = 0;
100  Double_t px = 0.;
101  Double_t py = 0.;
102  Double_t pz = 0.;
103  Int_t pdgType = 0;
104 
105  // Read event header line from input file
106  *fInputFile >> eventId >> nTracks >> pBeam >> b;
107 
108  // If end of input file is reached : close it and abort run
109  if ( fInputFile->eof() ) {
110  LOG(INFO) << "FairShieldGenerator: End of input file reached "
111  << FairLogger::endl;
112  CloseInput();
113  return kFALSE;
114  }
115 
116  LOG(INFO) << "FairShieldGenerator: Event " << eventId << ", pBeam = "
117  << pBeam << "GeV, b = " << b << " fm, multiplicity " << nTracks
118  << FairLogger::endl;
119 
120  // Loop over tracks in the current event
121  for (Int_t itrack=0; itrack<nTracks; itrack++) {
122 
123  // Read PID and momentum from file
124  *fInputFile >> iPid >> iMass >> iCharge >> px >> py >> pz;
125 
126  // Case ion
127  if ( iPid == 1000 ) {
128  char ionName[20];
129  sprintf(ionName, "Ion_%d_%d", iMass, iCharge);
130  TParticlePDG* part = fPDG->GetParticle(ionName);
131  if ( ! part ) {
132  LOG(WARNING) << "FairShieldGenerator::ReadEvent: Cannot find "
133  << ionName << " in database!"
134  << FairLogger::endl;
135  continue;
136  }
137  pdgType = part->PdgCode();
138  } else { pdgType = fPDG->ConvertGeant3ToPdg(iPid); } // "normal" particle
139 
140  // Give track to PrimaryGenerator
141  primGen->AddTrack(pdgType, px, py, pz, 0., 0., 0.);
142 
143  }
144 
145 
146  return kTRUE;
147 }

169 // ----- Private method RegisterIons ----------------------------------
170 Int_t FairShieldGenerator::RegisterIons()
171 {
172 
173  Int_t nIons = 0;
174  Int_t eventId, nTracks, iPid, iMass, iCharge;
175  Double_t pBeam, b, px, py, pz;
176  fIonMap.clear();
177 
178  while ( ! fInputFile->eof()) {
179 
180  *fInputFile >> eventId >> nTracks >> pBeam >> b;
181  if ( fInputFile->eof() ) { continue; }
182  for (Int_t iTrack=0; iTrack<nTracks; iTrack++) {
183  *fInputFile >> iPid >> iMass >> iCharge >> px >> py >> pz;
184  if ( iPid == 1000 ) { // ion
185  char buffer[20];
186  sprintf(buffer, "Ion_%d_%d", iMass, iCharge);
187  TString ionName(buffer);
188  if ( fIonMap.find(ionName) == fIonMap.end() ) { // new ion
189  FairIon* ion = new FairIon(ionName, iCharge, iMass, iCharge);
190  fIonMap[ionName] = ion;
191  nIons++;
192  } // new ion
193  } // ion
194  } // track loop
195 
196  } // event loop
197 
198  FairRunSim* run = FairRunSim::Instance();
199  map<TString, FairIon*>::iterator mapIt;
200  for (mapIt=fIonMap.begin(); mapIt!=fIonMap.end(); mapIt++) {
201  FairIon* ion = (*mapIt).second;
202  run->AddNewIon(ion);
203  }
204 
205  return nIons;
206 }
