168 void FairMQExampleShmSampler::ListenForAcks()
169 {
170  while (CheckCurrentState(RUNNING))
171  {
172  unique_ptr<FairMQMessage> msg(NewMessage());
173 
174  if (Receive(msg, "ack") >= 0)
175  {
176  string key(static_cast<char*>(msg->GetData()), msg->GetSize());
177  LOG(TRACE) << "Received ack for: " << key;
178  {
179  unique_lock<mutex> containerLock(fContainerMutex);
180  if (fLocalPtrs.find(key) != fLocalPtrs.end())
181  {
182  fLocalPtrs.erase(key);
183  }
184  else
185  {
186  LOG(WARN) << "Received ack for a key not contained in the container";
187  }
188  LOG(TRACE) << fLocalPtrs.size();
189  }
190  fAckCV.notify_all();
191  }
192  }

195 void FairMQExampleShmSampler::Log(const int intervalInMs)
196 {
197  timestamp_t t0 = get_timestamp();
198  timestamp_t t1;
199  timestamp_t msSinceLastLog;
200 
201  double mbPerSecOut = 0;
202  double msgPerSecOut = 0;
203 
204  while (true)
205  {
206  try
207  {
208  t1 = get_timestamp();
209 
210  msSinceLastLog = (t1 - t0) / 1000.0L;
211 
212  mbPerSecOut = (static_cast<double>(fBytesOutNew - fBytesOut) / (1024. * 1024.)) / static_cast<double>(msSinceLastLog) * 1000.;
213  fBytesOut = fBytesOutNew;
214 
215  msgPerSecOut = static_cast<double>(fMsgOutNew - fMsgOut) / static_cast<double>(msSinceLastLog) * 1000.;
216  fMsgOut = fMsgOutNew;
217 
218  LOG(DEBUG) << fixed
219  << setprecision(0) << "out: " << msgPerSecOut << " msg ("
220  << setprecision(2) << mbPerSecOut << " MB)\t("
221  << SegmentManager::Instance().Segment()->get_free_memory() / (1024. * 1024.) << " MB free)";
222 
223  t0 = t1;
224  boost::this_thread::sleep(boost::posix_time::milliseconds(intervalInMs));
225  }
226  catch (boost::thread_interrupted&)
227  {
228  break;
229  }
230  }
231 }
