75 void FairTrackParam::Momentum(TVector3& mom) const
76 {
77  Double_t p = ( TMath::Abs(fQp)>1.e-4 ) ?1./TMath::Abs(fQp) :1.e4;
78  Double_t pz = TMath::Sqrt( p*p / ( fTx*fTx + fTy*fTy + 1 ) );
79  Double_t px = fTx * pz;
80  Double_t py = fTy * pz;
81  mom.SetXYZ(px, py, pz);
82 
83 }

102 Double_t FairTrackParam::GetCovariance(Int_t i, Int_t j) const
103 {
104  if ( i<0 || j<0 || i>4 || j>4 ) {
105  LOG(error) << "FairTrackParam::GetCovariance: Invalid index pair (" << i << "," << j << ") !";
106  return 0;
107  }
108  if (i>j) {
109  Int_t k = i;
110  i = j;
111  j = k;
112  }
113  Int_t index = 0;
114  if (i==0) { index = j; }
115  else if (i==1) { index = 4 + j; }
116  else if (i==2) { index = 7 + j; }
117  else if (i==3) { index = 9 + j; }
118  else if (i==4) { index = 10 + j; }
119  return fCovMatrix[index];
120 }
