54 void PixelDigiWriteToFile::Exec(Option_t* /*opt*/)
55 {
56  Reset();
57 
58  Int_t nofDigis = fDigis->GetEntriesFast();
59 
60  for (Int_t ifile = 0; ifile < fNofOutputFiles; ifile++) {
61  fOutputFiles[ifile] << "EVENT BEGIN; RUNID = " << fRunId << "; MCENTRYNO = " << fMCEntryNo << "; PARTNO = " << ifile << ";" << std::endl;
62  }
63 
64  for (Int_t iDigi = 0; iDigi < nofDigis; iDigi++) {
65  PixelDigi* currentDigi = static_cast<PixelDigi*>(fDigis->At(iDigi));
66 
67  Int_t detId = currentDigi->GetDetectorID();
68  Int_t feId = currentDigi->GetFeID();
69  Int_t col = currentDigi->GetCol();
70  Int_t row = currentDigi->GetRow();
71  Double_t charge = currentDigi->GetCharge();
72 
73  Int_t fileToSave = 0;
74  if (fDivideLevel == 1) {
75  fileToSave = detId/256 - 1;
76  } else if (fDivideLevel == 2) {
77  fileToSave = (detId/256 - 1)*4 + (detId%256 -1);
78  }
79  fOutputFiles[fileToSave] << detId << " " << feId << " " << col << " " << row << " " << charge << std::endl;
80  }
81 
82  for (Int_t ifile = 0; ifile < fNofOutputFiles; ifile++) {
83  fOutputFiles[ifile] << "EVENT END" << std::endl;
84  }
85 
86  fMCEntryNo ++;
87 }

89 InitStatus PixelDigiWriteToFile::Init()
90 {
91  // Get input array
92  FairRootManager* ioman = FairRootManager::Instance();
93 
94  if (!ioman) LOG(fatal) << "No FairRootManager";
95  fDigis = static_cast<TClonesArray*>(ioman->GetObject("PixelDigis"));
96 
97  if (!fDigis)
98  LOG(warn) << "PixelDigiWriteToFile::Init() No input PixelDigis array!";
99 
100  LOG(info) << "-I- " << fName.Data() << "::Init(). Initialization succesfull.";
101 
102  fRunId = ioman->GetRunId();
103 
104  if (fDivideLevel == 0) {
105  fNofOutputFiles = 1;
106  fOutputFiles[0].open(fOutputFileName.Data(),std::fstream::out);
107  }
108  else {
109  if (fDivideLevel == 1) {
110  fNofOutputFiles = 3; // 1 file per station (3 stations)
111  } else if (fDivideLevel == 2) {
112  fNofOutputFiles = 12; // 1 file per sensor (3 stations times 4 sensors)
113  } else {
114  LOG(fatal) << "PixelDigiWriteToFile::Init(), fDivideLevel = " << fDivideLevel << " unknown, it has to be in the range <0,2>";
115  return kFATAL;
116  }
117 
118  for (Int_t ifile = 0; ifile < fNofOutputFiles; ifile++) {
119  TString fileName = fOutputFileName;
120  TString uniqFile = Form(".p%d.",ifile);
121  fileName.Replace(fileName.Last('.'),1,uniqFile.Data());
122  fOutputFiles[ifile].open(fileName.Data(),std::fstream::out);
123  }
124  }
125 
126  return kSUCCESS;
127 }
