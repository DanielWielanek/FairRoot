75 void PixelDigitize::Exec(Option_t* /*opt*/)
76 {
77  Reset();
78 
79  LOG(debug) << "PixelDigitize::Exec() EVENT " << fTNofEvents;
80 
81  fTNofEvents++;
82 
83  fNPoints = fPoints->GetEntriesFast();
84  fTNofPoints+= fNPoints;
85 
86  for ( Int_t iPoint = 0 ; iPoint < fNPoints ; iPoint++ ) {
87  PixelPoint* currentPixelPoint = static_cast<PixelPoint*>(fPoints->At(iPoint));
88 
89  Double_t posIn[3] = { currentPixelPoint->GetX(),
90  currentPixelPoint->GetY(),
91  currentPixelPoint->GetZ() };
92 
93  Int_t detId = currentPixelPoint->GetDetectorID();
94  TString nodeName = Form("/cave/Pixel%d_%d",detId/256,detId%256);
95 
96  gGeoManager->cd(nodeName.Data());
97  TGeoNode* curNode = gGeoManager->GetCurrentNode();
98 
99  if (!curNode) LOG(fatal) << "PixelDigitize::Exec() node \"" << nodeName.Data() << "\" unknown!";
100 
101  Double_t locPosIn[3];
102 
103  curNode->MasterToLocal(posIn,locPosIn);
104 
105  // The local coordinate system has the origin set to the center of the volume.
106  // To calculate the pixel number, it is necessary to move the origin to the bottom left corner.
107  TGeoVolume* actVolume = gGeoManager->GetCurrentVolume();
108  TGeoBBox* actBox = static_cast<TGeoBBox*>(actVolume->GetShape());
109  // Double_t sensorWidth = 2.*actBox->GetDX();
110  // Double_t sensorHeight = 2.*actBox->GetDY();
111  locPosIn[0] += actBox->GetDX();
112  locPosIn[1] += actBox->GetDY();
113 
114  LOG(debug) << "GLOB PNT " << detId << " POSITION: " << posIn[0] << " / " << posIn[1] << " / " << posIn[2];
115  LOG(debug) << "POINT ON " << detId << " POSITION: " << locPosIn[0] << " / " << locPosIn[1];
116 
117  Int_t col = static_cast<Int_t>(locPosIn[0]/fPitchX);
118  Int_t ocol = col;
119  Int_t row = static_cast<Int_t>(locPosIn[1]/fPitchY);
120  Int_t orow = row;
121  Int_t feCol = col/fFeCols;
122  col = col % fFeCols;
123  Int_t feRow = row/fFeRows;
124  row = row % fFeRows;
125  Int_t feId = 1+feCol*fMaxFEperCol+feRow;
126 
127  LOG(debug) << "COL = " << feCol << " FE + " << col << " ( " << ocol << " ) /// "
128  << "ROW = " << feRow << " FE + " << row << " ( " << orow << " ) /// "
129  << " /// PIXEL = " << feId;
130 
131  ActivatePixel(iPoint,detId,feId,col,row,1.,0.);
132  }
133 
134  fTNofDigis += fNDigis;
135 }

137 void PixelDigitize::ActivatePixel(Int_t index, Int_t detId, Int_t feId, Int_t col, Int_t row, Double_t charge, Double_t timestamp)
138 {
139  Bool_t pixelAlreadyFired = kFALSE;
140  PixelDigi* tempPixel = nullptr;
141  for ( Int_t ipixel = 0 ; ipixel < fNDigis ; ipixel++ ) {
142  tempPixel = static_cast<PixelDigi*>(fDigis->At(ipixel));
143  if ( tempPixel->GetDetectorID() == detId &&
144  tempPixel->GetFeID() == feId &&
145  tempPixel->GetCol() == col &&
146  tempPixel->GetRow() == row ) {
147  pixelAlreadyFired = kTRUE;
148  tempPixel->SetCharge(tempPixel->GetCharge()+charge);
149  }
150  }
151  // TODO: order the PixelDigi in the output array by station, not by track...
152  if ( !pixelAlreadyFired ) {
153  new ((*fDigis)[fNDigis]) PixelDigi(index,detId,feId,col,row,charge,timestamp);
154  fNDigis++;
155  }
156 }
