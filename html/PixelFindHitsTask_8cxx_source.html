125 void PixelFindHitsTask::Exec(TClonesArray* digis, TClonesArray* hits)
126 {
127 
128  Reset(hits);
129 
130  LOG(info) << "PixelFindHits::Exec(TCA) EVENT " << fTNofEvents;
131 
132  fTNofEvents++;
133  LOG(debug)<<"PixelFindHits::Exec() ok 0 ";
134  fNDigis = digis->GetEntriesFast();
135  LOG(debug)<<"PixelFindHits::Exec() fNDigis = "<<fNDigis;
136  fTNofDigis+= fNDigis;
137 //*
138  for ( Int_t iDigi = 0 ; iDigi < fNDigis ; iDigi++ )
139  {
140  PixelDigi* currentDigi = (PixelDigi*)digis->At(iDigi);
141 
142  Int_t detId = currentDigi->GetDetectorID();
143  TString nodeName = Form("/cave/Pixel%d_%d",detId/256,detId%256);
144  LOG(debug)<<"PixelFindHits::Exec() ok 1 node name = "<<nodeName.Data();
145  fGeoParSet->GetGeometry()->cd(nodeName.Data());
146  LOG(debug)<<"PixelFindHits::Exec() ok 2 ";
147  TGeoNode* curNode = fGeoParSet->GetGeometry()->GetCurrentNode();
148  LOG(debug)<<"PixelFindHits::Exec() ok 3 ";
149  //TGeoMatrix* matrix = curNode->GetMatrix();
150 
151  TGeoVolume* actVolume = fGeoParSet->GetGeometry()->GetCurrentVolume();
152  TGeoBBox* actBox = (TGeoBBox*)(actVolume->GetShape());
153 
154  Int_t feId = currentDigi->GetFeID();
155  Int_t col = currentDigi->GetCol();
156  Int_t row = currentDigi->GetRow();
157 
158  Double_t locPosCalc[3];
159  locPosCalc[0] = ( ((feId-1)/fMaxFEperCol)*fFeCols + col + 0.5 )*fPitchX;
160  locPosCalc[1] = ( ((feId-1)%fMaxFEperCol)*fFeRows + row + 0.5 )*fPitchY;
161  locPosCalc[2] = 0.;
162 
163  locPosCalc[0] -= actBox->GetDX();
164  locPosCalc[1] -= actBox->GetDY();
165 
166  Double_t globPos[3];
167 
168  curNode->LocalToMaster(locPosCalc,globPos);
169 
170  LOG(debug) << "HIT ON " << detId << " POSITION: " << locPosCalc[0] << " / " << locPosCalc[1];
171  LOG(debug) << "GLOB HIT " << detId << " POSITION: " << globPos[0] << " / " << globPos[1] << " / " << globPos[2];
172 
173  TVector3 pos (globPos[0],globPos[1],globPos[2]);
174  TVector3 posErr(fPitchX/TMath::Sqrt(12.),fPitchY/TMath::Sqrt(12.),actBox->GetDZ());
175 
176  new ((*hits)[fNHits]) PixelHit(detId,currentDigi->GetIndex(),pos,posErr);
177 
178  fNHits++;
179  }
180 
181 
182  fTNofHits += fNHits;
183  //return fHits;
184 }
