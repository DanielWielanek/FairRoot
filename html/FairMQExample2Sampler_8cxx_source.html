38 bool FairMQExample2Sampler::ConditionalRun()

44  LOG(info) << "Sending \"" << fText << "\"";
45 
46  // in case of error or transfer interruption, return false to go to IDLE state
47  // successfull transfer will return number of bytes transfered (can be 0 if sending an empty message).
48  if (Send(msg, "data1") < 0)

50  return false;

53  {
54  LOG(info) << "Configured maximum number of iterations reached. Leaving RUNNING state.";
55  return false;
56  }
57
