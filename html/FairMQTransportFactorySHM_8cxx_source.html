146 void FairMQTransportFactorySHM::StartMonitor()
147 {
148  int numTries = 0;
149 
150  auto env = boost::this_process::environment();
151 
152  boost::filesystem::path p = boost::process::search_path("shmmonitor");
153 
154  if (!p.empty())
155  {
156  boost::process::spawn(p, "-x", "-s", fSessionName, "-d", "-t", "2000", env);
157 
158  do
159  {
160  MonitorStatus* monitorStatus = fManager->ManagementSegment().find<MonitorStatus>(bipc::unique_instance).first;
161  if (monitorStatus)
162  {
163  LOG(debug) << "shmmonitor started";
164  break;
165  }
166  else
167  {
168  this_thread::sleep_for(std::chrono::milliseconds(10));
169  if (++numTries > 1000)
170  {
171  LOG(error) << "Did not get response from shmmonitor after " << 10 * 1000 << " milliseconds. Exiting.";
172  exit(EXIT_FAILURE);
173  }
174  }
175  }
176  while (true);
177  }
178  else
179  {
180  LOG(WARN) << "could not find shmmonitor in the path";
181  }
182 }

184 void FairMQTransportFactorySHM::SendHeartbeats()
185 {
186  string controlQueueName("fmq_shm_" + fSessionName + "_control_queue");
187  while (fSendHeartbeats)
188  {
189  try
190  {
191  bipc::message_queue mq(bipc::open_only, controlQueueName.c_str());
192  bpt::ptime sndTill = bpt::microsec_clock::universal_time() + bpt::milliseconds(100);
193  if (mq.timed_send(fDeviceId.c_str(), fDeviceId.size(), 0, sndTill))
194  {
195  this_thread::sleep_for(chrono::milliseconds(100));
196  }
197  else
198  {
199  LOG(debug) << "control queue timeout";
200  }
201  }
202  catch (bipc::interprocess_exception& ie)
203  {
204  this_thread::sleep_for(chrono::milliseconds(500));
205  // LOG(warn) << "no " << controlQueueName << " found";
206  }
207  }
208 }

261 FairMQTransportFactorySHM::~FairMQTransportFactorySHM()
262 {
263  fSendHeartbeats = false;
264  fHeartbeatThread.join();
265 
266  if (fContext)
267  {
268  if (zmq_ctx_term(fContext) != 0)
269  {
270  if (errno == EINTR)
271  {
272  LOG(error) << "failed closing context, reason: " << zmq_strerror(errno);
273  }
274  else
275  {
276  fContext = nullptr;
277  return;
278  }
279  }
280  }
281  else
282  {
283  LOG(error) << "context not available for shutdown";
284  }
285 
286  bool lastRemoved = false;
287 
288  { // mutex scope
289  bipc::scoped_lock<bipc::named_mutex> lock(*fShMutex);
290 
291  (fDeviceCounter->fCount)--;
292 
293  if (fDeviceCounter->fCount == 0)
294  {
295  LOG(debug) << "last segment user, removing segment.";
296 
297  fManager->RemoveSegment();
298  lastRemoved = true;
299  }
300  else
301  {
302  LOG(debug) << "other segment users present (" << fDeviceCounter->fCount << "), not removing it.";
303  }
304  }
305 
306  if (lastRemoved)
307  {
308  boost::interprocess::named_mutex::remove(std::string("fmq_shm_" + fSessionName + "_mutex").c_str());
309  }
310 }
