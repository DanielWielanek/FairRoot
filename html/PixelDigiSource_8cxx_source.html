96 Int_t PixelDigiSource::ReadEvent(UInt_t i)
97 {
98  fDigis->Delete();
99  fNDigis = 0;
100 
101  if ( i == 0 ) {
102  fInputFile.clear();
103  fInputFile.seekg(0, std::ios::beg);
104  }
105 
106  fCurrentEntryNo = i;
107 
108  std::string buffer;
109  LOG(debug) << "PixelDigiSource::ReadEvent() Begin of (" << fDigis->GetEntries() << ")";
110  do {
111  getline(fInputFile,buffer);
112  LOG(debug) << "read from file: \"" << buffer << "\"";
113  if ( buffer.find("EVENT BEGIN") == 0 ) {
114  fRunId = ReadIntFromString(buffer,"RUNID");
115  fMCEntryNo = ReadIntFromString(buffer,"MCENTRYNO");
116  fPartNo = ReadIntFromString(buffer,"PARTNO");
117  fEventHeader->SetRunId(fRunId);
118  fEventHeader->SetMCEntryNumber(fMCEntryNo);
119  fEventHeader->SetPartNo(fPartNo);
120 
121  LOG(debug) << "GOT NEW EVENT " << fMCEntryNo << " (part " << fPartNo << ") with run id = " << fRunId;
122  }
123  if ( buffer.find("EVENT") == 0 ) continue;
124  Int_t detId = atoi(buffer.c_str());
125  buffer.erase(0,buffer.find(' ')+1);
126  Int_t feId = atoi(buffer.c_str());
127  buffer.erase(0,buffer.find(' ')+1);
128  Int_t col = atoi(buffer.c_str());
129  buffer.erase(0,buffer.find(' ')+1);
130  Int_t row = atoi(buffer.c_str());
131  buffer.erase(0,buffer.find(' ')+1);
132  Double_t charge = atof(buffer.c_str());
133  LOG(debug) << " --/" << fNDigis << "/--> " << detId << " / " << feId << " / " << col << " / " << row << " / " << charge;
134  new ((*fDigis)[fNDigis]) PixelDigi(-1,detId,feId,col,row,charge,0.);
135  fNDigis++;
136  }
137  while ( fInputFile && buffer.compare("EVENT END") );
138  LOG(debug) << "PixelDigiSource::ReadEvent() End of";
139 
140  if ( !fInputFile ) {
141  return 1;
142  }
143 
144  return 0;
145 }
