117

121  Send(parts, fOutputChannelName);
122  }
123  return false;
124  }

126  std::string buffer;
127

129  fInputFile.read((char*)head,sizeof(head));
130 
131  if (fInputFile.eof()) {

133  fInputFile.close();
134  if (fCurrentFile == (int)fFileNames.size()) { // this is the last file
