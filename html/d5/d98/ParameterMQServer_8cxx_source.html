136 bool ParameterMQServer::ProcessRequest(FairMQMessagePtr& req, int /*index*/)
137 {
138  string parameterName = "";
139  FairParGenericSet* par = nullptr;
140 
141  string reqStr(static_cast<char*>(req->GetData()), req->GetSize());
142  LOG(info) << "Received parameter request from client: \"" << reqStr << "\"";
143 
144  size_t pos = reqStr.rfind(",");
145  string newParameterName = reqStr.substr(0, pos);
146  int runId = stoi(reqStr.substr(pos + 1));
147  LOG(info) << "Parameter name: " << newParameterName;
148  LOG(info) << "Run ID: " << runId;
149 
150  LOG(info) << "Retrieving parameter...";
151  // Check if the parameter name has changed to avoid getting same container repeatedly
152  if (newParameterName != parameterName)
153  {
154  parameterName = newParameterName;
155  par = static_cast<FairParGenericSet*>(fRtdb->getContainer(parameterName.c_str()));
156  }
157  fRtdb->initContainers(runId);
158 
159  LOG(info) << "Sending following parameter to the client:";
160  if (par)
161  {
162  par->print();
163 
164  FairMQMessagePtr rep(NewMessage());
165  Serialize<RootSerializer>(*rep, par);
166 
167  if (Send(rep, fRequestChannelName, 0) < 0)
168  {
169  LOG(ERROR) << "failed sending reply";
170  return false;
171  }
172  }
173  else
174  {
175  LOG(ERROR) << "Parameter uninitialized! Sending empty message back";
176  // Send an empty message back to keep the REQ/REP cycle
177  FairMQMessagePtr rep(NewMessage());
178  if (Send(rep, fRequestChannelName, 0) < 0)
179  {
180  LOG(ERROR) << "failed sending reply";
181  return false;
182  }
183  }
184 
185  return true;
186 }

195  if (update->GetSize() < 20)

198  LOG(info) << "Received string " << repString << " !";

208  }
209  else

226
