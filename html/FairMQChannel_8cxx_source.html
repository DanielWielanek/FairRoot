469 bool FairMQChannel::ValidateChannel()
470 {
471  try
472  {
473  unique_lock<mutex> lock(fChannelMutex);
474 
475  stringstream ss;
476  ss << "Validating channel \"" << fName << "\"... ";
477 
478  if (fIsValid)
479  {
480  ss << "ALREADY VALID";
481  LOG(debug) << ss.str();
482  return true;
483  }
484 
485  // validate socket type
486  const string socketTypeNames[] = { "sub", "pub", "pull", "push", "req", "rep", "xsub", "xpub", "dealer", "router", "pair" };
487  const set<string> socketTypes(socketTypeNames, socketTypeNames + sizeof(socketTypeNames) / sizeof(string));
488  if (socketTypes.find(fType) == socketTypes.end())
489  {
490  ss << "INVALID";
491  LOG(debug) << ss.str();
492  LOG(error) << "Invalid channel type: \"" << fType << "\"";
493  exit(EXIT_FAILURE);
494  }
495 
496  // validate socket address
497  if (fAddress == "unspecified" || fAddress == "")
498  {
499  ss << "INVALID";
500  LOG(debug) << ss.str();
501  LOG(debug) << "invalid channel address: \"" << fAddress << "\"";
502  return false;
503  }
504  else
505  {
506  vector<string> endpoints;
507  boost::algorithm::split(endpoints, fAddress, boost::algorithm::is_any_of(","));
508  for (const auto endpoint : endpoints)
509  {
510  string address;
511  if (endpoint[0] == '@' || endpoint[0] == '+' || endpoint[0] == '>')
512  {
513  address = endpoint.substr(1);
514  }
515  else
516  {
517  // we don't have a method modifier, check if the default method is set
518  const string socketMethodNames[] = { "bind", "connect" };
519  const set<string> socketMethods(socketMethodNames, socketMethodNames + sizeof(socketMethodNames) / sizeof(string));
520  if (socketMethods.find(fMethod) == socketMethods.end())
521  {
522  ss << "INVALID";
523  LOG(debug) << ss.str();
524  LOG(error) << "Invalid endpoint connection method: \"" << fMethod << "\" for " << endpoint;
525  exit(EXIT_FAILURE);
526  }
527  address = endpoint;
528  }
529  // check if address is a tcp or ipc address
530  if (address.compare(0, 6, "tcp://") == 0)
531  {
532  // check if TCP address contains port delimiter
533  string addressString = address.substr(6);
534  if (addressString.find(":") == string::npos)
535  {
536  ss << "INVALID";
537  LOG(debug) << ss.str();
538  LOG(error) << "invalid channel address: \"" << address << "\" (missing port?)";
539  return false;
540  }
541  }
542  else if (address.compare(0, 6, "ipc://") == 0)
543  {
544  // check if IPC address is not empty
545  string addressString = address.substr(6);
546  if (addressString == "")
547  {
548  ss << "INVALID";
549  LOG(debug) << ss.str();
550  LOG(error) << "invalid channel address: \"" << address << "\" (empty IPC address?)";
551  return false;
552  }
553  }
554  else if (address.compare(0, 9, "inproc://") == 0)
555  {
556  // check if IPC address is not empty
557  string addressString = address.substr(9);
558  if (addressString == "")
559  {
560  ss << "INVALID";
561  LOG(debug) << ss.str();
562  LOG(error) << "invalid channel address: \"" << address << "\" (empty inproc address?)";
563  return false;
564  }
565  }
566  else if (address.compare(0, 8, "verbs://") == 0)
567  {
568  // check if IPC address is not empty
569  string addressString = address.substr(9);
570  if (addressString == "")
571  {
572  ss << "INVALID";
573  LOG(debug) << ss.str();
574  LOG(error) << "invalid channel address: \"" << address << "\" (empty verbs address?)";
575  return false;
576  }
577  }
578  else
579  {
580  // if neither TCP or IPC is specified, return invalid
581  ss << "INVALID";
582  LOG(debug) << ss.str();
583  LOG(error) << "invalid channel address: \"" << address << "\" (missing protocol specifier?)";
584  return false;
585  }
586  }
587  }
588 
589  // validate channel transport
590  // const string channelTransportNames[] = { "default", "zeromq", "nanomsg", "shmem" };
591  // const set<string> channelTransports(channelTransportNames, channelTransportNames + sizeof(channelTransportNames) / sizeof(string));
592  if (FairMQ::TransportTypes.find(fTransport) == FairMQ::TransportTypes.end())
593  {
594  ss << "INVALID";
595  LOG(debug) << ss.str();
596  LOG(error) << "Invalid channel transport: \"" << fTransport << "\"";
597  exit(EXIT_FAILURE);
598  }
599 
600  // validate socket buffer size for sending
601  if (fSndBufSize < 0)
602  {
603  ss << "INVALID";
604  LOG(debug) << ss.str();
605  LOG(error) << "invalid channel send buffer size (cannot be negative): \"" << fSndBufSize << "\"";
606  exit(EXIT_FAILURE);
607  }
608 
609  // validate socket buffer size for receiving
610  if (fRcvBufSize < 0)
611  {
612  ss << "INVALID";
613  LOG(debug) << ss.str();
614  LOG(error) << "invalid channel receive buffer size (cannot be negative): \"" << fRcvBufSize << "\"";
615  exit(EXIT_FAILURE);
616  }
617 
618  // validate socket kernel transmit size for sending
619  if (fSndKernelSize < 0)
620  {
621  ss << "INVALID";
622  LOG(debug) << ss.str();
623  LOG(error) << "invalid channel send kernel transmit size (cannot be negative): \"" << fSndKernelSize << "\"";
624  exit(EXIT_FAILURE);
625  }
626 
627  // validate socket kernel transmit size for receiving
628  if (fRcvKernelSize < 0)
629  {
630  ss << "INVALID";
631  LOG(debug) << ss.str();
632  LOG(error) << "invalid channel receive kernel transmit size (cannot be negative): \"" << fRcvKernelSize << "\"";
633  exit(EXIT_FAILURE);
634  }
635 
636  // validate socket rate logging interval
637  if (fRateLogging < 0)
638  {
639  ss << "INVALID";
640  LOG(debug) << ss.str();
641  LOG(error) << "invalid socket rate logging interval (cannot be negative): \"" << fRateLogging << "\"";
642  exit(EXIT_FAILURE);
643  }
644 
645  fIsValid = true;
646  ss << "VALID";
647  LOG(debug) << ss.str();
648  return true;
649  }
650  catch (exception& e)
651  {
652  LOG(error) << "Exception caught in FairMQChannel::ValidateChannel: " << e.what();
653  exit(EXIT_FAILURE);
654  }
655 }

767 bool FairMQChannel::CheckCompatibility(unique_ptr<FairMQMessage>& msg) const
768 {
769  if (fTransportType == msg->GetType())
770  {
771  return true;
772  }
773  else
774  {
775  // LOG(warn) << "Channel type does not match message type. Copying...";
776  FairMQMessagePtr msgCopy(fTransportFactory->CreateMessage(msg->GetSize()));
777  memcpy(msgCopy->GetData(), msg->GetData(), msg->GetSize());
778  msg = move(msgCopy);
779  return false;
780  }

783 bool FairMQChannel::CheckCompatibility(vector<unique_ptr<FairMQMessage>>& msgVec) const
784 {
785  bool match = true;
786
