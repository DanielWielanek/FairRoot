58 int FairProgOptions::ParseCmdLine(const int argc, char const* const* argv, bool allowUnregistered)
59 {
60  // get options from cmd line and store in variable map
61  // here we use command_line_parser instead of parse_command_line, to allow unregistered and positional options
62  if (allowUnregistered)
63  {
64  po::command_line_parser parser{argc, argv};
65  parser.options(fAllOptions).allow_unregistered();
66  po::parsed_options parsedOptions = parser.run();
67  po::store(parsedOptions, fVarMap);
68  }
69  else
70  {
71  po::store(po::parse_command_line(argc, argv, fAllOptions), fVarMap);
72  }
73 
74  // Handles options like "--help" or "--version"
75  // return 1 if switch options found in fVarMap
76  if (ImmediateOptions())
77  {
78  return 1;
79  }
80 
81  po::notify(fVarMap);
82  return 0;
83 }

100 int FairProgOptions::PrintOptions()
101 {
102  // -> loop over variable map and print its content
103  // -> In this example the following types are supported:
104  // string, int, float, double, short, boost::filesystem::path
105  // vector<string>, vector<int>, vector<float>, vector<double>, vector<short>
106 
107  map<string, VarValInfo> mapinfo;
108 
109  // get string length for formatting and convert varmap values into string
110  int maxLen1st = 0;
111  int maxLen2nd = 0;
112  int maxLenTypeInfo = 0;
113  int maxLenDefault = 0;
114  int maxLenEmpty = 0;
115  for (const auto& m : fVarMap)
116  {
117  Max(maxLen1st, m.first.length());
118 
119  VarValInfo valinfo = GetVariableValueInfo(m.second);
120  mapinfo[m.first] = valinfo;
121 
122  Max(maxLen2nd, valinfo.value.length());
123  Max(maxLenTypeInfo, valinfo.type.length());
124  Max(maxLenDefault, valinfo.defaulted.length());
125  Max(maxLenEmpty, valinfo.empty.length());
126  }
127 
128  // TODO : limit the value len field in a better way
129  if (maxLen2nd > 100)
130  {
131  maxLen2nd = 100;
132  }
133 
134  stringstream ss;
135  ss << "Configuration: \n";
136 
137  for (const auto& p : mapinfo)
138  {
139  ss << setfill(' ')
140  << setw(maxLen1st) << left << p.first << " = "
141  << setw(maxLen2nd) << p.second.value
142  << setw(maxLenTypeInfo) << p.second.type
143  << setw(maxLenDefault) << p.second.defaulted
144  << setw(maxLenEmpty) << p.second.empty
145  << "\n";
146  }
147 
148  LOG(info) << ss.str();
149 
150  return 0;
151 }
