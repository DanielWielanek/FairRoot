161 Bool_t FairTutorialDet4::ProcessHits(FairVolume* /*vol*/)
162 {
165  //Set parameters at entrance of volume. Reset ELoss.
166  if (TVirtualMC::GetMC()->IsTrackEntering()) {
167  fELoss = 0.;
168  fTime = TVirtualMC::GetMC()->TrackTime() * 1.0e09;
169  fLength = TVirtualMC::GetMC()->TrackLength();
170  TVirtualMC::GetMC()->TrackPosition(fPos);
171  TVirtualMC::GetMC()->TrackMomentum(fMom);
172  }
173 
174  // Sum energy loss for all steps in the active volume
175  fELoss += TVirtualMC::GetMC()->Edep();
176 
177  // Create FairTutorialDet4Point at exit of active volume
178  if (TVirtualMC::GetMC()->IsTrackExiting() ||
179  TVirtualMC::GetMC()->IsTrackStop() ||
180  TVirtualMC::GetMC()->IsTrackDisappeared() ) {
181  fTrackID = TVirtualMC::GetMC()->GetStack()->GetCurrentTrackNumber();
182  // fVolumeID = vol->getMCid();
183  fVolumeID = fGeoHandler->GetUniqueDetectorId();
184  if (fELoss == 0.) { return kFALSE; }
185 
186  if (!fGlobalCoordinates) {
187  // Save positions in local coordinate system, so transform the
188  // global coordinates into local ones.
189  Double_t master[3] = {fPos.X(), fPos.Y(), fPos.Z()};
190  Double_t local[3];
191 
192  TVirtualMC::GetMC()->Gmtod(master, local, 1);
193  AddHit(fTrackID, fVolumeID, TVector3(local[0], local[1], local[2]),
194  TVector3(fMom.Px(), fMom.Py(), fMom.Pz()), fTime, fLength, fELoss);
195  } else {
196  AddHit(fTrackID, fVolumeID, TVector3(fPos.X(), fPos.Y(), fPos.Z()),
197  TVector3(fMom.Px(), fMom.Py(), fMom.Pz()), fTime, fLength, fELoss);
198  }
199 
200  // Increment number of tutorial det points in TParticle
201  FairStack* stack = static_cast<FairStack*>(TVirtualMC::GetMC()->GetStack());
202  stack->AddPoint(kTutDet);
203  }
204 
205  return kTRUE;
206 }

267 std::map<std::string, TGeoHMatrix> FairTutorialDet4::getMisalignmentMatrices()
268 {
269  TString volPath;
270  TString volStr = "/cave_1/tutorial4_0/tut4_det_";
271 
272  std::map<std::string, TGeoHMatrix> matrices;
273 
274  for (Int_t iDet = 0; iDet < fNrOfDetectors; ++iDet) {
275  LOG(debug)<<"Create Matrix for detector nr "<<iDet;
276  volPath = volStr;
277  volPath += iDet;
278 
279  LOG(debug) << "Path: "<< volPath;
280 
281  //we have to express the displacements as regards the old local RS (non misaligned BTOF)
282  Double_t dx = fShiftX[iDet];
283  Double_t dy = fShiftY[iDet];
284  Double_t dz = fShiftZ[iDet];
285  Double_t dphi = fRotX[iDet];
286  Double_t dtheta = fRotY[iDet];
287  Double_t dpsi = fRotZ[iDet];
288 
289  TGeoRotation* rrot = new TGeoRotation("rot",dphi,dtheta,dpsi);
290  TGeoCombiTrans localdelta = *(new TGeoCombiTrans(dx,dy,dz, rrot));
291  TGeoHMatrix ldm = TGeoHMatrix(localdelta);
292 
293  std::string thisPath(volPath);
294  matrices[thisPath] = ldm;
295  }
296 
297  LOG(info) << fNrOfDetectors << " misalignment matrices created!";
298 
299  return matrices;
300 }
