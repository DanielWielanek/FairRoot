151  ofi_hints->rx_attr->op_flags = FI_COMPLETION;
152  if (type == ConnectionType::Bind) {
153  ofi_hints->src_addr = sa;
154  ofi_hints->src_addrlen = sizeof(sockaddr_in);

314 auto Context::ConvertAddress(std::string address) -> Address
315 {
316  string protocol, ip;
317  unsigned int port = 0;
318  regex address_regex("^([a-z]+)://([0-9]+\\.[0-9]+\\.[0-9]+\\.[0-9]+):([0-9]+).*");
319  smatch address_result;
320  if (regex_match(address, address_result, address_regex)) {
321  protocol = address_result[1];
322  ip = address_result[2];
323  port = stoul(address_result[3]);
324  // LOG(debug) << "Parsed '" << protocol << "', '" << ip << "', '" << port << "' fields from '" << address << "'";
325  } else {
326  throw ContextError(tools::ToString("Wrong format: Address must be in format prot://ip:port"));
327  }
328 
329  return { protocol, ip, port };
330 }

332 auto Context::ConvertAddress(Address address) -> sockaddr_in
333 {
