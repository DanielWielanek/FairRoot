114 int FairMQSocketSHM::SendImpl(FairMQMessagePtr& msg, const int flags, const int timeout)
115 {
116  int nbytes = -1;
117  int elapsed = 0;
118 
119  while (true && !fInterrupted)
120  {
121  nbytes = zmq_msg_send(static_cast<FairMQMessageSHM*>(msg.get())->GetMessage(), fSocket, flags);
122  if (nbytes == 0)
123  {
124  return nbytes;
125  }
126  else if (nbytes > 0)
127  {
128  static_cast<FairMQMessageSHM*>(msg.get())->fQueued = true;
129 
130  size_t size = msg->GetSize();
131  fBytesTx += size;
132  ++fMessagesTx;
133 
134  return size;
135  }
136  else if (zmq_errno() == EAGAIN)
137  {
138  if (!fInterrupted && ((flags & ZMQ_DONTWAIT) == 0))
139  {
140  if (timeout)
141  {
142  elapsed += fSndTimeout;
143  if (elapsed >= timeout)
144  {
145  return -2;
146  }
147  }
148  continue;
149  }
150  else
151  {
152  return -2;
153  }
154  }
155  else if (zmq_errno() == ETERM)
156  {
157  LOG(info) << "terminating socket " << fId;
158  return -1;
159  }
160  else
161  {
162  LOG(error) << "Failed sending on socket " << fId << ", reason: " << zmq_strerror(errno);
163  return nbytes;
164  }
165  }
166 
167  return -1;
168 }

170 int FairMQSocketSHM::ReceiveImpl(FairMQMessagePtr& msg, const int flags, const int timeout)
171 {
172  int nbytes = -1;
173  int elapsed = 0;
174 
175  zmq_msg_t* msgPtr = static_cast<FairMQMessageSHM*>(msg.get())->GetMessage();
176  while (true)
177  {
178  nbytes = zmq_msg_recv(msgPtr, fSocket, flags);
179  if (nbytes == 0)
180  {
181  ++fMessagesRx;
182 
183  return nbytes;
184  }
185  else if (nbytes > 0)
186  {
187  // check for number of receiving messages. must be 1
188  const auto numMsgs = nbytes / sizeof(MetaHeader);
189  if (numMsgs > 1)
190  {
191  LOG(ERROR) << "Receiving SHM multipart with a single message receive call";
192  }
193 
194  assert (numMsgs == 1);
195 
196  MetaHeader* hdr = static_cast<MetaHeader*>(zmq_msg_data(msgPtr));
197  size_t size = 0;
198  static_cast<FairMQMessageSHM*>(msg.get())->fHandle = hdr->fHandle;
199  static_cast<FairMQMessageSHM*>(msg.get())->fSize = hdr->fSize;
200  static_cast<FairMQMessageSHM*>(msg.get())->fRegionId = hdr->fRegionId;
201  static_cast<FairMQMessageSHM*>(msg.get())->fHint = hdr->fHint;
202  size = msg->GetSize();
203 
204  fBytesRx += size;
205  ++fMessagesRx;
206 
207  return size;
208  }
209  else if (zmq_errno() == EAGAIN)
210  {
211  if (!fInterrupted && ((flags & ZMQ_DONTWAIT) == 0))
212  {
213  if (timeout)
214  {
215  elapsed += fRcvTimeout;
216  if (elapsed >= timeout)
217  {
218  return -2;
219  }
220  }
221  continue;
222  }
223  else
224  {
225  return -2;
226  }
227  }
228  else if (zmq_errno() == ETERM)
229  {
230  LOG(info) << "terminating socket " << fId;
231  return -1;
232  }
233  else
234  {
235  LOG(error) << "Failed receiving on socket " << fId << ", reason: " << zmq_strerror(errno);

241 int64_t FairMQSocketSHM::SendImpl(vector<FairMQMessagePtr>& msgVec, const int flags, const int timeout)
242 {
243  const unsigned int vecSize = msgVec.size();
244  int64_t totalSize = 0;
245 
246  if (vecSize == 1) {
247  return Send(msgVec.back(), flags);
248  }
249 
250  // put it into zmq message
251  zmq_msg_t lZmqMsg;
252  zmq_msg_init_size(&lZmqMsg, vecSize * sizeof(MetaHeader));
253 
254  // prepare the message with shm metas
255  MetaHeader *lMetas = static_cast<MetaHeader*>(zmq_msg_data(&lZmqMsg));
256 
257  for (auto &lMsg : msgVec)
258  {
259  zmq_msg_t *lMetaMsg = static_cast<FairMQMessageSHM*>(lMsg.get())->GetMessage();
260  memcpy(lMetas++, zmq_msg_data(lMetaMsg), sizeof(MetaHeader));
261  }
262 
263  while (!fInterrupted)
264  {
265  int nbytes = -1;
266  nbytes = zmq_msg_send(&lZmqMsg, fSocket, flags);
267 
268  if (nbytes == 0)
269  {
270  zmq_msg_close (&lZmqMsg);
271  return nbytes;
272  }
273  else if (nbytes > 0)
274  {
275  assert(nbytes == (vecSize * sizeof(MetaHeader))); // all or nothing
276 
277  for (auto &lMsg : msgVec)
278  {
279  FairMQMessageSHM *lShmMsg = static_cast<FairMQMessageSHM*>(lMsg.get());
280  lShmMsg->fQueued = true;
281  totalSize += lShmMsg->fSize;
282  }
283 
284  // store statistics on how many messages have been sent
285  fMessagesTx++;
286  fBytesTx += totalSize;
287 
288  zmq_msg_close (&lZmqMsg);
289  return totalSize;
290  }
291  else if (zmq_errno() == EAGAIN)
292  {
293  if (!fInterrupted && ((flags & ZMQ_DONTWAIT) == 0))
294  {
295  continue;
296  }
297  else
298  {
299  zmq_msg_close (&lZmqMsg);
300  return -2;
301  }
302  }
303  else if (zmq_errno() == ETERM)
304  {
305  zmq_msg_close (&lZmqMsg);
306  LOG(INFO) << "terminating socket " << fId;
307  return -1;
308  }
309  else
310  {
311  zmq_msg_close (&lZmqMsg);
312  LOG(ERROR) << "Failed sending on socket " << fId << ", reason: " << zmq_strerror(errno);
313  return nbytes;
314  }
315  }
316 
317  return -1;

321 int64_t FairMQSocketSHM::ReceiveImpl(vector<FairMQMessagePtr>& msgVec, const int flags, const int timeout)
322 {
323  int64_t totalSize = 0;
324 
325  while (!fInterrupted)
326  {
327  zmq_msg_t lRcvMsg;
328  zmq_msg_init(&lRcvMsg);
329  int nbytes = zmq_msg_recv(&lRcvMsg, fSocket, flags);
330  if (nbytes == 0)
331  {
332  zmq_msg_close (&lRcvMsg);
333  return 0;
334  }
335  else if (nbytes > 0)
336  {
337  MetaHeader* lHdrVec = static_cast<MetaHeader*>(zmq_msg_data(&lRcvMsg));
338  const auto lHdrVecSize = zmq_msg_size(&lRcvMsg);
339  assert(lHdrVecSize > 0);
340  assert(lHdrVecSize % sizeof(MetaHeader) == 0);
341 
342  const auto lNumMessages = lHdrVecSize / sizeof (MetaHeader);
343 
344  msgVec.reserve(lNumMessages);
345 
346  for (auto m = 0; m < lNumMessages; m++)
347  {
348  MetaHeader lMetaHeader;
349  memcpy(&lMetaHeader, &lHdrVec[m], sizeof(MetaHeader));
350 
351  msgVec.emplace_back(fair::mq::tools::make_unique<FairMQMessageSHM>(fManager));
352 
353  FairMQMessageSHM *lMsg = static_cast<FairMQMessageSHM*>(msgVec.back().get());
354  MetaHeader *lMsgHdr = static_cast<MetaHeader*>(zmq_msg_data(lMsg->GetMessage()));
355 
356  memcpy(lMsgHdr, &lMetaHeader, sizeof(MetaHeader));
357 
358  lMsg->fHandle = lMetaHeader.fHandle;
359  lMsg->fSize = lMetaHeader.fSize;
360  lMsg->fRegionId = lMetaHeader.fRegionId;
361  lMsg->fHint = lMetaHeader.fHint;
362 
363  totalSize += lMsg->GetSize();
364  }
365 
366  // store statistics on how many messages have been received (handle all parts as a single message)
367  fMessagesRx++;
368  fBytesRx += totalSize;
369 
370  zmq_msg_close (&lRcvMsg);
371  return totalSize;
372  }
373  else if (zmq_errno() == EAGAIN)
374  {
375  zmq_msg_close(&lRcvMsg);
376  if (!fInterrupted && ((flags & ZMQ_DONTWAIT) == 0))
377  {
378  continue;
379  }
380  else
381  {
382  return -2;
383  }
384  }
385  else
386  {
387  zmq_msg_close (&lRcvMsg);
388  return nbytes;
389  }
390  }
391 
392  return -1;
