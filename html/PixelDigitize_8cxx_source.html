96 // ----- Public method Exec --------------------------------------------
97 void PixelDigitize::Exec(Option_t* /*opt*/) {
98 
99  Reset();
100 
101  LOG(debug) << "PixelDigitize::Exec() EVENT " << fTNofEvents;
102 
103  fTNofEvents++;
104 
105  fNPoints = fPoints->GetEntriesFast();
106  fTNofPoints+= fNPoints;
107 
108  for ( Int_t iPoint = 0 ; iPoint < fNPoints ; iPoint++ ) {
109  PixelPoint* currentPixelPoint = static_cast<PixelPoint*>(fPoints->At(iPoint));
110 
111  Double_t posIn[3] = {currentPixelPoint->GetX(),
112  currentPixelPoint->GetY(),
113  currentPixelPoint->GetZ()};
114 
115  Int_t detId = currentPixelPoint->GetDetectorID();
116  TString nodeName = Form("/cave/Pixel%d_%d",detId/256,detId%256);
117 
118 
119  gGeoManager->cd(nodeName.Data());
120  TGeoNode* curNode = gGeoManager->GetCurrentNode();
121 
122  if ( !curNode ) LOG(fatal) << "PixelDigitize::Exec() node \"" << nodeName.Data() << "\" unknown!";
123 
124  Double_t locPosIn[3];
125 
126  curNode->MasterToLocal(posIn,locPosIn);
127 
128  // The local coordinate system has the origin set to the center of the volume.
129  // To calculate the pixel number, it is necessary to move the origin to the bottom left corner.
130  TGeoVolume* actVolume = gGeoManager->GetCurrentVolume();
131  TGeoBBox* actBox = static_cast<TGeoBBox*>(actVolume->GetShape());
132 // Double_t sensorWidth = 2.*actBox->GetDX();
133 // Double_t sensorHeight = 2.*actBox->GetDY();
134  locPosIn[0] += actBox->GetDX();
135  locPosIn[1] += actBox->GetDY();
136 
137  LOG(debug) << "GLOB PNT " << detId << " POSITION: " << posIn[0] << " / " << posIn[1] << " / " << posIn[2];
138  LOG(debug) << "POINT ON " << detId << " POSITION: " << locPosIn[0] << " / " << locPosIn[1];
139 
140  Int_t col = static_cast<Int_t>(locPosIn[0]/fPitchX);
141  Int_t ocol = col;
142  Int_t row = static_cast<Int_t>(locPosIn[1]/fPitchY);
143  Int_t orow = row;
144  Int_t feCol = col/fFeCols;
145  col = col%fFeCols;
146  Int_t feRow = row/fFeRows;
147  row = row%fFeRows;
148  Int_t feId = 1+feCol*fMaxFEperCol+feRow;
149 
150  LOG(debug) << "COL = " << feCol << " FE + " << col << " ( " << ocol << " ) /// "
151  << "ROW = " << feRow << " FE + " << row << " ( " << orow << " ) /// "
152  << " /// PIXEL = " << feId;
153 
154  ActivatePixel(iPoint,detId,feId,col,row,1.,0.);
155  }
156 
157  fTNofDigis += fNDigis;
158 }

161 // ----- Private method ActivatePixel ----------------------------------
162 void PixelDigitize::ActivatePixel(Int_t index, Int_t detId, Int_t feId, Int_t col, Int_t row, Double_t charge, Double_t timestamp) {
163 
164  Bool_t pixelAlreadyFired = kFALSE;
165  PixelDigi* tempPixel = NULL;
166  for ( Int_t ipixel = 0 ; ipixel < fNDigis ; ipixel++ ) {
167  tempPixel = static_cast<PixelDigi*>(fDigis->At(ipixel));
168  if ( tempPixel->GetDetectorID() == detId &&
169  tempPixel->GetFeID() == feId &&
170  tempPixel->GetCol() == col &&
171  tempPixel->GetRow() == row ) {
172  pixelAlreadyFired = kTRUE;
173  tempPixel->SetCharge(tempPixel->GetCharge()+charge);
174  }
175  }
176  // TODO: order the PixelDigi in the output array by station, not by track...
177  if ( !pixelAlreadyFired ) {
178  new ((*fDigis)[fNDigis]) PixelDigi(index,detId,feId,col,row,charge,timestamp);
179  fNDigis++;
180  }
181 }
