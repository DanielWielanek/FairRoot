128 bool FairMQPixelFileSink::StoreData(FairMQParts& parts, int /*index*/)
129 {
130  std::vector<TObject*> tempObjects;
131  for ( int ipart = 0 ; ipart < parts.Size() ; ipart++ )
132  {
133  PixelTMessage tm(parts.At(ipart)->GetData(), parts.At(ipart)->GetSize());
134  tempObjects.push_back((TObject*)tm.ReadObject(tm.GetClass()));
135  for ( unsigned int ibr = 0 ; ibr < fBranchNames.size() ; ibr++ )
136  {
137  if ( strcmp(tempObjects.back()->GetName(),fBranchNames[ibr].data()) == 0 )
138  {
139  fOutputObjects[ibr] = tempObjects.back();
140  fTree->SetBranchAddress(fBranchNames[ibr].c_str(),&fOutputObjects[ibr]);
141  }
142  }
143  }
144  fTree->Fill();
145 
146  for ( int ipart = 0 ; ipart < tempObjects.size() ; ipart++ )
147  {
148  if ( tempObjects[ipart] )
149  delete tempObjects[ipart];
150  }
151  tempObjects.clear();
152 
153  if ( fAckChannelName != "" )
154  {
155  unique_ptr<FairMQMessage> msg(NewMessage());
156  Send(msg, fAckChannelName);
157  }
158  return true;
159 }
