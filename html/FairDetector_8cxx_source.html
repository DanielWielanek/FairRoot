102 void FairDetector::Initialize()

104 // Registers hits collection in Root manager;
105 // sets sensitive volumes.
106 // ---

108  // Define sensitive volumes if in MT
109  if ( gMC->IsMT() ) {
110  std::cout << "Define sensitive volume " << std::endl;
111  DefineSensitiveVolumes();

113 
114  Int_t NoOfEntries=svList->GetEntries();
115  Int_t fMCid;
116  FairGeoNode* fN;
117  TString cutName;
118  TString copysign="#";
119  for (Int_t i = 0 ; i < NoOfEntries ; i++ ) {
120  FairVolume* aVol = static_cast<FairVolume*>(svList->At(i));
121  cutName = aVol->GetName();
122  Ssiz_t pos = cutName.Index (copysign, 1);
123  if(pos>1) { cutName.Resize(pos); }
124  if ( aVol->getModId() == GetModId() ) {
125  fMCid=TVirtualMC::GetMC()->VolId(cutName.Data());
126  aVol->setMCid(fMCid);
127  fN=aVol->getGeoNode();
128  if (fN) { fN->setMCid(fMCid); }
129  }
130  }
131 
132  // Initialize cached pointer to MC (on master in sequential mode)
133  fMC = TVirtualMC::GetMC();
