90 Bool_t Pixel::ProcessHits(FairVolume* vol)
91 {
92 
94  //Set parameters at entrance of volume. Reset ELoss.
95  if ( TVirtualMC::GetMC()->IsTrackEntering() ) {
96  fELoss = 0.;
97  fTime = TVirtualMC::GetMC()->TrackTime() * 1.0e09;
98  fLength = TVirtualMC::GetMC()->TrackLength();
99  TVirtualMC::GetMC()->TrackPosition(fPos);
100  TVirtualMC::GetMC()->TrackMomentum(fMom);
101  }
102 
103  // Sum energy loss for all steps in the active volume
104  fELoss += TVirtualMC::GetMC()->Edep();
105 
106  // Create PixelPoint at exit of active volume
107  if ( TVirtualMC::GetMC()->IsTrackExiting() ||
108  TVirtualMC::GetMC()->IsTrackStop() ||
109  TVirtualMC::GetMC()->IsTrackDisappeared() ) {
110  fTrackID = TVirtualMC::GetMC()->GetStack()->GetCurrentTrackNumber();
111  fVolumeID = vol->getMCid();
112 
113 
114  if (fELoss == 0. ) { return kFALSE; }
115 
116  // Taking stationNr and sectorNr from string is almost effortless.
117  // Simulation of 100k events with 5 pions without magnetic field takes:
118  // - Real time 142.366 s, CPU time 140.32s WITH USING VolPath TO GET fVolumeID
119  // - Real time 142.407 s, CPU time 140.64s WITHOUT THE FOLLOWING TString OPERATIONS
120  {
121  TString detPath = TVirtualMC::GetMC()->CurrentVolPath();
122  detPath.Remove (0,detPath.Last('/')+1);
123  detPath.Remove (0,detPath.First("Pixel")+5);
124  Int_t stationNr = detPath.Atoi();
125  detPath.Remove (0,detPath.First("_")+1);
126  Int_t sectorNr = detPath.Atoi();
127  fVolumeID = stationNr*256+sectorNr;
128  }
129 
130  AddHit(fTrackID, fVolumeID, TVector3(fPos.X(), fPos.Y(), fPos.Z()),
131  TVector3(fMom.Px(), fMom.Py(), fMom.Pz()), fTime, fLength,
132  fELoss);
133 
134  // Increment number of Pixel det points in TParticle
135  FairStack* stack = static_cast<FairStack*>(TVirtualMC::GetMC()->GetStack());
136  stack->AddPoint(kPixel);
137  }
138 
139  return kTRUE;
140 }

177 void Pixel::ConstructGeometry()
178 {
183  FairGeoLoader* geoLoad = FairGeoLoader::Instance();
184  FairGeoInterface* geoFace = geoLoad->getGeoInterface();
185  PixelGeo* Geo = new PixelGeo();
186  Geo->setGeomFile(GetGeometryFileName());
187  geoFace->addGeoModule(Geo);
188 
189  Bool_t rc = geoFace->readSet(Geo);
190  if (rc) { Geo->create(geoLoad->getGeoBuilder()); }
191  TList* volList = Geo->getListOfVolumes();
192 
193  // store geo parameter
194  FairRun* fRun = FairRun::Instance();
195  FairRuntimeDb* rtdb= FairRun::Instance()->GetRuntimeDb();
196  PixelGeoPar* par=static_cast<PixelGeoPar*>(rtdb->getContainer("PixelGeoPar"));
197  TObjArray* fSensNodes = par->GetGeoSensitiveNodes();
198  TObjArray* fPassNodes = par->GetGeoPassiveNodes();
199 
200  TListIter iter(volList);
201  FairGeoNode* node = NULL;
202  FairGeoVolume* aVol=NULL;
203 
204  while( (node = static_cast<FairGeoNode*>(iter.Next())) ) {
205  aVol = dynamic_cast<FairGeoVolume*> ( node );
206  if ( node->isSensitive() ) {
207  fSensNodes->AddLast( aVol );
208  } else {
209  fPassNodes->AddLast( aVol );
210  }
211  }
212  par->setChanged();
213  par->setInputVersion(fRun->GetRunId(),1);
214 
215  ProcessNodes ( volList );
216 }
