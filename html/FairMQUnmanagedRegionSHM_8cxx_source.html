18 FairMQUnmanagedRegionSHM::FairMQUnmanagedRegionSHM(Manager& manager, const size_t size, FairMQRegionCallback callback)
19  : fManager(manager)
20  , fRegion(nullptr)
21  , fRegionId(0)
22 {
23  try
24  {
25  RegionCounter* rc = fManager.ManagementSegment().find<RegionCounter>(bipc::unique_instance).first;
26  if (rc)
27  {
28  LOG(debug) << "region counter found, with value of " << rc->fCount << ". incrementing.";
29  (rc->fCount)++;
30  LOG(debug) << "incremented region counter, now: " << rc->fCount;
31  }
32  else
33  {
34  LOG(debug) << "no region counter found, creating one and initializing with 1";
35  rc = fManager.ManagementSegment().construct<RegionCounter>(bipc::unique_instance)(1);
36  LOG(debug) << "initialized region counter with: " << rc->fCount;
37  }
38 
39  fRegionId = rc->fCount;
40 
41  fRegion = fManager.CreateRegion(size, fRegionId, callback);
42  }
43  catch (bipc::interprocess_exception& e)
44  {
45  LOG(error) << "cannot create region. Already created/not cleaned up?";
46  LOG(error) << e.what();
47  exit(EXIT_FAILURE);
48  }
