86 Bool_t FairBoxGenerator::Init()
87 {
88  // Initialize parent generator
89  if(FairBaseMCGenerator::Init()==kFALSE) return kFALSE;
90  // Check for particle type
91  TDatabasePDG* pdgBase = TDatabasePDG::Instance();
92  TParticlePDG* particle = pdgBase->GetParticle(GetPDGType());
93 
94  if (! particle) {
95  LOG(fatal)<<"FairBoxGenerator: PDG "<< GetPDGType()<< " not defined";
96  }
97 
98  if (fPhiMax-fPhiMin>360){
99  LOG(fatal)<<"FairBoxGenerator:Init(): phi range is too wide: "
100  <<fPhiMin<<"<phi<"<<fPhiMax;
101  }
102  if (fEkinRangeIsSet){
103  if (fPRangeIsSet){
104  LOG(fatal)<<"FairBoxGenerator:Init(): Cannot set P and Ekin ranges simultaneously";
105  } else {
106  // Transform EkinRange to PRange, calculate momentum in GeV, p = √(K² + 2Kmc²)
107  fPMin = TMath::Sqrt(fEkinMin*fEkinMin + 2*fEkinMin*GetPDGMass());
108  fPMax = TMath::Sqrt(fEkinMax*fEkinMax + 2*fEkinMax*GetPDGMass());
109  fPRangeIsSet = kTRUE;
110  fEkinRangeIsSet = kFALSE;
111  }
112  }
113  if (fPRangeIsSet && fPtRangeIsSet) {
114  LOG(fatal)<<"FairBoxGenerator:Init(): Cannot set P and Pt ranges simultaneously";
115  }
116  if (fPRangeIsSet && fYRangeIsSet) {
117  LOG(fatal)<<"FairBoxGenerator:Init(): Cannot set P and Y ranges simultaneously";
118  }
119  if ( (fThetaRangeIsSet && fYRangeIsSet) ||
120  (fThetaRangeIsSet && fEtaRangeIsSet) ||
121  (fYRangeIsSet && fEtaRangeIsSet) ) {
122  LOG(fatal)<<"FairBoxGenerator:Init(): Cannot set Y, Theta or Eta ranges simultaneously";
123  }
124  return kTRUE;
125 }

127 Bool_t FairBoxGenerator::ReadEvent(FairPrimaryGenerator* primGen)
128 {
129  // Generate one event: produce primary particles emitted from one vertex.
130  // Primary particles are distributed uniformly along
131  // those kinematics variables which were limitted by setters.
132  // if SetCosTheta() function is used, the distribution will be uniform in
133  // cos(theta)
134 
135  Double32_t pabs=0, phi, pt=0, theta=0, eta, y, mt, px, py, pz=0;
136  GenerateEventParameters();
137  // Generate particles
138  for (Int_t k = 0; k < GetMultiplicity(); k++) {
139  phi = gRandom->Uniform(fPhiMin,fPhiMax) * TMath::DegToRad();
140 
141  if (fPRangeIsSet ) { pabs = gRandom->Uniform(fPMin,fPMax); }
142  else if (fPtRangeIsSet) { pt = gRandom->Uniform(fPtMin,fPtMax); }
143 
144  if (fThetaRangeIsSet) {
145  if (fCosThetaIsSet)
146  theta = acos(gRandom->Uniform(cos(fThetaMin* TMath::DegToRad()),
147  cos(fThetaMax* TMath::DegToRad())));
148  else {
149  theta = gRandom->Uniform(fThetaMin,fThetaMax) * TMath::DegToRad();
150  }
151  } else if (fEtaRangeIsSet) {
152  eta = gRandom->Uniform(fEtaMin,fEtaMax);
153  theta = 2*TMath::ATan(TMath::Exp(-eta));
154  } else if (fYRangeIsSet) {
155  y = gRandom->Uniform(fYMin,fYMax);
156  mt = TMath::Sqrt(GetPDGMass()*GetPDGMass() + pt*pt);
157  pz = mt * TMath::SinH(y);
158  }
159 
160  if (fThetaRangeIsSet || fEtaRangeIsSet) {
161  if (fPRangeIsSet ) {
162  pz = pabs*TMath::Cos(theta);
163  pt = pabs*TMath::Sin(theta);
164  } else if (fPtRangeIsSet) {
165  pz = pt/TMath::Tan(theta);
166  }
167  }
168 
169  px = pt*TMath::Cos(phi);
170  py = pt*TMath::Sin(phi);
171 
172  LOG(debug)<<"FairBoxGen: "<<
173  Form("PDG %i p=(%.2f, %.2f, %.2f) GeV,",GetPDGType(),px,py,pz);
174  primGen->AddTrack(GetPDGType(), px, py, pz, fX, fY, fZ);
175  }
176  return kTRUE;
177 
178 }
