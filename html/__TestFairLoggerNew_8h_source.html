158  bool CheckVerboseOutput(std::string logLevel, std::string outputString, std::string verbosity, std::string resultString)
159  {
160  // Example Output for logging level INFO and different verbosity levels
161  // LOW :[INFO] I am here.
162  // MEDIUM :[19:07:13][INFO] I am here.
163  // HIGH :[process_name][19:07:13:123456][INFO] I am here.
164  // VERYHIGH :[process_name][19:07:13:123456][INFO][_TestFairTools.cxx:78:LogNoArguments] I am here.
165  std::string regexString;
166  if (verbosity == "HIGH" || verbosity == "VERYHIGH")
167  {
168  regexString += "\\[.*]"; // process_name
169  regexString += "\\[\\d{2}:\\d{2}:\\d{2}.\\d{6}]"; // timestamp with ns
170  }
171  else if (verbosity == "MEDIUM")
172  {
173  regexString += "\\[\\d{2}:\\d{2}:\\d{2}]"; // timestamp without ns
174  }
175  regexString += "\\[" + logLevel + "]"; // severity
176 
177  if (verbosity == "VERYHIGH")
178  {
179  regexString += "\\[.*:\\d{2}:.*]"; // file:line:function
180  }
181 
182  regexString += "[ ]" + outputString + ".*";
183 
184  const std::regex e(regexString);
185  return std::regex_match(resultString, e);
186  }
