184 void SaveDataToFile(ManagerType& outMan, RooDataSet* dataset, bool printval)
185 {
186  vector<T> DataBunch;
187  for (int i = 0; i < dataset->numEntries(); i++)
188  {
189  T data;
190  data.SetTimeStamp(dataset->get(i)->getRealValue("t"));
191  data.SetTimeStampError(dataset->get(i)->getRealValue("tErr"));
192 
193  data.SetX(static_cast<Int_t>(round(dataset->get(i)->getRealValue("x"))));
194  data.SetY(static_cast<Int_t>(round(dataset->get(i)->getRealValue("y"))));
195  data.SetZ(static_cast<Int_t>(round(dataset->get(i)->getRealValue("z"))));
196 
197  if (printval)
198  {
199  LOG(info) << "x=" << data.GetX()
200  << " y=" << data.GetY()
201  << " z=" << data.GetZ()
202  << " t=" << data.GetTimeStamp()
203  << " tErr=" << data.GetTimeStampError();
204  }
205  DataBunch.push_back(data);
206  }
207 
208  outMan.AddToFile(DataBunch);
