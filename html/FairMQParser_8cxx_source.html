133 void ChannelParser(const boost::property_tree::ptree& tree, FairMQMap& channelMap)
134 {
135  string channelKey;
136 
137  for (const auto& p : tree)
138  {
139  if (p.first == "channels")
140  {
141  for (const auto& q : p.second)
142  {
143  channelKey = q.second.get<string>("name");
144 
145  int numSockets = q.second.get<int>("numSockets", 0);
146 
147  // try to get common properties to use for all subChannels
148  FairMQChannel commonChannel;
149  commonChannel.UpdateType(q.second.get<string>("type", commonChannel.GetType()));
150  commonChannel.UpdateMethod(q.second.get<string>("method", commonChannel.GetMethod()));
151  commonChannel.UpdateAddress(q.second.get<string>("address", commonChannel.GetAddress()));
152  commonChannel.UpdateTransport(q.second.get<string>("transport", commonChannel.GetTransport()));
153  commonChannel.UpdateSndBufSize(q.second.get<int>("sndBufSize", commonChannel.GetSndBufSize()));
154  commonChannel.UpdateRcvBufSize(q.second.get<int>("rcvBufSize", commonChannel.GetRcvBufSize()));
155  commonChannel.UpdateSndKernelSize(q.second.get<int>("sndKernelSize", commonChannel.GetSndKernelSize()));
156  commonChannel.UpdateRcvKernelSize(q.second.get<int>("rcvKernelSize", commonChannel.GetRcvKernelSize()));
157  commonChannel.UpdateRateLogging(q.second.get<int>("rateLogging", commonChannel.GetRateLogging()));
158 
159  // temporary FairMQChannel container
160  vector<FairMQChannel> channelList;
161 
162  if (numSockets > 0)
163  {
164  LOG(debug) << "" << channelKey << ":";
165  LOG(debug) << "\tnumSockets of " << numSockets << " specified,";
166  LOG(debug) << "\tapplying common settings to each:";
167 
168  LOG(debug) << "\ttype = " << commonChannel.GetType();
169  LOG(debug) << "\tmethod = " << commonChannel.GetMethod();
170  LOG(debug) << "\taddress = " << commonChannel.GetAddress();
171  LOG(debug) << "\ttransport = " << commonChannel.GetTransport();
172  LOG(debug) << "\tsndBufSize = " << commonChannel.GetSndBufSize();
173  LOG(debug) << "\trcvBufSize = " << commonChannel.GetRcvBufSize();
174  LOG(debug) << "\tsndKernelSize = " << commonChannel.GetSndKernelSize();
175  LOG(debug) << "\trcvKernelSize = " << commonChannel.GetRcvKernelSize();
176  LOG(debug) << "\trateLogging = " << commonChannel.GetRateLogging();
177 
178  for (int i = 0; i < numSockets; ++i)
179  {
180  FairMQChannel channel(commonChannel);
181  channelList.push_back(channel);
182  }
183  }
184  else
185  {
186  SocketParser(q.second.get_child(""), channelList, channelKey, commonChannel);
187  }
188 
189  channelMap.insert(make_pair(channelKey, move(channelList)));
190  }
191  }
192  }
