114 bool FairMQEx9FileSink::StoreData(FairMQParts& parts, int /*index*/)
115 {
116  std::vector<TObject*> tempObjects;
117  for ( int ipart = 0 ; ipart < parts.Size() ; ipart++ )
118  {
119  Ex9TMessage tm(parts.At(ipart)->GetData(), parts.At(ipart)->GetSize());
120  tempObjects.push_back((TObject*)tm.ReadObject(tm.GetClass()));
121  for ( unsigned int ibr = 0 ; ibr < fBranchNames.size() ; ibr++ )
122  {
123  if ( strcmp(tempObjects.back()->GetName(),fBranchNames[ibr].data()) == 0 )
124  {
125  fOutputObjects[ibr] = tempObjects.back();
126  fTree->SetBranchAddress(fBranchNames[ibr].c_str(),&fOutputObjects[ibr]);
127  }
128  }
129  }
130  fTree->Fill();
131 
132  for ( int ipart = 0 ; ipart < tempObjects.size() ; ipart++ )
133  {
134  if ( tempObjects[ipart] )
135  delete tempObjects[ipart];
136  }
137  tempObjects.clear();
138 
139  if ( fAckChannelName != "" )
140  {
141  unique_ptr<FairMQMessage> msg(NewMessage());
142  Send(msg, fAckChannelName);
143  }
144  return true;
145 }
