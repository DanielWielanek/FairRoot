182 void FairMultiLinkedData::InsertLink(FairLink link)
183 {

200 void FairMultiLinkedData::InsertHistory(FairLink link)
201 {
202  FairRootManager* ioman = FairRootManager::Instance();
203  FairMultiLinkedData* pointerToLinks = 0;
204 
205  if (fVerbose > 1)
206  std::cout << "FairMultiLinkedData::InsertHistory for Link " << link << " Type: " << ioman->GetBranchName(link.GetType()) << std::endl;
207 
208  if (link.GetType() < 0)
209  return;
210  if (link.GetType() == ioman->GetMCTrackBranchId())
211  return;
212  if(ioman->GetBranchName(link.GetType()).Contains("."))
213  return;
214  if(link.GetEntry() != -1 && link.GetEntry() != ioman->GetEntryNr())
215  return;
216 
217  if (link.GetIndex() < 0) { //if index is -1 then this is not a TClonesArray so only the Object is returned
218  TObject* myObject = ioman->GetObject(ioman->GetBranchName(link.GetType()));
219  if (myObject->InheritsFrom("FairMultiLinkedData_Interface")){
220  FairMultiLinkedData_Interface* interface = static_cast<FairMultiLinkedData_Interface*>(myObject);
221  pointerToLinks = interface->GetPointerToLinks();
222  } else {
223  std::cout << "FairMultiLinkedData::InsertHistory Link to wrong Class: " << ioman->GetBranchName(link.GetType()) << std::endl;
224  return;
225  }
226  } else {
227  TClonesArray* dataArray = static_cast<TClonesArray*>( ioman->GetObject(ioman->GetBranchName(link.GetType())));
228  if (dataArray != 0 && link.GetIndex() < dataArray->GetEntriesFast()) {
229  TObject* myObject = dataArray->At(link.GetIndex());
230  if (myObject->InheritsFrom("FairMultiLinkedData_Interface")){
231  FairMultiLinkedData_Interface* interface = static_cast<FairMultiLinkedData_Interface*>(myObject);
232  pointerToLinks = interface->GetPointerToLinks();
233  }
234  else {
235  std::cout << "FairMultiLinkedData::InsertHistory Link to wrong Class: " << ioman->GetBranchName(link.GetType()) << std::endl;
236  return;
237  }
238  }
239  }
240  if (pointerToLinks != 0){
241  std::set<FairLink> linkSet = pointerToLinks->GetLinks();
242  for (std::set<FairLink>::const_iterator iter = linkSet.begin(); iter!= linkSet.end(); iter++){
243  if (fVerbose > 1)
244  std::cout << "FairMultiLinkedData::InsertHistory inserting " << *iter << std::endl;
245  InsertLink(*iter);
246  }
247  }
