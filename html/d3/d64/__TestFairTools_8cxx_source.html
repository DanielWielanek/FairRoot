101  levelNames.erase(position + 1, levelNames.end());
102  levelNames.erase(levelNames.begin()); // don't use FATAL loglevel in this test
103 
104  std::vector<std::string> retVal;
105  std::string fullOutputString;
106  if (filename.compare("") != 0) {
107  fullOutputString = "[INFO ] Openning log file: " + filename;
108  retVal.push_back(fullOutputString);
109  retVal.push_back(" ");
110  }
111 
112  std::vector<std::string>::iterator iter;
113  for (iter = levelNames.begin(); iter < levelNames.end(); ++iter) {
114  fullOutputString = "[" + *iter + "] " + outString + " ";
115  retVal.push_back(fullOutputString);

156  bool CheckVerboseOutput(const std::string& logLevel,
157  const std::string& outputString,
158  const std::string& verboseLevel,
159  const std::string& resultString)
160  {
161  // Example Output for logging level INFO and different verbosity levels
162  // verbosity Level:Output
163  // LOW :[INFO ] I am here.
164  // MEDIUM :[INFO ] [_TestFairTools.cxx::LogNoArguments:78] I am here.
165  // HIGH :[INFO ] [05.12.2011 12:00:10] [_TestFairTools.cxx::LogNoArguments:78] I am here.
166  std::string regexString = "\\[" + logLevel + "[ ]*][ ]";
167  if (verboseLevel.compare("HIGH") == 0) {
168  regexString = regexString + "\\[\\d{2}\\.\\d{2}\\.\\d{4}[ ]\\d{2}:\\d{2}:\\d{2}][ ]";
169  regexString = regexString + "\\[.*::.*:\\d{2}][ ]";
170  }
171  if (verboseLevel.compare("MEDIUM") == 0) {
172  regexString = regexString + "\\[.*::.*:\\d{2}][ ]";
173  }
174  regexString = regexString + outputString + ".*";
175 
176  const std::regex e(regexString);
177  return std::regex_match(resultString, e);
178  }
