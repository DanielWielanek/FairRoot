81 Bool_t FairTestDetector::ProcessHits(FairVolume* vol)
82 {
85  // Set parameters at entrance of volume. Reset ELoss.
86  if (TVirtualMC::GetMC()->IsTrackEntering())
87  {
88  fELoss = 0.;
89  fTime = TVirtualMC::GetMC()->TrackTime() * 1.0e09;
90  fLength = TVirtualMC::GetMC()->TrackLength();
91  TVirtualMC::GetMC()->TrackPosition(fPos);
92  TVirtualMC::GetMC()->TrackMomentum(fMom);
93  }
94 
95  // Sum energy loss for all steps in the active volume
96  fELoss += TVirtualMC::GetMC()->Edep();
97 
98  // Create FairTestDetectorPoint at exit of active volume
99  if (TVirtualMC::GetMC()->IsTrackExiting() || TVirtualMC::GetMC()->IsTrackStop() || TVirtualMC::GetMC()->IsTrackDisappeared())
100  {
101  fTrackID = TVirtualMC::GetMC()->GetStack()->GetCurrentTrackNumber();
102  fVolumeID = vol->getMCid();
103  TVirtualMC::GetMC()->TrackPosition(fPosOut);
104  TVirtualMC::GetMC()->TrackMomentum(fMomOut);
105  if (fELoss == 0.)
106  {
107  return kFALSE;
108  }
109  AddHit(fTrackID,
110  fVolumeID,
111  TVector3(fPos.X(), fPos.Y(), fPos.Z()),
112  TVector3(fMom.Px(), fMom.Py(), fMom.Pz()),
113  TVector3(fPosOut.X(), fPosOut.Y(), fPosOut.Z()),
114  TVector3(fMomOut.Px(), fMomOut.Py(), fMomOut.Pz()),
115  fTime,
116  fLength,
117  fELoss);
118 
119  // Increment number of FairTestDetector det points in TParticle
120  FairStack* stack = static_cast<FairStack*>(TVirtualMC::GetMC()->GetStack());
121  stack->AddPoint(kTutDet);
122  }
123 
124  return kTRUE;
125 }

163 void FairTestDetector::ConstructGeometry()
164 {
169  FairGeoLoader* geoLoad = FairGeoLoader::Instance();
170  FairGeoInterface* geoFace = geoLoad->getGeoInterface();
171  FairTestDetectorGeo* Geo = new FairTestDetectorGeo();
172  Geo->setGeomFile(GetGeometryFileName());
173  geoFace->addGeoModule(Geo);
174 
175  Bool_t rc = geoFace->readSet(Geo);
176  if (rc)
177  {
178  Geo->create(geoLoad->getGeoBuilder());
179  }
180  TList* volList = Geo->getListOfVolumes();
181 
182  // store geo parameter
183  FairRun* fRun = FairRun::Instance();
184  FairRuntimeDb* rtdb = FairRun::Instance()->GetRuntimeDb();
185  FairTestDetectorGeoPar* par = static_cast<FairTestDetectorGeoPar*>(rtdb->getContainer("FairTestDetectorGeoPar"));
186  TObjArray* fSensNodes = par->GetGeoSensitiveNodes();
187  TObjArray* fPassNodes = par->GetGeoPassiveNodes();
188 
189  TListIter iter(volList);
190  FairGeoNode* node = NULL;
191  FairGeoVolume* aVol = NULL;
192 
193  while ((node = static_cast<FairGeoNode*>(iter.Next())))
194  {
195  aVol = dynamic_cast<FairGeoVolume*>(node);
196  if (node->isSensitive())
197  {
198  fSensNodes->AddLast(aVol);
199  }
200  else
201  {
202  fPassNodes->AddLast(aVol);
203  }
204  }
205  par->setChanged();
206  par->setInputVersion(fRun->GetRunId(), 1);
207 
208  ProcessNodes(volList);
209 }
