80 Bool_t FairRutherford::ProcessHits(FairVolume* vol)
81 {
84  //Set parameters at entrance of volume. Reset ELoss.
85  if ( TVirtualMC::GetMC()->IsTrackEntering() ) {
86  fELoss = 0.;
87  fTime = TVirtualMC::GetMC()->TrackTime() * 1.0e09;
88  fLength = TVirtualMC::GetMC()->TrackLength();
89  TVirtualMC::GetMC()->TrackPosition(fPos);
90  TVirtualMC::GetMC()->TrackMomentum(fMom);
91  }
92 
93  // Sum energy loss for all steps in the active volume
94  fELoss += TVirtualMC::GetMC()->Edep();
95 
96  // Create FairRutherfordPoint at exit of active volume
97  if ( TVirtualMC::GetMC()->IsTrackExiting() ||
98  TVirtualMC::GetMC()->IsTrackStop() ||
99  TVirtualMC::GetMC()->IsTrackDisappeared() ) {
100  fTrackID = TVirtualMC::GetMC()->GetStack()->GetCurrentTrackNumber();
101  fVolumeID = vol->getMCid();
102  if (fELoss == 0. ) { return kFALSE; }
103  AddHit(fTrackID, fVolumeID, TVector3(fPos.X(), fPos.Y(), fPos.Z()),
104  TVector3(fMom.Px(), fMom.Py(), fMom.Pz()), fTime, fLength,
105  fELoss);
106 
107  // Increment number of FairRutherford points in TParticle
108  FairStack* stack = static_cast<FairStack*>(TVirtualMC::GetMC()->GetStack());
109  stack->AddPoint(kFairRutherford);
110  }
111 
112  return kTRUE;
113 }
