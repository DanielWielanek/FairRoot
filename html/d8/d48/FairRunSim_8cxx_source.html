146 void FairRunSim::Init()
147 {
149  fRootManager->InitSink();
150 
151  CheckFlukaExec();
152 
153 // fOutFile=fRootManager->OpenOutFile(fOutname);
154  LOG(info) << "============== FairRunSim: Initialising simulation run ==============";
155 
156  FairGeoLoader* loader=new FairGeoLoader(fLoaderName->Data(), "Geo Loader");
157  FairGeoInterface* GeoInterFace=loader->getGeoInterface();
158  GeoInterFace->SetNoOfSets(ListOfModules->GetEntries());
159  GeoInterFace->setMediaFile(MatFname.Data());
160  GeoInterFace->readMedia();
161 
162  // gSystem->cd(flout.Data());
163 
164  fApp= new FairMCApplication("Fair","The Fair VMC App",ListOfModules, MatFname);
165  fApp->SetGenerator(fGen);
166 
167  // Add a Generated run ID to the FairRunTimeDb
168  if ( fRunId == 0 ) {
169  FairRunIdGenerator genid;
170  // FairRuntimeDb *rtdb= GetRuntimeDb();
171  fRunId = genid.generateId();
172  }
173  fRtdb->addRun(fRunId);
174 
175  fFileHeader->SetRunId(fRunId);
177  fApp->AddTask(fTask);
179  FairBaseParSet* par=dynamic_cast<FairBaseParSet*>(fRtdb->getContainer("FairBaseParSet"));
180  if (par) {
181  par->SetDetList(GetListOfModules());
182  par->SetGen(GetPrimaryGenerator());
183  par->SetBeamMom(fBeamMom);
184  }
185 
187  FairGeoParSet* geopar=dynamic_cast<FairGeoParSet*>(fRtdb->getContainer("FairGeoParSet"));
188  if (geopar) {
189  geopar->SetGeometry(gGeoManager);
190  }
191 
192  // Set global Parameter Info
193 
194  if(fPythiaDecayer) {
195  fApp->SetPythiaDecayer(fPythiaDecayer);
196  if (fPythiaDecayerConfig) {
197  fApp->SetPythiaDecayerConfig(fPythiaDecayerConfig);
198  }
199  }
200  if(fUserDecay) {
201  fApp->SetUserDecay(fUserDecay);
202  if (fUserDecayConfig) {
203  fApp->SetUserDecayConfig(fUserDecayConfig);
204  }
205  }
206  // on/off visualisation
207  if( fStoreTraj ) {
208  LOG(info) << "Create visualisation manager ";
209  new FairTrajFilter();
210  }
211  if(fRadLength) {
212  fApp->SetRadiationLengthReg(fRadLength);
213  }
214  if(fRadMap) {
215  fApp->SetRadiationMapReg(fRadMap);
216  }
217  if(fRadGrid) {
218  fApp->AddMeshList(fMeshList);
219  }
220 
221  if(fField) { fField->Init(); }
222  fApp->SetField(fField);
223  SetFieldContainer();
224 
225  TList* containerList=fRtdb->getListOfContainers();
226  TIter next(containerList);
227  FairParSet* cont;
228  TObjArray* ContList= new TObjArray();
229  while ((cont=dynamic_cast<FairParSet*>(next()))) {
230  ContList->Add(new TObjString(cont->GetName()));
231  }
232  if(par){
233  par->SetContListStr(ContList);
234  par->SetRndSeed(gRandom->GetSeed());
235  par->setChanged();
236  par->setInputVersion(fRunId,1);
237  }
238  if(geopar){
239  geopar->setChanged();
240  geopar->setInputVersion(fRunId,1);
241  }
242 
244  SetMCConfig();
245  fRootManager->WriteFileHeader(fFileHeader);
246 }

255 void FairRunSim::CheckFlukaExec()
256 {
258  TString work = getenv("VMCWORKDIR");
259  TString work_config=work+"/gconfig/";
260  work_config.ReplaceAll("//","/");
261 
262  TString config_dir= getenv("CONFIG_DIR");
263  config_dir.ReplaceAll("//","/");
264  if (!config_dir.EndsWith("/")) { config_dir+="/"; }
265 
266  TString flout;
267  if(strcmp(GetName(),"TFluka") == 0 ) {
268  TString flexec="run_fluka.sh";
269  if (TString(gSystem->FindFile(config_dir.Data(),flexec)) != TString("")) {
270  LOG(info) << "---User path for Configuration is used: "
271  << config_dir.Data();
272  } else {
273  flexec=work_config+"run_fluka.sh";
274  }
275  gSystem->Exec(flexec.Data());
276  // More generic for different macro
277  // dir layout <D.B>
278  TString macrodir = getenv("PWD");
279  cout << macrodir << endl;
280  flout = macrodir + "/fluka_out" ;
281  flout.ReplaceAll("//","/");
282  gSystem->cd(flout.Data());
283  }
