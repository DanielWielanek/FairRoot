178  bool CheckVerboseOutput(std::string logLevel, std::string outputString, std::string verbosity, std::string resultString)
179  {
180  // Example Output for logging level INFO and different verbosity levels
181  // LOW :[INFO] I am here.
182  // MEDIUM :[19:07:13][INFO] I am here.
183  // HIGH :[process_name][19:07:13:123456][INFO] I am here.
184  // VERYHIGH :[process_name][19:07:13:123456][INFO][_TestFairTools.cxx:78:LogNoArguments] I am here.
185  std::string regexString;
186  if (verbosity == "HIGH" || verbosity == "VERYHIGH")
187  {
188  regexString += "\\[.*]"; // process_name
189  regexString += "\\[\\d{2}:\\d{2}:\\d{2}.\\d{6}]"; // timestamp with ns
190  }
191  else if (verbosity == "MEDIUM")
192  {
193  regexString += "\\[\\d{2}:\\d{2}:\\d{2}]"; // timestamp without ns
194  }
195  regexString += "\\[" + logLevel + "]"; // severity
196 
197  if (verbosity == "VERYHIGH")
198  {
199  regexString += "\\[.*:\\d{2}:.*]"; // file:line:function
200  }
201 
202  regexString += "[ ]" + outputString + ".*";
203 
204  const std::regex e(regexString);
205  return std::regex_match(resultString, e);
206  }
