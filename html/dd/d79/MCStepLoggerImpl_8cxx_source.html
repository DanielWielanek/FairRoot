50 class StepLogger
51 {
52  int stepcounter = 0;
53 
54  std::set<int> trackset;
55  std::set<int> pdgset;
56  std::map<int, int> volumetosteps;

58  std::map<int, char const*> idtovolname;
59 
60  // TODO: consider writing to a TTree/TFile
61  public:
62  void addStep(TVirtualMC* mc)
63  {
64  assert(mc);
65  stepcounter++;
66  auto stack=mc->GetStack();
67  assert(stack);
68  trackset.insert(stack->GetCurrentTrackNumber());
69  pdgset.insert(mc->TrackPid());
70  int copyNo;
71  auto id = mc->CurrentVolID(copyNo);
72  if (volumetosteps.find(id) == volumetosteps.end()){
73  volumetosteps.insert(std::pair<int, int>(id, 0));
74  } else {
75  volumetosteps[id]++;
76  }
77  if (volumetostepsGlobal.find(id) == volumetostepsGlobal.end()){
78  volumetostepsGlobal.insert(std::pair<int, int>(id, 0));
79  } else {
80  volumetostepsGlobal[id]++;
81  }
82  if (idtovolname.find(id) == idtovolname.end()) {
83  idtovolname.insert(std::pair<int, char const*>(id, mc->CurrentVolName()));
84  }
85  }
86 
87  void clear() {
88  stepcounter = 0;
89  trackset.clear();
90  pdgset.clear();
91  volumetosteps.clear();
92  }
93 
94  void flush() {
95  std::cerr << "did " << stepcounter << " steps \n";
96  std::cerr << "transported " << trackset.size() << " different tracks \n";
97  std::cerr << "transported " << pdgset.size() << " different types \n";
98  // summarize steps per volume
99  for (auto& p : volumetosteps) {
100  std::cout << " VolName " << idtovolname[p.first] << " COUNT " << p.second << "\n";
101  }
102  clear();
103  }
104 
105  void flushFinal() {
106  // summarize steps per volume
107  TFile* stepLoggerFile = TFile::Open("stepLogger.root","recreate");
108  TH1F* stepLoggerHistogram = new TH1F("stepLoggerHistogram","Step logger results;volume number;number of steps",volumetostepsGlobal.size(),0.5,volumetostepsGlobal.size()+0.5);
109  uint binNumber = 0;
110  for (auto& p : volumetostepsGlobal) {
111  std::cout << " VolName " << idtovolname[p.first] << " COUNT " << p.second << "\n";
112  stepLoggerHistogram->SetBinContent(++binNumber,p.second);
113  stepLoggerHistogram->GetXaxis()->SetBinLabel ( binNumber,idtovolname[p.first]);
114  }
115  stepLoggerHistogram->DrawClone();
116  stepLoggerFile->Write();
117  stepLoggerFile->Close();
118  }
119 };

123 // a generic function that can dispatch to the original method of a FairMCApplication
124 // (for functions of type void FairMCApplication::Method(void); )
125 // RK: did two versions, because on the first call while Stepping the origMethod was set to avoid calling it over and over again
126 // RK: but this setting also prevented the FinishEvent ever be calling - instead Stepping was also called
127 extern "C" void dispatchStepping(FairMCApplication* app, char const* libname, char const* origFunctionName)
128 {
129  typedef void (FairMCApplication::*StepMethodType)();
130  static StepMethodType origMethod = nullptr;
131  if (origMethod == nullptr) {
132  auto libHandle = dlopen(libname, RTLD_NOW);
133  // try to make the library loading a bit more portable:
134  if (!libHandle){
135  // try appending *.so
136  std::stringstream stream;
137  stream << libname << ".so";
138  libHandle = dlopen(stream.str().c_str(), RTLD_NOW);
139  }
140  if (!libHandle){
141  // try appending *.dylib
142  std::stringstream stream;
143  stream << libname << ".dylib";
144  libHandle = dlopen(stream.str().c_str(), RTLD_NOW);
145  }
146  assert(libHandle);
147  void* symbolAddress = dlsym(libHandle, origFunctionName);
148  assert(symbolAddress);
149  // hack since C++ does not allow casting to C++ member function pointers
150  // thanks to gist.github.com/mooware/1174572
151  memcpy(&origMethod, &symbolAddress, sizeof(symbolAddress));
152  }
153  (app->*origMethod)();
154 }
