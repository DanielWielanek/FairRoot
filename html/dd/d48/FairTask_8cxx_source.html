103 void FairTask::ExecuteTask(Option_t *option)
104 {
105  // Execute main task and its subtasks.
106  // When calling this function, the Exec function of the corresponding class
107  // is invoked, then the list of its subtasks is executed calling recursively
108  // all the subtasks, etc.
109  //
110  // The option parameter may be used to select different execution steps
111  // within a task. This parameter is passed also to all the subtasks.
112 
113  if (fgBeginTask) {
114  Error("ExecuteTask","Cannot execute task:%s, already running task: %s",GetName(),fgBeginTask->GetName());
115  return;
116  }
117  if (!IsActive()) return;
118 
119  fOption = option;
120  fgBeginTask = this;
121  fgBreakPoint = 0;
122 
123  if (fBreakin) return;
124  if (gDebug > 1) {
125  LOG(info)<<"Execute task:"<<GetName()<<" : "<<GetTitle();
126  }
127  FairMonitor::GetMonitor()->StartMonitoring(this,"EXEC");
128  Exec(option);
129  FairMonitor::GetMonitor()->StopMonitoring(this,"EXEC");
130 
131 
132  fHasExecuted = kTRUE;
133  ExecuteTasks(option);
134 
135  if (fBreakout) return;
136 
137  if (!fgBreakPoint) {
138  fgBeginTask->CleanTasks();
139  fgBeginTask = 0;
140  }
141 }

143 void FairTask::ExecuteTasks(Option_t *option)
144 {
145  // Execute all the subtasks of a task.
146 
147  TIter next(fTasks);
148  FairTask *task;
149  while((task=static_cast<FairTask*>(next()))) {
150  if (fgBreakPoint) return;
151  if (!task->IsActive()) continue;
152  if (task->fHasExecuted) {
153  task->ExecuteTasks(option);
154  continue;
155  }
156  if (task->fBreakin == 1) {
157  printf("Break at entry of task: %s\n",task->GetName());
158  fgBreakPoint = this;
159  task->fBreakin++;
160  return;
161  }
162 
163  if (gDebug > 1) {
164  LOG(info)<<"Execute task:"<<task->GetName()<<" : "<<task->GetTitle();
165  }
166  FairMonitor::GetMonitor()->StartMonitoring(task,"EXEC");
167  task->Exec(option);
168  FairMonitor::GetMonitor()->StopMonitoring(task,"EXEC");
169 
170  task->fHasExecuted = kTRUE;
171  task->ExecuteTasks(option);
172  if (task->fBreakout == 1) {
173  printf("Break at exit of task: %s\n",task->GetName());
174  fgBreakPoint = this;
175  task->fBreakout++;
176  return;
177  }
178  }
179 }
