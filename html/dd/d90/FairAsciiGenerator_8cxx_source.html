48 {
49  // Check for input file

63  // Read event header line from input file
64  *fInputFile >> ntracks;
65  if (fInputFile->fail() || ntracks < 0 || ntracks > (INT_MAX-1)) LOG(fatal) << "Error reading the number of events from event header.";
66  *fInputFile >> eventID >> vx >> vy >> vz;
67 
68  // If end of input file is reached : close it and abort run
69  if (fInputFile->eof()) {

71  CloseInput();

73  }
74 
75  LOG(info) << "FairAsciiGenerator: Event " << eventID << ", vertex = ("
76  << vx << "," << vy << "," << vz << ") cm, multiplicity " << ntracks;
77 
78  // Loop over tracks in the current event
79  for (Int_t itrack=0; itrack<ntracks; itrack++) {
80 
81  // Read PID and momentum from file
82  *fInputFile >> pdgID >> px >> py >> pz;
83  // convert Geant3 code to PDG code
84 
85  // Int_t pdg= fPDG->ConvertGeant3ToPdg(pdgID);
86
