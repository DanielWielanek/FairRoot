367 void FairStack::SelectTracks()
368 {
369 
370  // --> Clear storage map
371  fStoreMap.clear();
372 
373  // loop over tracks created by FastSim to order the tracks
374  for ( fFSTrackIter = fFSTrackMap.begin(); fFSTrackIter != fFSTrackMap.end(); ++fFSTrackIter ) {
375  for (Int_t i=0; i<fNParticles; i++) {
376  TParticle* thisPart = GetParticle(i);
377  if ( thisPart->GetMother(0) == fFSTrackIter->first )
378  thisPart->SetMother(0,fFSTrackIter->second);
379  }
380  }
381 
382  // --> Check particles in the fParticle array
383  for (Int_t i=0; i<fNParticles; i++) {
384 
385  TParticle* thisPart = GetParticle(i);
386  Bool_t store = kTRUE;
387 
388  // --> Get track parameters
389  Int_t iMother = thisPart->GetMother(0);
390  TLorentzVector p;
391  thisPart->Momentum(p);
392  Double_t energy = p.E();
393  Double_t mass = p.M();
394 // Double_t mass = thisPart->GetMass();
395  Double_t eKin = energy - mass;
396 
397  // --> Calculate number of points
398  Int_t nPoints = 0;
399  for (Int_t iDet=kREF; iDet<kSTOPHERE; iDet++) {
400  pair<Int_t, Int_t> a(i, iDet);
401  if ( fPointsMap.find(a) != fPointsMap.end() ) {
402  nPoints += fPointsMap[a];
403  }
404  }
405 
406  // check if the track was created by the FastSim mechanism. Do not store such tracks
407  Bool_t fastSimTrack = kFALSE;
408  fFSTrackIter = fFSTrackMap.find(i);
409  if ( fFSTrackIter != fFSTrackMap.end() )
410  fastSimTrack = kTRUE;
411 
412  // --> Check for cuts (store primaries in any case)
413  if (iMother < 0) { store = kTRUE; }
414  else {
415  if (!fStoreSecondaries) { store = kFALSE; }
416  if (nPoints < fMinPoints) { store = kFALSE; }
417  if (eKin < fEnergyCut) { store = kFALSE; }
418  if (fastSimTrack) { store = kFALSE; }
419  }
420 
421  // --> Set storage flag
422  fStoreMap[i] = store;
423  }
424 
425  // --> If flag is set, flag recursively mothers of selected tracks
426  if (fStoreMothers) {
427  for (Int_t i=0; i<fNParticles; i++) {
428  if (fStoreMap[i]) {
429  Int_t iMother = GetParticle(i)->GetMother(0);
430  while(iMother >= 0) {
431  fStoreMap[iMother] = kTRUE;
432  iMother = GetParticle(iMother)->GetMother(0);
433  }
434  }
435  }
436  }
437 }
