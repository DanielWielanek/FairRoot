32 bool FairMQExampleMTSampler1::ConditionalRun()
33 {
34  // Creates a message using the transport of channel data1
35  FairMQMessagePtr msg(NewMessageFor("data1", 0, 1000000));
36 
37  // in case of error or transfer interruption, return false to go to IDLE state
38  // successfull transfer will return number of bytes transfered (can be 0 if sending an empty message).
39  if (Send(msg, "data1") < 0)
40  {
41  return false;
42  }
43 
44  if (fMaxIterations > 0 && ++fNumIterations >= fMaxIterations)
45  {
46  LOG(info) << "Configured maximum number of iterations reached. Leaving RUNNING state.";
47  return false;
48  }
49 
50  return true;
51 }
