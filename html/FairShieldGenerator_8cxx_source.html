76 Bool_t FairShieldGenerator::ReadEvent(FairPrimaryGenerator* primGen)
77 {
78 
79  // Check for input file
80  if ( ! fInputFile->is_open() ) {
81  LOG(error) << "FairShieldGenerator: Input file not open!";
82  return kFALSE;
83  }
84 
85 
86  // Define event variable to be read from file
87  Int_t eventId = 0;
88  Int_t nTracks = 0;
89  Double_t pBeam = 0.;
90  Double_t b = 0.;
91 
92  // Define track variables to be read from file
93  Int_t iPid = 0;
94  Int_t iMass = 0;
95  Int_t iCharge = 0;
96  Double_t px = 0.;
97  Double_t py = 0.;
98  Double_t pz = 0.;
99  Int_t pdgType = 0;
100 
101  // Read event header line from input file
102  *fInputFile >> eventId;
103  *fInputFile >> nTracks;
104  if (nTracks < 0 || nTracks > (INT_MAX-1)) LOG(fatal) << "Error reading the number of events from event header.";
105  *fInputFile >> pBeam >> b;
106 
107 
108  // If end of input file is reached : close it and abort run
109  if ( fInputFile->eof() ) {
110  LOG(info) << "FairShieldGenerator: End of input file reached ";
111  CloseInput();
112  return kFALSE;
113  }
114 
115  LOG(info) << "FairShieldGenerator: Event " << eventId << ", pBeam = "
116  << pBeam << "GeV, b = " << b << " fm, multiplicity " << nTracks;
117 
118  // Loop over tracks in the current event
119  for (Int_t itrack=0; itrack<nTracks; itrack++) {
120 
121  // Read PID and momentum from file
122  *fInputFile >> iPid >> iMass >> iCharge >> px >> py >> pz;
123 
124  // Case ion
125  if ( iPid == 1000 ) {
126  char ionName[20];
127  sprintf(ionName, "Ion_%d_%d", iMass, iCharge);
128  TParticlePDG* part = fPDG->GetParticle(ionName);
129  if ( ! part ) {
130  LOG(warn) << "FairShieldGenerator::ReadEvent: Cannot find "
131  << ionName << " in database!";
132  continue;
133  }
134  pdgType = part->PdgCode();
135  } else { pdgType = fPDG->ConvertGeant3ToPdg(iPid); } // "normal" particle
136 
137  // Give track to PrimaryGenerator
138  primGen->AddTrack(pdgType, px, py, pz, 0., 0., 0.);
139 
140  }
141 
142 
143  return kTRUE;
144 }

165 // ----- Private method RegisterIons ----------------------------------
166 Int_t FairShieldGenerator::RegisterIons()
167 {
168 
169  Int_t nIons = 0;
170  Int_t eventId, nTracks, iPid, iMass, iCharge;
171  Double_t pBeam, b, px, py, pz;
172  fIonMap.clear();
173 
174  while ( ! fInputFile->eof()) {
175 
176  *fInputFile >> eventId;
177  *fInputFile >> nTracks;
178  if (nTracks < 0 || nTracks > (INT_MAX-1)) LOG(fatal) << "Error reading the number of events from event header.";
179  *fInputFile >> pBeam >> b;
180  if ( fInputFile->eof() ) { continue; }
181  for (Int_t iTrack=0; iTrack<nTracks; iTrack++) {
182  *fInputFile >> iPid >> iMass >> iCharge >> px >> py >> pz;
183  if ( iPid == 1000 ) { // ion
184  char buffer[20];
185  sprintf(buffer, "Ion_%d_%d", iMass, iCharge);
186  TString ionName(buffer);
187  if ( fIonMap.find(ionName) == fIonMap.end() ) { // new ion
188  FairIon* ion = new FairIon(ionName, iCharge, iMass, iCharge);
189  fIonMap[ionName] = ion;
190  nIons++;
191  } // new ion
192  } // ion
193  } // track loop
194 
195  } // event loop
196 
197  FairRunSim* run = FairRunSim::Instance();
198  map<TString, FairIon*>::iterator mapIt;
199  for (mapIt=fIonMap.begin(); mapIt!=fIonMap.end(); mapIt++) {
200  FairIon* ion = (*mapIt).second;
201  run->AddNewIon(ion);
202  }
203 
204  return nIons;
205 }
