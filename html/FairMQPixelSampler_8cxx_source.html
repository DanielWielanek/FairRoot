95 bool FairMQPixelSampler::ConditionalRun()
96 {
97  if ( fEventCounter == fMaxIndex ) return false;
98 
99  Int_t readEventReturn = fSource->ReadEvent(fEventCounter);
100 
101  if ( readEventReturn != 0 ) return false;
102 
103  TMessage* message[1000];
104  FairMQParts parts;
105 
106  for ( int iobj = 0 ; iobj < fNObjects ; iobj++ ) {
107  message[iobj] = new TMessage(kMESS_OBJECT);
108  message[iobj]->WriteObject(fInputObjects[iobj]);
109  parts.AddPart(NewMessage(message[iobj]->Buffer(), message[iobj]->BufferSize(), free_tmessage2, message[iobj]));
110  }
111 
112  Send(parts, fOutputChannelName);
113 
114  fEventCounter++;
115 
116  return true;
117 }
