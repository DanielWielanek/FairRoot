78  {
79  // LOG(debug) << "message received with " << parts.Size() << " parts!";
80  fReceivedMsgs++;
81 
82  if (parts.Size() == 0) return 0; // probably impossible, but still check
83 
84  // the first part should be the event header
85  PixelPayload::EventHeader* payloadE = static_cast<PixelPayload::EventHeader*>(parts.At(0)->GetData());
86  // LOG(debug) << "GOT EVENT " << payloadE->fMCEntryNo << " OF RUN " << payloadE->fRunId << " (part " << payloadE->fPartNo << ")";
87 
88  if ( fStaticParameters == false || fCurrentRunId == -1 ) {
89  fNewRunId = payloadE->fRunId;
90  if (fNewRunId != fCurrentRunId) {
91  fCurrentRunId = fNewRunId;
92  UpdateParameters();
93  fFairTask->InitMQ(fParCList);
94 
95  LOG(info) << "Parameters updated, back to ProcessData(" << parts.Size() << " parts!)";
96  }
97  }
98 
99  // the second part should the TClonesArray with necessary data... now assuming Digi
100  PixelPayload::Digi* payloadD = static_cast<PixelPayload::Digi*>(parts.At(1)->GetData());
101  int digiArraySize = parts.At(1)->GetSize();
102  int nofDigis = digiArraySize / sizeof(PixelPayload::Digi);
103 
104  fInputArray->Clear();
105  for (int idigi = 0; idigi < nofDigis; idigi++) {
106  new ((*fInputArray)[idigi]) PixelDigi(-1,
107  payloadD[idigi].fDetectorID,
108  payloadD[idigi].fFeID,
109  payloadD[idigi].fCol,
110  payloadD[idigi].fRow,
111  payloadD[idigi].fCharge);
112  }
113 
114  // LOG(debug) << " EVENT HAS " << nofDigis << " DIGIS!!!";
115 
116  // Execute hit finder task
117  fOutput->Clear();
118  // LOG(info) << " The blocking line... analyzing event " << fEventHeader->GetMCEntryNumber();
119  fFairTask->ExecMQ(fInput, fOutput);
120 
121  FairMQParts partsOut;
122 
123  PixelPayload::EventHeader* header = new PixelPayload::EventHeader();
124  header->fRunId = payloadE->fRunId;
125  header->fMCEntryNo = payloadE->fMCEntryNo;
126  header->fPartNo = payloadE->fPartNo;
127 
128  FairMQMessagePtr msgHeader(NewMessage(header,
129  sizeof(PixelPayload::EventHeader),
130  [](void* data, void* /*hint*/) { delete static_cast<PixelPayload::EventHeader*>(data); },
131  nullptr));
132  partsOut.AddPart(std::move(msgHeader));
133 
134  for (int iobj = 0; iobj < fOutput->GetEntries(); iobj++) {
135  if (strcmp(fOutput->At(iobj)->GetName(),"PixelHits") == 0) {
136  Int_t nofEntries = ((TClonesArray*)fOutput->At(iobj))->GetEntries();
137  size_t hitsSize = nofEntries * sizeof(PixelPayload::Hit);
138 
139  FairMQMessagePtr msgTCA(NewMessage(hitsSize));
140 
141  PixelPayload::Hit* hitPayload = static_cast<PixelPayload::Hit*>(msgTCA->GetData());
142 
143  for (int ihit = 0; ihit < nofEntries; ihit++) {
144  PixelHit* hit = static_cast<PixelHit*>(((TClonesArray*)fOutput->At(iobj))->At(ihit));
145  if (!hit) {
146  continue;
147  }
148  new (&hitPayload[ihit]) PixelPayload::Hit();
149  hitPayload[ihit].fDetectorID = hit->GetDetectorID();
150  hitPayload[ihit].posX = hit->GetX();
151  hitPayload[ihit].posY = hit->GetY();
152  hitPayload[ihit].posZ = hit->GetZ();
153  hitPayload[ihit].dposX = hit->GetDx();
154  hitPayload[ihit].dposY = hit->GetDy();
155  hitPayload[ihit].dposZ = hit->GetDz();
156  }
157  // LOG(debug) << "second part has size = " << hitsSize;
158  partsOut.AddPart(std::move(msgTCA));
159 
160  // LOG(debug) << "Output array should have " << nofEntries << "hits";
161  }
162  }
163 
164  Send(partsOut, fOutputChannelName);
165  fSentMsgs++;
166

218  {
219  std::string paramName = thisPar->GetName();
220 
221  std::string* reqStr = new std::string(paramName + "," + std::to_string(fCurrentRunId));
222  LOG(warn) << "Requesting parameter \"" << paramName << "\" for Run ID " << fCurrentRunId << " (" << thisPar << ")";
223  FairMQMessagePtr req(NewMessage(const_cast<char*>(reqStr->c_str()),
224  reqStr->length(),
225  [](void* /* data */, void* hint){ delete static_cast<std::string*>(hint); },
226  reqStr));
227  FairMQMessagePtr rep(NewMessage());
228 
229  if (Send(req,fParamChannelName) > 0) {
230  if (Receive(rep,fParamChannelName) > 0) {
231  thisPar = nullptr;
232  Deserialize<RootSerializer>(*rep,thisPar);
233  LOG(info) << "Received parameter"<< paramName <<" from the server (" << thisPar << ")";
234  return thisPar;
235  }
236  }
