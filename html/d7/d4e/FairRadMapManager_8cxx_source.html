84 void FairRadMapManager::Init()
85 {
87  FairRootManager::Instance()->Register("RadMap","RadMapPoint", fPointCollection, kTRUE);
88  cout << "RadMapMan initialized" << endl;
89 
90  // compute once the masses of the volumes in this simulation and store them in a TMap object
91 
92  Int_t volumeiterator=0,lastvolume=0;
93  Double_t vmass;
94  TObjArray* volumelist;
95  TGeoVolume* myvolume;
96  const char* volumename;
97  fMassMap = new TMap(lastvolume+1,0);
98 
99  volumelist = gGeoManager->GetListOfVolumes();
100 
101  lastvolume = volumelist->GetLast();
102 
103  volumeiterator=0;
104 
105  cout << "RadMapMan: calculating the masses for " << lastvolume << " volumes in this simulation" << endl;
106 
107  while ( volumeiterator<=lastvolume ) {
108  volumename = volumelist->At(volumeiterator)->GetName();
109  myvolume = gGeoManager->GetVolume(volumename);
110  vmass = myvolume->WeightA(); // calculate weight analytically
111  TVectorD* vomass= new TVectorD(1,1,vmass,"END"); // 1-dim vector
112 
113  fMassMap->Add(myvolume,vomass);
114 
115  cout << myvolume->GetName() << " has " << vmass << " kg" << endl;
116 
117  volumeiterator++;
118  }

132  if ( TVirtualMC::GetMC()->IsTrackEntering() ) {

144  TVirtualMC::GetMC()->TrackPosition(fPosIn);
145  TVirtualMC::GetMC()->TrackMomentum(fMomIn);

151  TGeoVolume* actVolume = gGeoManager->GetVolume(fVolumeID);
152 
153  TVectorD* ActMass = static_cast<TVectorD*>(fMassMap->GetValue(actVolume));
154 
155  fActMass = ActMass->Min(); // read from TVectorD
156 
157  // cout << actVolume->GetName() << " has " << fActMass << " kg" << endl;
158 
159  }
161  fELoss += TVirtualMC::GetMC()->Edep();
162  fStep += TVirtualMC::GetMC()->TrackStep();
163 
164  fPdg = TVirtualMC::GetMC()->TrackPid();
165 
166  // calculate the energy dose
167  // exclude fragments with PDG code >= 10000
168 
169  if (fPdg<10000 && fStep>0 && fActMass>0 ) {
170  // fDose = fELoss*1.602e-10/(fDensity*actvol*0.001); // per mass/kg
171  fDose = fELoss*1.602e-10/(fActMass); // per mass/kg
172  } else {
173  fDose = -.00001;
174  }
175 
176  // exclude very high, probably wrong, energy doses
177  if ( fDose>0.02 ) {
178  fDose = -.00001;
179  }
180 
182  if ( TVirtualMC::GetMC()->IsTrackExiting() ||
183  TVirtualMC::GetMC()->IsTrackStop() ||
184  TVirtualMC::GetMC()->IsTrackDisappeared() ) {
185 
186 // FairRadMapPoint* p=0;
187  fTrackID = TVirtualMC::GetMC()->GetStack()->GetCurrentTrackNumber();
188  Int_t copyNo;
189  Int_t fVolID = TVirtualMC::GetMC()->CurrentVolID(copyNo); // CAVEAT: fVolID is NOT an unique identifier!!
190  TVirtualMC::GetMC()->TrackPosition(fPosOut);
191  TVirtualMC::GetMC()->TrackMomentum(fMomOut);
192 
193  TClonesArray& clref = *fPointCollection;
194  Int_t tsize = clref.GetEntriesFast();
195 
196 // p=new(clref[tsize]) FairRadMapPoint(fTrackID, fVolID,
197  new(clref[tsize]) FairRadMapPoint(fTrackID, fVolID,
198  TVector3(fPosIn.X(),fPosIn.Y(),fPosIn.Z()),
199  TVector3(fMomIn.X(),fMomIn.Y(),fMomIn.Z()),
