45 bool FairMQExHistoServer::ReceiveData(FairMQMessagePtr& msg, int index)
46 {
47  TObject* tempObject = nullptr;
48  Deserialize<RootSerializer>(*msg, tempObject);
49 
50  if (TString(tempObject->ClassName()).EqualTo("TObjArray"))
51  {
52  std::lock_guard<std::mutex> lk(mtx);
53  TObjArray* arrayHisto = static_cast<TObjArray*>(tempObject);
54  TH1* histogram_new;
55  TH1* histogram_existing;
56  for (int i = 0; i < arrayHisto->GetEntriesFast(); i++)
57  {

61  if (-1 == index1)
62  {

66  }

68  {

70  histogram_existing->Add(histogram);

72  }
73

75  }
