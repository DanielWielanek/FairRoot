390 void FairRootManager::AddMemoryBranchAny(const char* brname, T** obj) {
391  if (fAnyBranchMap.find(brname) == fAnyBranchMap.end()) {
392  auto& ot = typeid(T*);
393  auto& pt = typeid(T);
394  fAnyBranchMap[brname]=std::unique_ptr<TypeAddressPair const> (new TypeAddressPair(ot, pt,(void*)obj));
395  }

398 // try to retrieve an object address from the registered branches/names
399 template<typename T>
400 T FairRootManager::GetMemoryBranchAny(const char* brname) const {
401  static_assert(std::is_pointer<T>::value, "Return type of GetMemoryBranchAny has to be a pointer");
402  using P = typename std::remove_pointer<T>::type;
403  auto iter = fAnyBranchMap.find(brname);
404  if(iter != fAnyBranchMap.end()) {
405  // verify type consistency
406  if(typeid(P).hash_code() != iter->second->origtypeinfo.hash_code()) {
407  EmitMemoryBranchWrongTypeWarning(brname, typeid(P).name(), iter->second->origtypeinfo.name());
408  return nullptr;
409  }
410  return static_cast<T>(iter->second->ptraddr);
411  }
412  return nullptr;
413 }

430 // this function serves as a factory (or lookup) for memory managed
431 // instances associated to branches
432 // it returns a pointer to unmodifiable instance of T
433 template<typename TPtr>
434 TPtr FairRootManager::InitObjectAs(const char* brname) {
435  static_assert(std::is_pointer<TPtr>::value, "Return type of GetObjectAs has to be a pointer");
436  using X = typename std::remove_pointer<TPtr>::type;
437  static_assert(std::is_const<X>::value, "Return type of GetObjectAs has to be pointer to const class");
438  using T = typename std::remove_const<X>::type;
439 
440  // is there already an object associated to the branch in memory??
441  // then just return
442  T** obj = GetMemoryBranchAny<T**>(brname);
443  // obj is some address/instance holding TPtr instances
444  if(obj!=nullptr) return *obj;
445 
446  // it does not seem to be the case, let us create the pointer which will be initialized
447  // with the data (pointer to T)
448  T** addr = new T*;
449  // init the pointee to a default obj which we can return
450  (*addr) = new T;
451  // try to find and activate in the source
452  auto succeeded = fSource->ActivateObjectAny((void**)addr, typeid(T), brname);
453 
454  if(!succeeded) {
455  delete (*addr);
456  delete addr;
457  return nullptr;
458  }
459  // add into branch list
460  AddMemoryBranchAny<T>(brname, addr);
461  // register as a **requested** branch
462  // (duplications are explicitely allowed)
463  fReqBrNames.emplace_back(brname);
464 
465  // NOTE: ideally we would do proper resource management for addr and *addr
466  // since the FairRootManager becomes owner of these pointers/instances; Unfortunately this
467  // is quite a difficult task since we would have to store something like std::unique_ptr<T> in a member
468  // container which we cannot know a priori; Some solutions we could think of in the future are
469  // a) use the Destructor mechanism of ROOT::TClass since we still have the type info.
470  // b) investigate if boost::any could be of help here
471  // In any case, this problem is not very critical in the sense that FairRootManager is a singleton and hence
472  // cannot really leak memory (Assuming that the destructors of T are not doing something non-trivial).
473  return *addr;
474 }
