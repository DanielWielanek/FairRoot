93 void PixelFindHitsTask::Exec(TClonesArray* digis, TClonesArray* hits)
94 {
95  Reset(hits);
96 
97  LOG(info) << "PixelFindHits::Exec(TCA) EVENT " << fTNofEvents;
98 
99  fTNofEvents++;
100  LOG(debug)<<"PixelFindHits::Exec() ok 0 ";
101  fNDigis = digis->GetEntriesFast();
102  LOG(debug)<<"PixelFindHits::Exec() fNDigis = "<<fNDigis;
103  fTNofDigis+= fNDigis;
104 //*
105  for (Int_t iDigi = 0 ; iDigi < fNDigis ; iDigi++)
106  {
107  PixelDigi* currentDigi = (PixelDigi*)digis->At(iDigi);
108 
109  Int_t detId = currentDigi->GetDetectorID();
110  TString nodeName = Form("/cave/Pixel%d_%d",detId/256,detId%256);
111  LOG(debug)<<"PixelFindHits::Exec() ok 1 node name = "<<nodeName.Data();
112  fGeoParSet->GetGeometry()->cd(nodeName.Data());
113  LOG(debug)<<"PixelFindHits::Exec() ok 2 ";
114  TGeoNode* curNode = fGeoParSet->GetGeometry()->GetCurrentNode();
115  LOG(debug)<<"PixelFindHits::Exec() ok 3 ";
116  //TGeoMatrix* matrix = curNode->GetMatrix();
117 
118  TGeoVolume* actVolume = fGeoParSet->GetGeometry()->GetCurrentVolume();
119  TGeoBBox* actBox = (TGeoBBox*)(actVolume->GetShape());
120 
121  Int_t feId = currentDigi->GetFeID();
122  Int_t col = currentDigi->GetCol();
123  Int_t row = currentDigi->GetRow();
124 
125  Double_t locPosCalc[3];
126  locPosCalc[0] = (((feId-1)/fMaxFEperCol)*fFeCols + col + 0.5)*fPitchX;
127  locPosCalc[1] = (((feId-1)%fMaxFEperCol)*fFeRows + row + 0.5)*fPitchY;
128  locPosCalc[2] = 0.;
129 
130  locPosCalc[0] -= actBox->GetDX();
131  locPosCalc[1] -= actBox->GetDY();
132 
133  Double_t globPos[3];
134 
135  curNode->LocalToMaster(locPosCalc,globPos);
136 
137  LOG(debug) << "HIT ON " << detId << " POSITION: " << locPosCalc[0] << " / " << locPosCalc[1];
138  LOG(debug) << "GLOB HIT " << detId << " POSITION: " << globPos[0] << " / " << globPos[1] << " / " << globPos[2];
139 
140  TVector3 pos(globPos[0],globPos[1],globPos[2]);
141  TVector3 posErr(fPitchX/TMath::Sqrt(12.),fPitchY/TMath::Sqrt(12.),actBox->GetDZ());
142 
143  new ((*hits)[fNHits]) PixelHit(detId,currentDigi->GetIndex(),pos,posErr);
144 
145  fNHits++;
146  }
147 
148  fTNofHits += fNHits;
149  //return fHits;
150 }
