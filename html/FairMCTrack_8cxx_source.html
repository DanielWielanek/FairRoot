162 // ----- Public method SetNPoints --------------------------------------
163 void FairMCTrack::SetNPoints(Int_t iDet, Int_t nPoints)
164 {
165 
166  if ( iDet == kREF ) {
167  if ( nPoints < 0 ) { nPoints = 0; }
168  else if ( nPoints > 1 ) { nPoints = 1; }
169  fNPoints = ( fNPoints & ( ~ 1 ) ) | nPoints;
170  }
171 
172  else if ( iDet == kTutDet ) {
173  if ( nPoints < 0 ) { nPoints = 0; }
174  else if ( nPoints > 7 ) { nPoints = 7; }
175  fNPoints = ( fNPoints & ( ~ ( 7 << 1 ) ) ) | ( nPoints << 1 );
176  }
177 
178  else if ( iDet == kFairRutherford ) {
179  if ( nPoints < 0 ) { nPoints = 0; }
180  else if ( nPoints > 31 ) { nPoints = 31; }
181  fNPoints = ( fNPoints & ( ~ ( 31 << 4 ) ) ) | ( nPoints << 4 );
182  }
183 
184  else LOG(error) << "Unknown detector ID "
185  << iDet;
186 
187 }
