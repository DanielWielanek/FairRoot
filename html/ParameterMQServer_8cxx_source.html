143 bool ParameterMQServer::ProcessRequest(FairMQMessagePtr& req, int /*index*/)
144 {
145  string parameterName = "";
146  FairParGenericSet* par = nullptr;
147 
148  string reqStr(static_cast<char*>(req->GetData()), req->GetSize());
149  LOG(INFO) << "Received parameter request from client: \"" << reqStr << "\"";
150 
151  size_t pos = reqStr.rfind(",");
152  string newParameterName = reqStr.substr(0, pos);
153  int runId = stoi(reqStr.substr(pos + 1));
154  LOG(INFO) << "Parameter name: " << newParameterName;
155  LOG(INFO) << "Run ID: " << runId;
156 
157  LOG(INFO) << "Retrieving parameter...";
158  // Check if the parameter name has changed to avoid getting same container repeatedly
159  if (newParameterName != parameterName)
160  {
161  parameterName = newParameterName;
162  par = static_cast<FairParGenericSet*>(fRtdb->getContainer(parameterName.c_str()));
163  }
164  fRtdb->initContainers(runId);
165 
166  LOG(INFO) << "Sending following parameter to the client:";
167  if (par)
168  {
169  par->print();
170 
171  TMessage* tmsg = new TMessage(kMESS_OBJECT);
172  tmsg->WriteObject(par);
173  FairMQMessagePtr rep(NewMessage(tmsg->Buffer(),
174  tmsg->BufferSize(),
175  [](void* /*data*/, void* object){ delete static_cast<TMessage*>(object); },
176  tmsg));
177 
178  if (Send(rep, fRequestChannelName, 0) < 0)
179  {
180  LOG(ERROR) << "failed sending reply";
181  return false;
182  }
183  }
184  else
185  {
186  LOG(ERROR) << "Parameter uninitialized!";
187  // Send an empty message back to keep the REQ/REP cycle
188  FairMQMessagePtr rep(NewMessage());
189  if (Send(rep, fRequestChannelName, 0) < 0)
190  {
191  LOG(ERROR) << "failed sending reply";
192  return false;
193  }
194  }
195 
196  return true;
197 }

199 bool ParameterMQServer::ProcessUpdate(FairMQMessagePtr& update, int /*index*/)
200 {
201  gGeoManager = NULL; // FairGeoParSet update deletes previous geometry because of resetting gGeoManager, so let's NULL it
202 
203  std::string* text;
204 
205  LOG(DEBUG) << "got process update message with size = " << update->GetSize() << " !";
206  if ( update->GetSize() < 20 )
207  {
208  std::string repString = string(static_cast<char*>(update->GetData()), update->GetSize());
209  LOG(INFO) << "Received string " << repString << " !";
210  if ( fNofSimDevices == 0 ) {

212  fRunId = genid.generateId();
213  }
214  string messageToSend = to_string(fRunId) + "_" + to_string(fNofSimDevices);
215  text = new string(messageToSend);
216  fNofSimDevices += 1;
217  LOG(INFO) << "Replying with \"" << messageToSend << "\"";
218  }
219  else
220  {
221  ParMQTMessage tm(update->GetData(), update->GetSize());
222 
223  // get the run id coded in the description of FairParSet
224  FairParGenericSet* newPar = (FairParGenericSet*)tm.ReadObject(tm.GetClass());
225  std::string parDescr = std::string(newPar->getDescription());
226  uint runId = 0;
227  if ( parDescr.find("RUNID") != std::string::npos )
228  {
229  parDescr.erase(0,parDescr.find("RUNID")+5);
230  runId = atoi(parDescr.data());
231  if ( parDescr.find("RUNID") != std::string::npos )
232  parDescr.erase(0,parDescr.find("RUNID")+5);
233  }
234  fRtdb->initContainers(runId);
235 
236  newPar->setChanged(true); // trigger writing to file
237  newPar->setStatic(true); // to get rid of error
238  newPar->Print();
239 
240  if ( fRtdb->addContainer(newPar) )
241  {
242  text = new string("SUCCESS");
243  }
244  else
245  {
246  text = new string("FAIL");
247  }
248 
249  Bool_t kParameterMerged = kTRUE;
250  FairParRootFileIo* parOut = new FairParRootFileIo(kParameterMerged);
251  parOut->open(fOutputName.data());
252  fRtdb->setOutput(parOut);
253  fRtdb->saveOutput();
254  fRtdb->closeOutput();
255  }
256 
257  FairMQMessagePtr msg(NewMessage(const_cast<char*>(text->c_str()),
258  text->length(),
259  [](void* /*data*/, void* object) { delete static_cast<string*>(object); },
260  text));
261 
262  if (Send(msg, fUpdateChannelName) < 0)
263  {
264  return false;
265  }
266  return true;
