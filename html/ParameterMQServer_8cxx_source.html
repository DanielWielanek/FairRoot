99 void ParameterMQServer::Run()
100 {
101  string parameterName = "";
102  FairParGenericSet* par = nullptr;
103 
104  while (CheckCurrentState(RUNNING))
105  {
106  FairMQMessagePtr req(NewMessage());
107 
108  if (Receive(req, fChannelName, 0) > 0)
109  {
110  string reqStr(static_cast<char*>(req->GetData()), req->GetSize());
111  LOG(info) << "Received parameter request from client: \"" << reqStr << "\"";
112 
113  size_t pos = reqStr.rfind(",");
114  string newParameterName = reqStr.substr(0, pos);
115  int runId = stoi(reqStr.substr(pos + 1));
116  LOG(info) << "Parameter name: " << newParameterName;
117  LOG(info) << "Run ID: " << runId;
118 
119  LOG(info) << "Retrieving parameter...";
120  // Check if the parameter name has changed to avoid getting same container repeatedly
121  if (newParameterName != parameterName)
122  {
123  parameterName = newParameterName;
124  par = static_cast<FairParGenericSet*>(fRtdb->getContainer(parameterName.c_str()));
125  }
126  fRtdb->initContainers(runId);
127 
128  LOG(info) << "Sending following parameter to the client:";
129  if (par)
130  {
131  par->print();
132 
133  TMessage* tmsg = new TMessage(kMESS_OBJECT);
134  tmsg->WriteObject(par);
135 
136  FairMQMessagePtr rep(NewMessage(tmsg->Buffer(),
137  tmsg->BufferSize(),
138  [](void* /*data*/, void* object){ delete static_cast<TMessage*>(object); },
139  tmsg));
140 
141  if (Send(rep, fChannelName, 0) < 0)
142  {
143  LOG(error) << "failed sending reply";
144  break;
145  }
146  }
147  else
148  {
149  LOG(error) << "Parameter uninitialized!";
150  // Send an empty message back to keep the REQ/REP cycle
151  FairMQMessagePtr rep(NewMessage());
152  if (Send(rep, fChannelName, 0) < 0)
153  {
154  LOG(error) << "failed sending reply";
155  break;
156  }
157  }
158  }
159  }
160 }
