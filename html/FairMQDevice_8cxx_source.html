95 void FairMQDevice::InitWrapper()
96 {
97  if (!fTransportFactory)
98  {
99  LOG(error) << "Transport not initialized. Did you call SetTransport()?";
100  exit(EXIT_FAILURE);
101  }
102 
103  // Containers to store the uninitialized channels.
104  vector<FairMQChannel*> uninitializedBindingChannels;
105  vector<FairMQChannel*> uninitializedConnectingChannels;
106 
107  // Fill the uninitialized channel containers
108  for (auto& mi : fChannels)
109  {
110  for (auto vi = mi.second.begin(); vi != mi.second.end(); ++vi)
111  {
112  // if (vi->fModified)
113  // {
114  // if (vi->fReset)
115  // {
116  // vi->fSocket.reset();
117  // }
118  // set channel name: name + vector index
119  vi->fName = fair::mq::tools::ToString(mi.first, "[", vi - (mi.second).begin(), "]");
120 
121  if (vi->fMethod == "bind")
122  {
123  // if binding address is not specified, try getting it from the configured network interface
124  if (vi->fAddress == "unspecified" || vi->fAddress == "")
125  {
126  // if the configured network interface is default, get its name from the default route
127  if (fNetworkInterface == "default")
128  {
129  fNetworkInterface = fair::mq::tools::getDefaultRouteNetworkInterface();
130  }
131  vi->fAddress = "tcp://" + fair::mq::tools::getInterfaceIP(fNetworkInterface) + ":1";
132  }
133  // fill the uninitialized list
134  uninitializedBindingChannels.push_back(&(*vi));
135  }
136  else if (vi->fMethod == "connect")
137  {
138  // fill the uninitialized list
139  uninitializedConnectingChannels.push_back(&(*vi));
140  }
141  else if (vi->fAddress.find_first_of("@+>") != string::npos)
142  {
143  // fill the uninitialized list
144  uninitializedConnectingChannels.push_back(&(*vi));
145  }
146  else
147  {
148  LOG(error) << "Cannot update configuration. Socket method (bind/connect) not specified.";
149  throw runtime_error("Cannot update configuration. Socket method (bind/connect) not specified.");
150  }
151  // }
152  }
153  }
154 
155  // Bind channels. Here one run is enough, because bind settings should be available locally
156  // If necessary this could be handled in the same way as the connecting channels
157  AttachChannels(uninitializedBindingChannels);
158 
159  if (uninitializedBindingChannels.size() > 0)
160  {
161  LOG(error) << uninitializedBindingChannels.size() << " of the binding channels could not initialize. Initial configuration incomplete.";
162  throw runtime_error(fair::mq::tools::ToString(uninitializedBindingChannels.size(), " of the binding channels could not initialize. Initial configuration incomplete."));
163  }
164 
165  CallStateChangeCallbacks(INITIALIZING_DEVICE);
166 
167  // notify parent thread about completion of first validation.
168  {
169  lock_guard<mutex> lock(fInitialValidationMutex);
170  fInitialValidationFinished = true;
171  fInitialValidationCondition.notify_one();
172  }
173 
174  // go over the list of channels until all are initialized (and removed from the uninitialized list)
175  int numAttempts = 1;
176  auto sleepTimeInMS = 50;
177  auto maxAttempts = fInitializationTimeoutInS * 1000 / sleepTimeInMS;
178  // first attempt
179  AttachChannels(uninitializedConnectingChannels);
180  // if not all channels could be connected, update their address values from config and retry
181  while (!uninitializedConnectingChannels.empty())
182  {
183  this_thread::sleep_for(chrono::milliseconds(sleepTimeInMS));
184 
185  if (fConfig)
186  {
187  for (auto& chan : uninitializedConnectingChannels)
188  {
189  string key{"chans." + chan->GetChannelPrefix() + "." + chan->GetChannelIndex() + ".address"};
190  string newAddress = fConfig->GetValue<string>(key);
191  if (newAddress != chan->GetAddress())
192  {
193  chan->UpdateAddress(newAddress);
194  }
195  }
196  }
197 
198  if (numAttempts++ > maxAttempts)
199  {
200  LOG(error) << "could not connect all channels after " << fInitializationTimeoutInS << " attempts";
201  throw runtime_error(fair::mq::tools::ToString("could not connect all channels after ", fInitializationTimeoutInS, " attempts"));
202  }
203 
204  AttachChannels(uninitializedConnectingChannels);
205  }
206 
207  Init();
208 
209  ChangeState(internal_DEVICE_READY);
210 }

248 bool FairMQDevice::AttachChannel(FairMQChannel& ch)
249 {
250  if (!ch.fTransportFactory)
251  {
252  if (ch.fTransport == "default" || ch.fTransport == fDefaultTransport)
253  {
254  LOG(debug) << ch.fName << ": using default transport";
255  ch.InitTransport(fTransportFactory);
256  }
257  else
258  {
259  LOG(debug) << ch.fName << ": channel transport (" << fDefaultTransport << ") overriden to " << ch.fTransport;
260  ch.InitTransport(AddTransport(ch.fTransport));
261  }
262  ch.fTransportType = ch.fTransportFactory->GetType();
263  }
264 
265  vector<string> endpoints;
266  boost::algorithm::split(endpoints, ch.fAddress, boost::algorithm::is_any_of(","));
267  for (auto& endpoint : endpoints)
268  {
269  //(re-)init socket
270  if (!ch.fSocket)
271  {
272  ch.fSocket = ch.fTransportFactory->CreateSocket(ch.fType, ch.fName);
273  }
274 
275  // set high water marks
276  ch.fSocket->SetOption("snd-hwm", &(ch.fSndBufSize), sizeof(ch.fSndBufSize));
277  ch.fSocket->SetOption("rcv-hwm", &(ch.fRcvBufSize), sizeof(ch.fRcvBufSize));
278 
279  // set kernel transmit size
280  if (ch.fSndKernelSize != 0)
281  {
282  ch.fSocket->SetOption("snd-size", &(ch.fSndKernelSize), sizeof(ch.fSndKernelSize));
283  }
284  if (ch.fRcvKernelSize != 0)
285  {
286  ch.fSocket->SetOption("rcv-size", &(ch.fRcvKernelSize), sizeof(ch.fRcvKernelSize));
287  }
288 
289  // attach
290  bool bind = (ch.fMethod == "bind");
291  bool connectionModifier = false;
292  string address = endpoint;
293 
294  // check if the default fMethod is overridden by a modifier
295  if (endpoint[0] == '+' || endpoint[0] == '>')
296  {
297  connectionModifier = true;
298  bind = false;
299  address = endpoint.substr(1);
300  }
301  else if (endpoint[0] == '@')
302  {
303  connectionModifier = true;
304  bind = true;
305  address = endpoint.substr(1);
306  }
307 
308  bool rc = true;
309  // make the connection
310  if (bind)
311  {
312  rc = BindEndpoint(*ch.fSocket, address);
313  }
314  else
315  {
316  rc = ConnectEndpoint(*ch.fSocket, address);
317  }
318 
319  // bind might bind to an address different than requested,
320  // put the actual address back in the config
321  endpoint.clear();
322  if (connectionModifier)
323  {
324  endpoint.push_back(bind?'@':'+');
325  }
326  endpoint += address;
327 
328  LOG(debug) << "Attached channel " << ch.fName << " to " << endpoint << (bind ? " (bind) " : " (connect) ");
329 
330  // after the book keeping is done, exit in case of errors
331  if (!rc)
332  {
333  return rc;
334  }
335  }
336 
337  // put the (possibly) modified address back in the channel object and config
338  string newAddress{boost::algorithm::join(endpoints, ",")};
339  if (newAddress != ch.fAddress)
340  {
341  ch.UpdateAddress(newAddress);
342  if (fConfig)
343  {
344  string key{"chans." + ch.GetChannelPrefix() + "." + ch.GetChannelIndex() + ".address"};
345  fConfig->SetValue(key, newAddress);
346  }
347  }
348 
349  return true;
350 }

359 bool FairMQDevice::BindEndpoint(FairMQSocket& socket, string& endpoint)
360 {
361  // number of attempts when choosing a random port
362  int maxAttempts = 1000;
363  int numAttempts = 0;
364 
365  // initialize random generator
366  default_random_engine generator(chrono::system_clock::now().time_since_epoch().count());
367  uniform_int_distribution<int> randomPort(fPortRangeMin, fPortRangeMax);
368 
369  // try to bind to the saved port. In case of failure, try random one.
370  while (!socket.Bind(endpoint))
371  {
372  LOG(debug) << "Could not bind to configured (TCP) port, trying random port in range " << fPortRangeMin << "-" << fPortRangeMax;
373  ++numAttempts;
374 
375  if (numAttempts > maxAttempts)
376  {
377  LOG(error) << "could not bind to any (TCP) port in the given range after " << maxAttempts << " attempts";
378  return false;
379  }
380 
381  size_t pos = endpoint.rfind(":");
382  endpoint = endpoint.substr(0, pos + 1) + fair::mq::tools::ToString(static_cast<int>(randomPort(generator)));
383  }
384 
385  return true;
386 }

448 void FairMQDevice::RunWrapper()
449 {
450  CallStateChangeCallbacks(RUNNING);
451 
452  LOG(info) << "DEVICE: Running...";
453 
454  // start the rate logger thread
455  thread rateLogger(&FairMQDevice::LogSocketRates, this);
456 
457  // notify transports to resume transfers
458  for (auto& t : fTransports)
459  {
460  t.second->Resume();
461  }
462 
463  try
464  {
465  PreRun();
466 
467  // process either data callbacks or ConditionalRun/Run
468  if (fDataCallbacks)
469  {
470  // if only one input channel, do lightweight handling without additional polling.
471  if (fInputChannelKeys.size() == 1 && fChannels.at(fInputChannelKeys.at(0)).size() == 1)
472  {
473  HandleSingleChannelInput();
474  }
475  else // otherwise do full handling with polling
476  {
477  HandleMultipleChannelInput();
478  }
479  }
480  else
481  {
482  using Clock = chrono::steady_clock;
483  using TimeScale = chrono::microseconds;
484  const TimeScale::rep period = TimeScale::period::den / fRate;
485  const auto reftime = Clock::now();
486  while (CheckCurrentState(RUNNING) && ConditionalRun())
487  {
488  if (fRate > 0.001) {
489  auto timespan = chrono::duration_cast<TimeScale>(Clock::now() - reftime).count() - fLastTime;
490  if (timespan < period) {
491  TimeScale sleepfor(period - timespan);
492  this_thread::sleep_for(sleepfor);
493  }
494  fLastTime = chrono::duration_cast<TimeScale>(Clock::now() - reftime).count();
495  }
496  }
497 
498  Run();
499  }
500  }
501  catch (const out_of_range& oor)
502  {
503  LOG(error) << "out of range: " << oor.what();
504  LOG(error) << "incorrect/incomplete channel configuration?";
505  }
506 
507  // if Run() exited and the state is still RUNNING, transition to READY.
508  if (CheckCurrentState(RUNNING))
509  {
510  ChangeState(internal_READY);
511  }
512 
513  PostRun();
514 
515  rateLogger.join();
516 }

538 void FairMQDevice::HandleMultipleChannelInput()
539 {
540  // check if more than one transport is used
541  fMultitransportInputs.clear();
542  for (const auto& k : fInputChannelKeys)
543  {
544  FairMQ::Transport t = fChannels.at(k).at(0).fTransportType;
545  if (fMultitransportInputs.find(t) == fMultitransportInputs.end())
546  {
547  fMultitransportInputs.insert(pair<FairMQ::Transport, vector<string>>(t, vector<string>()));
548  fMultitransportInputs.at(t).push_back(k);
549  }
550  else
551  {
552  fMultitransportInputs.at(t).push_back(k);
553  }
554  }
555 
556  for (const auto& mi : fMsgInputs)
557  {
558  for (unsigned int i = 0; i < fChannels.at(mi.first).size(); ++i)
559  {
560  fChannels.at(mi.first).at(i).fMultipart = false;
561  }
562  }
563 
564  for (const auto& mi : fMultipartInputs)
565  {
566  for (unsigned int i = 0; i < fChannels.at(mi.first).size(); ++i)
567  {
568  fChannels.at(mi.first).at(i).fMultipart = true;
569  }
570  }
571 
572  // if more than one transport is used, handle poll of each in a separate thread
573  if (fMultitransportInputs.size() > 1)
574  {
575  HandleMultipleTransportInput();
576  }
577  else // otherwise poll directly
578  {
579  bool proceed = true;
580 
581  FairMQPollerPtr poller(fChannels.at(fInputChannelKeys.at(0)).at(0).fTransportFactory->CreatePoller(fChannels, fInputChannelKeys));
582 
583  while (CheckCurrentState(RUNNING) && proceed)
584  {
585  poller->Poll(200);
586 
587  // check which inputs are ready and call their data handlers if they are.
588  for (const auto& ch : fInputChannelKeys)
589  {
590  for (unsigned int i = 0; i < fChannels.at(ch).size(); ++i)
591  {
592  if (poller->CheckInput(ch, i))
593  {
594  if (fChannels.at(ch).at(i).fMultipart)
595  {
596  proceed = HandleMultipartInput(ch, fMultipartInputs.at(ch), i);
597  }
598  else
599  {
600  proceed = HandleMsgInput(ch, fMsgInputs.at(ch), i);
601  }
602 
603  if (!proceed)
604  {
605  break;
606  }
607  }
608  }
609  if (!proceed)
610  {
611  break;
612  }
613  }
614  }
615  }

635 void FairMQDevice::PollForTransport(const FairMQTransportFactory* factory, const vector<string>& channelKeys)
636 {
637  try
638  {
639  FairMQPollerPtr poller(factory->CreatePoller(fChannels, channelKeys));
640 
641  while (CheckCurrentState(RUNNING) && fMultitransportProceed)
642  {
643  poller->Poll(500);
644 
645  for (const auto& ch : channelKeys)
646  {
647  for (unsigned int i = 0; i < fChannels.at(ch).size(); ++i)
648  {
649  if (poller->CheckInput(ch, i))
650  {
651  lock_guard<mutex> lock(fMultitransportMutex);
652 
653  if (!fMultitransportProceed)
654  {
655  break;
656  }
657 
658  if (fChannels.at(ch).at(i).fMultipart)
659  {
660  fMultitransportProceed = HandleMultipartInput(ch, fMultipartInputs.at(ch), i);
661  }
662  else
663  {
664  fMultitransportProceed = HandleMsgInput(ch, fMsgInputs.at(ch), i);
665  }
666 
667  if (!fMultitransportProceed)
668  {
669  break;
670  }
671  }
672  }
673  if (!fMultitransportProceed)
674  {
675  break;
676  }
677  }
678  }
679  }
680  catch (exception& e)
681  {
682  LOG(error) << "FairMQDevice::PollForTransport() failed: " << e.what() << ", going to ERROR state.";
683  ChangeState(ERROR_FOUND);

835 void FairMQDevice::LogSocketRates()
836 {
837  chrono::time_point<chrono::high_resolution_clock> t0;
838  chrono::time_point<chrono::high_resolution_clock> t1;
839 
840  unsigned long long msSinceLastLog;
841 
842  vector<FairMQSocket*> filteredSockets;
843  vector<string> filteredChannelNames;
844  vector<int> logIntervals;
845  vector<int> intervalCounters;
846 
847  // iterate over the channels map
848  for (const auto& mi : fChannels)
849  {
850  // iterate over the channels vector
851  for (auto vi = (mi.second).begin(); vi != (mi.second).end(); ++vi)
852  {
853  if (vi->fRateLogging > 0)
854  {
855  filteredSockets.push_back(vi->fSocket.get());
856  logIntervals.push_back(vi->fRateLogging);
857  intervalCounters.push_back(0);
858  filteredChannelNames.push_back(fair::mq::tools::ToString(mi.first, "[", vi - (mi.second).begin(), "]"));
859  }
860  }
861  }
862 
863  unsigned int numFilteredSockets = filteredSockets.size();
864 
865  if (numFilteredSockets > 0)
866  {
867  vector<unsigned long> bytesIn(numFilteredSockets);
868  vector<unsigned long> msgIn(numFilteredSockets);
869  vector<unsigned long> bytesOut(numFilteredSockets);
870  vector<unsigned long> msgOut(numFilteredSockets);
871 
872  vector<unsigned long> bytesInNew(numFilteredSockets);
873  vector<unsigned long> msgInNew(numFilteredSockets);
874  vector<unsigned long> bytesOutNew(numFilteredSockets);
875  vector<unsigned long> msgOutNew(numFilteredSockets);
876 
877  vector<double> mbPerSecIn(numFilteredSockets);
878  vector<double> msgPerSecIn(numFilteredSockets);
879  vector<double> mbPerSecOut(numFilteredSockets);
880  vector<double> msgPerSecOut(numFilteredSockets);
881 
882  int i = 0;
883  for (const auto& vi : filteredSockets)
884  {
885  bytesIn.at(i) = vi->GetBytesRx();
886  bytesOut.at(i) = vi->GetBytesTx();
887  msgIn.at(i) = vi->GetMessagesRx();
888  msgOut.at(i) = vi->GetMessagesTx();
889  ++i;
890  }
891 
892  t0 = chrono::high_resolution_clock::now();
893 
894  LOG(debug) << "<channel>: in: <#msgs> (<MB>) out: <#msgs> (<MB>)";
895 
896  while (CheckCurrentState(RUNNING))
897  {
898  t1 = chrono::high_resolution_clock::now();
899 
900  msSinceLastLog = chrono::duration_cast<chrono::milliseconds>(t1 - t0).count();
901 
902  i = 0;
903 
904  for (const auto& vi : filteredSockets)
905  {
906  intervalCounters.at(i)++;
907 
908  if (intervalCounters.at(i) == logIntervals.at(i))
909  {
910  intervalCounters.at(i) = 0;
911 
912  bytesInNew.at(i) = vi->GetBytesRx();
913  msgInNew.at(i) = vi->GetMessagesRx();
914  bytesOutNew.at(i) = vi->GetBytesTx();
915  msgOutNew.at(i) = vi->GetMessagesTx();
916 
917  mbPerSecIn.at(i) = (static_cast<double>(bytesInNew.at(i) - bytesIn.at(i)) / (1000. * 1000.)) / static_cast<double>(msSinceLastLog) * 1000.;
918  msgPerSecIn.at(i) = static_cast<double>(msgInNew.at(i) - msgIn.at(i)) / static_cast<double>(msSinceLastLog) * 1000.;
919  mbPerSecOut.at(i) = (static_cast<double>(bytesOutNew.at(i) - bytesOut.at(i)) / (1000. * 1000.)) / static_cast<double>(msSinceLastLog) * 1000.;
920  msgPerSecOut.at(i) = static_cast<double>(msgOutNew.at(i) - msgOut.at(i)) / static_cast<double>(msSinceLastLog) * 1000.;
921 
922  bytesIn.at(i) = bytesInNew.at(i);
923  msgIn.at(i) = msgInNew.at(i);
924  bytesOut.at(i) = bytesOutNew.at(i);
925  msgOut.at(i) = msgOutNew.at(i);
926 
927  LOG(info) << filteredChannelNames.at(i) << ": "
928  << "in: " << msgPerSecIn.at(i) << " (" << mbPerSecIn.at(i) << " MB) "
929  << "out: " << msgPerSecOut.at(i) << " (" << mbPerSecOut.at(i) << " MB)";
930  }
931 
932  ++i;
933  }
934 
935  t0 = t1;
936  this_thread::sleep_for(chrono::milliseconds(1000));
937  }
938  }
939 }
