172  ret = fi_enable(fDataEndpoint);
173  if (ret != FI_SUCCESS)
174  throw SocketError(tools::ToString("Failed enabling ofi endpoint, reason: ", fi_strerror(ret)));

253 auto Socket::WaitForControlPeer() -> void
254 {
255  assert(fWaitingForControlPeer);
256 
257  // First frame in message contains event number and value
258  zmq_msg_t msg;
259  zmq_msg_init(&msg);
260  if (zmq_msg_recv(&msg, fMonitorSocket, 0) == -1)
261  throw SocketError(tools::ToString("Failed to get monitor event, reason: ", zmq_strerror(errno)));
262 
263  uint8_t* data = (uint8_t*) zmq_msg_data(&msg);
264  uint16_t event = *(uint16_t*)(data);
265  int value = *(uint32_t *)(data + 2);
266 
267  // Second frame in message contains event address
268  zmq_msg_init(&msg);
269  if (zmq_msg_recv(&msg, fMonitorSocket, 0) == -1)
270  throw SocketError(tools::ToString("Failed to get monitor event, reason: ", zmq_strerror(errno)));
271 
272  if (event == ZMQ_EVENT_ACCEPTED) {
273  // string localAddress = string(static_cast<char*>(zmq_msg_data(&msg)), zmq_msg_size(&msg));
274  sockaddr_in remoteAddr;
275  socklen_t addrSize = sizeof(sockaddr_in);
276  int ret = getpeername(value, (sockaddr*)&remoteAddr, &addrSize);
277  if (ret != 0)
278  throw SocketError(tools::ToString("Failed retrieving remote address, reason: ", strerror(errno)));
279  string remoteIp(inet_ntoa(remoteAddr.sin_addr));
280  int remotePort = ntohs(remoteAddr.sin_port);
281  LOG(debug) << "Accepted control peer connection from " << remoteIp << ":" << remotePort;
282  } else if (event == ZMQ_EVENT_CONNECTED) {
283  LOG(debug) << "Connected successfully to control peer";
284  } else {
285  LOG(debug) << "Unknown monitor event received: " << event << ". Ignoring.";
286  }
287 
288  fWaitingForControlPeer = false;
289 }
