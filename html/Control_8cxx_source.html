96 auto Control::InteractiveMode() -> void
97 {
98  try
99  {
100  RunStartupSequence();
101 
102  char input; // hold the user console input
103  pollfd cinfd[1];
104  cinfd[0].fd = fileno(stdin);
105  cinfd[0].events = POLLIN;
106 
107  struct termios t;
108  tcgetattr(STDIN_FILENO, &t); // get the current terminal I/O structure
109  t.c_lflag &= ~ICANON; // disable canonical input
110  tcsetattr(STDIN_FILENO, TCSANOW, &t); // apply the new settings
111 
112  PrintInteractiveHelp();
113 
114  bool keepRunning = true;
115 
116  while (keepRunning)
117  {
118  if (poll(cinfd, 1, 500))
119  {
120  if (fDeviceTerminationRequested)
121  {
122  break;
123  }
124 
125  cin >> input;
126 
127  switch (input)
128  {
129  case 'i':
130  LOG(info) << "\n\n --> [i] init device\n";
131  ChangeDeviceState(DeviceStateTransition::InitDevice);
132  break;
133  case 'j':
134  LOG(info) << "\n\n --> [j] init task\n";
135  ChangeDeviceState(DeviceStateTransition::InitTask);
136  break;
137  case 'p':
138  LOG(info) << "\n\n --> [p] pause\n";
139  ChangeDeviceState(DeviceStateTransition::Pause);
140  break;
141  case 'r':
142  LOG(info) << "\n\n --> [r] run\n";
143  ChangeDeviceState(DeviceStateTransition::Run);
144  break;
145  case 's':
146  LOG(info) << "\n\n --> [s] stop\n";
147  ChangeDeviceState(DeviceStateTransition::Stop);
148  break;
149  case 't':
150  LOG(info) << "\n\n --> [t] reset task\n";
151  ChangeDeviceState(DeviceStateTransition::ResetTask);
152  break;
153  case 'd':
154  LOG(info) << "\n\n --> [d] reset device\n";
155  ChangeDeviceState(DeviceStateTransition::ResetDevice);
156  break;
157  case 'h':
158  LOG(info) << "\n\n --> [h] help\n";
159  PrintInteractiveHelp();
160  break;
161  // case 'x':
162  // LOG(info) << "\n\n --> [x] ERROR\n";
163  // ChangeDeviceState(DeviceStateTransition::ERROR_FOUND);
164  // break;
165  case 'q':
166  LOG(info) << "\n\n --> [q] end\n";
167  keepRunning = false;
168  break;
169  default:
170  LOG(info) << "Invalid input: [" << input << "]";
171  PrintInteractiveHelp();
172  break;
173  }
174  }
175 
176  if (fDeviceTerminationRequested)
177  {
178  keepRunning = false;
179  }
180  }
181 
182  tcgetattr(STDIN_FILENO, &t); // get the current terminal I/O structure
183  t.c_lflag |= ICANON; // re-enable canonical input
184  tcsetattr(STDIN_FILENO, TCSANOW, &t); // apply the new settings
185 
186  if (!fDeviceTerminationRequested)
187  {
188  RunShutdownSequence();
189  }
190  }
191  catch (PluginServices::DeviceControlError& e)
192  {
193  // If we are here, it means another plugin has taken control. That's fine, just print the exception message and do nothing else.
194  LOG(debug) << e.what();
195  }
196 }

245 auto Control::SignalHandler(int signal) -> void
246 {
247  if (!fDeviceTerminationRequested)
248  {
249  fDeviceTerminationRequested = true;
250 
251  StealDeviceControl();
252 
253  LOG(info) << "Received device shutdown request (signal " << signal << ").";
254  LOG(info) << "Waiting for graceful device shutdown. Hit Ctrl-C again to abort immediately.";
255 
256  UnsubscribeFromDeviceStateChange(); // In case, static or interactive mode have subscribed already
257  SubscribeToDeviceStateChange([&](DeviceState newState)
258  {
259  {
260  lock_guard<mutex> lock{fEventsMutex};
261  fEvents.push(newState);
262  }
263  fNewEvent.notify_one();
264  });
265 
266  fSignalHandlerThread = thread(&Control::RunShutdownSequence, this);
267  }
268  else
269  {
270  LOG(warn) << "Received 2nd device shutdown request (signal " << signal << ").";
271  LOG(warn) << "Aborting immediately !";
272  abort();
273  }

276 auto Control::RunShutdownSequence() -> void
277 {
278  auto nextState = GetCurrentDeviceState();
279  EmptyEventQueue();
280  while (nextState != DeviceState::Exiting)
281  {
282  switch (nextState)
283  {
284  case DeviceState::Idle:
285  ChangeDeviceState(DeviceStateTransition::End);
286  break;
287  case DeviceState::DeviceReady:
288  ChangeDeviceState(DeviceStateTransition::ResetDevice);
289  break;
290  case DeviceState::Ready:
291  ChangeDeviceState(DeviceStateTransition::ResetTask);
292  break;
293  case DeviceState::Running:
294  ChangeDeviceState(DeviceStateTransition::Stop);
295  break;
296  case DeviceState::Paused:
297  ChangeDeviceState(DeviceStateTransition::Resume);
298  break;
299  default:
300  break;
301  }
302 
303  nextState = WaitForNextState();
304  }
305 
306  UnsubscribeFromDeviceStateChange();
307  ReleaseDeviceControl();
308 }

310 auto Control::RunStartupSequence() -> void
311 {
312  SubscribeToDeviceStateChange([&](DeviceState newState)
313  {
314  {
315  lock_guard<mutex> lock{fEventsMutex};
316  fEvents.push(newState);
317  }
318  fNewEvent.notify_one();
319  });
320 
321  ChangeDeviceState(DeviceStateTransition::InitDevice);
322  while (WaitForNextState() != DeviceState::DeviceReady) {}
323  ChangeDeviceState(DeviceStateTransition::InitTask);
324  while (WaitForNextState() != DeviceState::Ready) {}
325  ChangeDeviceState(DeviceStateTransition::Run);
326  while (WaitForNextState() != DeviceState::Running) {}
327 }
