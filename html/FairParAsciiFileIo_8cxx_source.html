57 Bool_t FairParAsciiFileIo::open(const Text_t* fname, const Text_t* status)
58 {
59  // opens file
60  // if a file is already open, this file will be closed
61  // activates detector I/Os
62  close();
63  if (!((strcmp(status,"in")==0) || (strcmp(status,"out")==0))) {
64  cout<<"Put the right stream option for file "<<fname
65  <<"\n writing state : out\n reading state : in \nopen aborted \n";
66  return kFALSE;
67  }
68  file=new std::fstream();
69  if(strcmp(status,"in")==0) {file->open( fname, ios::in);};
70  if(strcmp(status,"out")==0) {file->open( fname, ios::out);};
71  filebuf* buf = file->rdbuf();
72  if (file && (buf->is_open()==1)) {
73  filename=fname;
74  FairRuntimeDb::instance()->activateParIo(this);
75  return kTRUE;
76  }
77  cerr << "-E- Could not open input file " << fname << endl;
78  Fatal("open","Could not open input file");
79  return kFALSE;
80 }

82 Bool_t FairParAsciiFileIo::open(const TList* fnamelist, const Text_t* status)
83 {
84  if ( 0 == fnamelist->GetEntries() ) {
85  LOG(error) << "The defined list of parameter files is empty. There are no parameters initialized from the ASCII files.";
86  return kFALSE;
87  }
88  TString outFileName = gSystem->WorkingDirectory();
89 
90  outFileName += "/all_";
91  Int_t pid = gSystem->GetPid();
92  outFileName += pid;
93  outFileName += ".par";
94  TString catCommand = "cat ";
95  TObjString* string;
96  TListIter myIter(fnamelist);
97  while((string = static_cast<TObjString*>(myIter.Next()))) {
98  // check if the file exist
99  // if file exist return value is false
100  TString strParPath = string->GetString();
101  gSystem->ExpandPathName(strParPath);
102  if (gSystem->AccessPathName(strParPath))
103  LOG(FATAL) << "Parameter file " << strParPath << " does not exist." << FairLogger::endl;
104  // cout << string->GetString() <<endl;
105  catCommand += string->GetString();
106  catCommand += " ";
107  }
108  catCommand += "> ";
109  catCommand += outFileName;
110 
111  // cout <<"CAT: "<<catCommand.Data()<<endl;
112 
113  gSystem->Exec(catCommand);
114 
115  return open(outFileName, status);
