112 bool FairMQEx9FileSinkBin::StoreData(FairMQParts& parts, int /*index*/)
113 {
114  if ( parts.Size() == 0 ) return true; // probably impossible, but still check
115 
116  // expecting even number of parts in the form: header,data,header,data,header,data and so on...
117  int nPPE = 2; // nof parts per event
118 
119  if ( parts.Size()%nPPE >= 1 )
120  LOG(info) << "received " << parts.Size() << " parts, will ignore last part!!!";
121 
122  for ( int ievent = 0 ; ievent < parts.Size()/nPPE ; ievent++ ) {
123 
124  // the first part should be the event header
125  PixelPayload::EventHeader* payloadE = static_cast<PixelPayload::EventHeader*>(parts.At(nPPE*ievent)->GetData());
126  // LOG(debug) << "GOT EVENT " << payloadE->fMCEntryNo << " OF RUN " << payloadE->fRunId << " (part " << payloadE->fPartNo << ")";
127 
128  for ( unsigned int ibr = 0 ; ibr < fBranchNames.size() ; ibr++ )
129  {
130  if ( "EventHeader." == fBranchNames[ibr] )
131  {
132  ((FairEventHeader*)fOutputObjects[ibr])->SetRunId (payloadE->fRunId);
133  ((FairEventHeader*)fOutputObjects[ibr])->SetMCEntryNumber(payloadE->fMCEntryNo);
134  }
135  }
136 
137  // the second part should the TClonesArray with necessary data... now assuming Digi
138  PixelPayload::Hit* payloadH = static_cast<PixelPayload::Hit*>(parts.At(nPPE*ievent+1)->GetData());
139  int hitArraySize = parts.At(nPPE*ievent+1)->GetSize();
140  int nofHits = hitArraySize / sizeof(PixelPayload::Hit);
141 
142  for ( unsigned int ibr = 0 ; ibr < fBranchNames.size() ; ibr++ )
143  {
144  if ( "PixelHits" == fBranchNames[ibr] )
145  {
146  ((TClonesArray*)fOutputObjects[ibr])->Clear();
147  for ( int ihit = 0 ; ihit < nofHits ; ihit++ )
148  {
149  TVector3 pos (payloadH[ihit].posX, payloadH[ihit].posY, payloadH[ihit].posZ);
150  TVector3 posErr(payloadH[ihit].dposX,payloadH[ihit].dposY,payloadH[ihit].dposZ);
151  new ((*((TClonesArray*)fOutputObjects[ibr]))[ihit]) PixelHit(payloadH[ihit].fDetectorID,-1,pos,posErr);
152  // new ((*fHits)[fNHits]) PixelHit(detId,currentDigi->GetIndex(),pos,posErr);
153  }
154  }
155  }
156 
157  fTree->Fill();
158  }
159 
160  if ( fAckChannelName != "" ) {
161  unique_ptr<FairMQMessage> msg(NewMessage());
162  Send(msg, fAckChannelName);
163  }
164  return true;
165 }
