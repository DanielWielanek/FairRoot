87 Int_t PixelDigiBinSource::ReadEvent(UInt_t i)
88 {
89 
90  fDigis->Clear();
91  fNDigis = 0;
92 
93  if ( !fInputFile ) {
94  return 1;
95  }
96 
97  if ( i == 0 ) {
98  fInputFile.clear();
99  fInputFile.seekg(0, std::ios::beg);
100  }
101 
102  fCurrentEntryNo = i;
103 
104  std::string buffer;
105  LOG(debug) << "PixelDigiBinSource::ReadEvent() Begin of (" << fDigis->GetEntries() << ")";
106 
107  Int_t head[4]; // runId, MCEntryNo, PartNo, NofDigis
108  fInputFile.read((char*)head,sizeof(head));
109 
110  if ( fInputFile.eof() ) {
111  LOG(info) << "End of file reached!";
112  return 1;
113  }
114 
115  Int_t dataSize = 4; // detId, feId, col, row
116 
117  const Int_t constNofData = head[3]*dataSize;
118  short int dataCont[constNofData];
119  fInputFile.read((char*)dataCont,sizeof(dataCont));
120 
121  fRunId = head[0];
122  fMCEntryNo = head[1];
123  fPartNo = head[2];
124  fEventHeader->SetRunId(fRunId);
125  fEventHeader->SetMCEntryNumber(fMCEntryNo);
126  fEventHeader->SetPartNo(fPartNo);
127 
128  for ( Int_t idata = 0 ; idata < head[3] ; idata++ ) {
129  LOG(debug) << " --/" << idata << "/--> "
130  << dataCont[idata*dataSize+0] << " / "
131  << dataCont[idata*dataSize+1] << " / "
132  << dataCont[idata*dataSize+2] << " / "
133  << dataCont[idata*dataSize+3] << " / "
134  << " 0.";
135  new ((*fDigis)[fNDigis]) PixelDigi(-1,
136  (Int_t)dataCont[idata*dataSize+0],
137  (Int_t)dataCont[idata*dataSize+1],
138  (Int_t)dataCont[idata*dataSize+2],
139  (Int_t)dataCont[idata*dataSize+3],
140  0.,0.);
141  fNDigis++;
142  }
143  LOG(debug) << "PixelDigiBinSource::ReadEvent() End of";
144 
145  if ( !fInputFile ) {
146  return 1;
147  }
148 
149  return 0;
150 }
