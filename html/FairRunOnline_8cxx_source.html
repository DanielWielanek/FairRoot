129 void FairRunOnline::Init()
130 {
131  LOG(info)<<"FairRunOnline::Init";
132  if (fIsInitialized) {
133  LOG(fatal) << "Error Init is already called before!";
134  exit(-1);
135  } else {
136  fIsInitialized = kTRUE;
137  }
138 
139  fRootManager->InitSource();
140  fRootManager->InitSink();
141 
142  // FairGeoLoader* loader = new FairGeoLoader("TGeo", "Geo Loader");
143  // FairGeoInterface* GeoInterFace = loader->getGeoInterface();
144  // GeoInterFace->SetNoOfSets(ListOfModules->GetEntries());
145  // GeoInterFace->setMediaFile(MatFname.Data());
146  // GeoInterFace->readMedia();
147 
148  // Add a Generated run ID to the FairRunTimeDb for input data which contain a FairMCEventHeader
149  // The call doesn't make sense for online sources which doesn't contain a FairMCEventHeader
150 
151  if(kONLINE != fRootManager->GetSource()->GetSourceType())
152  {
153  fRootManager->ReadEvent(0);
154  }
155 
156  GetEventHeader();
157 
158  fRootManager->FillEventHeader(fEvtHeader);
159 
160  if(0 == fRunId) // Run ID was not set in run manager
161  {
162  if(0 == fEvtHeader->GetRunId()) // Run ID was not set in source
163  {
164  // Generate unique Run ID
165  FairRunIdGenerator genid;
166  fRunId = genid.generateId();
167  fRootManager->GetSource()->SetRunId(fRunId);
168  }
169  else
170  {
171  // Use Run ID from source
172  fRunId = fEvtHeader->GetRunId();
173  }
174  }
175  else
176  {
177  // Run ID was set in the run manager - propagate to source
178  fRootManager->GetSource()->SetRunId(fRunId);
179  }
180 
181  fRtdb->addRun(fRunId);
182  fFileHeader->SetRunId(fRunId);
183  FairBaseParSet* par = dynamic_cast<FairBaseParSet*>(fRtdb->getContainer("FairBaseParSet"));
184  FairGeoParSet* geopar = dynamic_cast<FairGeoParSet*>(fRtdb->getContainer("FairGeoParSet"));
185  if (geopar) {
186  geopar->SetGeometry(gGeoManager);
187  }
188  if(fField) {
189  fField->Init();
190  fField->FillParContainer();
191  }
192  TList* containerList = fRtdb->getListOfContainers();
193  TIter next(containerList);
194  FairParSet* cont;
195  TObjArray* ContList= new TObjArray();
196  while ((cont=dynamic_cast<FairParSet*>(next()))) {
197  ContList->Add(new TObjString(cont->GetName()));
198  }
199  if (par) {
200  par->SetContListStr(ContList);
201  par->setChanged();
202  par->setInputVersion(fRunId,1);
203  }
204  if (geopar) {
205  geopar->setChanged();
206  geopar->setInputVersion(fRunId,1);
207  }
208 
209  fRootManager->WriteFileHeader(fFileHeader);
210 
211  fRootManager->GetSource()->SetParUnpackers();
212  fTask->SetParTask();
213  fRtdb->initContainers(fRunId);
214 
215  // InitContainers();
216  // --- Get event header from Run
217  if ( ! fEvtHeader ) {
218  LOG(fatal) << "FairRunOnline::InitContainers:No event header in run!";
219  return;
220  }
221  LOG(info) << "FairRunOnline::InitContainers: event header at " << fEvtHeader;
222  fRootManager->Register("EventHeader.", "Event", fEvtHeader, kTRUE);
223  fEvtHeader->SetRunId(fRunId);
224 
225  fRootManager->GetSource()->InitUnpackers();
226 
227  // Now call the User initialize for Tasks
228  fTask->InitTask();
229 
230  // create the output tree after tasks initialisation
231  fRootManager->InitSink();
232  fRootManager->WriteFolder();
233  fRootManager->WriteFileHeader(fFileHeader);
234 }

279 Int_t FairRunOnline::EventLoop()
280 {
281  gSystem->IgnoreInterrupt();
282  signal(SIGINT, handler_ctrlc);
283 
284  fRootManager->FillEventHeader(fEvtHeader);
285  Int_t tmpId = fEvtHeader->GetRunId();
286 
287  if ( tmpId != -1 && tmpId != fRunId ) {
288  LOG(info) << "FairRunOnline::EventLoop() Call Reinit due to changed RunID (from " << fRunId << " to " << tmpId;
289  fRunId = tmpId;
290  Reinit( fRunId );
291  fRootManager->GetSource()->ReInitUnpackers();
292  fTask->ReInitTask();
293  }
294 
295  fRootManager->StoreWriteoutBufferData(fRootManager->GetEventTime());
296  fTask->ExecuteTask("");
297  fRootManager->FillEventHeader(fEvtHeader);
298  Fill();
299  fRootManager->DeleteOldWriteoutBufferData();
300  fTask->FinishEvent();
301  fNevents += 1;
302  if(fServer && 0 == (fNevents%fServerRefreshRate))
303  {
304  fServer->ProcessRequests();
305  }
306 
307  if(gIsInterrupted)
308  {
309  return 1;
310  }
311 
312  return 0;
313 }

316 void FairRunOnline::Run(Int_t Ev_start, Int_t Ev_end)
317 {
318  fNevents = 0;
319 
320  gIsInterrupted = kFALSE;
321 
322  Int_t MaxAllowed=fRootManager->CheckMaxEventNo(Ev_end);
323  if ( MaxAllowed != -1 ) {
324  if (Ev_end==0) {
325  if (Ev_start==0) {
326  Ev_end=MaxAllowed;
327  } else {
328  Ev_end = Ev_start;
329  if ( Ev_end > MaxAllowed ) {
330  Ev_end = MaxAllowed;
331  }
332  Ev_start=0;
333  }
334  } else {
335  if (Ev_end > MaxAllowed) {
336  cout << "-------------------Warning---------------------------" << endl;
337  cout << " -W FairRunAna : File has less events than requested!!" << endl;
338  cout << " File contains : " << MaxAllowed << " Events" << endl;
339  cout << " Requested number of events = " << Ev_end << " Events"<< endl;
340  cout << " The number of events is set to " << MaxAllowed << " Events"<< endl;
341  cout << "-----------------------------------------------------" << endl;
342  Ev_end = MaxAllowed;
343  }
344  }
345  }
346  Int_t status;
347  if(Ev_start < 0) {
348  while(kTRUE) {
349  status = fRootManager->ReadEvent();
350  if(0 == status) {
351  status = EventLoop();
352  }
353  if(1 == status) {
354  break;
355  } else if(2 == status) {
356  continue;
357  }
358  if(gIsInterrupted)
359  {
360  break;
361  }
362  }
363  } else {
364  for (Int_t i = Ev_start; i < Ev_end; i++) {
365  status = fRootManager->ReadEvent(i);
366  if(0 == status) {
367  status = EventLoop();
368  }
369  if(1 == status) {
370  break;
371  } else if(2 == status) {
372  i -= 1;
373  continue;
374  }
375 
376  if(gIsInterrupted)
377  {
378  break;
379  }
380  }
381  }
382 
383  fRootManager->StoreAllWriteoutBufferData();
384  if (fAutomaticFinish) {
385  Finish();
386  }
387 }
