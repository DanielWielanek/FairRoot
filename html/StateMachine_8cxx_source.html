63 auto StateMachine::Run() -> void
64 {
65  LOG(state) << "Starting FairMQ state machine";
66 
67  LOG(debug) << "Entering initial " << fErrorState << " state (orthogonal error state machine)";
68  LOG(state) << "Entering initial " << fState << " state";
69 
70  std::unique_lock<std::mutex> lock{fMutex};
71  while (true)
72  {
73  while (fNextStates.empty())
74  {
75  fNewState.wait(lock);
76  }
77 
78  State lastState;
79 
80  if (fNextStates.front() == State::Error)
81  {
82  // advance error FSM
83  lastState = fErrorState;
84  fErrorState = fNextStates.front();
85  fNextStates.pop_front();
86  LOG(error) << "Entering " << fErrorState << " state (orthogonal error state machine)";
87  }
88  else
89  {
90  // advance regular FSM
91  lastState = fState;
92  fState = fNextStates.front();
93  fNextStates.pop_front();
94  LOG(state) << "Entering " << fState << " state";
95  }
96  lock.unlock();
97 
98  fCallbacks.Emit<StateChange, State>(fState, lastState);
99 
100  lock.lock();
101  if (fState == State::Exiting || fErrorState == State::Error) break;
102  }
103 
104  LOG(state) << "Exiting FairMQ state machine";
105 }

107 auto StateMachine::ChangeState(StateTransition transition) -> void
108 {
109  State lastState;
110 
111  std::unique_lock<std::mutex> lock{fMutex};
112 
113  if (transition == StateTransition::ErrorFound)
114  {
115  lastState = fErrorState;
116  }
117  else if (fNextStates.empty())
118  {
119  lastState = fState;
120  }
121  else
122  {
123  lastState = fNextStates.back();
124  }
125 
126  const State nextState{Transition(lastState, transition)};
127  fNextStates.push_back(nextState);
128  lock.unlock();
129 
130  fCallbacks.Emit<StateQueued, State>(nextState, lastState);
131  fNewState.notify_one();
132 }

134 auto StateMachine::Transition(const State currentState, const StateTransition transition) -> State
135 {
136  switch (currentState) {
137  case State::Idle:
138  if (transition == StateTransition::InitDevice ) return State::InitializingDevice;
139  if (transition == StateTransition::End ) return State::Exiting;
140  break;
141  case State::InitializingDevice:
142  if (transition == StateTransition::Automatic ) return State::DeviceReady;
143  break;
144  case State::DeviceReady:
145  if (transition == StateTransition::InitTask ) return State::InitializingTask;
146  if (transition == StateTransition::ResetDevice) return State::ResettingDevice;
147  break;
148  case State::InitializingTask:
149  if (transition == StateTransition::Automatic ) return State::Ready;
150  break;
151  case State::Ready:
152  if (transition == StateTransition::Run ) return State::Running;
153  if (transition == StateTransition::ResetTask ) return State::ResettingTask;
154  break;
155  case State::Running:
156  if (transition == StateTransition::Stop ) return State::Ready;
157  break;
158  case State::ResettingTask:
159  if (transition == StateTransition::Automatic ) return State::DeviceReady;
160  break;
161  case State::ResettingDevice:
162  if (transition == StateTransition::Automatic ) return State::Idle;
163  break;
164  case State::Exiting:
165  break;
166  case State::Ok:
167  if (transition == StateTransition::ErrorFound ) return State::Error;
168  break;
169  case State::Error:
170  break;
171  }
172  throw IllegalTransition{tools::ToString("No transition ", transition, " from state ", currentState, ".")};
173 }
