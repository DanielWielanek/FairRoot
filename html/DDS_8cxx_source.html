66 auto DDS::HandleControl() -> void
67 {
68  try
69  {
70  // subscribe for state changes from DDS (subscriptions start firing after fService.start() is called)
71  SubscribeForCustomCommands();
72 
73  // subscribe for DDS service errors.
74  fService.subscribeOnError([](const dds::intercom_api::EErrorCode errorCode, const string& errorMsg) {
75  LOG(error) << "DDS Error received: error code: " << errorCode << ", error message: " << errorMsg << endl;
76  });
77 
78  LOG(debug) << "Subscribing for DDS properties.";
79  SubscribeForConnectingChannels();
80 
81  // subscribe to device state changes, pushing new state chenges into the event queue
82  SubscribeToDeviceStateChange([&](DeviceState newState)
83  {
84  {
85  lock_guard<mutex> lock{fEventsMutex};
86  fEvents.push(newState);
87  }
88  fNewEvent.notify_one();
89  if (newState == DeviceState::Exiting)
90  {
91  fDeviceTerminationRequested = true;
92  }
93 
94  {
95  lock_guard<mutex> lock{fStateChangeSubscriberMutex};
96  string id = GetProperty<string>("id");
97  for (auto subscriberId : fStateChangeSubscribers)
98  {
99  LOG(debug) << "Publishing state-change: " << newState << " to " << subscriberId;
100  fDDSCustomCmd.send("state-change: " + id + "," + ToStr(newState), to_string(subscriberId));
101  }
102  }
103  });
104 
105  ChangeDeviceState(DeviceStateTransition::InitDevice);
106  while (WaitForNextState() != DeviceState::InitializingDevice) {}
107 
108  // in the Initializing state subscribe to receive addresses of connecting channels from DDS
109  // and propagate addresses of bound channels to DDS.
110  FillChannelContainers();
111 
112  LOG(DEBUG) << "$DDS_TASK_PATH: " << getenv("DDS_TASK_PATH");
113  LOG(DEBUG) << "$DDS_GROUP_NAME: " << getenv("DDS_GROUP_NAME");
114  LOG(DEBUG) << "$DDS_COLLECTION_NAME: " << getenv("DDS_COLLECTION_NAME");
115  LOG(DEBUG) << "$DDS_TASK_NAME: " << getenv("DDS_TASK_NAME");
116  LOG(DEBUG) << "$DDS_TASK_INDEX: " << getenv("DDS_TASK_INDEX");
117  LOG(DEBUG) << "$DDS_COLLECTION_INDEX: " << getenv("DDS_COLLECTION_INDEX");
118 
119  // start DDS service - subscriptions will only start firing after this step
120  fService.start();
121 
122  // publish bound addresses via DDS at keys corresponding to the channel prefixes, e.g. 'data' in data[i]
123  PublishBoundChannels();
124 
125  while (WaitForNextState() != DeviceState::DeviceReady) {}
126 
127  ChangeDeviceState(DeviceStateTransition::InitTask);
128  while (WaitForNextState() != DeviceState::Ready) {}
129  ChangeDeviceState(DeviceStateTransition::Run);
130 
131  // wait until stop signal
132  unique_lock<mutex> lock(fStopMutex);
133  while (!fDeviceTerminationRequested)
134  {
135  fStopCondition.wait_for(lock, chrono::seconds(1));
136  }
137  LOG(debug) << "Stopping DDS control plugin";
138  }
139  catch (exception& e)
140  {
141  LOG(error) << "Error: " << e.what() << endl;
142  return;
143  }
144 
145  fDDSKeyValue.unsubscribe();
146  fDDSCustomCmd.unsubscribe();
147 
148  try
149  {
150  UnsubscribeFromDeviceStateChange();
151  ReleaseDeviceControl();
152  }
153  catch (fair::mq::PluginServices::DeviceControlError& e)
154  {
155  LOG(error) << e.what();
156  }

159 auto DDS::FillChannelContainers() -> void
160 {
161  try {
162  unordered_map<string, int> channelInfo(GetChannelInfo());
163 
164  // fill binding and connecting chans
165  for (const auto& c : channelInfo) {
166  string methodKey{"chans." + c.first + "." + to_string(c.second - 1) + ".method"};
167  if (GetProperty<string>(methodKey) == "bind") {
168  fBindingChans.insert(make_pair(c.first, vector<string>()));
169  for (int i = 0; i < c.second; ++i) {
170  fBindingChans.at(c.first).push_back(GetProperty<string>(string{"chans." + c.first + "." + to_string(i) + ".address"}));
171  }
172  } else if (GetProperty<string>(methodKey) == "connect") {
173  fConnectingChans.insert(make_pair(c.first, DDSConfig()));
174  LOG(debug) << "preparing to connect: " << c.first << " with " << c.second << " sub-channels.";
175  for (int i = 0; i < c.second; ++i) {
176  fConnectingChans.at(c.first).fSubChannelAddresses.push_back(string());
177  }
178  } else {
179  LOG(error) << "Cannot update address configuration. Channel method (bind/connect) not specified.";
180  return;
181  }
182  }
183 
184  // save properties that will have multiple values arriving (with only some of them to be used)
185  vector<string> iValues;
186  if (PropertyExists("dds-i")) {
187  iValues = GetProperty<vector<string>>("dds-i");
188  }
189  vector<string> inValues;
190  if (PropertyExists("dds-i-n")) {
191  inValues = GetProperty<vector<string>>("dds-i-n");
192  }
193 
194  for (const auto& vi : iValues) {
195  size_t pos = vi.find(":");
196  string chanName = vi.substr(0, pos );
197 
198  // check if provided name is a valid channel name
199  if (fConnectingChans.find(chanName) == fConnectingChans.end()) {
200  throw invalid_argument(ToString("channel provided to dds-i is not an actual connecting channel of this device: ", chanName));
201  }
202 
203  int i = stoi(vi.substr(pos + 1));
204  LOG(debug) << "dds-i: adding " << chanName << " -> i of " << i;
205  fI.insert(make_pair(chanName, i));
206  }
207 
208  for (const auto& vi : inValues) {
209  size_t pos = vi.find(":");
210  string chanName = vi.substr(0, pos);
211 
212  // check if provided name is a valid channel name
213  if (fConnectingChans.find(chanName) == fConnectingChans.end()) {
214  throw invalid_argument(ToString("channel provided to dds-i-n is not an actual connecting channel of this device: ", chanName));
215  }
216 
217  string i_n = vi.substr(pos + 1);
218  pos = i_n.find("-");
219  int i = stoi(i_n.substr(0, pos));
220  int n = stoi(i_n.substr(pos + 1));
221  LOG(debug) << "dds-i-n: adding " << chanName << " -> i: " << i << " n: " << n;
222  fIofN.insert(make_pair(chanName, IofN(i, n)));
223  }
224  } catch (const exception& e) {
225  LOG(error) << "Error filling channel containers: " << e.what();
226  }

229 auto DDS::SubscribeForConnectingChannels() -> void
230 {
231  fDDSKeyValue.subscribe([&] (const string& propertyId, const string& key, const string& value)
232  {
233  try
234  {
235  LOG(debug) << "Received update for " << propertyId << ": key=" << key << " value=" << value;
236  string val = value;
237  // check if it is to handle as one out of multiple values
238  auto it = fIofN.find(propertyId);
239  if (it != fIofN.end()) {
240  it->second.fEntries.push_back(value);
241  if (it->second.fEntries.size() == it->second.fN) {
242  sort(it->second.fEntries.begin(), it->second.fEntries.end());
243  val = it->second.fEntries.at(it->second.fI);
244  } else {
245  LOG(debug) << "received " << it->second.fEntries.size() << " values for " << propertyId << ", expecting total of " << it->second.fN;
246  return;
247  }
248  }
249 
250  vector<string> connectionStrings;
251  boost::algorithm::split(connectionStrings, val, boost::algorithm::is_any_of(","));
252  if (connectionStrings.size() > 1) { // multiple bound channels received
253  auto it2 = fI.find(propertyId);
254  if (it2 != fI.end()) {
255  LOG(debug) << "adding connecting channel " << propertyId << " : " << connectionStrings.at(it2->second);
256  fConnectingChans.at(propertyId).fDDSValues.insert(make_pair<string, string>(key.c_str(), connectionStrings.at(it2->second).c_str()));
257  } else {
258  LOG(error) << "multiple bound channels received, but no task index specified, only assigning the first";
259  fConnectingChans.at(propertyId).fDDSValues.insert(make_pair<string, string>(key.c_str(), connectionStrings.at(0).c_str()));
260  }
261  } else { // only one bound channel received
262  fConnectingChans.at(propertyId).fDDSValues.insert(make_pair<string, string>(key.c_str(), val.c_str()));
263  }
264 
265  // update channels and remove them from unfinished container
266  for (auto mi = fConnectingChans.begin(); mi != fConnectingChans.end(); /* no increment */)
267  {
268  if (mi->second.fSubChannelAddresses.size() == mi->second.fDDSValues.size())
269  {
270  // when multiple subChannels are used, their order on every device should be the same, irregardless of arrival order from DDS.
271  sort(mi->second.fSubChannelAddresses.begin(), mi->second.fSubChannelAddresses.end());
272  auto it3 = mi->second.fDDSValues.begin();
273  for (unsigned int i = 0; i < mi->second.fSubChannelAddresses.size(); ++i)
274  {
275  SetProperty<string>(string{"chans." + mi->first + "." + to_string(i) + ".address"}, it3->second);
276  ++it3;
277  }
278  fConnectingChans.erase(mi++);
279  }
280  else
281  {
282  ++mi;
283  }
284  }
285  }
286  catch (const exception& e)
287  {
288  LOG(error) << "Error on handling DDS property update for " << propertyId << ": key=" << key << " value=" << value << ": " << e.what();
289  }
290  });
291 }

323 auto DDS::SubscribeForCustomCommands() -> void
324 {
325  string id = GetProperty<string>("id");
326  string pid(to_string(getpid()));
327 
328  fDDSCustomCmd.subscribe([id, pid, this](const string& cmd, const string& cond, uint64_t senderId)
329  {
330  LOG(info) << "Received command: " << cmd;
331 
332  if (cmd == "check-state")
333  {
334  fDDSCustomCmd.send(id + ": " + ToStr(GetCurrentDeviceState()) + " (pid: " + pid + ")", to_string(senderId));
335  }
336  else if (fCommands.find(cmd) != fCommands.end())
337  {
338  fDDSCustomCmd.send(id + ": attempting to " + cmd, to_string(senderId));
339  ChangeDeviceState(ToDeviceStateTransition(cmd));
340  }
341  else if (cmd == "END")
342  {
343  fDDSCustomCmd.send(id + ": attempting to " + cmd, to_string(senderId));
344  ChangeDeviceState(ToDeviceStateTransition(cmd));
345  fDDSCustomCmd.send(id + ": " + ToStr(GetCurrentDeviceState()), to_string(senderId));
346  if (ToStr(GetCurrentDeviceState()) == "EXITING")
347  {
348  unique_lock<mutex> lock(fStopMutex);
349  fStopCondition.notify_one();
350  }
351  }
352  else if (cmd == "dump-config")
353  {
354  stringstream ss;
355  for (const auto pKey: GetPropertyKeys())
356  {
357  ss << id << ": " << pKey << " -> " << GetPropertyAsString(pKey) << endl;
358  }
359  fDDSCustomCmd.send(ss.str(), to_string(senderId));
360  }
361  else if (cmd == "subscribe-to-heartbeats")
362  {
363  {
364  // auto size = fHeartbeatSubscribers.size();
365  lock_guard<mutex> lock{fHeartbeatSubscriberMutex};
366  fHeartbeatSubscribers.insert(senderId);
367  }
368  fDDSCustomCmd.send("heartbeat-subscription: " + id + ",OK", to_string(senderId));
369  }
370  else if (cmd == "unsubscribe-from-heartbeats")
371  {
372  {
373  lock_guard<mutex> lock{fHeartbeatSubscriberMutex};
374  fHeartbeatSubscribers.erase(senderId);
375  }
376  fDDSCustomCmd.send("heartbeat-unsubscription: " + id + ",OK", to_string(senderId));
377  }
378  else if (cmd == "subscribe-to-state-changes")
379  {
380  {
381  // auto size = fStateChangeSubscribers.size();
382  lock_guard<mutex> lock{fStateChangeSubscriberMutex};
383  fStateChangeSubscribers.insert(senderId);
384  }
385  fDDSCustomCmd.send("state-changes-subscription: " + id + ",OK", to_string(senderId));
386  auto state = GetCurrentDeviceState();
387  LOG(debug) << "Publishing state-change: " << state << " to " << senderId;
388  fDDSCustomCmd.send("state-change: " + id + "," + ToStr(state), to_string(senderId));
389  }
390  else if (cmd == "unsubscribe-from-state-changes")
391  {
392  {
393  lock_guard<mutex> lock{fStateChangeSubscriberMutex};
394  fStateChangeSubscribers.erase(senderId);
395  }
396  fDDSCustomCmd.send("state-changes-unsubscription: " + id + ",OK", to_string(senderId));
397  }
398  else
399  {
400  LOG(warn) << "Unknown command: " << cmd;
401  LOG(warn) << "Origin: " << senderId;
402  LOG(warn) << "Destination: " << cond;
403  }
404  });
405 }
