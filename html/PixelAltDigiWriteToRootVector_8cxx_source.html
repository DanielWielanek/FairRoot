140 // ----- Private method Init -------------------------------------------
141 InitStatus PixelAltDigiWriteToRootVector::Init() {
142 
143  // Get input array
144  FairRootManager* ioman = FairRootManager::Instance();
145 
146  if ( ! ioman ) LOG(fatal) << "No FairRootManager";
147  fDigis = static_cast<TClonesArray*>(ioman->GetObject("PixelDigis"));
148 
149  if ( !fDigis )
150  LOG(warn) << "PixelAltDigiWriteToRootVector::Init() No input PixelDigis array!";
151 
152  LOG(info) << "-I- " << fName.Data() << "::Init(). Initialization succesfull.";
153 
154  fRunId = ioman->GetRunId();
155 
156  if ( fDivideLevel == 0 ) {
157  fNofOutputFiles = 1;
158  }
159  else {
160  if ( fDivideLevel == 1 ) {
161  fNofOutputFiles = 3; // 1 file per station (3 stations)
162  }
163  else if ( fDivideLevel == 2 ) {
164  fNofOutputFiles = 12; // 1 file per sensor (3 stations times 4 sensors)
165  }
166  else {
167  LOG(fatal) << "PixelAltDigiWriteToRootVector::Init(), fDivideLevel = " << fDivideLevel << " unknown, it has to be in the range <0,2>";
168  return kFATAL;
169  }
170  }
171 
172  fPixelEventHeader = new PixelPayload::EventHeader();
173  for ( Int_t ifile = 0 ; ifile < fNofOutputFiles ; ifile++ ) {
174  TString fileName = fOutputFileName;
175  TString uniqFile = (fDivideLevel==0?".":(fDivideLevel==1?Form(".s%d.",ifile):Form(".c%d.",ifile)));
176  fileName.Replace(fileName.Last('.'),1,uniqFile.Data());
177 
178  fOutputRootFile[ifile] = new TFile(fileName.Data(),"RECREATE");
179  fOutputRootTree[ifile] = new TTree("fairdata","example MQ 9 Pixel data");
180  LOG(debug) << "trying to create branch in file >>" << fOutputRootFile[ifile]->GetName() << "<<";
181  fOutputRootTree[ifile]->Branch("EventHeader.","PixelPayload::EventHeader",&fPixelEventHeader,0);
182  fOutputRootTree[ifile]->Branch("DigiVector.",&fPixelDigiVector,32000,0);
183  LOG(debug) << "branches created, fPixelDigiVector = >>" << &fPixelDigiVector << "<<";
184  }
185 
186  return kSUCCESS;
187 
188 }
