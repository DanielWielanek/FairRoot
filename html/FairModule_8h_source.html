158 void FairModule::ConstructASCIIGeometry(T dataType1, TString containerName, U)
159 {
160  FairGeoLoader* loader=FairGeoLoader::Instance();
161  FairGeoInterface* GeoInterface =loader->getGeoInterface();
162  T* MGeo=new T();
163  MGeo->print();
164  MGeo->setGeomFile(GetGeometryFileName());
165  GeoInterface->addGeoModule(MGeo);
166  Bool_t rc = GeoInterface->readSet(MGeo);
167  if ( rc ) { MGeo->create(loader->getGeoBuilder()); }
168 
169  TList* volList = MGeo->getListOfVolumes();
170  // store geo parameter
171  FairRun* fRun = FairRun::Instance();
172  FairRuntimeDb* rtdb= FairRun::Instance()->GetRuntimeDb();
173 
174  dataType1 = *MGeo;
175 
176  if ( "" != containerName) {
177  LOG(info) << "Add GeoNodes for "<< MGeo->getDescription()
178  << " to container " << containerName;
179 
180  // U par=(U)(rtdb->getContainer(containerName));
181  U* par=static_cast<U*>(rtdb->getContainer(containerName));
182  TObjArray* fSensNodes = par->GetGeoSensitiveNodes();
183  TObjArray* fPassNodes = par->GetGeoPassiveNodes();
184 
185  TListIter iter(volList);
186  FairGeoNode* node = NULL;
187  FairGeoVolume* aVol=NULL;
188 
189  while( (node = static_cast<FairGeoNode*>(iter.Next())) ) {
190  aVol = dynamic_cast<FairGeoVolume*> ( node );
191  if ( node->isSensitive() ) {
192  fSensNodes->AddLast( aVol );
193  } else {
194  fPassNodes->AddLast( aVol );
195  }
196  }
197  ProcessNodes( volList );
198  par->setChanged();
199  par->setInputVersion(fRun->GetRunId(),1);
200 
201  }
202 }
