66 FairMQPollerNN::FairMQPollerNN(const unordered_map<string, vector<FairMQChannel>>& channelsMap, const vector<string>& channelList)
67  : fItems()
68  , fNumItems(0)
69  , fOffsetMap()
70 {
71  int offset = 0;
72 
73  try
74  {
75  // calculate offsets and the total size of the poll item set
76  for (string channel : channelList)
77  {
78  fOffsetMap[channel] = offset;
79  offset += channelsMap.at(channel).size();
80  fNumItems += channelsMap.at(channel).size();
81  }
82 
83  fItems = new nn_pollfd[fNumItems];
84 
85  int index = 0;
86  for (string channel : channelList)
87  {
88  for (unsigned int i = 0; i < channelsMap.at(channel).size(); ++i)
89  {
90  index = fOffsetMap[channel] + i;
91  fItems[index].fd = channelsMap.at(channel).at(i).GetSocket().GetSocket(1);
92 
93  int type = 0;
94  size_t sz = sizeof(type);
95  nn_getsockopt(channelsMap.at(channel).at(i).GetSocket().GetSocket(1), NN_SOL_SOCKET, NN_PROTOCOL, &type, &sz);
96 
97  SetItemEvents(fItems[index], type);
98  }
99  }
100  }
101  catch (const std::out_of_range& oor)
102  {
103  LOG(error) << "at least one of the provided channel keys for poller initialization is invalid";
104  LOG(error) << "out of range error: " << oor.what() << '\n';
105  exit(EXIT_FAILURE);
106  }
107 }
