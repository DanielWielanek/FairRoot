92 {

106  if ( parts.Size() > 0 ) {
107  Send(parts, fOutputChannelName);
108  }
109  return false;

112  fCurrentFile++;
113  }
114 
115  if ( !fInputFile.is_open() ) { // wrong file name
116  LOG(error) << "FairMQEx9aSamplerBin::ConditionalRun fInputFile \"" << fFileNames[fCurrentFile] << "\" could not be open!";
117  if ( parts.Size() > 0 ) {
118  Send(parts, fOutputChannelName);
119  }
120  return false;
121  }
122 
123  std::string buffer;
124 
125  int head[4]; // runId, MCEntryNo, PartNo, NofDigis
126  fInputFile.read((char*)head,sizeof(head));
127 
128  if ( fInputFile.eof() ) {
129  LOG(info) << "End of file reached!";
130  fInputFile.close();
131  if ( fCurrentFile == (int)fFileNames.size() ) { // this is the last file
132  if ( parts.Size() > 0 ) {
133  Send(parts, fOutputChannelName);
134  }
135  return false;
136  }
137  else
138  return true;
139  }
140 
141  int dataSize = 4; // detId, feId, col, row

143  short int dataCont[constNofData];
144  fInputFile.read((char*)dataCont,sizeof(dataCont));
145

153  ));

167  digiPayload[idigi].fCol = (int)dataCont[idigi*dataSize+2];
168  digiPayload[idigi].fRow = (int)dataCont[idigi*dataSize+3];
169  }
170  parts.AddPart(std::move(msgDigis));
171 
172  fEventCounter++;
173  }
174 
175  Send(parts, fOutputChannelName);
176
