14 // FairGeoTransform
15 //
16 // Class to hold the orientation (rotation matrix) and the position
17 // (translation vector) of a coordinate system (system 2) relative to a
18 // reference coordinate system (system 1)
19 // It provides member functions to transform a vector or a point and an other
20 // coordinate system from its own coordinate system into the reference
21 // coordinate system and vice versa.
22 // Instances of this class can e.g. hold the lab or detector transformation of
23 // a geometry volume (see class FairGeoVolume)
24 //

72 FairGeoVector FairGeoTransform::transFrom(const FairGeoVector& p) const
73 {
74  // Transforms a vector (point) given in its own coordinate
75  // system (2) into the reference coordinate system (1)
76  // e.g. v2 is a vector (point) in the detector coordinate system;
77  // it can be transformed to a vector v2 the lab system with
78  // FairGeoVector v2=mo.transFrom(v1)
79  // where mo is the coordinate system of the mother
80  return rot*p+trans;
81 }

83 FairGeoVector FairGeoTransform::transTo(const FairGeoVector& p) const
84 {
85  // Transforms a vector (point) given in the reference system (1)
86  // into the local coordinate system (2)
87  // e.g. v1 is a vector (point) in the lab system; it can be transformed to
88  // a vector v2 the detector coordinate system with
89  // FairGeoVector v2=mo.transTo(v1)
90  // where mo is the coordinate system of the mother
91  return rot.inverse()*(p-trans);
92 }

94 void FairGeoTransform::transTo(const FairGeoTransform& s)
95 {
96  // Transforms the coordinate system into the coordinate system
97  // described by s. Both transformations must have the same reference
98  // system e.g. the lab system
99  // This function is e.g. used to transform a daughter coordinate system
100  // with a transformation relative to the lab into the detector coordinate
101  // system.
102  const FairGeoRotation& rm=s.getRotMatrix();
103  FairGeoRotation rt(rm.inverse());
104  if (rm.diff2(rot)<0.000001) { rot.setUnitMatrix(); }
105  else { rot.transform(rt); }
106  trans-=s.getTransVector();
107  trans=rt*trans;
108  // trans.round(3); // rounds to 3 digits (precision 1 micrometer)
109 }

111 void FairGeoTransform::transFrom(const FairGeoTransform& s)
112 {
113  // Transforms the coordinate system described by s into the local
114  // coordinate system
115  // This function is e.g. used to transform a daughter coordinate system
116  // with a transformation relative to its mother into the lab system.
117  // e.g. daughterDetTransform.transFrom(motherLabTransform)
118  const FairGeoRotation& r=s.getRotMatrix();
119  rot.transform(r);
120  trans=r*trans;
121  trans+=s.getTransVector();
122 }
