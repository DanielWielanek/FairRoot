108  void Run()
109  {
110  const FairMQChannel& inputChannel = fChannels.at(fInputChannelName).at(0);
111 
112  while (CheckCurrentState(RUNNING))
113  {
114  FairMQMessagePtr msgSize(NewMessage());
115  FairMQMessagePtr msg(NewMessage());
116 
117  if (inputChannel.Receive(msgSize) >= 0)
118  {
119  if (inputChannel.Receive(msg) >= 0)
120  {
121  int dataSize = *(static_cast<int*>(msgSize->GetData()));
122  int* subEvtPtr = static_cast<int*>(msg->GetData());
123 
124  // LOG(debug) << "array size = " << dataSize;
125  // if (dataSize > 0)
126  // {
127  // LOG(debug) << "first element in array = " << *subEvtPtr;
128  // }
129 
130  fUnpacker->DoUnpack(subEvtPtr, dataSize);
131  Serialize<SerializationType>(*msg, fUnpacker->GetOutputData());
132  Send(msg, "data-out");
133  fUnpacker->Reset();
134  }
135  }
