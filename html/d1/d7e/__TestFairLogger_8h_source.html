160  bool CheckVerboseOutput(std::string logLevel,
161  std::string outputString,
162  std::string verbosity,
163  std::string resultString)
164  {
165  // Example Output for logging level INFO and different verbosity levels
166  // LOW :[INFO] I am here.
167  // MEDIUM :[19:07:13][INFO] I am here.
168  // HIGH :[process_name][19:07:13:123456][INFO] I am here.
169  // VERYHIGH :[process_name][19:07:13:123456][INFO][_TestFairTools.cxx:78:LogNoArguments] I am here.
170  std::string regexString;
171  if (verbosity == "HIGH" || verbosity == "VERYHIGH") {
172  regexString += "\\[.*]"; // process_name
173  regexString += "\\[\\d{2}:\\d{2}:\\d{2}.\\d{6}]"; // timestamp with ns
174  } else if (verbosity == "MEDIUM") {
175  regexString += "\\[\\d{2}:\\d{2}:\\d{2}]"; // timestamp without ns
176  }
177  regexString += "\\[" + logLevel + "]"; // severity
178 
179  if (verbosity == "VERYHIGH") {
180  regexString += "\\[.*:\\d{2}:.*]"; // file:line:function
181  }
182 
183  regexString += "[ ]" + outputString + ".*";
184 
185  const std::regex e(regexString);
186  return std::regex_match(resultString, e);
187  }
