404 void FairRootManager::AddMemoryBranchAny(const char* brname, T** obj) {
405  if (fAnyBranchMap.find(brname) == fAnyBranchMap.end()) {
406  auto& ot = typeid(T*);
407  auto& pt = typeid(T);
408  fAnyBranchMap[brname]=std::unique_ptr<TypeAddressPair const> (new TypeAddressPair(ot, pt,(void*)obj));
409  }
410 }

412 // try to retrieve an object address from the registered branches/names
413 template<typename T>
414 T FairRootManager::GetMemoryBranchAny(const char* brname) const {
415  static_assert(std::is_pointer<T>::value, "Return type of GetMemoryBranchAny has to be a pointer");
416  using P = typename std::remove_pointer<T>::type;
417  auto iter = fAnyBranchMap.find(brname);
418  if(iter != fAnyBranchMap.end()) {
419  // verify type consistency
420  if(typeid(P).hash_code() != iter->second->origtypeinfo.hash_code()) {
421  EmitMemoryBranchWrongTypeWarning(brname, typeid(P).name(), iter->second->origtypeinfo.name());
422  return nullptr;
423  }
424  return static_cast<T>(iter->second->ptraddr);
425  }
426  return nullptr;
427 }

439 // this function serves as a factory (or lookup) for memory managed
440 // instances associated to branches
441 // it returns a pointer to unmodifiable instance of T
442 template<typename TPtr>
443 TPtr FairRootManager::InitObjectAs(const char* brname) {
444  static_assert(std::is_pointer<TPtr>::value, "Return type of GetObjectAs has to be a pointer");
445  using X = typename std::remove_pointer<TPtr>::type;
446  static_assert(std::is_const<X>::value, "Return type of GetObjectAs has to be pointer to const class");
447  using T = typename std::remove_const<X>::type;
448 
449  // is there already an object associated to the branch in memory??
450  // then just return
451  T** obj = GetMemoryBranchAny<T**>(brname);
452  // obj is some address/instance holding TPtr instances
453  if(obj!=nullptr) return *obj;
454 
455  // it does not seem to be the case, let us create the pointer which will be initialized
456  // with the data (pointer to T)
457  T** addr = new T*;
458  // init the pointee to a default obj which we can return
459  (*addr) = new T;
460  // try to find and activate in the source
461  auto succeeded = fSource->ActivateObjectAny((void**)addr, typeid(T), brname);
462 
463  if(!succeeded) {
464  delete (*addr);
465  delete addr;
466  return nullptr;
467  }
468  // add into branch list
469  AddMemoryBranchAny<T>(brname, addr);
470  // register as a **requested** branch
471  // (duplications are explicitely allowed)
472  fReqBrNames.emplace_back(brname);
473 
474  // NOTE: ideally we would do proper resource management for addr and *addr
475  // since the FairRootManager becomes owner of these pointers/instances; Unfortunately this
476  // is quite a difficult task since we would have to store something like std::unique_ptr<T> in a member
477  // container which we cannot know a priori; Some solutions we could think of in the future are
478  // a) use the Destructor mechanism of ROOT::TClass since we still have the type info.
479  // b) investigate if boost::any could be of help here
480  // In any case, this problem is not very critical in the sense that FairRootManager is a singleton and hence
481  // cannot really leak memory (Assuming that the destructors of T are not doing something non-trivial).
482  return *addr;
483 }
