41 // DoUnpack: Public method
42 Bool_t FairMBSUnpacker::DoUnpack(Int_t* data, Int_t size)
43 {
44  LOG(debug) << "FairMBSUnpacker: Unpacking... size = " << size;
45 
46  Int_t l_i = 0;
47 
48  Int_t n17 = 0;
49 
50  while (l_i < size)
51  {
52  n17 = 0;
53 
54  UInt_t* p1 = reinterpret_cast<UInt_t*>(data + l_i);
55  UInt_t l_sam_id = (p1[0] & 0xf0000000) >> 28; // identifies the sam
56  UInt_t l_gtb_id = (p1[0] & 0x0f000000) >> 24; // 0 or 1, identifies which of the 2 cables of the sam
57  UInt_t l_lec = (p1[0] & 0x00f00000) >> 20;
58  UInt_t l_da_siz = (p1[0] & 0x000001ff);
59 
60  LOG(debug) << "FairMBSUnpacker: SAM: " << l_sam_id << ", GTB: " << l_gtb_id << ", lec: " << l_lec << ", size: " << l_da_siz;
61 
62  l_i += 1;
63 
64  p1 = reinterpret_cast<UInt_t*>(data + l_i);
65 
66  for (UInt_t i1 = 0; i1 < l_da_siz; i1 += 2)
67  {
68  UInt_t tac_addr;
69  UInt_t tac_ch;
70  UInt_t cal;
71  UInt_t clock;
72  UInt_t tac_data;
73  UInt_t qdc_data;
74  tac_addr = (p1[i1] & 0xf8000000) >> 27;
75  tac_ch = (p1[i1] & 0x07c00000) >> 22;
76  cal = (p1[i1] & 0x003C0000) >> 18;
77  clock = 63 - ((p1[i1] & 0x0003f000) >> 12);
78  tac_data = 4095 - ((p1[i1] & 0x00000fff));
79  qdc_data = (p1[i1 + 1] & 0x00000fff);
80  l_i += 2;
81 
82  if (16 == tac_ch)
83  {
84  n17 += 1;
85  }
86  LOG(debug) << "FairMBSUnpacker: TAC ADDR: " << tac_addr << ", TAC CH: " << tac_ch << ", TAC Data: " << tac_data << ", QDC Data: " << qdc_data;
87  new ((*fRawData)[fNHits]) FairMBSRawItem(l_sam_id, l_gtb_id, tac_addr, tac_ch, cal, clock, tac_data, qdc_data);
88  fNHits++;
89  }
90 
91  LOG(debug) << "FairMBSUnpacker: n17=" << n17;
92  }
93 
94  LOG(debug) << "FairMBSUnpacker: Number of hits in LAND: " << fNHits;
95 
96  fNHitsTotal += fNHits;
97 
98  return kTRUE;
99 }
