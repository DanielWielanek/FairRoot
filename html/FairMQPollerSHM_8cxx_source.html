66 FairMQPollerSHM::FairMQPollerSHM(const unordered_map<string, vector<FairMQChannel>>& channelsMap, const vector<string>& channelList)
67  : fItems()
68  , fNumItems(0)
69  , fOffsetMap()
70 {
71  int offset = 0;
72 
73  try
74  {
75  // calculate offsets and the total size of the poll item set
76  for (string channel : channelList)
77  {
78  fOffsetMap[channel] = offset;
79  offset += channelsMap.at(channel).size();
80  fNumItems += channelsMap.at(channel).size();
81  }
82 
83  fItems = new zmq_pollitem_t[fNumItems];
84 
85  int index = 0;
86  for (string channel : channelList)
87  {
88  for (unsigned int i = 0; i < channelsMap.at(channel).size(); ++i)
89  {
90  index = fOffsetMap[channel] + i;
91 
92  fItems[index].socket = channelsMap.at(channel).at(i).GetSocket().GetSocket();
93  fItems[index].fd = 0;
94  fItems[index].revents = 0;
95 
96  int type = 0;
97  size_t size = sizeof(type);
98  zmq_getsockopt(channelsMap.at(channel).at(i).GetSocket().GetSocket(), ZMQ_TYPE, &type, &size);
99 
100  SetItemEvents(fItems[index], type);
101  }
102  }
103  }
104  catch (const std::out_of_range& oor)
105  {
106  LOG(error) << "at least one of the provided channel keys for poller initialization is invalid";
107  LOG(error) << "out of range error: " << oor.what() << '\n';
108  exit(EXIT_FAILURE);
109  }
110 }

135 void FairMQPollerSHM::SetItemEvents(zmq_pollitem_t& item, const int type)
136 {
137  if (type == ZMQ_REQ || type == ZMQ_REP || type == ZMQ_PAIR || type == ZMQ_DEALER || type == ZMQ_ROUTER)
138  {
139  item.events = ZMQ_POLLIN|ZMQ_POLLOUT;
140  }
141  else if (type == ZMQ_PUSH || type == ZMQ_PUB || type == ZMQ_XPUB)
142  {
143  item.events = ZMQ_POLLOUT;
144  }
145  else if (type == ZMQ_PULL || type == ZMQ_SUB || type == ZMQ_XSUB)
146  {
147  item.events = ZMQ_POLLIN;
148  }
149  else
150  {
151  LOG(error) << "invalid poller configuration, exiting.";
152  exit(EXIT_FAILURE);
153  }
