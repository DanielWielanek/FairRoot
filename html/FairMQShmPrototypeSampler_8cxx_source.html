189 void FairMQShmPrototypeSampler::Log(const int intervalInMs)
190 {

192  chrono::time_point<chrono::high_resolution_clock> t1;

214 
215  t0 = t1;
216  this_thread::sleep_for(chrono::milliseconds(intervalInMs));
