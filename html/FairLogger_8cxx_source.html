147 void FairLogger::Log(fair::Severity severity, const char* file, const char* line, const char* func, const char* format, va_list arglist)
148 {
149  // If the format string together with the argument list was used once it can't
150  // be used another time. Don't know why but if we do so the arguments are not
151  // correct any longer and the program may crash (On Mac OSX). Even if it does
152  // not crash it does not work as expected.
153  // To vercome the problem the output is written to a buffer which then can be
154  // used several times.
155 
156  while (true)
157  {
158  fBufferSizeNeeded = vsnprintf(fBufferPointer, fBufferSize, format, arglist);
159 
160  // Try to vsnprintf into our buffer.
161  // int needed = vsnprintf (buf, size, fmt, ap);
162  // NB. C99 (which modern Linux and OS X follow) says vsnprintf
163  // failure returns the length it would have needed. But older
164  // glibc and current Windows return -1 for failure, i.e., not
165  // telling us how much was needed.
166 
167  if (fBufferSizeNeeded <= static_cast<int>(fBufferSize) && fBufferSizeNeeded >= 0) {
168  // It fits fine so we're done.
169  LOGD(severity, file, line, func) << std::string(fBufferPointer, static_cast<size_t>(fBufferSizeNeeded));
170  break;
171  }
172 
173  // vsnprintf reported that it wanted to write more characters
174  // than we allotted. So try again using a dynamic buffer. This
175  // doesn't happen very often if we chose our initial size well.
176  fBufferSize = (fBufferSizeNeeded > 0) ? (fBufferSizeNeeded + 1) : (fBufferSize * 2);
177  fDynamicBuffer.resize(fBufferSize);
178  }
179 }

186 void FairLogger::LogFatalMessage()
187 {
188  // Since Fatal indicates a fatal error it is maybe usefull to have
189  // system information from the incident. Since the output on the screen
190  // does not help the noraml user, the stderr is redirected into a special
191  // core dump file for later usage.
192  // Fatal also indicates a problem which is so severe that the process should
193  // not go on, so the process is aborted.
194 
195  if (gSystem)
196  {
197  TString corefile = "core_dump_";
198  int pid = gSystem->GetPid();
199  corefile += pid;
200 
201  std::cerr << "For later analysis we write a core dump to " << corefile << std::endl;
202 
203  if (freopen(corefile, "w", stderr))
204  {
205  gSystem->StackTrace();
206  fclose(stderr);
207  }
208  }
209 
210  throw fair::FatalException("Fatal error occured. Exiting...");
211 }
