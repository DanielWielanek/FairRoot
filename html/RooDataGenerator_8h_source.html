174  z = new RooRealVar("z", "z", fOpt.z.min, fOpt.z.max) ;

176  tErr = new RooRealVar("tErr", "tErr", fOpt.tErr.min, fOpt.tErr.max);
177 
178  mean_t = new RooRealVar("mu_t", "mean of t-distribution", (tmin + tmax) / 2) ;
179  sigma_t = new RooRealVar("sigma_t", "width of t-distribution", 0.1) ;
180 
181  Gauss_x = new RooGaussian("Gauss_x", "gaussian PDF", *x, RooConst(fOpt.x.mean), RooConst(fOpt.x.sigma));
182  Gauss_y = new RooGaussian("Gauss_y", "gaussian PDF", *y, RooConst(fOpt.y.mean), RooConst(fOpt.y.sigma));
183  Gauss_z = new RooGaussian("Gauss_z", "gaussian PDF", *z, RooConst(fOpt.z.mean), RooConst(fOpt.z.sigma));
184  Gauss_t = new RooGaussian("Gauss_t", "gaussian PDF", *t, *mean_t, *sigma_t);
185  Gauss_tErr = new RooGaussian("Gauss_tErr", "gaussian PDF", *tErr, RooConst((tErr->getMin() + tErr->getMax()) / 2), RooConst(fOpt.tErr.sigma));
186 
187  fModel = new RooProdPdf("Gauss_xyzt_ter", "Gauss_x*Gauss_y*Gauss_z*Gauss_t*Gauss_tErr", RooArgList(*Gauss_x, *Gauss_y, *Gauss_z, *Gauss_t, *Gauss_tErr));
188  }
