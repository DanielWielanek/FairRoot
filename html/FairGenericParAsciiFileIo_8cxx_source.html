73 const UChar_t *FairGenericParAsciiFileIo::readData(type t, const Char_t *format,
74  TString &line,
75  Int_t &length) {
76  // reads c-type single data and arrays
77  const Int_t st = sizeof(t);
78  const Int_t maxbuf = 8000;
79  const Int_t bufSizeExt = 10000 * st;
80  const Char_t d[] = " \t";
81  Text_t buf[maxbuf];
82  TString s;
83  Int_t l = 0, bufSize = bufSizeExt;
84  UChar_t *val = new UChar_t[bufSize];
85  Ssiz_t m = line.Last('\\');
86  if (m < 0) {
87  sscanf(line.Data(), format, &t);
88  memcpy(&val[l], &t, st);
89  length = st;
90  } else {
91  do {
92  pFile->getline(buf, maxbuf);
93  if (buf[0] != '/' && buf[0] != '#') {
94  s = buf;
95  m = s.Last('\\');
96  if (m > 0 && m < s.Length()) {
97  s = s(0, m - 1);
98  }
99  if ((bufSize - 1000) < l) {
100  bufSize += bufSizeExt;
101  UChar_t *va = new UChar_t[bufSize];
102  memcpy(va, val, l);
103  delete[] val;
104  val = va;
105  }
106  Char_t *ss = strtok(const_cast<Char_t*>(s.Data()), d);
107  while (ss != 0) {
108  sscanf(ss, format, &t);
109  memcpy(&val[l], &t, st);
110  l += st;
111  ss = strtok(NULL, d);
112  }
113  }
114  } while (buf[0] != '#' && !pFile->eof() && m > 0);
115  length = l;
116  }
117  return val;
118 }

136 Bool_t FairGenericParAsciiFileIo::readGenericSet(FairParGenericSet *pPar) {
137  // reads condition-stype parameter containers from ASCII file
138  if (!pFile) {
139  return kFALSE;
140  }
141  pFile->clear();
142  pFile->seekg(0, ios::beg);
143  const Text_t *name = (pPar->GetName());
144 
145  if (!findContainer(name)) {
146  return kFALSE;
147  }
148 
149 
150  const Char_t d[] = " \t";
151  FairParamList *paramList = new FairParamList;
152  const Int_t maxbuf = 8000;
153  Text_t buf[maxbuf];
154  buf[0] = '\0';
155  TString s, pName, pVal, pType;
156  Ssiz_t n, m;
157  while (buf[0] != '#' && !pFile->eof()) {
158  pFile->getline(buf, maxbuf);
159  if (buf[0] != '/' && buf[0] != '#') {
160  s = buf;
161  n = s.First(':');
162  // <DB> Check if empty buffer
163  if (s.IsNull()) continue;
164  if (n == -1) {
165  Error("readCond(FairParGenericSet*)",
166  "%s:\n Missing backslash for parameter %s", name, pName.Data());
167  delete paramList;
168  return kFALSE;
169  }
170  pName = s(0, n);
171  s = s(n + 1, s.Length() - n - 1);
172  s = s.Strip(s.kLeading);
173  if (pName.CompareTo("author") == 0 ||
174  pName.CompareTo("description") == 0) {
175  m = s.Last('\\');
176  if (m <= 0) {
177  pVal = s;
178  } else {
179  pVal = s(0, m);
180  pVal += "\n";
181  while (m > 0) {
182  pFile->getline(buf, maxbuf);
183  if (buf[0] != '/') {
184  s = buf;
185  m = s.Last('\\');
186  if (m > 0) {
187  pVal += s(0, m);
188  pVal += "\n";
189  } else {
190  pVal += s;
191  }
192  }
193  }
194  }
195  if (pName.CompareTo("author") == 0) {
196  pPar->setAuthor(pVal.Data());
197  } else if (pName.CompareTo("description") == 0) {
198  pPar->setDescription(pVal.Data());
199  }
200  } else {
201  Char_t *ss = strtok(const_cast<Char_t*>(s.Data()), d);
202  pType = TString(ss);
203  Int_t mt = s.Index(pType);
204  s = s(mt + pType.Length() + 1, s.Length() - mt - pType.Length() - 1);
205  if (pType.CompareTo("Text_t") == 0) {
206  m = s.Last('\\');
207  if (m < 0) {
208  pVal = s;
209  } else {
210  pVal = "";
211  while (m >= 0) {
212  pFile->getline(buf, maxbuf);
213  if (buf[0] != '/') {
214  s = buf;
215  m = s.Last('\\');
216  if (m > 0) {
217  pVal += s(0, m);
218  pVal += "\n";
219  } else {
220  pVal += s;
221  }
222  }
223  }
224  }
225  pVal = pVal.Strip(pVal.kLeading);
226  if (pVal.Length() > 0) {
227  paramList->add(pName.Data(), pVal.Data());
228  }
229  } else {
230  const UChar_t *val = 0;
231  Int_t length = 0;
232  if (pType.CompareTo("Int_t") == 0) {
233  Int_t v = 0;
234  val = readData(v, "%i", s, length);
235  } else if (pType.CompareTo("Float_t") == 0) {
236  Float_t v = 0.F;
237  val = readData(v, "%f", s, length);
238  } else if (pType.CompareTo("Double_t") == 0) {
239  Double_t v = 0.;
240  val = readData(v, "%lf", s, length);
241  } else if (pType.CompareTo("Char_t") == 0) {
242  Char_t v = '0';
243  val = readData(v, "%c", s, length);
244  } else {
245  Error("readCond(FairParGenericSet*)",
246  "%s:\n Parameter %s with unsupported type %s", name,
247  pName.Data(), pType.Data());
248  delete paramList;
249  return kFALSE;
250  }
251  FairParamObj *obj = new FairParamObj(pName.Data());
252  obj->setParamType(pType.Data());
253  UChar_t *pValue = obj->setLength(length);
254  memcpy(pValue, val, length);
255  paramList->getList()->Add(obj);
256  delete[] val;
257  }
258  }
259  }
260  }
261  Bool_t allFound = pPar->getParams(paramList);
262  if (allFound) {
263  pPar->setInputVersion(1, inputNumber);
264  pPar->setChanged();
265  printf("%s initialized from Ascii file\n", name);
266  } else {
267  pPar->setInputVersion(-1, inputNumber);
268  }
269  delete paramList;
270  return allFound;
271 }

273 Int_t FairGenericParAsciiFileIo::writeGenericSet(FairParGenericSet *pPar) {
274  // writes condition-stype parameter containers to ASCII file
275  if (pFile) {
276  // Int_t version=1;
277  const Text_t *name = pPar->GetName();
278  const Text_t *context = pPar->getParamContext();
279  *pFile << sepLine;
280  *pFile << "# Class: " << name << "\n# Context: " << context << "\n";
281  *pFile << sepLine;
282  *pFile << "[" << name << "]\n";
283  writeComment(pPar);
284  *pFile << "//-----------------------------------------------------------"
285  "-----------------\n";
286  FairParamList *paramList = new FairParamList;
287  pPar->putParams(paramList);
288  TList *pList = paramList->getList();
289  TIter next(pList);
290  FairParamObj *po;
291  while ((po = static_cast<FairParamObj*>(next()))) {
292  const Char_t *pType = po->getParamType();
293  UChar_t *pValue = po->getParamValue();
294  if (po->isBasicType() && strcmp(pType, "UChar_t") != 0) {
295  if (strcmp(pType, "Text_t") == 0) {
296  TString val(reinterpret_cast<Char_t*>(pValue), po->getLength());
297  val.ReplaceAll("\n", " \\\n");
298  *pFile << po->GetName() << ": " << pType << " \\\n " << val.Data()
299  << std::endl;
300  } else {
301  Int_t nParams = po->getNumParams();
302  if (nParams == 1) {
303  *pFile << po->GetName() << ": " << pType << " ";
304  } else {
305  *pFile << po->GetName() << ": " << pType << " \\\n ";
306  }
307  if (strcmp(pType, "Char_t") == 0) {
308  writeData(reinterpret_cast<Char_t*>(pValue), nParams);
309  } else if (strcmp(pType, "Int_t") == 0) {
310  writeData(reinterpret_cast<Int_t*>(pValue), nParams);
311  } else if (strcmp(pType, "Float_t") == 0) {
312  writeData(reinterpret_cast<Float_t*>(pValue), nParams);
313  } else if (strcmp(pType, "Double_t") == 0) {
314  writeData(reinterpret_cast<Double_t*>(pValue), nParams);
315  }
316  }
317  } else {
318  Error("writeCond(FairParGenericSet*)",
319  "Type %s of parameter %s not supported by ASCII I/O", pType,
320  po->GetName());
321  // version=-1;
322  }
323  }
324  *pFile << sepLine;
325  delete paramList;
326  pPar->setChanged(kFALSE);
327  return 1;
328  }
329  Error("writeCond(FairParGenericSet*)", "Output is not writable");
330  return -1;
331 }
