107 {
108  LOG(info) << "FairYamlVMCConfig::SetupGeant4() called";
109 
110  if (!fYamlConfig["Geant4_UserGeometry"]) {
111  LOG(fatal) << "User geometry not provided";
112  }
113  if (!fYamlConfig["Geant4_PhysicsList"]) {
114  LOG(fatal) << "Physics list not provided";
115  }
116  if (!fYamlConfig["Geant4_SpecialProcess"]) {
117  LOG(fatal) << "Special processy not provided";
118  }
119  bool specialStacking = false;
120  if (fYamlConfig["Geant4_SpecialStacking"]) {
121  // LOG(info) << "Special stacking used";
122  specialStacking = fYamlConfig["Geant4_SpecialStacking"].as<bool>();
123  }
124  bool mtMode = FairRunSim::Instance()->IsMT();
125 
126  if (fYamlConfig["Geant4_Multithreaded"]) {
127  mtMode = fYamlConfig["Geant4_Multithreaded"].as<bool>();
128  // LOG(info) << "Setting Geant4 multithreaded to " << (mtMode?"true":"false");
129  }
130 
131  FairFastSimRunConfiguration* runConfiguration
132  = new FairFastSimRunConfiguration(fYamlConfig["Geant4_UserGeometry"] .as<std::string>(),
133  fYamlConfig["Geant4_PhysicsList"] .as<std::string>(),
134  fYamlConfig["Geant4_SpecialProcess"].as<std::string>(),
135  specialStacking,
136  mtMode);
137 
138  TGeant4* geant4 = new TGeant4("TGeant4", "The Geant4 Monte Carlo", runConfiguration);
139 
140  LOG(info) << geant4->GetName() << " MonteCarlo engine created!.";
141 }

165 TString FairYamlVMCConfig::ObtainYamlFileName(const char* mcEngine)
166 {
167  TString lUserConfig = FairRunSim::Instance()->GetUserConfig();
168 
169  TString work = getenv("VMCWORKDIR");
170  TString work_config=work+"/gconfig/";
171  work_config.ReplaceAll("//","/");
172 
173  TString config_dir= getenv("CONFIG_DIR");
174  config_dir.ReplaceAll("//","/");
175 
176  Bool_t AbsPath=kFALSE;
177 
178  TString configFileWithPath;
179 
180  TString configFile;
181  if (lUserConfig.IsNull()) {
182  if (strcmp(mcEngine,"TGeant4") == 0) {
183  configFile = "g4Config.yaml";
184  }
185  if (strcmp(mcEngine,"TGeant3") == 0) {
186  configFile = "g3Config.yaml";
187  }
188  lUserConfig = configFile;
189  } else {
190  if (lUserConfig.Contains("/")) { AbsPath=kTRUE; }
191  configFile = lUserConfig;
192  LOG(info) << "---------------User config is used: "
193  << configFile.Data();
194  }
195  if (!AbsPath && TString(gSystem->FindFile(config_dir.Data(),configFile)) != TString("")) {
196  LOG(info) << "---------------CONFIG_DIR is used: "
197  << config_dir.Data() << " to get \"" << configFile.Data() << "\"";
198  configFileWithPath=configFile;
199  } else {
200  if (AbsPath) { configFileWithPath = lUserConfig; }
201  else { configFileWithPath =work_config+lUserConfig; }
202  }
203  return configFileWithPath;
204 }
