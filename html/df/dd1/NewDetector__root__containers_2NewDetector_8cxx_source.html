99 Bool_t NewDetector::ProcessHits(FairVolume* vol)
100 {
103  //Set parameters at entrance of volume. Reset ELoss.
104  if ( TVirtualMC::GetMC()->IsTrackEntering() ) {
105  fELoss = 0.;
106  fTime = TVirtualMC::GetMC()->TrackTime() * 1.0e09;
107  fLength = TVirtualMC::GetMC()->TrackLength();
108  TVirtualMC::GetMC()->TrackPosition(fPos);
109  TVirtualMC::GetMC()->TrackMomentum(fMom);
110  }
111 
112  // Sum energy loss for all steps in the active volume
113  fELoss += TVirtualMC::GetMC()->Edep();
114 
115  // Create NewDetectorPoint at exit of active volume
116  if ( TVirtualMC::GetMC()->IsTrackExiting() ||
117  TVirtualMC::GetMC()->IsTrackStop() ||
118  TVirtualMC::GetMC()->IsTrackDisappeared() ) {
119  fTrackID = TVirtualMC::GetMC()->GetStack()->GetCurrentTrackNumber();
120  fVolumeID = vol->getMCid();
121  if (fELoss == 0. ) { return kFALSE; }
122  AddHit(fTrackID, fVolumeID, TVector3(fPos.X(), fPos.Y(), fPos.Z()),
123  TVector3(fMom.Px(), fMom.Py(), fMom.Pz()), fTime, fLength,
124  fELoss);
125 
126  // Increment number of NewDetector det points in TParticle
127  MyProjStack* stack = (MyProjStack*) TVirtualMC::GetMC()->GetStack();
128  stack->AddPoint(kNewDetector);
129  }
130 
131  return kTRUE;
132 }
