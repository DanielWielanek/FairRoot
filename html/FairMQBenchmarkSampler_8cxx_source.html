56 {
57  // store the channel reference to avoid traversing the map on every loop iteration
58  FairMQChannel& dataOutChannel = fChannels.at(fOutChannelName).at(0);
59 
60  FairMQMessagePtr baseMsg(dataOutChannel.Transport()->CreateMessage(fMsgSize));
61

63  auto tStart = chrono::high_resolution_clock::now();
64 
65  while (CheckCurrentState(RUNNING))
66  {
67  if (fSameMessage)
68  {
69  FairMQMessagePtr msg(dataOutChannel.Transport()->CreateMessage());
70  msg->Copy(*baseMsg);
71 
72  if (dataOutChannel.Send(msg) >= 0)
73  {
74  if (fMaxIterations > 0)
75  {
76  if (fNumIterations >= fMaxIterations)
77  {
78  break;
79  }
80  }
81  ++fNumIterations;
82  }
83  }
84  else
85  {
86  FairMQMessagePtr msg(dataOutChannel.Transport()->CreateMessage(fMsgSize));
87 
88  if (dataOutChannel.Send(msg) >= 0)
89  {

91  {

93  {
94  break;
95  }
96  }

98  }
99  }
100

102

106  }
107  }
108

110 
111  LOG(info) << "Done " << fNumIterations << " iterations in " << chrono::duration<double, milli>(tEnd - tStart).count() << "ms.";
112
