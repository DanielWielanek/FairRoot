86 bool FairMQPixelSamplerBin::ConditionalRun()
87 {
88  if ( fEventCounter == fMaxIndex ) return false;
89 
90  int readEventReturn = 0;
91  readEventReturn = fSource->ReadEvent(fEventCounter);
92 
93  if ( readEventReturn != 0 ) return false;
94 
95  FairMQParts parts;
96 
97  for ( int iobj = 0 ; iobj < fNObjects ; iobj++ ) {
98  if ( strcmp(fInputObjects[iobj]->GetName(),"EventHeader.") == 0 ) {
99  PixelPayload::EventHeader* header = new PixelPayload::EventHeader();
100  header->fRunId = ((FairEventHeader*)fInputObjects[iobj])->GetRunId();
101  header->fMCEntryNo = ((FairEventHeader*)fInputObjects[iobj])->GetMCEntryNumber();
102  header->fPartNo = 0;
103  FairMQMessagePtr msgHeader(NewMessage(header,
104  sizeof(PixelPayload::EventHeader),
105  [](void* data, void* /*hint*/) { delete static_cast<PixelPayload::EventHeader*>(data); }
106  ));
107  parts.AddPart(std::move(msgHeader));
108  // LOG(debug) << "-----------------------------";
109  // LOG(debug) << "first part has size = " << sizeof(PixelPayload::EventHeader);
110  }
111  else {
112  Int_t nofEntries = ((TClonesArray*)fInputObjects[iobj])->GetEntries();
113  size_t digisSize = nofEntries * sizeof(PixelPayload::Digi);
114 
115  FairMQMessagePtr msgTCA(NewMessage(digisSize));
116 
117  PixelPayload::Digi* digiPayload = static_cast<PixelPayload::Digi*>(msgTCA->GetData());
118 
119 
120  for ( int idigi = 0 ; idigi < nofEntries ; idigi++ ) {
121  PixelDigi* digi = static_cast<PixelDigi*>(((TClonesArray*)fInputObjects[iobj])->At(idigi));
122  if ( !digi ) {
123  continue;
124  }
125  new (&digiPayload[idigi]) PixelPayload::Digi();
126  digiPayload[idigi].fDetectorID = digi->GetDetectorID();
127  digiPayload[idigi].fFeID = digi->GetFeID();
128  digiPayload[idigi].fCharge = digi->GetCharge();
129  digiPayload[idigi].fCol = digi->GetCol();
130  digiPayload[idigi].fRow = digi->GetRow();
131  }
132  // LOG(debug) << "second part has size = " << digisSize;
133  parts.AddPart(std::move(msgTCA));
134  }
135  }
136 
137  // LOG(debug) << "sending data with " << parts.Size() << " parts";
138  Send(parts, fOutputChannelName);
139 
140  fEventCounter++;
141 
142  return true;
143 }
