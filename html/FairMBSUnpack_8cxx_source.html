57 // DoUnpack: Public method
58 Bool_t FairMBSUnpack::DoUnpack(Int_t* data, Int_t size)
59 {
60  LOG(debug) << "FairMBSUnpack : Unpacking... size = " << size;
61 
62  Int_t l_i = 0;
63 
64  Int_t n17 = 0;
65 
66  while (l_i < size)
67  {
68  n17 = 0;
69 
70  UInt_t* p1 = reinterpret_cast<UInt_t*>(data + l_i);
71  UInt_t l_sam_id = (p1[0] & 0xf0000000) >> 28; // identifies the sam
72  UInt_t l_gtb_id = (p1[0] & 0x0f000000) >> 24; // 0 or 1, identifies which of the 2 cables of the sam
73  UInt_t l_lec = (p1[0] & 0x00f00000) >> 20;
74  UInt_t l_da_siz = (p1[0] & 0x000001ff);
75 
76  LOG(debug) << "FairMBSUnpack : SAM:" << l_sam_id << ", GTB:" << l_gtb_id << ", lec:" << l_lec
77  << ", size:" << l_da_siz;
78 
79  l_i += 1;
80 
81  p1 = reinterpret_cast<UInt_t*>((data + l_i));
82 
83  for (UInt_t i1 = 0; i1 < l_da_siz; i1 += 2)
84  {
85  UInt_t tac_addr;
86  UInt_t tac_ch;
87  UInt_t cal;
88  UInt_t clock;
89  UInt_t tac_data;
90  UInt_t qdc_data;
91  tac_addr = (p1[i1] & 0xf8000000) >> 27;
92  tac_ch = (p1[i1] & 0x07c00000) >> 22;
93  cal = (p1[i1] & 0x003C0000) >> 18;
94  clock = 63 - ((p1[i1] & 0x0003f000) >> 12);
95  tac_data = 4095 - ((p1[i1] & 0x00000fff));
96  qdc_data = (p1[i1 + 1] & 0x00000fff);
97  l_i += 2;
98 
99  if (16 == tac_ch)
100  {
101  n17 += 1;
102  }
103  LOG(debug) << "FairMBSUnpack : TAC ADDR IS " << tac_addr << ", TAC CH IS " << tac_ch << ", TAC Data IS "
104  << tac_data << ", QDC Data IS " << qdc_data;
105  new ((*fRawData)[fNHits])
106  FairMBSRawItem(l_sam_id, l_gtb_id, tac_addr, tac_ch, cal, clock, tac_data, qdc_data);
107  fNHits++;
108  }
109 
110  LOG(debug) << "FairMBSUnpack : n17=" << n17;
111  }
112 
113  LOG(debug) << "FairMBSUnpack : Number of hits in LAND: " << fNHits;
114 
115  fNHitsTotal += fNHits;
116 
117  return kTRUE;
118 }
