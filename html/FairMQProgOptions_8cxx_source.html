53 int FairMQProgOptions::ParseAll(const int argc, char const* const* argv, bool allowUnregistered)
54 {
55  if (FairProgOptions::ParseCmdLine(argc, argv, allowUnregistered))
56  {
57  // ParseCmdLine returns 0 if no immediate switches found.
58  return 1;
59  }
60 
61  // if these options are provided, do no further checks and let the device handle them
62  if (fVarMap.count("print-channels") || fVarMap.count("version"))
63  {
64  fair::Logger::SetConsoleSeverity("nolog");
65  return 0;
66  }
67 
68  string severity = GetValue<string>("severity");
69  string logFile = GetValue<string>("log-to-file");
70  bool color = GetValue<bool>("color");
71 
72  string verbosity = GetValue<string>("verbosity");
73  fair::Logger::SetVerbosity(verbosity);
74 
75  if (logFile != "")
76  {
77  fair::Logger::InitFileSink(severity, logFile);
78  fair::Logger::SetConsoleSeverity("nolog");
79  }
80  else
81  {
82  fair::Logger::SetConsoleColor(color);
83  fair::Logger::SetConsoleSeverity(severity);
84  }
85 
86  string id;
87 
88  // check if config-key for config parser is provided
89  if (fVarMap.count("config-key"))
90  {
91  id = fVarMap["config-key"].as<string>();
92  }
93  else
94  {
95  id = fVarMap["id"].as<string>();
96  }
97 
98  // check if any config parser is selected
99  try
100  {
101  if (fVarMap.count("mq-config"))
102  {
103  LOG(debug) << "mq-config: Using default JSON parser";
104  Store(fair::mq::parser::JSON().UserParser(fVarMap["mq-config"].as<string>(), id));
105  }
106  else if (fVarMap.count("config-json-string"))
107  {
108  LOG(debug) << "config-json-string: Parsing JSON string";
109  string value = fair::mq::ConvertVariableValue<fair::mq::VarInfoToString>()(fVarMap.at("config-json-string"));
110  stringstream ss;
111  ss << value;
112  Store(fair::mq::parser::JSON().UserParser(ss, id));
113  }
114  else if (fVarMap.count("channel-config"))
115  {
116  LOG(debug) << "channel-config: Parsing channel configuration";
117  Store(fair::mq::parser::SUBOPT().UserParser(fVarMap.at("channel-config").as<vector<string>>(), id));
118  }
119  else
120  {
121  LOG(warn) << "FairMQProgOptions: no channels configuration provided via neither of:";
122  for (const auto& p : fMQParserOptions.options())
123  {
124  LOG(warn) << "--" << p->canonical_display_name();
125  }
126  LOG(warn) << "No channels will be created (You can create them manually).";
127  }
128  }
129  catch (std::exception& e)
130  {
131  LOG(error) << e.what();
132  return 1;
133  }
134 
135  FairProgOptions::PrintOptions();
136 
137  return 0;
138 }

166 // read FairMQChannelMap and insert/update corresponding values in variable map
167 // create key for variable map as follow : channelName.index.memberName
168 void FairMQProgOptions::UpdateMQValues()
169 {
170  for (const auto& p : fFairMQMap)
171  {
172  int index = 0;
173 
174  for (const auto& channel : p.second)
175  {
176  string typeKey = "chans." + p.first + "." + to_string(index) + ".type";
177  string methodKey = "chans." + p.first + "." + to_string(index) + ".method";
178  string addressKey = "chans." + p.first + "." + to_string(index) + ".address";
179  string transportKey = "chans." + p.first + "." + to_string(index) + ".transport";
180  string sndBufSizeKey = "chans." + p.first + "." + to_string(index) + ".sndBufSize";
181  string rcvBufSizeKey = "chans." + p.first + "." + to_string(index) + ".rcvBufSize";
182  string sndKernelSizeKey = "chans." + p.first + "." + to_string(index) + ".sndKernelSize";
183  string rcvKernelSizeKey = "chans." + p.first + "." + to_string(index) + ".rcvKernelSize";
184  string rateLoggingKey = "chans." + p.first + "." + to_string(index) + ".rateLogging";
185 
186  fMQKeyMap[typeKey] = MQKey{p.first, index, "type"};
187  fMQKeyMap[methodKey] = MQKey{p.first, index, "method"};
188  fMQKeyMap[addressKey] = MQKey{p.first, index, "address"};
189  fMQKeyMap[transportKey] = MQKey{p.first, index, "transport"};
190  fMQKeyMap[sndBufSizeKey] = MQKey{p.first, index, "sndBufSize"};
191  fMQKeyMap[rcvBufSizeKey] = MQKey{p.first, index, "rcvBufSize"};
192  fMQKeyMap[sndKernelSizeKey] = MQKey{p.first, index, "sndKernelSize"};
193  fMQKeyMap[rcvKernelSizeKey] = MQKey{p.first, index, "rcvkernelSize"};
194  fMQKeyMap[rateLoggingKey] = MQKey{p.first, index, "rateLogging"};
195 
196  UpdateVarMap<string>(typeKey, channel.GetType());
197  UpdateVarMap<string>(methodKey, channel.GetMethod());
198  UpdateVarMap<string>(addressKey, channel.GetAddress());
199  UpdateVarMap<string>(transportKey, channel.GetTransport());
200  UpdateVarMap<int>(sndBufSizeKey, channel.GetSndBufSize());
201  UpdateVarMap<int>(rcvBufSizeKey, channel.GetRcvBufSize());
202  UpdateVarMap<int>(sndKernelSizeKey, channel.GetSndKernelSize());
203  UpdateVarMap<int>(rcvKernelSizeKey, channel.GetRcvKernelSize());
204  UpdateVarMap<int>(rateLoggingKey, channel.GetRateLogging());
205 
206  index++;
207  }
208  UpdateVarMap<int>("chans." + p.first + ".numSockets", index);
209  }
210 }
