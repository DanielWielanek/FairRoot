38 // returns a map with network interface names as keys and their IP addresses as values
39 inline int getHostIPs(std::map<std::string, std::string>& addressMap)
40 {
41  struct ifaddrs *ifaddr, *ifa;
42  int s;
43  char host[NI_MAXHOST];
44 
45  if (getifaddrs(&ifaddr) == -1)
46  {
47  perror("getifaddrs");
48  return -1;
49  }
50 
51  for (ifa = ifaddr; ifa != NULL; ifa = ifa->ifa_next)
52  {
53  if (ifa->ifa_addr == NULL)
54  {
55  continue;
56  }
57 
58  if (ifa->ifa_addr->sa_family == AF_INET)
59  {
60  s = getnameinfo(ifa->ifa_addr, sizeof(struct sockaddr_in), host, NI_MAXHOST, NULL, 0, NI_NUMERICHOST);
61  if (s != 0)
62  {
63  std::cout << "getnameinfo() failed: " << gai_strerror(s) << std::endl;
64  return -1;
65  }
66 
67  addressMap.insert(std::pair<std::string, std::string>(ifa->ifa_name, host));
68  }
69  }
70  freeifaddrs(ifaddr);
71 
72  return 0;
73 }

93 {

99 #else // Linux

101 #endif
102 
103  if (!file)

106  return "";
107  }
108 
109  while (!feof(file.get()))

114  }

122  }

125  LOG(debug) << "Detected network interface name for the default route: " << interfaceName;
126  }
