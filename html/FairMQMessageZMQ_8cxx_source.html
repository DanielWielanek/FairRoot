151 // To emulate shrinking, a new message is created with the new size (ViewMsg), that points to the original buffer with the new size.
152 // Once the "view message" is transfered, the original is destroyed.
153 // Used size is applied only once in ApplyUsedSize, which is called by the socket before sending.
154 // This function just updates the desired size until the actual "resizing" happens.

170 void FairMQMessageZMQ::ApplyUsedSize()
171 {
172  // Apply only once (before actual send).
173  // The check is needed because a send could fail and can be reattempted by the user, in this case we do not want to modify buffer again.
174  if (fUsedSizeModified && !fViewMsg)
175  {
176  fViewMsg = fair::mq::tools::make_unique<zmq_msg_t>();
177  void* ptr = zmq_msg_data(fMsg.get());
178  if (zmq_msg_init_data(fViewMsg.get(),
179  ptr,
180  fUsedSize,
181  [](void* /* data */, void* obj)
182  {
183  zmq_msg_close(static_cast<zmq_msg_t*>(obj));
184  delete static_cast<zmq_msg_t*>(obj);
185  },
186  fMsg.release()) != 0)
187  {
188  LOG(error) << "failed initializing view message, reason: " << zmq_strerror(errno);
189  }
190  }
191 }
