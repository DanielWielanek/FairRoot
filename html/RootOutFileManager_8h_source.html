293  {

297 
298  std::vector<std::vector<DataType>> allObj;

304  }

308  }
309 
310  if (fTree)
311  {
312  if (fUseClonesArray)
313  {
314  fOutput = new TClonesArray(fClassName.c_str());
315  fTree->SetBranchAddress(fBranchName.c_str(), &fOutput);
316 
317  for (Long64_t i = 0; i < fTree->GetEntries(); i++)
318  {
319  tempObj.clear();
320  fTree->GetEntry(i);
321  for (Int_t iobj = 0; iobj < fOutput->GetEntriesFast(); ++iobj)
322  {
323  DataType* data = reinterpret_cast<DataType*>(fOutput->At(iobj));
324  if (!data)
325  {
326  continue;
327  }
328  tempObj.push_back(*data);
329  }
330  allObj.push_back(tempObj);
331  }
332  }
333  else
334  {
335  fTree->SetBranchAddress(branchname.c_str(), &fOutputData);
336  for (Long64_t i = 0; i < fTree->GetEntries(); i++)
337  {
338  fTree->GetEntry(i);
339  DataType data = *fOutputData;
340  tempObj.push_back(data);
341  }
342  allObj.push_back(tempObj);
343  }
344  }
345  else

347  LOG(error) << "Could not find tree " << treename.c_str();
348  }
349 
350  if (file)
351  {
352  file->Close();
353  }
354  return allObj;
