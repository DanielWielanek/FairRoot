68 
69  while (!fStop) // end thread condition (should exist until region is destroyed)
70  {
71  auto rcvTill = bpt::microsec_clock::universal_time() + bpt::milliseconds(200);
72  RegionBlock block;
73  if (fQueue->timed_receive(&block, sizeof(RegionBlock), recvdSize, priority, rcvTill))
74  {
75  // LOG(debug) << "received: " << block.fHandle << " " << block.fSize << " " << block.fMessageId;
76  if (fCallback)
77  {
78  fCallback(reinterpret_cast<char*>(fRegion.get_address()) + block.fHandle, block.fSize, reinterpret_cast<void*>(block.fHint));
79  }
80  }
81  else
82  {
83  // LOG(debug) << "queue " << fQueueName << " timeout!";
84  }
85  } // while !fStop
86
