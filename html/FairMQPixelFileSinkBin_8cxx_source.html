70  LOG(info) << "SHOULD CREATE THE FILE AND TREE";

72  fTreeName = "cbmsim";
73 
74  fOutFile = TFile::Open(fFileName.c_str(),fFileOption.c_str());
75 
76  fTree = new TTree(fTreeName.c_str(), "/cbmout");
77 
78  fFolder = new TFolder("cbmout", "Main Output Folder");
79  TFolder* foldEventHeader = fFolder->AddFolder("EvtHeader","EvtHeader");
80  TFolder* foldPixel = fFolder->AddFolder("Pixel","Pixel");
81 
82  TList* BranchNameList = new TList();
83 
84  for ( fNObjects = 0 ; fNObjects < fBranchNames.size() ; fNObjects++ ) {
85  if ( fClassNames[fNObjects].find("TClonesArray(") == 0 ) {
86  fClassNames [fNObjects] = fClassNames[fNObjects].substr(13,fClassNames[fNObjects].length()-12-2);
87  fOutputObjects [fNObjects] = new TClonesArray(fClassNames[fNObjects].c_str());
88  fTree->Branch(fBranchNames[fNObjects].c_str(),"TClonesArray", &fOutputObjects[fNObjects]);
89  foldPixel->Add(fOutputObjects[fNObjects]);
90  BranchNameList->AddLast(new TObjString(fBranchNames[fNObjects].c_str()));
91  }
92  else if ( fClassNames[fNObjects].find("FairEventHeader") == 0 ) {
93  fOutputObjects [fNObjects] = new FairEventHeader();
94  fTree->Branch(fBranchNames[fNObjects].c_str(),"FairEventHeader", &fOutputObjects[fNObjects]);
95  foldEventHeader->Add(fOutputObjects[fNObjects]);
96  BranchNameList->AddLast(new TObjString(fBranchNames[fNObjects].c_str()));
97  }
98  else {

100  }
101  }
102 
103  fFolder->Write();
104  BranchNameList->Write("BranchList", TObject::kSingleKey);

111 bool FairMQPixelFileSinkBin::StoreData(FairMQParts& parts, int /*index*/)
112 {
113  if ( parts.Size() == 0 ) return true; // probably impossible, but still check
114 
115  // expecting even number of parts in the form: header,data,header,data,header,data and so on...
116  int nPPE = 2; // nof parts per event
117 
118  if ( parts.Size()%nPPE >= 1 )
119  LOG(info) << "received " << parts.Size() << " parts, will ignore last part!!!";
120 
121  for ( int ievent = 0 ; ievent < parts.Size()/nPPE ; ievent++ ) {
122 
123  // the first part should be the event header
124  PixelPayload::EventHeader* payloadE = static_cast<PixelPayload::EventHeader*>(parts.At(nPPE*ievent)->GetData());
125  // LOG(debug) << "GOT EVENT " << payloadE->fMCEntryNo << " OF RUN " << payloadE->fRunId << " (part " << payloadE->fPartNo << ")";
126 
127  for ( unsigned int ibr = 0 ; ibr < fBranchNames.size() ; ibr++ )
128  {
129  if ( "EventHeader." == fBranchNames[ibr] )
130  {
131  ((FairEventHeader*)fOutputObjects[ibr])->SetRunId (payloadE->fRunId);
132  ((FairEventHeader*)fOutputObjects[ibr])->SetMCEntryNumber(payloadE->fMCEntryNo);
133  }
134  }
135 
136  // the second part should the TClonesArray with necessary data... now assuming Digi
137  PixelPayload::Hit* payloadH = static_cast<PixelPayload::Hit*>(parts.At(nPPE*ievent+1)->GetData());
138  int hitArraySize = parts.At(nPPE*ievent+1)->GetSize();
139  int nofHits = hitArraySize / sizeof(PixelPayload::Hit);
140 
141  for ( unsigned int ibr = 0 ; ibr < fBranchNames.size() ; ibr++ )
142  {
143  if ( "PixelHits" == fBranchNames[ibr] )
144  {
145  ((TClonesArray*)fOutputObjects[ibr])->Clear();
146  for ( int ihit = 0 ; ihit < nofHits ; ihit++ )
147  {
148  TVector3 pos (payloadH[ihit].posX, payloadH[ihit].posY, payloadH[ihit].posZ);
149  TVector3 posErr(payloadH[ihit].dposX,payloadH[ihit].dposY,payloadH[ihit].dposZ);
150  new ((*((TClonesArray*)fOutputObjects[ibr]))[ihit]) PixelHit(payloadH[ihit].fDetectorID,-1,pos,posErr);
151  // new ((*fHits)[fNHits]) PixelHit(detId,currentDigi->GetIndex(),pos,posErr);
152  }
153  }
154  }
155 
156  fTree->Fill();
157  }
158 
159  if ( fAckChannelName != "" ) {
160  unique_ptr<FairMQMessage> msg(NewMessage());
161  Send(msg, fAckChannelName);
162  }
163  return true;
164 }
