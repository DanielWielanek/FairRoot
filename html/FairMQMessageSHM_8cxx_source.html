128 bool FairMQMessageSHM::InitializeChunk(const size_t size)
129 {
130  while (fHandle < 0)
131  {
132  try
133  {
134  bipc::managed_shared_memory::size_type actualSize = size;
135  char* hint = 0; // unused for bipc::allocate_new
136  fLocalPtr = fManager.Segment().allocation_command<char>(bipc::allocate_new, size, actualSize, hint);
137  }
138  catch (bipc::bad_alloc& ba)
139  {
140  // LOG(warn) << "Shared memory full...";
141  this_thread::sleep_for(chrono::milliseconds(50));
142  if (fInterrupted)
143  {
144  return false;
145  }
146  else
147  {
148  continue;
149  }
150  }
151  fHandle = fManager.Segment().get_handle_from_address(fLocalPtr);
152  }
153 
154  fSize = size;
155 
156  if (zmq_msg_init_size(&fMessage, sizeof(MetaHeader)) != 0)
157  {
158  LOG(error) << "failed initializing meta message, reason: " << zmq_strerror(errno);
159  return false;
160  }
161  MetaHeader header;
162  header.fSize = size;
163  header.fHandle = fHandle;
164  header.fRegionId = fRegionId;
165  header.fHint = fHint;
166  memcpy(zmq_msg_data(&fMessage), &header, sizeof(MetaHeader));
167 
168  fMetaCreated = true;
169 
170  return true;
171 }

254 bool FairMQMessageSHM::SetUsedSize(const size_t size)
255 {
256  if (size == fSize)
257  {
258  return true;
259  }
260  else if (size <= fSize)
261  {
262  try
263  {
264 
265  bipc::managed_shared_memory::size_type shrunkSize = size;
266  fLocalPtr = fManager.Segment().allocation_command<char>(bipc::shrink_in_place, fSize + 128, shrunkSize, fLocalPtr);
267  fSize = size;
268 
269  // update meta header
270  MetaHeader* hdrPtr = static_cast<MetaHeader*>(zmq_msg_data(&fMessage));
271  hdrPtr->fSize = fSize;
272  return true;
273  }
274  catch (bipc::interprocess_exception& e)
275  {
276  LOG(info) << "could not set used size: " << e.what();
277  return false;
278  }
279  }
280  else
281  {
282  LOG(error) << "cannot set used size higher than original.";
283  return false;

338 void FairMQMessageSHM::CloseMessage()
339 {
340  if (fHandle >= 0 && !fQueued)
341  {
342  if (fRegionId == 0)
343  {
344  fManager.Segment().deallocate(fManager.Segment().get_address_from_handle(fHandle));
345  fHandle = 0;
346  }
347  else
348  {
349  // send notification back to the receiver
350  // RegionBlock block(fHandle, fSize);
351  // if (fManager.GetRegionQueue(fRegionId).try_send(static_cast<void*>(&block), sizeof(RegionBlock), 0))
352  // {
353  // // LOG(info) << "true";
354  // }
355  // // else
356  // // {
357  // // LOG(debug) << "could not send ack";
358  // // }
359 
360  // timed version
361  RegionBlock block(fHandle, fSize, fHint);
362  bool success = false;
363  do
364  {
365  auto sndTill = bpt::microsec_clock::universal_time() + bpt::milliseconds(200);
366  if (!fRegionPtr)
367  {
368  fRegionPtr = fManager.GetRemoteRegion(fRegionId);
369  }
370  if (fRegionPtr)
371  {
372  if (fRegionPtr->fQueue->timed_send(&block, sizeof(RegionBlock), 0, sndTill))
373  {
374  success = true;
375  }
376  else
377  {
378  if (fInterrupted)
379  {
380  break;
381  }
382  LOG(debug) << "region ack queue is full, retrying...";
383  }
384  }
385  else
386  {
387  // LOG(warn) << "region ack queue for id " << fRegionId << " no longer exist. Not sending ack";
388  success = true;
389  }
390  }
391  while (!success);
392  }
393  }
394 
395  if (fMetaCreated)
396  {
397  if (zmq_msg_close(&fMessage) != 0)
398  {
399  LOG(error) << "failed closing message, reason: " << zmq_strerror(errno);
400  }
401  }
