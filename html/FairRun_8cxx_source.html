85 void FairRun::StoreTaskNames(const FairTask* task)
86 {
87  if(task) {
88  fFileHeader->AddTaskClassName(task->ClassName());
89  fNTasks++;
90 
91  const TList* subtasks = task->GetListOfTasks();
92  if(subtasks) {
93  for(const auto subtask : *subtasks) {
94  StoreTaskNames(static_cast<FairTask*>(subtask));
95  }
96  }
97  }
98 }
