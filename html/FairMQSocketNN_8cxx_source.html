131 int FairMQSocketNN::SendImpl(FairMQMessagePtr& msg, const int flags, const int timeout)
132 {
133  int nbytes = -1;
134  int elapsed = 0;
135 
136  FairMQMessageNN* msgPtr = static_cast<FairMQMessageNN*>(msg.get());
137  void* bufPtr = msgPtr->GetMessage();
138 
139  while (true)
140  {
141  if (msgPtr->fRegionPtr == nullptr)
142  {
143  nbytes = nn_send(fSocket, &bufPtr, NN_MSG, flags);
144  }
145  else
146  {
147  nbytes = nn_send(fSocket, bufPtr, msg->GetSize(), flags);
148  // nn_send copies the data, safe to call region callback here
149  static_cast<FairMQUnmanagedRegionNN*>(msgPtr->fRegionPtr)->fCallback(bufPtr, msg->GetSize(), reinterpret_cast<void*>(msgPtr->fHint));
150  }
151 
152  if (nbytes >= 0)
153  {
154  fBytesTx += nbytes;
155  ++fMessagesTx;
156  static_cast<FairMQMessageNN*>(msg.get())->fReceiving = false;
157 
158  return nbytes;
159  }
160 #if NN_VERSION_CURRENT>2 // backwards-compatibility with nanomsg version<=0.6
161  else if (nn_errno() == ETIMEDOUT)
162 #else
163  else if (nn_errno() == EAGAIN)
164 #endif
165  {
166  if (!fInterrupted && ((flags & NN_DONTWAIT) == 0))
167  {
168  if (timeout)
169  {
170  elapsed += fSndTimeout;
171  if (elapsed >= timeout)
172  {
173  return -2;
174  }
175  }
176  continue;
177  }
178  else
179  {
180  return -2;
181  }
182  }
183  else if (nn_errno() == EAGAIN)
184  {
185  return -2;
186  }
187  else if (nn_errno() == ETERM)
188  {
189  LOG(info) << "terminating socket " << fId;
190  return -1;
191  }
192  else
193  {
194  LOG(error) << "Failed sending on socket " << fId << ", reason: " << nn_strerror(errno);
195  return nbytes;

261  const unsigned int vecSize = msgVec.size();
262 #ifdef MSGPACK_FOUND

268  msgpack::packer<msgpack::sbuffer> packer(&sbuf);
269 
270  // pack all parts into a single msgpack simple buffer

273  FairMQMessageNN* partPtr = static_cast<FairMQMessageNN*>(msgVec[i].get());

340 int64_t FairMQSocketNN::ReceiveImpl(vector<FairMQMessagePtr>& msgVec, const int flags, const int timeout)
341 {
342 #ifdef MSGPACK_FOUND
343  // Warn if the vector is filled before Receive() and empty it.
344  // if (msgVec.size() > 0)
345  // {
346  // LOG(warn) << "Message vector contains elements before Receive(), they will be deleted!";
347  // msgVec.clear();
348  // }
349 
350  int elapsed = 0;
351 
352  while (true)
353  {
354  // pointer to point to received message buffer
355  char* ptr = nullptr;
356  // receive the message into a buffer allocated by nanomsg and let ptr point to it
357  int nbytes = nn_recv(fSocket, &ptr, NN_MSG, flags);
358  if (nbytes >= 0) // if no errors or non-blocking timeouts
359  {
360  // store statistics on how many bytes received
361  fBytesRx += nbytes;
362  // store statistics on how many messages received (count messages instead of parts)
363  ++fMessagesRx;
364 
365  // offset to be used by msgpack to handle separate chunks
366  size_t offset = 0;
367  while (offset != static_cast<size_t>(nbytes)) // continue until all parts have been read
368  {
369  // vector of chars to hold blob (unlike char*/void* this type can be converted to by msgpack)
370  vector<char> buf;
371 
372  // unpack and convert chunk
373  msgpack::unpacked result;
374  unpack(result, ptr, nbytes, offset);
375  msgpack::object object(result.get());
376  object.convert(buf);
377  // get the single message size
378  size_t size = buf.size() * sizeof(char);
379  FairMQMessagePtr part(new FairMQMessageNN(size));
380  static_cast<FairMQMessageNN*>(part.get())->fReceiving = true;
381  memcpy(part->GetData(), buf.data(), size);
382  msgVec.push_back(move(part));
383  }
384 
385  nn_freemsg(ptr);
386  return nbytes;
387  }
388 #if NN_VERSION_CURRENT>2 // backwards-compatibility with nanomsg version<=0.6
389  else if (nn_errno() == ETIMEDOUT)
390 #else
391  else if (nn_errno() == EAGAIN)
392 #endif
393  {
394  if (!fInterrupted && ((flags & NN_DONTWAIT) == 0))
395  {
396  if (timeout)
397  {
398  elapsed += fRcvTimeout;
399  if (elapsed >= timeout)
400  {
401  return -2;
402  }
403  }
404  continue;
405  }
406  else
407  {
408  return -2;
409  }
410  }
411  else if (nn_errno() == EAGAIN)
412  {
413  return -2;
414  }
415  else if (nn_errno() == ETERM)
416  {
417  LOG(info) << "terminating socket " << fId;
418  return -1;
419  }
420  else
421  {
422  LOG(error) << "Failed receiving on socket " << fId << ", reason: " << nn_strerror(errno);
423  return nbytes;
424  }
425  }
426 #else /*MSGPACK_FOUND*/
427  LOG(error) << "Cannot receive message into vector of size " << msgVec.size() << " and flags " << flags << " with nanomsg multipart because MessagePack is not available.";
428  exit(EXIT_FAILURE);
