353 void FairEventManager::LoadXMLSettings() {
354  TDOMParser *Parser = new TDOMParser();
355  Parser->SetValidate(kFALSE);
356  Parser->ParseFile(fXMLConfig);
357  TXMLNode *MainNode = Parser->GetXMLDocument()->GetRootNode();
358  FairXMLNode *xml = new FairXMLNode(MainNode);
359  for(int i =0;i<xml->GetNChildren();i++){
360  TString nodename = xml->GetChild(i)->GetName();
361  if(nodename.EqualTo("Detectors")){
362  TGeoNode *top = gGeoManager->GetTopNode();
363  FairXMLNode *top_xml = xml->GetChild(i)->GetChild(0);
364  if(!top_xml->IsNull())
365  LoadXMLDetector(top,top_xml);
366  }else if(nodename.EqualTo("MCTracksColors")){
367  FairXMLNode *colors = xml->GetChild(i);
368  for(int j=0;j<colors->GetNChildren();j++){
369  FairXMLNode *color = colors->GetChild(j);
370  TString pgd_code = color->GetAttribValue("pdg");
371  TString color_code = color->GetAttribValue("color");
372  fPDGToColor[pgd_code.Atoi()] = StringToColor(color_code);
373  }
374  }
375  }
376  delete xml;
377  delete Parser;
378  gEve->Redraw3D();
379 }

381 void FairEventManager::LoadXMLDetector(TGeoNode* node, FairXMLNode* xml,Int_t depth) {
382  TString name = xml->GetAttribValue("name");
383  TString node_name = node->GetName();
384  Bool_t recursive = (xml->GetAttribValue("recursive").Length()!=0&&!name.EqualTo(node_name));
385  if(recursive&&depth==0) return;
386  TString transparency = xml->GetAttribValue("transparency");
387  TString color = xml->GetAttribValue("color");
388  if(!color.EqualTo("")){
389  node->GetVolume()->SetFillColor(StringToColor(color));
390  node->GetVolume()->SetLineColor(StringToColor(color));
391  }
392  if(!transparency.EqualTo("")){
393  node->GetVolume()->SetTransparency((Char_t)(transparency.Atoi()));
394  }
395  if(xml->GetAttribValue("recursive").Length()>0){
396  TString val = xml->GetAttribValue("recursive");
397  Int_t xml_depth = val.Atoi();
398  if(recursive){
399  xml_depth =depth-1;
400  }
401  for(int i=0;i<node->GetNdaughters();i++){
402  TGeoNode *daughter_node = node->GetDaughter(i);
403  LoadXMLDetector(daughter_node,xml,xml_depth);
404  }
405  }
406  if(xml->GetNChildren()>0&&!recursive){
407  for(int i=0;i<node->GetNdaughters();i++){
408  TString subdetector_name = node->GetDaughter(i)->GetName();
409  for(int j=0;j<xml->GetNChildren();j++){
410  FairXMLNode *subnode = xml->GetChild(j);
411  TString subnode_name = subnode->GetAttribValue("name");
412  if(subnode_name==subdetector_name){
413  LoadXMLDetector(node->GetDaughter(i),subnode);
414  }
415  }
416  }
417  }
