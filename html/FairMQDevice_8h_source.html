246  template<typename ...Ts>
247  FairMQPollerPtr NewPoller(const Ts&... inputs)
248  {
249  std::vector<std::string> chans{inputs...};
250 
251  // if more than one channel provided, check compatibility
252  if (chans.size() > 1)
253  {
254  FairMQ::Transport type = fChannels.at(chans.at(0)).at(0).Transport()->GetType();
255 
256  for (unsigned int i = 1; i < chans.size(); ++i)
257  {
258  if (type != fChannels.at(chans.at(i)).at(0).Transport()->GetType())
259  {
260  LOG(error) << "poller failed: different transports within same poller are not yet supported. Going to ERROR state.";
261  ChangeState(ERROR_FOUND);
262  }
263  }
264  }
265 
266  return fChannels.at(chans.at(0)).at(0).Transport()->CreatePoller(fChannels, chans);
267  }
