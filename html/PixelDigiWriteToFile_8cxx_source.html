84 
85  Reset();

87  Int_t nofDigis = fDigis->GetEntriesFast();
88 
89  for ( Int_t ifile = 0 ; ifile < fNofOutputFiles ; ifile++ ) {
90  fOutputFiles[ifile] << "EVENT BEGIN; RUNID = " << fRunId << "; MCENTRYNO = " << fMCEntryNo << "; PARTNO = " << ifile << ";" << std::endl;
91  }
92 
93  for ( Int_t iDigi = 0 ; iDigi < nofDigis ; iDigi++ ) {
94  PixelDigi* currentDigi = static_cast<PixelDigi*>(fDigis->At(iDigi));

101 
102  Int_t fileToSave = 0;
103  if ( fDivideLevel == 1 ) {
104  fileToSave = detId/256 - 1;

106  else if ( fDivideLevel == 2 ) {
107  fileToSave = (detId/256 - 1)*4 + (detId%256 -1);
108  }
109  fOutputFiles[fileToSave] << detId << " " << feId << " " << col << " " << row << " " << charge << std::endl;
110  }
111

113  fOutputFiles[ifile] << "EVENT END" << std::endl;
114  }
115 
116  fMCEntryNo ++;
117 }

126 InitStatus PixelDigiWriteToFile::Init() {
127 
128  // Get input array
129  FairRootManager* ioman = FairRootManager::Instance();
130 
131  if ( ! ioman ) LOG(fatal) << "No FairRootManager";
132  fDigis = static_cast<TClonesArray*>(ioman->GetObject("PixelDigis"));
133 
134  if ( !fDigis )
135  LOG(warn) << "PixelDigiWriteToFile::Init() No input PixelDigis array!";
136 
137  LOG(info) << "-I- " << fName.Data() << "::Init(). Initialization succesfull.";
138 
139  fRunId = ioman->GetRunId();
140 
141  if ( fDivideLevel == 0 ) {
142  fNofOutputFiles = 1;
143  fOutputFiles[0].open(fOutputFileName.Data(),std::fstream::out);
144  }
145  else {
146  if ( fDivideLevel == 1 ) {
147  fNofOutputFiles = 3; // 1 file per station (3 stations)
148  }
149  else if ( fDivideLevel == 2 ) {
150  fNofOutputFiles = 12; // 1 file per sensor (3 stations times 4 sensors)
151  }
152  else {
153  LOG(fatal) << "PixelDigiWriteToFile::Init(), fDivideLevel = " << fDivideLevel << " unknown, it has to be in the range <0,2>";
154  return kFATAL;
155  }
156  for ( Int_t ifile = 0 ; ifile < fNofOutputFiles ; ifile++ ) {
157  TString fileName = fOutputFileName;
158  TString uniqFile = Form(".p%d.",ifile);
159  fileName.Replace(fileName.Last('.'),1,uniqFile.Data());
160  fOutputFiles[ifile].open(fileName.Data(),std::fstream::out);
161  }
162  }
163 
164  return kSUCCESS;
165 
166 }
