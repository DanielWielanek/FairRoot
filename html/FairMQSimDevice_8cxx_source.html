75 void FairMQSimDevice::InitTask()
76 {
77  fRunSim = new FairRunSim();
78 
79  fRunSim->SetSink(fSink);
80 
81  if ( fFirstParameter || fSecondParameter ) {
82  FairRuntimeDb *rtdb=fRunSim->GetRuntimeDb();
83  if ( fFirstParameter )
84  rtdb->setFirstInput(fFirstParameter);
85  if ( fSecondParameter )
86  rtdb->setSecondInput(fSecondParameter);
87  }
88 
89  fRunSim->SetName(fTransportName.data());
90 
91  if ( fUserConfig.Length() > 0 )
92  fRunSim->SetUserConfig(fUserConfig);
93  if ( fUserCuts.Length() > 0 )
94  fRunSim->SetUserCuts(fUserCuts);
95  // ------------------------------------------------------------------------
96 
97  // ----- Create media -------------------------------------------------
98  fRunSim->SetMaterials(fMaterialsFile.data());
99  // ------------------------------------------------------------------------
100 
101  // ----- Magnetic field -------------------------------------------
102  if ( fMagneticField )
103  fRunSim->SetField(fMagneticField);
104  // --------------------------------------------------------------------
105 
106  // ----- Create geometry ----------------------------------------------
107  for ( int idet = 0 ; idet < fDetectorArray->GetEntries() ; idet++ ) {
108  fRunSim->AddModule((FairModule*)(fDetectorArray->At(idet)));
109  }
110  // ------------------------------------------------------------------------
111 
112  // ----- Negotiate the run number -------------------------------------
113  // ----- via the fUpdateChannelName --------------------------------
114  // ----- ask the fParamMQServer ------------------------------------
115  // ----- receive the run number and sampler id ---------------------
116  std::string* askForRunNumber = new string("ReportSimDevice");
117  FairMQMessagePtr req(NewMessage(const_cast<char*>(askForRunNumber->c_str()),
118  askForRunNumber->length(),
119  [](void* /*data*/, void* object) { delete static_cast<string*>(object); },
120  askForRunNumber));
121  std::unique_ptr<FairMQMessage> rep(NewMessage());
122 
123  unsigned int runId = 0;
124  if (Send(req, fUpdateChannelName) > 0)
125  {
126  if (Receive(rep, fUpdateChannelName) > 0)
127  {
128  std::string repString = string(static_cast<char*>(rep->GetData()), rep->GetSize());
129  LOG(INFO) << " -> " << repString.data();
130  runId = stoi(repString);
131  repString = repString.substr(repString.find_first_of('_')+1,repString.length());
132  fSimDeviceId = stoi(repString);
133  LOG(INFO) << "runId = " << runId << " /// fSimDeviceId = " << fSimDeviceId;
134  }
135  }
136  // ------------------------------------------------------------------------
137 
138  // ------------------------------------------------------------------------
139  if ( fPrimaryGenerator )
140  {
141  fPrimaryGenerator->SetEventNr(fSimDeviceId*fNofEvents); // run n simulations with same run id - offset the event number
142  fRunSim->SetGenerator(fPrimaryGenerator);
143  }
144  // ------------------------------------------------------------------------
145 
146  fRunSim->SetStoreTraj(fStoreTrajFlag);
147 
148  // ----- Set tasks ----------------------------------------------------
149  if ( fTaskArray ) {
150  for ( int itask = 0 ; itask < fTaskArray->GetEntries() ; itask++ ) {
151  fRunSim->AddTask((FairTask*)(fTaskArray->At(itask)));
152  }
153  }
154  // ------------------------------------------------------------------------
155 
156  // ----- Initialize simulation run ------------------------------------
157  fRunSim->SetRunId(runId); // run n simulations with same run id - offset the event number
158  fRunSim->Init();
159  // ------------------------------------------------------------------------
160 
161 }

227  if ( !CheckCurrentState(RUNNING) )

229  fRunSim->StopMCRun();

233  TObjString* ObjStr;
234 
235  for (auto& mi : fChannels)
236  {
237  LOG(DEBUG) << "trying channel >" << mi.first.data() << "<";
238 
239  FairMQParts parts;
240 
241  for(Int_t t=0; t<branchNameList->GetEntries(); t++)
242  {
243  ObjStr= static_cast<TObjString*>(branchNameList->TList::At(t));
244  LOG(DEBUG) << " branch >" << ObjStr->GetString().Data() << "<";
245  std::string modifiedBranchName = std::string("#") + ObjStr->GetString().Data() + "#";
246  if ( mi.first.find(modifiedBranchName) != std::string::npos || mi.first.find("#all#") != std::string::npos )
247  {
248  TObject* object = (FairRootManager::Instance()->GetObject(ObjStr->GetString()))->Clone();
249  TMessage* message = new TMessage(kMESS_OBJECT);
250  message->WriteObject(object);
251  parts.AddPart(NewMessage(message->Buffer(),
252  message->BufferSize(),
253  [](void* /*data*/, void* hint) { delete (TMessage*)hint;},
254  message));
255  LOG(DEBUG) << "channel >" << mi.first.data() << "< --> >" << ObjStr->GetString().Data() << "<";
256  }
257  }
258  if ( parts.Size() > 0 )
259  Send(parts,mi.first.data());
260  }
