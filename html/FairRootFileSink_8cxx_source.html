134 void FairRootFileSink::TruncateBranchNames()
135 {
146  TCollection* lf = fOutFolder->GetListOfFolders();
147  TIterator* iter= lf->MakeIterator();
148  TObjArray* Br= fOutTree->GetListOfBranches();
149  TIterator* BrIter= Br->MakeIterator();
150  TObject* obj;
151  TObject* BrObj;
152 
154  while((obj=iter->Next())) {
160  TString ffn=(fOutFolder->GetName());
161  ffn=ffn+".";
162  ffn=ffn+obj->GetName();
163  ffn=ffn+".";
164 
171  BrIter->Reset();
172 
173  while((BrObj=BrIter->Next())) {
174  TBranch* b=static_cast<TBranch*>(BrObj);
175  TruncateBranchNames(b, ffn);
176  }
177  }
178 
179  // Remove all occurence of FairMCEventHeader and from
180  // all branches containing that string.
181  // This is not the correct way to do it, but up tonow we don't understand
182  // why this part comes in when storing a derrived class from
183  // FairMCEventHeader or FairEventHeader.
184 
185  iter->Reset();
186  while((obj=iter->Next())) {
187 
188  TString ffn=".FairMCEventHeader";
189  BrIter->Reset();
190 
191  while((BrObj=BrIter->Next())) {
192  TBranch* b=static_cast<TBranch*>(BrObj);
193  TruncateBranchNames(b, ffn);
194  }
195  }
196 
197  iter->Reset();
198  while((obj=iter->Next())) {
199 
200  TString ffn=".FairEventHeader";
201  BrIter->Reset();
202 
203  while((BrObj=BrIter->Next())) {
204  TBranch* b=static_cast<TBranch*>(BrObj);
205  TruncateBranchNames(b, ffn);
206  }
207  }
208 
209  delete iter;
210  delete BrIter;
211 }

313 bool FairRootFileSink::CreatePersistentBranchesAny()
314 {
315  for(auto& iter : fPersistentBranchesMap)
316  {
317  LOG(INFO) << "CREATING BRANCH " << iter.first;
318  auto &tinfo = iter.second->persistenttypeinfo;
319  auto tname = impl::demangle(tinfo.name());
320 
321  // for the branch creation we need a TClass describing the object
322  // get it from ROOT via the type name
323  auto cl = TClass::GetClass(tinfo);
324  if(!cl) {
325  LOG(FATAL) << "No TClass found for " << tname << "\n";
326  return false;
327  }
328 
329  if(!cl->HasDictionary()) {
330  LOG(FATAL) << "No dictionary found for " << tname << "\n";
331  return false;
332  }
333 
334  // create the branch
335  auto obj = iter.second->ptraddr;
336 
337  LOG(INFO) << "Creating branch for " << iter.first.c_str() << " with address " << obj;
338  fOutTree->Branch(iter.first.c_str(), tname.c_str(), obj);
339  }
340  return true;
341 }
