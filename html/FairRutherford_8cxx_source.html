97 Bool_t FairRutherford::ProcessHits(FairVolume* vol)
98 {
101  //Set parameters at entrance of volume. Reset ELoss.
102  if ( TVirtualMC::GetMC()->IsTrackEntering() ) {
103  fELoss = 0.;
104  fTime = TVirtualMC::GetMC()->TrackTime() * 1.0e09;
105  fLength = TVirtualMC::GetMC()->TrackLength();
106  TVirtualMC::GetMC()->TrackPosition(fPos);
107  TVirtualMC::GetMC()->TrackMomentum(fMom);
108  }
109 
110  // Sum energy loss for all steps in the active volume
111  fELoss += TVirtualMC::GetMC()->Edep();
112 
113  // Create FairRutherfordPoint at exit of active volume
114  if ( TVirtualMC::GetMC()->IsTrackExiting() ||
115  TVirtualMC::GetMC()->IsTrackStop() ||
116  TVirtualMC::GetMC()->IsTrackDisappeared() ) {
117  fTrackID = TVirtualMC::GetMC()->GetStack()->GetCurrentTrackNumber();
118  fVolumeID = vol->getMCid();
119  if (fELoss == 0. ) { return kFALSE; }
120  AddHit(fTrackID, fVolumeID, TVector3(fPos.X(), fPos.Y(), fPos.Z()),
121  TVector3(fMom.Px(), fMom.Py(), fMom.Pz()), fTime, fLength,
122  fELoss);
123 
124  // Increment number of FairRutherford points in TParticle
125  FairStack* stack = static_cast<FairStack*>(TVirtualMC::GetMC()->GetStack());
126  stack->AddPoint(kFairRutherford);
127  }
128 
129  return kTRUE;
130 }

203 void FairRutherford::SetSensitiveVolumes()
204 {
205  TList* volList = fgGeo->getListOfVolumes();
206 
207  // store geo parameter
208  FairRuntimeDb* rtdb= FairRun::Instance()->GetRuntimeDb();
209  FairRutherfordGeoPar* par=static_cast<FairRutherfordGeoPar*>(rtdb->getContainer("FairRutherfordGeoPar"));
210  TObjArray* sensNodes = par->GetGeoSensitiveNodes();
211  TObjArray* passNodes = par->GetGeoPassiveNodes();
212 
213  TListIter iter(volList);
214  FairGeoNode* node = NULL;
215  FairGeoVolume* aVol=NULL;
216 
217  while( (node = static_cast<FairGeoNode*>(iter.Next())) ) {
218  aVol = dynamic_cast<FairGeoVolume*> ( node );
219  if ( node->isSensitive() ) {
220  sensNodes->AddLast( aVol );
221  } else {
222  passNodes->AddLast( aVol );
223  }
224  }
225  par->setChanged();
226  par->setInputVersion(FairRun::Instance()->GetRunId(),1);
227 
228  ProcessNodes ( volList );
229 }
