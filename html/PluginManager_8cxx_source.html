83 auto fair::mq::PluginManager::MakeFromCommandLineOptions(const vector<string> args) -> shared_ptr<PluginManager>
84 {
85  // Parse command line options
86  auto options = ProgramOptions();
87  auto vm = po::variables_map{};
88  try
89  {
90  auto parsed = po::command_line_parser(args).options(options).allow_unregistered().run();
91  po::store(parsed, vm);
92  po::notify(vm);
93  } catch (const po::error& e)
94  {
95  throw ProgramOptionsParseError{ToString("Error occured while parsing the 'Plugin Manager' program options: ", e.what())};
96  }
97 
98  // Process plugin search paths
99  auto append = vector<fs::path>{};
100  auto prepend = vector<fs::path>{};
101  auto searchPaths = vector<fs::path>{};
102  if (vm.count("plugin-search-path"))
103  {
104  for (const auto& path : vm["plugin-search-path"].as<vector<string>>())
105  {
106  if (path.substr(0, 1) == "<") { prepend.emplace_back(path.substr(1)); }
107  else if (path.substr(0, 1) == ">") { append.emplace_back(path.substr(1)); }
108  else { searchPaths.emplace_back(path); }
109  }
110  }
111 
112  // Create PluginManager with supplied options
113  auto mgr = make_shared<PluginManager>();
114  mgr->SetSearchPaths(searchPaths);
115  for(const auto& path : prepend) { mgr->PrependSearchPath(path); }
116  for(const auto& path : append) { mgr->AppendSearchPath(path); }
117  if (vm.count("plugin")) { mgr->LoadPlugins(vm["plugin"].as<vector<string>>()); }
118 
119  // Return the plugin manager and command line options, that have not been recognized.
120  return mgr;
121 }

123 auto fair::mq::PluginManager::LoadPlugin(const string& pluginName) -> void
124 {
125  if (pluginName.substr(0,2) == "p:")
126  {
127  // Mechanism A: prelinked dynamic
128  LoadPluginPrelinkedDynamic(pluginName.substr(2));
129  }
130  else if (pluginName.substr(0,2) == "d:")
131  {
132  // Mechanism B: dynamic
133  LoadPluginDynamic(pluginName.substr(2));
134  }
135  else if (pluginName.substr(0,2) == "s:")
136  {
137  // Mechanism C: static (builtin)
138  LoadPluginStatic(pluginName.substr(2));
139  }
140  else
141  {
142  // Mechanism B: dynamic (default)
143  LoadPluginDynamic(pluginName);
144  }
145 }

164 auto fair::mq::PluginManager::LoadPluginDynamic(const string& pluginName) -> void
165 {
166  // Search plugin and load, if found
167  if (fPluginFactories.find(pluginName) == fPluginFactories.end())
168  {
169  auto success = false;
170  for(const auto& searchPath : SearchPaths())
171  {
172  try
173  {
174  LoadSymbols(
175  pluginName,
176  searchPath / ToString(LibPrefix(), pluginName),
177  dll::load_mode::append_decorations
178  );
179 
180  fPluginOrder.push_back(pluginName);
181  success = true;
182  break;
183  }
184  catch (boost::system::system_error& e)
185  {
186  if(string{e.what()}.find("No such file or directory") == string::npos)
187  {
188  throw PluginLoadError(ToString("An error occurred while loading dynamic plugin ", pluginName, ": ", e.what()));
189  }
190  }
191  }
192  if(!success) { throw PluginLoadError(ToString("The plugin ", pluginName, " could not be found in the plugin search paths.")); }
193  }
194 }
