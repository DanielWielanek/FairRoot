103 // ----- Public method Exec --------------------------------------------
104 void PixelAltFindHits::Exec(Option_t* /*opt*/) {
105  Reset();
106 
107  LOG(debug) << "PixelAltFindHits::Exec() EVENT " << fTNofEvents;
108 
109  fTNofEvents++;
110 
111  fNDigis = fDigis->GetEntriesFast();
112  fTNofDigis+= fNDigis;
113 
114  for ( Int_t iDigi = 0 ; iDigi < fNDigis ; iDigi++ ) {
115  PixelDigi* currentDigi = static_cast<PixelDigi*>(fDigis->At(iDigi));
116 
117  Int_t detId = currentDigi->GetDetectorID();
118 
119  Double_t posX, posY, posZ;
120  Double_t errX, errY, errZ;
121  Int_t hitIndex = FindHit(detId,currentDigi->GetFeID(),currentDigi->GetCol(),currentDigi->GetRow(),currentDigi->GetIndex(),
122  posX,posY,posZ,errX,errY,errZ);
123 
124  TVector3 pos (posX,posY,posZ);
125  TVector3 posErr(errX,errY,errZ);
126 
127  if ( hitIndex == -666 ) continue; // hit not found
128 
129  new ((*fHits)[fNHits]) PixelHit(detId,hitIndex,pos,posErr);
130 
131  fNHits++;
132  }
133 
134  fTNofHits += fNHits;
135 }

138 // ----- Private method FindHit ----------------------------------------
139 // should return -666 if hit not found
140 Int_t PixelAltFindHits::FindHit(Int_t detId, Int_t feId, Int_t col, Int_t row, Int_t index,
141  Double_t& posX, Double_t& posY, Double_t& posZ,
142  Double_t& errX, Double_t& errY, Double_t& errZ) {
143  TString nodeName = Form("/cave/Pixel%d_%d",detId/256,detId%256);
144 
145  gGeoManager->cd(nodeName.Data());
146  TGeoNode* curNode = gGeoManager->GetCurrentNode();
147  TGeoVolume* actVolume = gGeoManager->GetCurrentVolume();
148  TGeoBBox* actBox = static_cast<TGeoBBox*>(actVolume->GetShape());
149 
150  Double_t locPosCalc[3];
151  locPosCalc[0] = ( ((feId-1)/fMaxFEperCol)*fFeCols + col + 0.5 )*fPitchX;
152  locPosCalc[1] = ( ((feId-1)%fMaxFEperCol)*fFeRows + row + 0.5 )*fPitchY;
153  locPosCalc[2] = 0.;
154 
155  locPosCalc[0] -= actBox->GetDX();
156  locPosCalc[1] -= actBox->GetDY();
157 
158  Double_t globPos[3];
159 
160  curNode->LocalToMaster(locPosCalc,globPos);
161 
162  LOG(debug) << "HIT ON " << detId << " POSITION: " << locPosCalc[0] << " / " << locPosCalc[1];
163  LOG(debug) << "GLOB HIT " << detId << " POSITION: " << globPos[0] << " / " << globPos[1] << " / " << globPos[2];
164 
165  posX = globPos[0];
166  posY = globPos[1];
167  posZ = globPos[2];
168  errX = fPitchX/TMath::Sqrt(12.);
169  errY = fPitchY/TMath::Sqrt(12.);
170  errZ = actBox->GetDZ();
171 
172 // should return -666 if hit not found
173  return index;
174 }
