94 void FairTestDetectorHitProducerSmearing::Exec(Option_t* option)
95 {
96  LOG(DEBUG) << "Exec of FairTestDetectorHitProducerSmearing"
97  << FairLogger::endl;
98 
99  fHitsArray->Delete();
100 
101  // fill the map
102  FairTestDetectorPoint* point = NULL;
103  FairTestDetectorHit* hit = NULL;
104  for (int iPoint = 0; iPoint < fPointsArray->GetEntriesFast(); iPoint++)
105  {
106  point = (FairTestDetectorPoint*)fPointsArray->At(iPoint);
107  if (!point)
108  {
109  continue;
110  }
111 
112  TVector3 position;
113  point->Position(position);
114 
115  TVector3 dposition;
116 
117  // simulate the detector response: smear position with gaussians
118  dposition.SetXYZ(0.05, 0.05, 0.);
119  position.SetXYZ(gRandom->Gaus(position.X(), 0.05), gRandom->Gaus(position.Y(), 0.05), position.Z());
120 
121  hit = new ((*fHitsArray)[iPoint]) FairTestDetectorHit(point->GetDetectorID(), iPoint, position, dposition);
122  }
123 }
