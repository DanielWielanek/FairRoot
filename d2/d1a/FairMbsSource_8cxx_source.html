56  FairUnpack *unpack;
57  Bool_t seen = kFALSE;
58  for (Int_t i = 0; i < fUnpackers->GetEntriesFast(); i++) {
59  unpack = (FairUnpack *)fUnpackers->At(i);
60 
61  if (unpack->GetSubCrate() < 0) { // All sub-crates
62  if (type != unpack->GetType() || subType != unpack->GetSubType() ||
63  procId != unpack->GetProcId() || control != unpack->GetControl()) {
64  continue;
65  }
66  } else { // specified sub-crate
67  if (type != unpack->GetType() || subType != unpack->GetSubType() ||
68  procId != unpack->GetProcId() || subCrate != unpack->GetSubCrate() ||
69  control != unpack->GetControl()) {
70  continue;
71  }
72  }
73 
74  if (!unpack->DoUnpack(data, size)) {
75  return kFALSE;
76  }
77 
78  seen = kTRUE;
79  }
