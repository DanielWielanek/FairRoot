93 // ----- Accessor to momentum -------------------------------------------
94 void FairTrackParam::Momentum(TVector3& mom) const
95 {
96  Double_t p = ( TMath::Abs(fQp)>1.e-4 ) ?1./TMath::Abs(fQp) :1.e4;
97  Double_t pz = TMath::Sqrt( p*p / ( fTx*fTx + fTy*fTy + 1 ) );
98  Double_t px = fTx * pz;
99  Double_t py = fTy * pz;
100  mom.SetXYZ(px, py, pz);
101 
102 }

132 // ----- Accessor to covariance matrix elements ------------------------
133 Double_t FairTrackParam::GetCovariance(Int_t i, Int_t j) const
134 {
135  if ( i<0 || j<0 || i>4 || j>4 ) {
136  LOG(ERROR) << "FairTrackParam::GetCovariance: Invalid index pair ("
137  << i << "," << j << ") !" << FairLogger::endl;
138  return 0;
139  }
140  if (i>j) {
141  Int_t k = i;
142  i = j;
143  j = k;
144  }
145  Int_t index = 0;
146  if (i==0) { index = j; }
147  else if (i==1) { index = 4 + j; }
148  else if (i==2) { index = 7 + j; }
149  else if (i==3) { index = 9 + j; }
150  else if (i==4) { index = 10 + j; }
151  return fCovMatrix[index];
152 }
