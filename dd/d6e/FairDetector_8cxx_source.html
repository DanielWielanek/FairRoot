87 void FairDetector::Initialize()
88 {
89 // Registers hits collection in Root manager;
90 // sets sensitive volumes.
91 // ---
92  Int_t NoOfEntries=svList->GetEntries();
93  Int_t fMCid;
94  FairGeoNode* fN;
95  TString cutName;
96  TString copysign="#";
97  for (Int_t i = 0 ; i < NoOfEntries ; i++ ) {
98  FairVolume* aVol = (FairVolume*) svList->At(i);
99  cutName = aVol->GetName();
100  Ssiz_t pos = cutName.Index (copysign, 1);
101  if(pos>1) { cutName.Resize(pos); }
102  if ( aVol->getModId() == GetModId() ) {
103  fMCid=gMC->VolId(cutName.Data());
104  aVol->setMCid(fMCid);
105  fN=aVol->getGeoNode();
106  if (fN) { fN->setMCid(fMCid); }
107  }
108  }
109 
110 }
