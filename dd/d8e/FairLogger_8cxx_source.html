164 void FairLogger::Log(FairLogLevel level, const char* file, const char* line,
165  const char* func, const char* format, va_list arglist)
166 {
167  // If the format string together with the argument list was used once it can't
168  // be used another time. Don't know why but if we do so the arguments are not
169  // correct any longer and the program may crash (On Mac OSX). Even if it does
170  // not crash it does not work as expected.
171  // To vercome the problem the output is written to a buffer which then can be
172  // used several times.
173 
174  if(fLogToFile && !fLogFileOpen) {
175  OpenLogFile();
176  }
177 
178  fLevel = level;
179  while (1) {
180  fBufferSizeNeeded = vsnprintf(fBufferPointer, fBufferSize, format, arglist);
181 
182  // Try to vsnprintf into our buffer.
183  // int needed = vsnprintf (buf, size, fmt, ap);
184  // NB. C99 (which modern Linux and OS X follow) says vsnprintf
185  // failure returns the length it would have needed. But older
186  // glibc and current Windows return -1 for failure, i.e., not
187  // telling us how much was needed.
188 
189  if (fBufferSizeNeeded <= static_cast<int>(fBufferSize) && fBufferSizeNeeded >= 0) {
190  // It fit fine so we're done.
191  GetOutputStream(level, file, line, func) <<
192  std::string(fBufferPointer, static_cast<size_t>(fBufferSizeNeeded))<<
193  " " << FairLogger::endl;
194 
195  break;
196  }
197 
198  // vsnprintf reported that it wanted to write more characters
199  // than we allotted. So try again using a dynamic buffer. This
200  // doesn't happen very often if we chose our initial size well.
201  fBufferSize = (fBufferSizeNeeded > 0)
202  ? (fBufferSizeNeeded+1) : (fBufferSize*2);
203 
204  fDynamicBuffer.resize (fBufferSize);
205  }
206 }

240 {
241  // Check if the log file name is set. If not use a default logfile with
242  // the name FairLogfile_<pid>.log. <pid> is the pid of the runing job.
243  TString logfile = fLogFileName;
244  if (logfile.IsNull()) {
245  Int_t PID = gSystem->GetPid();
246  logfile = "FairLogfile_";
247  logfile += PID;
248  logfile += ".log";
249 
250  }
251 
252  // Check if filename contains a full path. If not create logfile
253  // in the current directory.
254  // TODO(F.U.): Check if the path exists and is writeable.
255  if (!logfile.Contains('/')) {
256  logfile = "./" + logfile;
257  }
258 
259  if (fFileStream) {
260  CloseLogFile();
261  delete fFileStream;
262  fFileStream = NULL;
263  }
264 
265  fFileStream = new std::ofstream(logfile.Data());
266 
267  fLogFileOpen = kTRUE;

270 FairLogLevel FairLogger::ConvertToLogLevel(const char* levelc) const
271 {
272  // Convert string to upper case
273  // Return the according log level
274  // in case the level is not known return info level
275  TString level = levelc;
276  level.ToUpper();
277  if (level == "FATAL") { return FATAL; }
278  if (level == "ERROR") { return ERROR; }
279  if (level == "WARNING") { return WARNING; }
280  if (level == "INFO") { return INFO; }
281  if (level == "DEBUG") { return DEBUG; }
282  if (level == "DEBUG1") { return DEBUG1; }
283  if (level == "DEBUG2") { return DEBUG2; }
284  if (level == "DEBUG3") { return DEBUG3; }
285  if (level == "DEBUG4") { return DEBUG4; }
286  LOG(ERROR)<<"Log level \""<<level<<"\" not supported. Use default level \"INFO\"."<<FairLogger::endl;
287  return INFO;

520 void FairLogger::LogFatalMessage(std::ostream& strm)
521 {
522  // Since Fatal indicates a fatal error it is maybe usefull to have
523  // system information from the incident. Since the output on the screen
524  // does not help the noraml user, the stderr is redirected into a special
525  // core dump file for later usage.
526  // Fatal also indicates a problem which is so severe that the process should
527  // not go on, so the process is aborted.
528 
529  *fScreenStream << LogLevelColor[FATAL];
530 
531  *fScreenStream << "[" << std::setw(7) << std::left << LogLevelString[FATAL] <<"] ";
532 
533  GetTime();
534  *fScreenStream << fTimeBuffer;
535 
536  *fScreenStream << "We stop the execution of the process at this point.\n";
537 
538  if ( fLogToFile ) {
539  *fFileStream << "[" << std::setw(7) << std::left << LogLevelString[FATAL] <<"] ";
540 
541  *fFileStream << fTimeBuffer;
542  *fFileStream << "We stop the execution of the process at this point.\n";
543  }
544 
545  if (gSystem) {
546  TString corefile = "core_dump_";
547  Int_t PID = gSystem->GetPid();
548  corefile += PID;
549 
550 
551  *fScreenStream << LogLevelColor[FATAL];
552 
553  *fScreenStream << "[" << std::setw(7) << std::left << LogLevelString[FATAL] <<"] ";
554 
555  GetTime();
556  *fScreenStream << fTimeBuffer;
557 
558  *fScreenStream << "For later analysis we write a core dump to " <<
559  corefile << " \n";
560 
561 
562  if ( fLogToFile ) {
563  *fFileStream << "[" << std::setw(7) << std::left << LogLevelString[FATAL] <<"] ";
564 
565  *fFileStream << fTimeBuffer;
566  *fFileStream << "For later analysis we write a core dump to" <<
567  corefile << "\n";
568  }
569 
570  // reset format to default before exiting
571  *(gLogger->fScreenStream) << "\33[0m" << std::endl;
572 
573  flush(strm);
574  freopen(corefile, "w", stderr);
575  gSystem->StackTrace();
576  fclose(stderr);
577  gSystem->Abort(1);
578  } else {
579  abort();
580  }
581 
582 }
