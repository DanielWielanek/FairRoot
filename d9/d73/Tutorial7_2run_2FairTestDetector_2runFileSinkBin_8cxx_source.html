163 int main(int argc, char** argv)
164 {
165  s_catch_signals();
166  DeviceOptions_t options;
167  try
168  {
169  if (!parse_cmd_line(argc, argv, &options))
170  return 0;
171  }
172  catch (exception& e)
173  {
174  MQLOG(ERROR) << e.what();
175  return 1;
176  }
177 
178  MQLOG(INFO) << "PID: " << getpid();
179 
180 #ifdef NANOMSG
181  FairMQTransportFactory* transportFactory = new FairMQTransportFactoryNN();
182 #else
183  FairMQTransportFactory* transportFactory = new FairMQTransportFactoryZMQ();
184 #endif
185 
186  filesink.SetTransport(transportFactory);
187 
188  filesink.SetProperty(TSink::Id, options.id);
189  filesink.SetProperty(TSink::NumIoThreads, options.ioThreads);
190 
191  filesink.SetProperty(TSink::NumInputs, 1);
192  filesink.SetProperty(TSink::NumOutputs, 0);
193 
194  filesink.InitInputContainer( options.classname.c_str() );
195  filesink.SetFileProperties(options.filename,options.treename,options.branchname,options.classname,
196  options.fileoption,options.useTClonesArray);
197 
198 
199  filesink.ChangeState(TSink::INIT);
200  filesink.SetProperty(TSink::InputSocketType, options.inputSocketType);
201  filesink.SetProperty(TSink::InputRcvBufSize, options.inputBufSize);
202  filesink.SetProperty(TSink::InputMethod, options.inputMethod);
203  filesink.SetProperty(TSink::InputAddress, options.inputAddress);
204 
205  filesink.ChangeState(TSink::SETOUTPUT);
206  filesink.ChangeState(TSink::SETINPUT);
207  filesink.ChangeState(TSink::BIND);
208  filesink.ChangeState(TSink::CONNECT);
209  filesink.ChangeState(TSink::RUN);
210 
211  // wait until the running thread has finished processing.
212  boost::unique_lock<boost::mutex> lock(filesink.fRunningMutex);
213  while (!filesink.fRunningFinished)
214  {
215  filesink.fRunningCondition.wait(lock);
216  }
217 
218  filesink.ChangeState(TSink::STOP);
219  filesink.ChangeState(TSink::END);
220 
221  return 0;
222 }
