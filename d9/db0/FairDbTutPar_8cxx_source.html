137 void FairDbTutPar::fill(UInt_t rid)
138 {
139  // Get Reader Meta Class
140  fParam_Reader=GetParamReader();
141 
142  // Define a Context
143  ValTimeStamp ts(rid);
144  ValCondition context(FairDbDetector::kGfi,DataType::kData,ts);
145 
146  // Activate reading for this Context
147  fParam_Reader->Activate(context, GetVersion());
148 
149  // By default use the latest row entry
150  // (Other rows would correspond to outdated versions)
151  Int_t numRows = fParam_Reader->GetNumRows();
152  if ( numRows > 1 ) { numRows = 1; }
153 
154  for (int i = 0; i < numRows; ++i) {
155  FairDbTutPar* cgd = (FairDbTutPar*) fParam_Reader->GetRow(i);
156  if (!cgd) { continue; }
157  fTopPitch = cgd->GetTopPitch();
158  fTopAnchor = cgd->GetTopAnchor();
159  fTopNrFE = cgd->GetNrTopFE();
160  fFeType = cgd->GetFeType();
161  }

166 void FairDbTutPar::store(UInt_t rid)
167 {
168  // Boolean IO test variable
169  Bool_t fail= kFALSE;
170 
171  // Create a unique statement on choosen DB entry
172  auto_ptr<FairDbStatement> stmtDbn(fMultConn->CreateStatement(GetDbEntry()));
173  if ( ! stmtDbn.get() ) {
174  cout << "-E- FairDbTutPar::Store() Cannot create statement for Database_id: " << GetDbEntry()
175  << "\n Please check the FAIRDB_TSQL_* environment. Quitting ... " << endl;
176  exit(1);
177  }
178 
179  // Check if for this DB entry the table already exists.
180  // If not call the corresponding Table Definition Function
181  std::vector<std::string> sql_cmds;
182  TString atr(GetName());
183  atr.ToUpper();
184 
185  if (! fMultConn->GetConnection(GetDbEntry())->TableExists("FAIRDBTUTPAR") ) {
186  sql_cmds.push_back(FairDb::GetValDefinition("FAIRDBTUTPAR").Data());
187  sql_cmds.push_back(FairDbTutPar::GetTableDefinition());
188  }
189 
190  // Packed SQL commands executed internally via SQL processor
191  std::vector<std::string>::iterator itr(sql_cmds.begin()), itrEnd(sql_cmds.end());
192  while( itr != itrEnd ) {
193  std::string& sql_cmd(*itr++);
194  stmtDbn->Commit(sql_cmd.c_str());
195  if ( stmtDbn->PrintExceptions() ) {
196  fail = true;
197  cout << "-E- FairDbTutPar::Store() ******* Error Executing SQL commands *********** " << endl;
198  }
199 
200  }
201 
202  // Refresh list of tables in connected database
203  // for the choosen DB entry
204  fMultConn->GetConnection(GetDbEntry())->SetTableExists();
205 
206  // Writer Meta-Class Instance
207  fParam_Writer = GetParamWriter();
208 
209 
210  // Activate Writer Meta-Class with the proper
211  // Validity Time Interval (run_id encapsulated)
212 
213  // Writer Activate() arguments list
214  // <Arguments> <Type> <Comments>
215  //
216  // Validity Interval, ValInterval
217  // Composition , Int_t set via cont. factory
218  // Version , Int_t set via cont. factory
219  // DbEntry , Int_t set via cont. factory
220  // LogTitle , std::string set via cont. factory
221 
222  fParam_Writer->Activate(GetValInterval(rid),GetComboNo(), GetVersion(),GetDbEntry(),GetLogTitle());
223 
224  // Object to Table mapping
225  *fParam_Writer<< (*this);
226 
227  // Check for eventual IO problems
228  if ( !fParam_Writer->Close() ) {
229  fail = true;
230  cout << "-E- FairDbTutPar::Store() ******** Cannot do IO on class: " << GetName() << endl;
231  }
232
